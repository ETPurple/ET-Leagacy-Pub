389 static void DynamicLightSinglePass(void)
390 {
391  int i, l, a, b, c, color, *intColors;
392  vec3_t origin;
393  byte *colors;
394  unsigned hitIndexes[SHADER_MAX_INDEXES];
395  int numIndexes;
396  float radius, radiusInverseCubed;
397  float intensity, remainder, modulate;
398  vec3_t floatColor, dir;
399  dlight_t *dl;
400 
401  // early out
402  if (backEnd.refdef.num_dlights == 0)
403  {
404  return;
405  }
406 
407  // clear colors
408  Com_Memset(tess.svars.colors, 0, sizeof(tess.svars.colors));
409 
410  // walk light list
411  for (l = 0; l < backEnd.refdef.num_dlights; l++)
412  {
413  // early out
414  if (!(tess.dlightBits & (1 << l)))
415  {
416  continue;
417  }
418 
419  // setup
420  dl = &backEnd.refdef.dlights[l];
421  VectorCopy(dl->transformed, origin);
422  radius = dl->radius;
423  radiusInverseCubed = dl->radiusInverseCubed;
424  intensity = dl->intensity;
425  floatColor[0] = dl->color[0] * 255.0f;
426  floatColor[1] = dl->color[1] * 255.0f;
427  floatColor[2] = dl->color[2] * 255.0f;
428 
429  // directional lights have max intensity and washout remainder intensity
430  if (dl->flags & REF_DIRECTED_DLIGHT)
431  {
432  remainder = intensity * 0.125f;
433  }
434  else
435  {
436  remainder = 0.0f;
437  }
438 
439  // illuminate vertexes
440  colors = tess.svars.colors[0];
441  for (i = 0; i < tess.numVertexes; i++, colors += 4)
442  {
443  backEnd.pc.c_dlightVertexes++;
444 
445  // directional dlight, origin is a directional normal
446  if (dl->flags & REF_DIRECTED_DLIGHT)
447  {
448  // twosided surfaces use absolute value of the calculated lighting
449  modulate = intensity * DotProduct(dl->origin, tess.normal[i]);
450  if (tess.shader->cullType == CT_TWO_SIDED)
451  {
452  modulate = Q_fabs(modulate);
453  }
454  modulate += remainder;
455  }
456  // ball dlight
457  else
458  {
459  dir[0] = radius - Q_fabs(origin[0] - tess.xyz[i][0]);
460  if (dir[0] <= 0.0f)
461  {
462  continue;
463  }
464  dir[1] = radius - Q_fabs(origin[1] - tess.xyz[i][1]);
465  if (dir[1] <= 0.0f)
466  {
467  continue;
468  }
469  dir[2] = radius - Q_fabs(origin[2] - tess.xyz[i][2]);
470  if (dir[2] <= 0.0f)
471  {
472  continue;
473  }
474 
475  modulate = intensity * dir[0] * dir[1] * dir[2] * radiusInverseCubed;
476  }
477 
478  // optimizations
479  if (modulate < (1.0f / 128.0f))
480  {
481  continue;
482  }
483  else if (modulate > 1.0f)
484  {
485  modulate = 1.0f;
486  }
487 
488  // add to color
489  color = colors[0] + (int)(floatColor[0] * modulate);
490  colors[0] = color > 255 ? 255 : (byte)color;
491  color = colors[1] + (int)(floatColor[1] * modulate);
492  colors[1] = color > 255 ? 255 : (byte)color;
493  color = colors[2] + (int)(floatColor[2] * modulate);
494  colors[2] = color > 255 ? 255 : (byte)color;
495  }
496  }
497 
498  // build a list of triangles that need light
499  intColors = (int *) tess.svars.colors;
500  numIndexes = 0;
501  for (i = 0; i < tess.numIndexes; i += 3)
502  {
503  a = tess.indexes[i];
504  b = tess.indexes[i + 1];
505  c = tess.indexes[i + 2];
506  if (!(intColors[a] | intColors[b] | intColors[c]))
507  {
508  continue;
509  }
510  hitIndexes[numIndexes++] = a;
511  hitIndexes[numIndexes++] = b;
512  hitIndexes[numIndexes++] = c;
513  }
514 
515  if (numIndexes == 0)
516  {
517  return;
518  }
519 
520  // debug code
521  //for( i = 0; i < numIndexes; i++ )
522  // intColors[ hitIndexes[ i ] ] = 0x000000FF;
523 
524  qglEnableClientState(GL_COLOR_ARRAY);
525  qglColorPointer(4, GL_UNSIGNED_BYTE, 0, tess.svars.colors);
526 
527  // render the dynamic light pass
528  R_FogOff();
529  GL_Bind(tr.whiteImage);
530  GL_State(GLS_SRCBLEND_DST_COLOR | GLS_DSTBLEND_ONE | GLS_DEPTHFUNC_EQUAL);
531  R_DrawElements(numIndexes, hitIndexes);
532  backEnd.pc.c_totalIndexes += numIndexes;
533  backEnd.pc.c_dlightIndexes += numIndexes;
534  R_FogOn();
535 }
