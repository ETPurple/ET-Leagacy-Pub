38 /*
39 packet header
40 -------------
41 4 outgoing sequence. high bit will be set if this is a fragmented message
42 [2 qport (only for client to server)]
43 [2 fragment start byte]
44 [2 fragment length. if < FRAGMENT_SIZE, this is the last fragment]
45 
46 if the sequence number is -1, the packet should be handled as an out-of-band
47 message instead of as part of a netcon.
48 
49 All fragments will have the same sequence numbers.
50 
51 The qport field is a workaround for bad address translating routers that
52 sometimes remap the client's source port on a packet during gameplay.
53 
54 If the base part of the net address matches and the qport matches, then the
55 channel matches even if the IP port differs. The IP port should be updated
56 to the new value before sending out any replies.
57 */

110 void Netchan_TransmitNextFragment(netchan_t *chan)
111 {
112  msg_t send;
113  byte send_buf[MAX_PACKETLEN];
114  int fragmentLength;
115 
116  // write the packet header
117  MSG_InitOOB(&send, send_buf, sizeof(send_buf)); // <-- only do the oob here
118 
119  MSG_WriteLong(&send, chan->outgoingSequence | FRAGMENT_BIT);
120 
121  // send the qport if we are a client
122  if (chan->sock == NS_CLIENT)
123  {
124  MSG_WriteShort(&send, qport->integer);
125  }
126 
127  // copy the reliable message to the packet first
128  fragmentLength = FRAGMENT_SIZE;
129  if (chan->unsentFragmentStart + fragmentLength > chan->unsentLength)
130  {
131  fragmentLength = chan->unsentLength - chan->unsentFragmentStart;
132  }
133 
134  MSG_WriteShort(&send, chan->unsentFragmentStart);
135  MSG_WriteShort(&send, fragmentLength);
136  MSG_WriteData(&send, chan->unsentBuffer + chan->unsentFragmentStart, fragmentLength);
137 
138  // send the datagram
139  NET_SendPacket(chan->sock, send.cursize, send.data, chan->remoteAddress);
140 
141  // Store send time and size of this packet for rate control
142  chan->lastSentTime = Sys_Milliseconds();
143  chan->lastSentSize = send.cursize;
144 
145  if (showpackets->integer)
146  {
147  Com_Printf("%s send %4i : s=%i fragment=%i,%i\n"
148  , netsrcString[chan->sock]
149  , send.cursize
150  , chan->outgoingSequence
151  , chan->unsentFragmentStart, fragmentLength);
152  }
153 
154  chan->unsentFragmentStart += fragmentLength;
155 
156  // this exit condition is a little tricky, because a packet
157  // that is exactly the fragment length still needs to send
158  // a second packet of zero length so that the other side
159  // can tell there aren't more to follow
160  if (chan->unsentFragmentStart == chan->unsentLength && fragmentLength != FRAGMENT_SIZE)
161  {
162  chan->outgoingSequence++;
163  chan->unsentFragments = qfalse;
164  }
165 }

175 {

239 qboolean Netchan_Process(netchan_t *chan, msg_t *msg)
240 {
241  int sequence;
242  int fragmentStart, fragmentLength;
243  qboolean fragmented;
244 
245  // get sequence numbers
246  MSG_BeginReadingOOB(msg);
247  sequence = MSG_ReadLong(msg);
248 
249  // check for fragment information
250  if (sequence & FRAGMENT_BIT)
251  {
252  sequence &= ~FRAGMENT_BIT;
253  fragmented = qtrue;
254  }
255  else
256  {
257  fragmented = qfalse;
258  }
259 
260  // read the qport if we are a server
261  if (chan->sock == NS_SERVER)
262  {
263  MSG_ReadShort(msg);
264  }
265 
266  // read the fragment information
267  if (fragmented)
268  {
269  fragmentStart = MSG_ReadShort(msg);
270  fragmentLength = MSG_ReadShort(msg);
271  }
272  else
273  {
274  fragmentStart = 0; // stop warning message
275  fragmentLength = 0;
276  }
277 
278  if (showpackets->integer)
279  {
280  if (fragmented)
281  {
282  Com_Printf("%s recv %4i : s=%i fragment=%i,%i\n"
283  , netsrcString[chan->sock]
284  , msg->cursize
285  , sequence
286  , fragmentStart, fragmentLength);
287  }
288  else
289  {
290  Com_Printf("%s recv %4i : s=%i\n"
291  , netsrcString[chan->sock]
292  , msg->cursize
293  , sequence);
294  }
295  }
296 
297  // discard out of order or duplicated packets
298  if (sequence <= chan->incomingSequence)
299  {
300  if (showdrop->integer || showpackets->integer)
301  {
302  Com_Printf("%s:Out of order packet %i at %i\n"
303  , NET_AdrToString(chan->remoteAddress)
304  , sequence
305  , chan->incomingSequence);
306  }
307  return qfalse;
308  }
309 
310  // dropped packets don't keep the message from being used
311  chan->dropped = sequence - (chan->incomingSequence + 1);
312  if (chan->dropped > 0)
313  {
314  if (showdrop->integer || showpackets->integer)
315  {
316  Com_Printf("%s:Dropped %i packets at %i\n"
317  , NET_AdrToString(chan->remoteAddress)
318  , chan->dropped
319  , sequence);
320  }
321  }
322 
323  // if this is the final framgent of a reliable message,
324  // bump incoming_reliable_sequence
325  if (fragmented)
326  {
327  // make sure we add the fragments in correct order
328  // either a packet was dropped, or we received this one too soon
329  // we don't reconstruct the fragments. we will wait till this fragment gets to us again
330  // (NOTE: we could probably try to rebuild by out of order chunks if needed)
331  if (sequence != chan->fragmentSequence)
332  {
333  chan->fragmentSequence = sequence;
334  chan->fragmentLength = 0;
335  }
336 
337  // if we missed a fragment, dump the message
338  if (fragmentStart != chan->fragmentLength)
339  {
340  if (showdrop->integer || showpackets->integer)
341  {
342  Com_Printf("%s:Dropped a message fragment\n"
343  , NET_AdrToString(chan->remoteAddress));
344  }
345  // we can still keep the part that we have so far,
346  // so we don't need to clear chan->fragmentLength
347  return qfalse;
348  }
349 
350  // copy the fragment to the fragment buffer
351  if (fragmentLength < 0 || msg->readcount + fragmentLength > msg->cursize ||
352  chan->fragmentLength + fragmentLength > sizeof(chan->fragmentBuffer))
353  {
354  if (showdrop->integer || showpackets->integer)
355  {
356  Com_Printf("%s:illegal fragment length\n"
357  , NET_AdrToString(chan->remoteAddress));
358  }
359  return qfalse;
360  }
361 
362  Com_Memcpy(chan->fragmentBuffer + chan->fragmentLength,
363  msg->data + msg->readcount, fragmentLength);
364 
365  chan->fragmentLength += fragmentLength;
366 
367  // if this wasn't the last fragment, don't process anything
368  if (fragmentLength == FRAGMENT_SIZE)
369  {
370  return qfalse;
371  }
372 
373  if (chan->fragmentLength > msg->maxsize)
374  {
375  Com_Printf("%s:fragmentLength %i > msg->maxsize\n"
376  , NET_AdrToString(chan->remoteAddress),
377  chan->fragmentLength);
378  return qfalse;
379  }
380 
381  // copy the full message over the partial fragment
382  // make sure the sequence number is still there
383  *(int *)msg->data = LittleLong(sequence);
384 
385  Com_Memcpy(msg->data + 4, chan->fragmentBuffer, chan->fragmentLength);
386  msg->cursize = chan->fragmentLength + 4;
387  chan->fragmentLength = 0;
388  msg->readcount = 4; // past the sequence number
389  msg->bit = 32; // past the sequence number
390 
391  // clients were not acking fragmented messages
392  chan->incomingSequence = sequence;
393 
394  return qtrue;
395  }
396 
397  // the message can now be read from the current message pointer
398  chan->incomingSequence = sequence;
399 
400  return qtrue;
401 }

526 void NET_SendPacket(netsrc_t sock, int length, const void *data, netadr_t to)
527 {
528  // network debugging
529  if (sv_packetloss->integer || sv_packetdelay->integer
530 #ifndef DEDICATED
531  || cl_packetloss->integer || cl_packetdelay->integer
532 #endif
533  )
534  {
535  int packetloss, packetdelay;
536  delaybuf_t **delaybuf_head, **delaybuf_tail;
537 
538  switch (sock)
539  {
540 #ifndef DEDICATED
541  case NS_CLIENT:
542  packetloss = cl_packetloss->integer;
543  packetdelay = cl_packetdelay->integer;
544  delaybuf_head = &cl_delaybuf_head;
545  delaybuf_tail = &cl_delaybuf_tail;
546  break;
547 #endif
548  case NS_SERVER:
549  packetloss = sv_packetloss->integer;
550  packetdelay = sv_packetdelay->integer;
551  delaybuf_head = &sv_delaybuf_head;
552  delaybuf_tail = &sv_delaybuf_tail;
553  break;
554  default:
555  // shut up compiler for dedicated
556  packetloss = 0;
557  packetdelay = 0;
558  delaybuf_head = NULL;
559  delaybuf_tail = NULL;
560  break;
561  }
562 
563  if (packetloss > 0)
564  {
565  if (((float)rand() / RAND_MAX) * 100 <= packetloss)
566  {
567  if (showpackets->integer)
568  {
569  Com_Printf("drop packet %4i\n", length);
570  }
571  return;
572  }
573  }
574 
575  if (packetdelay)
576  {
577  int curtime;
578  delaybuf_t *buf, *nextbuf;
579 
580  curtime = Sys_Milliseconds();
581 
582  //send any scheduled packets, starting from oldest
583  for (buf = *delaybuf_head; buf; buf = nextbuf)
584  {
585  if (buf->time + packetdelay > curtime)
586  {
587  break;
588  }
589 
590  if (showpackets->integer)
591  {
592  Com_Printf( "delayed packet(%dms) %4i\n", buf->time - curtime, buf->length);
593  }
594 
595  switch (buf->to.type)
596  {
597  case NA_BOT:
598  case NA_BAD:
599  break;
600  case NA_LOOPBACK:
601  NET_SendLoopPacket(buf->sock, buf->length, buf->data, buf->to);
602  break;
603  default:
604  Sys_SendPacket(buf->length, buf->data, buf->to);
605  break;
606  }
607 
608  // remove from queue
609  nextbuf = buf->next;
610  *delaybuf_head = nextbuf;
611  if (!*delaybuf_head)
612  {
613  *delaybuf_tail = NULL;
614  }
615  Z_Free(buf);
616  }
617 
618  // create buffer and add it to the queue
619  buf = (delaybuf_t *) Z_Malloc(sizeof(*buf));
620  if (!buf)
621  {
622  Com_Error(ERR_FATAL, "Couldn't allocate packet delay buffer\n" );
623  }
624 
625  buf->sock = sock;
626  buf->length = length;
627  Com_Memcpy(buf->data, data, length);
628  buf->to = to;
629  buf->time = curtime;
630  buf->next = NULL;
631 
632  if (*delaybuf_head)
633  {
634  (*delaybuf_tail)->next = buf;
635  }
636  else
637  {
638  *delaybuf_head = buf;
639  }
640  *delaybuf_tail = buf;
641 
642  return;
643  }
644  }
645 
646  // sequenced packets are shown in netchan, so just show oob
647  if (showpackets->integer && *(int *)data == -1)
648  {
649  Com_Printf("send packet %4i\n", length);
650  }
651 
652  if (to.type == NA_LOOPBACK)
653  {
654  NET_SendLoopPacket(sock, length, data, to);
655  return;
656  }
657  if (to.type == NA_BOT)
658  {
659  return;
660  }
661  if (to.type == NA_BAD)
662  {
663  return;
664  }
665 
666  Sys_SendPacket(length, data, to);
667 }
