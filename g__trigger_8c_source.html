335 void AimAtTarget(gentity_t *self)
336 {
337  gentity_t *ent;
338  vec3_t origin;
339  float height, gravity, time, forward;
340  float dist;
341 
342  VectorAdd(self->r.absmin, self->r.absmax, origin);
343  VectorScale(origin, 0.5f, origin);
344 
345  ent = G_PickTarget(self->target);
346  if (!ent)
347  {
348  G_FreeEntity(self);
349  return;
350  }
351 
352  height = ent->s.origin[2] - origin[2];
353  gravity = g_gravity.value;
354  time = (float)(sqrt(Q_fabs(height / (0.5f * gravity))));
355  if (time == 0.f)
356  {
357  G_FreeEntity(self);
358  return;
359  }
360 
361  // set s.origin2 to the push velocity
362  VectorSubtract(ent->s.origin, origin, self->s.origin2);
363  self->s.origin2[2] = 0;
364  dist = VectorNormalize(self->s.origin2);
365 
366  forward = dist / time;
367  VectorScale(self->s.origin2, forward, self->s.origin2);
368 
369  self->s.origin2[2] = time * gravity;
370 }
