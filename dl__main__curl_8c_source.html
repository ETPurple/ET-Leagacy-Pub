328 dlStatus_t DL_DownloadLoop(void)
329 {
330  CURLMcode status;
331  CURLMsg *msg;
332  int dls = 0;
333  const char *err = NULL;
334 
335  if (!dl_request)
336  {
337  Com_DPrintf("DL_DownloadLoop: Error - unexpected call with dl_request == NULL\n");
338  return DL_DONE;
339  }
340 
341  if ((status = curl_multi_perform(dl_multi, &dls)) == CURLM_CALL_MULTI_PERFORM && dls)
342  {
343  return DL_CONTINUE;
344  }
345 
346  while ((msg = curl_multi_info_read(dl_multi, &dls)) && msg->easy_handle != dl_request)
347  ;
348 
349  if (!msg || msg->msg != CURLMSG_DONE)
350  {
351  return DL_CONTINUE;
352  }
353 
354  if (msg->data.result != CURLE_OK)
355  {
356  err = curl_easy_strerror(msg->data.result);
357  }
358  else
359  {
360  err = NULL;
361  }
362 
363  curl_multi_remove_handle(dl_multi, dl_request);
364  curl_easy_cleanup(dl_request);
365 
366  fclose(dl_file);
367  dl_file = NULL;
368 
369  dl_request = NULL;
370 
371  Cvar_Set("ui_dl_running", "0");
372 
373  if (err)
374  {
375  Com_Printf(S_COLOR_RED "DL_DownloadLoop: Error - request terminated with failure status '%s'\n", err);
376  return DL_FAILED;
377  }
378 
379  return DL_DONE;
380 }
