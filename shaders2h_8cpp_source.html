91 static void make_c_string(string &in)
92 {
93  string out = "";
94 
95  for (size_t i = 0; i < in.size(); ++i)
96  {
97  LOOP_HEAD:
98 
99  char c = in[i];
100  char c2 = (i + 1 < in.size() ? in[i + 1] : 0);
101  if ('"' == c)
102  {
103  add_string_part(out, "\\\"", "\"");
104  }
105  else if ('\\' == c)
106  {
107  add_string_part(out, "\\\\", "\\");
108  }
109  else if ('/' == c && '/' == c2)
110  {
111  i += 2;
112  while (i < in.size())
113  {
114  if (in[i] == '\n')
115  {
116  goto LOOP_HEAD;
117  }
118  i++;
119  }
120  }
121  else if ('/' == c && '*' == c2)
122  {
123  while (i + 1 < in.size())
124  {
125  if (in[i] == '*' && in[i + 1] == '/')
126  {
127  break;
128  }
129  i++;
130  }
131  i += 2;
132  }
133  else if (c == '\n')
134  {
135  // Skip extra new lines
136  while (i + 1 < in.size() && in[i + 1] == '\n')
137  {
138  i++;
139  }
140 
141  for(size_t x = i + 1;x < in.size(); x++)
142  {
143  if (in[x] == '\n')
144  {
145  i = x;
146  goto LOOP_HEAD;
147  }
148 
149  if(in[x] == '/' && x + 1 < in.size() && (in[x + 1] == '/' || in[x + 1] == '*'))
150  {
151  i = x;
152  goto LOOP_HEAD;
153  }
154 
155  // yeah theres something available
156  if(in[x] > 32)
157  {
158  break;
159  }
160 
161  if(x + 1 == in.size())
162  {
163  i = x + 1;
164  }
165  }
166 
167  if (out.size() > 0)
168  {
169  if (i + 1 < in.size())
170  {
171  add_string_part(out, "\\n\"\n\t\"", "\n");
172  }
173  else
174  {
175  add_string_part(out, "\\n", "\n");
176  }
177  }
178  }
179  else if (c == ' ')
180  {
181  // Skip extra white spaces
182  while (i + 1 < in.size() && (in[i + 1] == ' ' || in[i + 1] == '\t'))
183  {
184  i++;
185  }
186  // Don't add white spaces before newline
187  if (i + 1 < in.size() && in[i + 1] != '\n')
188  {
189  if (i + 2 < in.size() && in[i + 1] != '/' && (in[i + 2] != '/' || in[i + 2] != '*'))
190  {
191  out += ' ';
192  }
193  }
194  }
195  else if (c == '\t')
196  {
197  if (out.size() > 0)
198  {
199  if (i + 1 < in.size() && in[i + 1] != '\n')
200  {
201  add_string_part(out, "\\t", "\t");
202  }
203  }
204  }
205  else
206  {
207  out += c;
208  }
209 
210  }
211 
212  // Make sure the output ends with a new line char
213  if(out.size() > 2)
214  {
215  bool needNewLine = false;
216  if (plain_text)
217  {
218  needNewLine = (out[out.size() - 2] != '\\' || out[out.size() - 1] != 'n');
219  }
220  else
221  {
222  needNewLine = (out[out.size() - 1] != '\n');
223  }
224 
225  if (needNewLine)
226  {
227  add_string_part(out, "\\n", "\n");
228  }
229  }
230 
231  if (plain_text)
232  {
233  in = "\"" + out + "\"";
234  }
235  else
236  {
237  in = out;
238  }
239 }

332 static void write_const_char(FILE *outputFile, const string& name, const string& text, const string& comment = "")
333 {
334  if (comment.size())
335  {
336  fprintf(outputFile, "//%s\n", comment.c_str());
337  }
338 
339  if (plain_text)
340  {
341  fprintf(outputFile, "const char *%s =\n\t%s;\n\n", name.c_str(), text.c_str());
342  }
343  else
344  {
345  fprintf(outputFile, "const char %s[%i] =\n{\n", name.c_str(), int(text.size() + 1)); // We need extra char for the ending 0x00
346  for (size_t i = 0; i < text.size();)
347  {
348  size_t x = 0;
349  for (x = 0; x < LINE_SIZE && i < text.size(); x++, i++)
350  {
351  if (x == 0)
352  {
353  fprintf(outputFile, "\t");
354  }
355  else
356  {
357  fprintf(outputFile, ", ");
358  }
359  const char ch = text[i];
360 
361  fprintf(outputFile, "0x%c%c", hexChars[(ch & 0xF0) >> 4], hexChars[ch & 0xF]);
362  }
363 
364  if (i < text.size())
365  {
366  fprintf(outputFile, ",\n");
367  }
368  else
369  {
370  if (x == LINE_SIZE)
371  {
372  fprintf(outputFile, ",\n\t0x00\n};");
373  }
374  else
375  {
376  fprintf(outputFile, ", 0x00\n};");
377  }
378  }
379  }
380 
381  fprintf(outputFile, "\n\n");
382  }

385 static int write_output(string &out_file, string &array_name, vector<shader_out> &shaders, const string &shader_def)
386 {
387  FILE *outputFile;

404  }
405 
406  fprintf(outputFile, "static const char* GetFallbackShader(const char *name)\n{\n");
407  bool first = true;
408  for (std::vector<shader_out>::iterator it = shaders.begin(); it != shaders.end(); ++it)
409  {
410  fprintf(outputFile, "\t");
411  if (!first)
412  {
413  fprintf(outputFile, "else ");
414  }
415  else
416  {
417  first = false;
418  }
419  fprintf(outputFile, "if(!Q_stricmp(name,\"%s\"))\n\t{\n\t\treturn %s;\n\t}\n", it->name.c_str(), it->varName.c_str());
420  }
421  fprintf(outputFile, "\treturn NULL;\n}\n");
422  fprintf(outputFile, "#endif // #ifdef SHADER_HEADER_%s\n", array_name.c_str());
