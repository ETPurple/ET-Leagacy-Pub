721 static void DoRailDiscs(int numSegs, const vec3_t start, const vec3_t dir, const vec3_t right, const vec3_t up)
722 {
723  int i, j;
724  vec3_t pos[4];
725  vec3_t v;
726  int spanWidth = r_railWidth->integer;
727  float c, s;
728  float scale;
729 
730  if (numSegs > 1)
731  {
732  numSegs--;
733  }
734  if (!numSegs)
735  {
736  return;
737  }
738 
739  scale = 0.25f;
740 
741  for (i = 0; i < 4; i++)
742  {
743  c = cos(DEG2RAD(45 + i * 90));
744  s = sin(DEG2RAD(45 + i * 90));
745  v[0] = (right[0] * c + up[0] * s) * scale * spanWidth;
746  v[1] = (right[1] * c + up[1] * s) * scale * spanWidth;
747  v[2] = (right[2] * c + up[2] * s) * scale * spanWidth;
748  VectorAdd(start, v, pos[i]);
749 
750  if (numSegs > 1)
751  {
752  // offset by 1 segment if we're doing a long distance shot
753  VectorAdd(pos[i], dir, pos[i]);
754  }
755  }
756 
757  for (i = 0; i < numSegs; i++)
758  {
759  RB_CHECKOVERFLOW(4, 6);
760 
761  for (j = 0; j < 4; j++)
762  {
763  VectorCopy(pos[j], tess.xyz[tess.numVertexes]);
764  tess.texCoords[tess.numVertexes][0][0] = (j < 2);
765  tess.texCoords[tess.numVertexes][0][1] = (j && j != 3);
766  tess.vertexColors[tess.numVertexes][0] = backEnd.currentEntity->e.shaderRGBA[0];
767  tess.vertexColors[tess.numVertexes][1] = backEnd.currentEntity->e.shaderRGBA[1];

770 
771  VectorAdd(pos[j], dir, pos[j]);
772  }
773

775  tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 1;
776  tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 3;
777  tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 3;
778  tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 1;
779  tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 2;
780  }
