177 void Com_NextDownload(void)
178 {
179  char *s;
180  char *remoteName, *localName;
181 
182  // We are looking to start a download here
183  if (*dld.downloadList)
184  {
185  s = dld.downloadList;
186 
187  // format is:
188  // @remotename@localname@remotename@localname, etc.
189 
190  if (*s == '@')
191  {
192  s++;
193  }
194  remoteName = s;
195 
196  if ((s = strchr(s, '@')) == NULL)
197  {
198  Com_DownloadsComplete();
199  return;
200  }
201 
202  *s++ = 0;
203  localName = s;
204  if ((s = strchr(s, '@')) != NULL)
205  {
206  *s++ = 0;
207  }
208  else
209  {
210  s = localName + strlen(localName); // point at the nul byte
211 
212  }
213 
214  checkDownloadName(remoteName);
215 
216  Com_BeginDownload(localName, remoteName);
217 
218  dld.downloadRestart = qtrue;
219 
220  // move over the rest
221  memmove(dld.downloadList, s, strlen(s) + 1);
222 
223  return;
224  }
225 
226  Com_DownloadsComplete();
227 }

329 void Com_WWWDownload(void)
330 {
331  const char *to_ospath;
332  dlStatus_t ret;
333  static qboolean bAbort = qfalse;
334 
335  if (dld.bWWWDlAborting)
336  {
337  if (!bAbort)
338  {
339  Com_DPrintf("Com_WWWDownload: WWWDlAborting\n");
340  bAbort = qtrue;
341  }
342  return;
343  }
344  if (bAbort)
345  {
346  Com_DPrintf("Com_WWWDownload: WWWDlAborting done\n");
347  bAbort = qfalse;
348  }
349 
350  ret = DL_DownloadLoop();
351 
352  if (ret == DL_CONTINUE)
353  {
354  return;
355  }
356  else if (ret == DL_DONE)
357  {
358  // taken from CL_ParseDownload
359  // we work with OS paths
360 
361  dld.download = 0;
362 #if defined(FEATURE_PAKISOLATION) && !defined(DEDICATED)
363  to_ospath = GetDownloadDestPath(dld.downloadTempName, dld.originalDownloadName);
364 #else
365  to_ospath = FS_BuildOSPath(Cvar_VariableString("fs_homepath"), dld.originalDownloadName, NULL);
366 #endif
367  if (rename(dld.downloadTempName, to_ospath))
368  {
369  FS_CopyFile(dld.downloadTempName, to_ospath);
370 
371  if (remove(dld.downloadTempName) != 0)
372  {
373  Com_Printf("WARNING: Com_WWWDownload - cannot remove file '%s'\n", dld.downloadTempName);
374  }
375  }
376  *dld.downloadTempName = *dld.downloadName = 0;
377  Cvar_Set("cl_downloadName", "");
378  if (dld.bWWWDlDisconnected)
379  {
380  // for an auto-update in disconnected mode, we'll be spawning the setup in CL_DownloadsComplete
381  if (!autoupdate.updateStarted && !dld.noReconnect)
382  {
383  // reconnect to the server, which might send us to a new disconnected download
384  Cbuf_ExecuteText(EXEC_APPEND, "reconnect\n");
385  }
386  }
387  else
388  {
389  Com_AddReliableCommand("wwwdl done");
390  // tracking potential web redirects leading us to wrong checksum - only works in connected mode
391  if (strlen(dld.redirectedList) + strlen(dld.originalDownloadName) + 1 >= sizeof(dld.redirectedList))
392  {
393  // just to be safe
394  Com_Printf("ERROR: redirectedList overflow (%s)\n", dld.redirectedList);
395  }
396  else
397  {
398  strcat(dld.redirectedList, "@");
399  strcat(dld.redirectedList, dld.originalDownloadName);
400  }
401  }
402  }
403  else
404  {
405  if (dld.bWWWDlDisconnected)
406  {
407  // in a connected download, we'd tell the server about failure and wait for a reply
408  // but in this case we can't get anything from server
409  // if we just reconnect it's likely we'll get the same disconnected download message, and error out again
410  // this may happen for a regular dl or an auto update
411  const char *error = va("Download failure while getting '%s'\n", dld.downloadName); // get the msg before clearing structs
412 
413  dld.bWWWDlDisconnected = qfalse; // need clearing structs before ERR_DROP, or it goes into endless reload
414  Com_ClearStaticDownload();
415  Com_Error(ERR_DROP, "%s", error);
416  }
417  else
418  {
419  // see CL_ParseDownload, same abort strategy
420  Com_Printf("Download failure while getting '%s'\n", dld.downloadName);
421  Com_AddReliableCommand("wwwdl fail");
422  dld.bWWWDlAborting = qtrue;
423  }
424  return;
425  }
426 
427  dld.bWWWDl = qfalse;
428  Com_NextDownload();
429 }
