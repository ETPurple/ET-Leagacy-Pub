219  return;

319  }
320

322  if (ent->client->sess.referee)
323  {
324  G_refPause_cmd(ent, fPause);
325  }
326  else
327  {
328  int tteam = ent->client->sess.sessionTeam;
329 
330  if (!G_cmdDebounce(ent, aCommandInfo[dwCommand].pszCommandName))
331  {
332  return;
333  }
334 
335  // Trigger the auto-handling of pauses
336  if (fPause)
337  {

339  {
340  CP("cpm \"^3Your team has no more timeouts remaining!\n\"");
341  return;
342  }
343  else

345  teamInfo[tteam].timeouts--;
346  level.match_pause = tteam + 128;
347  G_globalSoundEnum(GAMESOUND_MISC_REFEREE);
348  G_spawnPrintf(DP_PAUSEINFO, level.time + 15000, NULL);
349  AP(va("print \"^3Match is ^1PAUSED^3!\n^7[%s^7: - %d Timeouts Remaining]\n\"", aTeams[tteam], teamInfo[tteam].timeouts));
350  AP(va("cp \"^3Match is ^1PAUSED^3! (%s^3)\n\"", aTeams[tteam]));
351  level.server_settings |= CV_SVS_PAUSE;
352  trap_SetConfigstring(CS_SERVERTOGGLES, va("%d", level.server_settings));
353  }
354  }
355  else if (tteam + 128 != level.match_pause)
356  {
357  CP("cpm \"^3Your team didn't call the timeout!\n\"");
358  return;
359  }
360  else
361  {
362  AP("print \"^3Match is ^5UNPAUSED^3 ... resuming in 10 seconds!\n\"");
363  level.match_pause = PAUSE_UNPAUSING;
364  G_globalSound("sound/osp/prepare.wav");
365  G_spawnPrintf(DP_UNPAUSING, level.time + 10, NULL);
366  }
367  }
