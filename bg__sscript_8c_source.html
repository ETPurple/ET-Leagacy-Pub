155 static qboolean BG_SS_ParseSpeaker(int handle)
156 {
157  pc_token_t token;
158  bg_speaker_t speaker;
159 
160  Com_Memset(&speaker, 0, sizeof(speaker));
161 
162  speaker.volume = 127;
163  speaker.range = 1250;
164 
165  if (!trap_PC_ReadToken(handle, &token) || Q_stricmp(token.string, "{"))
166  {
167  return BG_SS_ParseError(handle, "expected '{'");
168  }
169 
170  while (1)
171  {
172  if (!trap_PC_ReadToken(handle, &token))
173  {
174  break;
175  }
176 
177  if (token.string[0] == '}')
178  {
179  break;
180  }
181 
182  if (!Q_stricmp(token.string, "noise"))
183  {
184  if (!PC_String_ParseNoAlloc(handle, speaker.filename, sizeof(speaker.filename)))
185  {
186  return BG_SS_ParseError(handle, "expected sound filename");
187  }
188  }
189  else if (!Q_stricmp(token.string, "origin"))
190  {
191  if (!PC_Vec_Parse(handle, &speaker.origin))
192  {
193  return BG_SS_ParseError(handle, "expected origin vector");
194  }
195  }
196  else if (!Q_stricmp(token.string, "targetname"))
197  {
198  if (!PC_String_ParseNoAlloc(handle, speaker.targetname, sizeof(speaker.targetname)))
199  {
200  return BG_SS_ParseError(handle, "expected targetname string");
201  }
202  else
203  {
204  speaker.targetnamehash = BG_StringHashValue(speaker.targetname);
205  }
206  }
207  else if (!Q_stricmp(token.string, "looped"))
208  {
209  if (!trap_PC_ReadToken(handle, &token))
210  {
211  return BG_SS_ParseError(handle, "expected loop value");
212  }
213  else
214  {
215  if (!Q_stricmp(token.string, "no"))
216  {
217  speaker.loop = S_LT_NOT_LOOPED;
218  }
219  else if (!Q_stricmp(token.string, "on"))
220  {
221  speaker.loop = S_LT_LOOPED_ON;
222  speaker.activated = qtrue;
223  }
224  else if (!Q_stricmp(token.string, "off"))
225  {
226  speaker.loop = S_LT_LOOPED_OFF;
227  }
228  else
229  {
230  return BG_SS_ParseError(handle, "unknown loop value '%s'", token.string);
231  }
232  }
233  }
234  else if (!Q_stricmp(token.string, "broadcast"))
235  {
236  if (!trap_PC_ReadToken(handle, &token))
237  {
238  return BG_SS_ParseError(handle, "expected broadcast value");
239  }
240  else
241  {
242  if (!Q_stricmp(token.string, "no"))
243  {
244  speaker.broadcast = S_BT_LOCAL;
245  }
246  else if (!Q_stricmp(token.string, "global"))
247  {
248  speaker.broadcast = S_BT_GLOBAL;
249  }
250  else if (!Q_stricmp(token.string, "nopvs"))
251  {
252  speaker.broadcast = S_BT_NOPVS;
253  }
254  else
255  {
256  return BG_SS_ParseError(handle, "unknown broadcast value '%s'", token.string);
257  }
258  }
259  }
260  else if (!Q_stricmp(token.string, "wait"))
261  {
262  if (!PC_Int_Parse(handle, &speaker.wait))
263  {
264  return BG_SS_ParseError(handle, "expected wait value");
265  }
266  else if (speaker.wait < 0)
267  {
268  return BG_SS_ParseError(handle, "wait value %i is invalid", speaker.wait);
269  }
270  }
271  else if (!Q_stricmp(token.string, "random"))
272  {
273  if (!PC_Int_Parse(handle, &speaker.random))
274  {
275  return BG_SS_ParseError(handle, "expected random value");
276  }
277  else if (speaker.random < 0)
278  {
279  return BG_SS_ParseError(handle, "random value %i is invalid", speaker.random);
280  }
281  }
282  else if (!Q_stricmp(token.string, "volume"))
283  {
284  if (!PC_Int_Parse(handle, &speaker.volume))
285  {
286  return BG_SS_ParseError(handle, "expected volume value");
287  }
288  else if (speaker.volume < 0 || speaker.volume > 65535)
289  {
290  return BG_SS_ParseError(handle, "volume value %i is invalid", speaker.volume);
291  }
292  }
293  else if (!Q_stricmp(token.string, "range"))
294  {
295  if (!PC_Int_Parse(handle, &speaker.range))
296  {
297  return BG_SS_ParseError(handle, "expected range value");
298  }
299  else if (speaker.range < 0)
300  {
301  return BG_SS_ParseError(handle, "range value %i is invalid", speaker.range);
302  }
303  }
304  else
305  {
306  return BG_SS_ParseError(handle, "unknown token '%s'", token.string);
307  }
308  }
309 
310  if (!BG_SS_StoreSpeaker(&speaker))
311  {
312  return BG_SS_ParseError(handle, "Failed to store speaker", token.string);
313  }
314 
315  return qtrue;
316 }
