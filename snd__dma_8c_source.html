289 {

293 
294  while (name[i] != '\0')

337  return NULL;

341

343  // see if already loaded
344  while (sfx)
345  {

347  {

351  }

353  // find a free sfx

355  {

359  }
360  }
361

365  {

367  }
368  numSfx++;
369  }
370 
371  sfx = &knownSfx[i];
372  Com_Memset(sfx, 0, sizeof(*sfx));
373  strcpy(sfx->soundName, name);
374

376  sfxHash[hash] = sfx;
377

541 void S_SpatializeOrigin(vec3_t origin, int master_vol, int *left_vol, int *right_vol, float range, int no_attenuation)
542 {
543  vec_t lscale, rscale;
544 
545  if (dma.channels == 1 || no_attenuation) // no attenuation = no spatialization
546  {
547  rscale = 1.0;
548  lscale = 1.0;
549  }
550  else
551  {
552  vec_t dist;
553  vec3_t source_vec;
554  vec3_t vec;
555  float dist_fullvol = range * 0.064f;
556 
557  // calculate stereo seperation and distance attenuation
558  VectorSubtract(origin, listener_origin, source_vec);
559 
560  dist = vec3_norm(source_vec);
561  dist -= dist_fullvol;
562  if (dist < 0.0f || no_attenuation)
563  {
564  dist = 0.0f; // close enough to be at full volume
565  }
566  else
567  {
568  dist /= range;
569  }
570 
571  vec3_rotate(source_vec, listener_axis, vec);
572 
573  rscale = (float)(sqrt((double)(1.0f - vec[1])));
574  lscale = (float)(sqrt((double)(1.0f + vec[1])));
575  if (rscale < 0)
576  {
577  rscale = 0;
578  }
579  else
580  {
581  rscale *= 1.0f - dist;
582  }
583  if (lscale < 0)
584  {
585  lscale = 0;
586  }
587  else
588  {
589  lscale *= 1.0f - dist;
590  }
591  }
592 
593  // add in distance effect
594  *right_vol = (int)(master_vol * rscale);
595  if (*right_vol < 0)
596  {
597  *right_vol = 0;
598  }
599 
600  *left_vol = (int)(master_vol * lscale);
601  if (*left_vol < 0)
602  {
603  *left_vol = 0;
604  }
605 }

624 {

627  int i, time;
628

630  {
631  //Com_Printf(S_COLOR_YELLOW "S_Base_StartSoundEx: sound not started or muted\n");
632  return;

634 
635  if (!origin && (entnum < 0 || entnum >= MAX_GENTITIES))
636  {
637  Com_Error(ERR_DROP, "S_Base_StartSoundEx: bad entitynum %i", entnum);
638  }
639 
640  if (sfxHandle < 0 || sfxHandle >= numSfx)
641  {
642  Com_Printf(S_COLOR_YELLOW "S_Base_StartSoundEx: handle %i out of range", sfxHandle);
643  return;
644  }
645 
646  sfx = &knownSfx[sfxHandle];
647 
648  if (sfx->inMemory == qfalse)
649  {
650  S_memoryLoad(sfx);
651  }
652 
653  if (s_show->integer == 1)
654  {
655  Com_Printf("S_Base_StartSoundEx: %i : %s\n", s_paintedtime, sfx->soundName);
656  }
657 
658  time = Sys_Milliseconds();
659  ch = s_channels;
660 
661  // shut off other sounds on this channel if necessary
662  for (i = 0; i < MAX_CHANNELS ; i++, ch++)
663  {
664  if (ch->entnum == entnum && ch->thesfx)
665  {
666  if (ch->thesfx == sfx && time - ch->allocTime < 50) // double played in one frame
667  {
668  return;
669  }
670  else if (ch->entchannel == entchannel)
671  {
672  if ((flags & SND_CUTOFF_ALL)) // cut the sounds that are flagged to be cut
673  {

675  }

677  {
678  continue;
679  }

681  {
682  S_ChannelFree(ch);
683  }
684  else if ((ch->flags & SND_REQUESTCUT) && (flags & SND_CUTOFF)) // cutoff 'weak' sounds on channel
685  {
686  S_ChannelFree(ch);
687  }
688  }
689  }
690  }
691 
692  sfx->lastTimeUsed = time;
693 
694  ch = S_ChannelMalloc();
695  if (!ch)
696  {
697  int oldest = sfx->lastTimeUsed;
698  int chosen = -1;
699 
700  ch = s_channels;
701 
702  for (i = 0 ; i < MAX_CHANNELS ; i++, ch++)
703  {
704  if (ch->entnum != listener_number && ch->entnum == entnum && ch->allocTime < oldest && ch->entchannel != CHAN_ANNOUNCER)
705  {
706  oldest = ch->allocTime;
707  chosen = i;
708  }
709  }
710  if (chosen == -1)
711  {
712  ch = s_channels;
713  for (i = 0 ; i < MAX_CHANNELS ; i++, ch++)
714  {
715  if (ch->entnum != listener_number && ch->allocTime < oldest && ch->entchannel != CHAN_ANNOUNCER)
716  {
717  oldest = ch->allocTime;
718  chosen = i;
719  }
720  }
721  if (chosen == -1)
722  {
723  ch = s_channels;
724  if (ch->entnum == listener_number)
725  {
726  for (i = 0 ; i < MAX_CHANNELS ; i++, ch++)
727  {
728  if (ch->allocTime < oldest)
729  {
730  oldest = ch->allocTime;
731  chosen = i;
732  }
733  }
734  }
735  if (chosen == -1)
736  {
737  Com_Printf("S_Base_StartSoundEx WARNING: dropping sound\n");
738  return;
739  }
740  }
741  }
742  ch = &s_channels[chosen];
743  ch->allocTime = sfx->lastTimeUsed;
744  }
745 
746  if (origin)
747  {
