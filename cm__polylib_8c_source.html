353 void ClipWindingEpsilon(winding_t *in, vec3_t normal, vec_t dist,
354  vec_t epsilon, winding_t **front, winding_t **back)
355 {
356  vec_t dists[MAX_POINTS_ON_WINDING + 4] = { 0 };
357  int sides[MAX_POINTS_ON_WINDING + 4] = { 0 };
358  int counts[3] = { 0, 0, 0 };
359  static vec_t dot; // VC 4.2 optimizer bug if not static
360  int i, j;
361  vec_t *p1, *p2;
362  vec3_t mid;
363  winding_t *f, *b;
364  int maxpts;
365 
366  // determine sides for each point
367  for (i = 0 ; i < in->numpoints ; i++)
368  {
369  dot = DotProduct(in->p[i], normal);
370  dot -= dist;
371  dists[i] = dot;
372  if (dot > epsilon)
373  {
374  sides[i] = SIDE_FRONT;
375  }
376  else if (dot < -epsilon)
377  {
378  sides[i] = SIDE_BACK;
379  }
380  else
381  {
382  sides[i] = SIDE_ON;
383  }
384 
385  counts[sides[i]]++;
386  }
387  sides[i] = sides[0];
388  dists[i] = dists[0];
389 
390  *front = *back = NULL;
391 
392  if (!counts[0])
393  {
394  *back = CopyWinding(in);
395  return;
396  }
397  if (!counts[1])
398  {
399  *front = CopyWinding(in);
400  return;
401  }
402 
403  maxpts = in->numpoints + 4; // cant use counts[0]+2 because
404  // of fp grouping errors
405 
406  *front = f = AllocWinding(maxpts);
407  *back = b = AllocWinding(maxpts);
408 
409  for (i = 0 ; i < in->numpoints ; i++)
410  {
411  p1 = in->p[i];
412 
413  if (sides[i] == SIDE_ON)
414  {
415  VectorCopy(p1, f->p[f->numpoints]);
416  f->numpoints++;
417  VectorCopy(p1, b->p[b->numpoints]);
418  b->numpoints++;
419  continue;
420  }
421 
422  if (sides[i] == SIDE_FRONT)
423  {
424  VectorCopy(p1, f->p[f->numpoints]);
425  f->numpoints++;
426  }
427  if (sides[i] == SIDE_BACK)
428  {
429  VectorCopy(p1, b->p[b->numpoints]);
430  b->numpoints++;
431  }
432 
433  if (sides[i + 1] == SIDE_ON || sides[i + 1] == sides[i])
434  {
435  continue;
436  }
437 
438  // generate a split point
439  p2 = in->p[(i + 1) % in->numpoints];
440 
441  dot = dists[i] / (dists[i] - dists[i + 1]);
442  for (j = 0 ; j < 3 ; j++) // avoid round off error when possible
443  {
444  if (normal[j] == 1.f)
445  {
446  mid[j] = dist;
447  }
448  else if (normal[j] == -1.f)
449  {
450  mid[j] = -dist;
451  }
452  else
453  {
454  mid[j] = p1[j] + dot * (p2[j] - p1[j]);
455  }
456  }
457 
458  VectorCopy(mid, f->p[f->numpoints]);
459  f->numpoints++;
460  VectorCopy(mid, b->p[b->numpoints]);
461  b->numpoints++;
462  }
463 
464  if (f->numpoints > maxpts || b->numpoints > maxpts)
465  {
466  Com_Error(ERR_DROP, "ClipWindingEpsilon: points exceeded estimate");
467  }
468  if (f->numpoints > MAX_POINTS_ON_WINDING || b->numpoints > MAX_POINTS_ON_WINDING)
469  {
470  Com_Error(ERR_DROP, "ClipWindingEpsilon: MAX_POINTS_ON_WINDING");
471  }
472 }

626 void CheckWinding(winding_t *w)
627 {
628  int i, j;
629  vec_t *p1, *p2;
630  vec_t d, edgedist;
631  vec3_t dir, edgenormal, facenormal;
632  vec_t area;
633  vec_t facedist;
634 
635  if (w->numpoints < 3)
636  {
637  Com_Error(ERR_DROP, "CheckWinding: %i points", w->numpoints);
638  }
639 
640  area = WindingArea(w);
641  if (area < 1)
642  {
643  Com_Error(ERR_DROP, "CheckWinding: %f area", (double)area);
644  }
645 
646  WindingPlane(w, facenormal, &facedist);
647 
648  for (i = 0 ; i < w->numpoints ; i++)
649  {
650  p1 = w->p[i];
651 
652  for (j = 0 ; j < 3 ; j++)
653  if (p1[j] > MAX_MAP_BOUNDS || p1[j] < -MAX_MAP_BOUNDS)
654  {
655  Com_Error(ERR_DROP, "CheckWinding: MAX_MAP_BOUNDS: %f", (double)p1[j]);
656  }
657 
658  j = i + 1 == w->numpoints ? 0 : i + 1;
659 
660  // check the point is on the face plane
661  d = DotProduct(p1, facenormal) - facedist;
662  if (d < -ON_EPSILON || d > ON_EPSILON)
663  {
664  Com_Error(ERR_DROP, "CheckWinding: point off plane");
665  }
666 
667  // check the edge isnt degenerate
668  p2 = w->p[j];
669  VectorSubtract(p2, p1, dir);
670 
671  if (vec3_length(dir) < ON_EPSILON)
672  {
673  Com_Error(ERR_DROP, "CheckWinding: degenerate edge");
674  }
675 
676  vec3_cross(facenormal, dir, edgenormal);
677  vec3_norm2(edgenormal, edgenormal);
678  edgedist = DotProduct(p1, edgenormal);
679  edgedist += ON_EPSILON;
680 
681  // all other points must be on front side
682  for (j = 0 ; j < w->numpoints ; j++)
683  {
684  if (j == i)
685  {
686  continue;
687  }
688  d = DotProduct(w->p[j], edgenormal);
689  if (d > edgedist)
690  {
691  Com_Error(ERR_DROP, "CheckWinding: non-convex");
692  }
693  }
694  }

793  }

797  {

810  hullSide[j] = qfalse;
811  }
812  }

815  if (!outside)
816  {
817  continue;
818  }

821  for (j = 0 ; j < numHullPoints ; j++)
822  {

824  {

826  }
827  }

829  {
830  continue;
831  }

833  // insert the point here
834  VectorCopy(p, newHullPoints[0]);
835  numNew = 1;
836 
837  // copy over all points that aren't double fronts
838  j = (j + 1) % numHullPoints;
839  for (k = 0 ; k < numHullPoints ; k++)
840  {

842  {

844  }
