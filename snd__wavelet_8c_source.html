48 void daub4(float b[], unsigned long n, int isign)
49 {
50  float wksp[4097];
51 #define a(x) b[(x) - 1] // numerical recipies so a[1] = b[0]
52  unsigned long nh, nh1, i, j;
53 
54  if (n < 4)
55  {
56  return;
57  }
58 
59  nh1 = (nh = n >> 1) + 1;
60  if (isign >= 0)
61  {
62  for (i = 1, j = 1; j <= n - 3; j += 2, i++)
63  {
64  wksp[i] = C0 * a(j) + C1 * a(j + 1) + C2 * a(j + 2) + C3 * a(j + 3);
65  wksp[i + nh] = C3 * a(j) - C2 * a(j + 1) + C1 * a(j + 2) - C0 * a(j + 3);
66  }
67  wksp[i] = C0 * a(n - 1) + C1 * a(n) + C2 * a(1) + C3 * a(2);
68  wksp[i + nh] = C3 * a(n - 1) - C2 * a(n) + C1 * a(1) - C0 * a(2);
69  }
70  else
71  {
72  wksp[1] = C2 * a(nh) + C1 * a(n) + C0 * a(1) + C3 * a(nh1);
73  wksp[2] = C3 * a(nh) - C0 * a(n) + C1 * a(1) - C2 * a(nh1);
74  for (i = 1, j = 3; i < nh; i++)
75  {
76  wksp[j++] = C2 * a(i) + C1 * a(i + nh) + C0 * a(i + 1) + C3 * a(i + nh1);
77  wksp[j++] = C3 * a(i) - C0 * a(i + nh) + C1 * a(i + 1) - C2 * a(i + nh1);
78  }
79  }
80  for (i = 1; i <= n; i++)
81  {
82  a(i) = wksp[i];
83  }
84 #undef a
85 }

132 byte MuLawEncode(short s)
133 {
134  unsigned long adjusted;
135  byte exponent, mantissa;
136  byte sign = (s < 0) ? 0 : 0x80;
137 
138  if (s < 0)
139  {
140  s = -s;
141  }
142  adjusted = (long)s << (16 - sizeof(short) * 8);
143  adjusted += 128L + 4L;
144  if (adjusted > 32767)
145  {
146  adjusted = 32767;
147  }
148  exponent = numBits[(adjusted >> 7) & 0xff] - 1;
149  mantissa = (adjusted >> (exponent + 3)) & 0xf;
150  return ~(sign | (exponent << 4) | mantissa);
151 }
