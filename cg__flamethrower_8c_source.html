417 flameChunk_t *CG_SpawnFlameChunk(flameChunk_t *headFlameChunk)
418 {
419  flameChunk_t *f;
420 
421  if (!freeFlameChunks)
422  {
423  return NULL;
424  }
425 
426  if (headFlameChunks && headFlameChunks->dead)
427  {
428  headFlameChunks = NULL;
429  }
430 
431  // select the first free trail, and remove it from the list
432  f = freeFlameChunks;
433  freeFlameChunks = f->nextGlobal;
434  if (freeFlameChunks)
435  {
436  freeFlameChunks->prevGlobal = NULL;
437  }
438 
439  f->nextGlobal = activeFlameChunks;
440  if (activeFlameChunks)
441  {
442  activeFlameChunks->prevGlobal = f;
443  }
444  activeFlameChunks = f;
445  f->prevGlobal = NULL;
446  f->inuse = qtrue;
447  f->dead = qfalse;
448 
449  // if this owner has a headJunc, add us to the start
450  if (headFlameChunk)
451  {
452  // remove the headJunc from the list of heads
453  if (headFlameChunk == headFlameChunks)
454  {
455  headFlameChunks = headFlameChunks->nextHead;
456  if (headFlameChunks)
457  {
458  headFlameChunks->prevHead = NULL;
459  }
460  }
461  else
462  {
463  if (headFlameChunk->nextHead)
464  {
465  headFlameChunk->nextHead->prevHead = headFlameChunk->prevHead;
466  }
467  if (headFlameChunk->prevHead)
468  {
469  headFlameChunk->prevHead->nextHead = headFlameChunk->nextHead;
470  }
471  }
472  headFlameChunk->prevHead = NULL;
473  headFlameChunk->nextHead = NULL;
474  }
475  // make us the headTrail
476  if (headFlameChunks)
477  {
478  headFlameChunks->prevHead = f;
479  }
480  f->nextHead = headFlameChunks;
481  f->prevHead = NULL;
482  headFlameChunks = f;
483 
484  f->nextFlameChunk = headFlameChunk; // if headJunc is NULL, then we'll just be the end of the list
485 
486  numFlameChunksInuse++;
487 
488  return f;
489 }

594 void CG_MoveFlameChunk(flameChunk_t *f)
595 {
596  vec3_t newOrigin, sOrg;
597  trace_t trace;
598  float dot;
599 
600  // subtract friction from speed
601  if (f->velSpeed > 1 && f->lastFrictionTake < cg.time - 50)
602  {
603  CG_FlameAdjustSpeed(f, -((float)(cg.time - f->lastFrictionTake) / 1000.0f) * FLAME_FRICTION_PER_SEC);
604  f->lastFrictionTake = cg.time;
605  }
606 
607  // adjust size
608  if (f->size < f->sizeMax)
609  {
610  if ((cg.time - f->timeStart) < f->blueLife)
611  {
612  f->sizeRate = GET_FLAME_BLUE_SIZE_SPEED(FLAME_START_MAX_SIZE); // use a constant so the blue flame doesn't distort
613  }
614  else
615  {
616  f->sizeRate = GET_FLAME_SIZE_SPEED(f->sizeMax);
617  }
618 
619  f->size += f->sizeRate * (float)(cg.time - f->baseOrgTime);
620  if (f->size > f->sizeMax)
621  {
622  f->size = f->sizeMax;
623  }
624  }
625 
626  VectorCopy(f->baseOrg, sOrg);
627  while (f->velSpeed > 1 && f->baseOrgTime != cg.time)
628  {
629  CG_FlameCalcOrg(f, cg.time, newOrigin);
630 
631  // trace a line from previous position to new position
632  CG_Trace(&trace, sOrg, flameChunkMins, flameChunkMaxs, newOrigin, f->ownerCent, MASK_SHOT | MASK_WATER); // JPW NERVE water fixes
633 
634  if (trace.startsolid)
635  {
636  f->velSpeed = 0;
637  f->dead = qtrue; // water fixes
638  break;
639  }
640 
641  if (trace.surfaceFlags & SURF_NOIMPACT)
642  {
643  break;
644  }
645 
646  // moved some distance
647  VectorCopy(trace.endpos, f->baseOrg);
648  f->baseOrgTime += (int)((float)(cg.time - f->baseOrgTime) * trace.fraction);
649 
650  if (trace.fraction == 1.0f)
651  {
652  // check for hitting client
653  if ((f->ownerCent != cg.snap->ps.clientNum) && !(cg.snap->ps.eFlags & EF_DEAD) && VectorDistance(newOrigin, cg.snap->ps.origin) < 32)
654  {
655  VectorNegate(f->velDir, trace.plane.normal);
656  }
657  else
658  {
659  break;
660  }
661  }
662 
663  // reflect off surface
664  dot = DotProduct(f->velDir, trace.plane.normal);
665  VectorMA(f->velDir, -2 * dot, trace.plane.normal, f->velDir);
666  VectorNormalize(f->velDir);
667  // subtract some speed
668  f->velSpeed *= 0.5f * (0.25f + 0.75f * ((dot + 1.0f) * 0.5f));
669  VectorCopy(f->velDir, f->parentFwd);
670 
671  VectorCopy(f->baseOrg, sOrg);
672  }
673 
674  CG_FlameCalcOrg(f, cg.time, f->org);
675  f->baseOrgTime = cg.time; // incase we skipped the movement
676 }
