423 void CG_LocateCampaign(void)
424 {
425  int numdirs;
426  char filename[MAX_QPATH];
427  char dirlist[1024];
428  char *dirptr;
429  int i;
430  unsigned int dirlen;
431  qboolean found = qfalse;
432 
433  // get all campaigns from .campaign files
434  numdirs = trap_FS_GetFileList("scripts", ".campaign", dirlist, 1024);
435  dirptr = dirlist;
436  for (i = 0; i < numdirs; i++, dirptr += dirlen + 1)
437  {
438  dirlen = strlen(dirptr);
439  Q_strncpyz(filename, "scripts/", MAX_QPATH);
440  Q_strcat(filename, MAX_QPATH, dirptr);
441  if (CG_FindCampaignInFile(filename, cgs.currentCampaign, &cgs.campaignData))
442  {
443  found = qtrue;
444  break;
445  }
446  }
447 
448  if (!found)
449  {
450  return;
451  }
452 
453  for (i = 0; i < cgs.campaignData.mapCount; i++)
454  {
455  Com_sprintf(filename, sizeof(filename), "scripts/%s.arena", cgs.campaignData.mapnames[i]);
456 
457  if (!CG_FindArenaInfo(filename, cgs.campaignData.mapnames[i], &cgs.campaignData.arenas[i]))
458  {
459  return;
460  }
461  }
462 
463  cgs.campaignInfoLoaded = qtrue;
464 }
