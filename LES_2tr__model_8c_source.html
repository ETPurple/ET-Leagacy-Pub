422 unsigned char R_MDC_GetAnorm(const vec3_t dir)
423 {
424  int i, best_start_i[3] = { 0 }, next_start, next_end;
425  int best = 0;
426  float best_diff, group_val, this_val, diff;
427  float *this_norm;
428 
429  // find best Z match
430  if (dir[2] > 0.097545f)
431  {
432  next_start = 144;
433  next_end = NUMMDCVERTEXNORMALS;
434  }
435  else
436  {
437  next_start = 0;
438  next_end = 144;
439  }
440 
441  best_diff = 999;
442  this_val = -999;
443 
444  for (i = next_start ; i < next_end ; i++)
445  {
446  if (r_anormals[i][2] == this_val)
447  {
448  continue;
449  }
450  else
451  {
452  this_val = r_anormals[i][2];
453  }
454 
455  if ((diff = Q_fabs(dir[2] - r_anormals[i][2])) < best_diff)
456  {
457  best_diff = diff;
458  best_start_i[2] = i;
459 
460  }
461 
462  if (next_start)
463  {
464  if (r_anormals[i][2] > dir[2])
465  {
466  break; // we've gone passed the dir[2], so we can't possibly find a better match now
467  }
468  }
469  else
470  {
471  if (r_anormals[i][2] < dir[2])
472  {
473  break; // we've gone passed the dir[2], so we can't possibly find a better match now
474  }
475  }
476  }
477 
478  best_diff = -999;
479 
480  // find best match within the Z group
481 
482  for (i = best_start_i[2], group_val = r_anormals[i][2]; i < NUMMDCVERTEXNORMALS; i++)
483  {
484  this_norm = r_anormals[i];
485 
486  if (this_norm[2] != group_val)
487  {
488  break; // done checking the group
489  }
490 
491  diff = DotProduct(dir, this_norm);
492 
493  if (diff > best_diff)
494  {
495  best_diff = diff;
496  best = i;
497  }
498  }
499 
500  return (unsigned char)best;
501 }
