49 polyBuffer_t *CG_PB_FindFreePolyBuffer(qhandle_t shader, int numVerts, int numIndicies)
50 {
51  int i;
52  int firstFree = -1;
53 
54  // first find one with the same shader if possible
55  for (i = 0; i < MAX_PB_BUFFERS; ++i)
56  {
57  if (!cg_polyBuffersInuse[i])
58  {
59  if (firstFree == -1)
60  {
61  firstFree = i;
62  }
63  continue;
64  }
65 
66  if (cg_polyBuffers[i].shader != shader)
67  {
68  continue;
69  }
70 
71  if (cg_polyBuffers[i].numIndicies + numIndicies >= MAX_PB_INDICIES)
72  {
73  continue;
74  }
75 
76  if (cg_polyBuffers[i].numVerts + numVerts >= MAX_PB_VERTS)
77  {
78  continue;
79  }
80 
81  cg_polyBuffersInuse[i] = qtrue;
82  cg_polyBuffers[i].shader = shader;
83 
84  return &cg_polyBuffers[i];
85  }
86 
87  // return the free pB we have already found above
88  if (firstFree != -1)
89  {
90  cg_polyBuffersInuse[firstFree] = qtrue;
91  cg_polyBuffers[firstFree].shader = shader;
92  cg_polyBuffers[firstFree].numIndicies = 0;
93  cg_polyBuffers[firstFree].numVerts = 0;
94  return &cg_polyBuffers[firstFree];
95  }
96 
97  // or return NULL if no free buffer was found..
98  return NULL;
99 }
