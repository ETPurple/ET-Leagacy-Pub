247  return G_ConfigError(handle, "expected a command value");

261  }
262  }
263 
264  return qtrue;

347 void G_configLoadAndSet(const char *name)
348 {
349  pc_token_t token;
350  int handle;
351  config_t *config;
352  qboolean parseOK = qtrue;
353 
354  handle = trap_PC_LoadSource(va("configs/%s.config", name));
355 
356  if (!handle)
357  {
358  Com_Printf(S_COLOR_RED "ERROR: File not found: %s\n", name);
359  return;
360  }
361 
362  Com_Memset(&level.config, 0, sizeof(config_t));
363 
364  G_wipeCvars();
365 
366  config = &level.config;
367 
368  config->publicConfig = qfalse;
369 
370  while (1)
371  {
372  if (!trap_PC_ReadToken(handle, &token))
373  {
374  break;
375  }
376 
377  if (!Q_stricmp(token.string, "configname"))
378  {
379  if (!PC_String_ParseNoAlloc(handle, config->name, sizeof(config->name)))
380  {
381  G_Printf("expected config name\n");
382  parseOK = qfalse;
383  break;
384  }
385  G_Printf("Config name is: %s\n", config->name);
386  }
387  else if (!Q_stricmp(token.string, "version"))
388  {
389  if (!PC_String_ParseNoAlloc(handle, config->version, sizeof(config->name)))
390  {
391  G_Printf("expected config version\n");
392  parseOK = qfalse;
393  break;
394  }
395  }
396  else if (!Q_stricmp(token.string, "init"))
397  {
398  if (!G_ParseSettings(handle, qtrue, config))
399  {
400  G_Printf("Reading settings failed\n");
401  parseOK = qfalse;
402  break;
403  }
404  }
405  else if (!Q_stricmp(token.string, "map"))
406  {
407  if (!G_ParseMapSettings(handle, config))
408  {
409  G_Printf("Reading map settings failed\n");
410  parseOK = qfalse;
411  break;
412  }
413  }
414  else if (!Q_stricmp(token.string, "signature"))
415  {
416  if (!PC_String_ParseNoAlloc(handle, config->signature, sizeof(config->signature)))
417  {
418  G_Printf("expected config signature\n");
419  parseOK = qfalse;
420  break;
421  }
422  }
423  else if (!Q_stricmp(token.string, "public"))
424  {
425  config->publicConfig = qtrue;
426  }
427  else
428  {
429  G_Printf("unknown token %s\n", token.string);
430  parseOK = qfalse;
431  break;
432  }
433  }
434  trap_PC_FreeSource(handle);
435 
436  if (parseOK)
437  {
438  trap_SetConfigstring(CS_CONFIGNAME, config->name);
439 
440  if (level.config.version[0] && level.config.name[0])
441  {
442  trap_SendServerCommand(-1, va("cp \"^7Config '%s^7' version '%s'^7 loaded\"", level.config.name, level.config.version));
443  }
444  else if (level.config.name[0])
445  {
446  trap_SendServerCommand(-1, va("cp \"^7Config '%s^7' loaded\"", level.config.name));
447  }
448  }
449  else
450  {
451  trap_SetConfigstring(CS_CONFIGNAME, "");
452  trap_SendServerCommand(-1, va("cp \"^7Config '%s^7' ^1FAILED ^7to load\"", name));
453  }
454 
455  G_UpdateCvars();
456 }
