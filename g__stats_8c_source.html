336 void G_ResetXP(gentity_t *ent)
337 {
338  int i = 0;
339  int ammo[MAX_WEAPONS], ammoclip[MAX_WEAPONS];
340  int oldWeapon; //, newWeapon;
341 
342  if (!ent || !ent->client)
343  {
344  return;
345  }
346 
347 #ifdef FEATURE_RATING
348  if (!g_skillRating.integer)
349 #endif
350  {
351  ent->client->sess.rank = 0;
352  }
353 
354  for (i = 0; i < SK_NUM_SKILLS; i++)
355  {
356  ent->client->sess.skillpoints[i] = 0.0f;
357  ent->client->sess.skill[i] = 0;
358  }
359 
360  G_CalcRank(ent->client);
361  ent->client->ps.stats[STAT_XP] = 0;
362  ent->client->ps.persistant[PERS_SCORE] = 0;
363 
364  // zero out all weapons and grab the default weapons for a player of this XP level.
365  // backup..
366  Com_Memcpy(ammo, ent->client->ps.ammo, sizeof(ammo));
367  Com_Memcpy(ammoclip, ent->client->ps.ammoclip, sizeof(ammoclip));
368  oldWeapon = ent->client->ps.weapon;
369  // Check weapon validity, maybe dump some weapons for this (now) unskilled player..
370  // It also sets the (possibly new) amounts of ammo for weapons.
371  SetWolfSpawnWeapons(ent->client);
372  // restore..
373  //newWeapon = ent->client->ps.weapon;
374 
375  for (i = WP_NONE; i < WP_NUM_WEAPONS; ++i) //i<MAX_WEAPONS
376  { // only restore ammo for valid weapons..
377  // ..they might have lost some weapons because of skill changes.
378  // Also restore to the old amount of ammo, because..
379  // ..SetWolfSpawnWeapons sets amount of ammo for a fresh spawning player,
380  // which is usually more than the player currently has left.
381  if (COM_BitCheck(ent->client->ps.weapons, i))
382  {
383  if (ammo[i] < ent->client->ps.ammo[i])
384  {
385  ent->client->ps.ammo[i] = ammo[i];
386  }
387  if (ammoclip[i] < ent->client->ps.ammoclip[i])
388  {
389  ent->client->ps.ammoclip[i] = ammoclip[i];
390  }
391  }
392  else
393  {
394  ent->client->ps.ammo[i] = 0;
395  ent->client->ps.ammoclip[i] = 0;
396  }
397  }
398  // check if the old weapon is still valid.
399  // If so, restore to the last used weapon..
400  if (COM_BitCheck(ent->client->ps.weapons, oldWeapon))
401  {
402  ent->client->ps.weapon = oldWeapon;
403  }
404  ClientUserinfoChanged(ent - g_entities);
405 }

528  {

531  }
