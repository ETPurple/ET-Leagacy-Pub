604 static int LAN_CompareServers(int source, int sortKey, int sortDir, int s1, int s2)
605 {
606  int res;
607  serverInfo_t *server1, *server2;
608  char name1[MAX_SERVER_NAME_LENGTH], name2[MAX_SERVER_NAME_LENGTH];
609

613  {
614  return 0;
615  }
616 
617  res = 0;
618  switch (sortKey)
619  {
620  case SORT_HOST:
621  //% res = Q_stricmp( server1->hostName, server2->hostName );
622  Q_strncpyz(name1, server1->hostName, sizeof(name1));

638  {

640  }

642  {
643  res = -1;
644  }

646  {
647  res = 1;
648  }
649  else
650  {
651  res = 0;
652  }
653  break;
654  case SORT_GAME:
655  if (server1->gameType < server2->gameType)

657  res = -1;
658  }
659  else if (server1->gameType > server2->gameType)
660  {
661  res = 1;

665  res = 0;
666  }
667  break;
