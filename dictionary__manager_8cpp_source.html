39 static bool has_suffix(const std::string& lhs, const std::string& rhs)
40 {
41  if (lhs.length() < rhs.length())
42  {
43  return false;
44  }
45  else
46  {
47  return lhs.compare(lhs.length() - rhs.length(), rhs.length(), rhs) == 0;

109 DictionaryManager::get_dictionary(const Language& language)
110 {
111  //log_debug << "Dictionary for language \"" << spec << "\" requested" << std::endl;
112  //log_debug << "...normalized as \"" << lang << "\"" << std::endl;
113  assert(language);
114 
115  Dictionaries::iterator i = dictionaries.find(language);
116  if (i != dictionaries.end())
117  {
118  return *i->second;
119  }
120  else // Dictionary for languages lang isn't loaded, so we load it
121  {
122  //log_debug << "get_dictionary: " << lang << std::endl;
123  Dictionary *dict = new Dictionary(charset);
124 
125  dictionaries[language] = dict;
126 
127  for (SearchPath::reverse_iterator p = search_path.rbegin(); p != search_path.rend(); ++p)
128  {
129  std::vector<std::string> files = filesystem->open_directory(*p);
130 
131  std::string best_filename;
132  int best_score = 0;
133 
134  for (std::vector<std::string>::iterator filename = files.begin(); filename != files.end(); ++filename)
135  {
136  // check if filename matches requested language
137  if (has_suffix(*filename, ".po"))
138  { // ignore anything that isn't a .po file
139 
140  Language po_language = Language::from_env(convertFilename2Language(*filename));
141 
142  if (!po_language)
143  {
144  log_warning << *filename << ": warning: ignoring, unknown language" << std::endl;
145  }
146  else
147  {
148  int score = Language::match(language, po_language);
149 
150  if (score > best_score)
151  {
152  best_score = score;
153  best_filename = *filename;
154  }
155  }
156  }
157  }
158 
159  if (!best_filename.empty())
160  {
161  std::string pofile = *p + "/" + best_filename;
162  try
163  {
164 #if __cplusplus >= 201103L // C++11
165  std::unique_ptr<std::istream> in = filesystem->open_file(pofile);
166 #else
167  std::auto_ptr<std::istream> in = filesystem->open_file(pofile);
168 #endif
169  if (!in.get())
170  {
171  log_error << "error: failure opening: " << pofile << std::endl;
172  }
173  else
174  {
175  POParser::parse(pofile, *in, *dict);
176  }
177  }
178  catch (std::exception& e)
179  {
180  log_error << "error: failure parsing: " << pofile << std::endl;
181  log_error << e.what() << "" << std::endl;
182  }
183  }
184  }
185 
186  if (!language.get_country().empty())
187  {
188  // printf("Adding language fallback %s\n", language.get_language().c_str());
189  dict->addFallback(&get_dictionary(Language::from_spec(language.get_language())));
190  }
191  return *dict;
192  }

298 std::string DictionaryManager::convertFilename2Language(const std::string &s_in) const
299 {
300  std::string s;
301  if (s_in.substr(s_in.size() - 3, 3) == ".po")
302  {
303  s = s_in.substr(0, s_in.size() - 3);
304  }
305  else
306  {
307  s = s_in;
308  }
309 
310  bool underscore_found = false;
311  for (unsigned int i = 0; i < s.size(); i++)
312  {
313  if (underscore_found)
314  {
315  // If we get a non-alphanumerical character/
316  // we are done (en_GB.UTF-8) - only convert
317  // the 'gb' part ... if we ever get this kind
318  // of filename.
319  if (!::isalpha(s[i]))
320  {
321  break;
322  }
323  s[i] = static_cast<char>(::toupper(s[i]));
324  }
325  else
326  {
327  underscore_found = s[i] == '_';
328  }
329  }
330  return s;
331 } // convertFilename2Language
