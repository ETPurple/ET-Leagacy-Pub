43 int CM_PointLeafnum_r(const vec3_t p, int num)
44 {
45  float d;

47  cplane_t *plane;
48 
49  while (num >= 0)
50  {
51  node = cm.nodes + num;
52  plane = node->plane;
53 
54  if (plane->type < 3)
55  {
56  d = p[plane->type] - plane->dist;
57  }

66  else

70  }
71 
72  c_pointcontents++; // optimize counter

347 int CM_TransformedPointContents(const vec3_t p, clipHandle_t model, const vec3_t origin, const vec3_t angles)
348 {
349  vec3_t p_l;
350 
351  // subtract origin offset
352  VectorSubtract(p, origin, p_l);
353 
354  // rotate start and end into the models frame of reference
355  if (model != BOX_MODEL_HANDLE &&
356  (angles[0] != 0.f || angles[1] != 0.f || angles[2] != 0.f))
357  {
358  vec3_t temp, forward, right, up;
359 
360  angles_vectors(angles, forward, right, up);
361 
362  VectorCopy(p_l, temp);
363  p_l[0] = DotProduct(temp, forward);
364  p_l[1] = -DotProduct(temp, right);
365  p_l[2] = DotProduct(temp, up);
366  }
367 
368  return CM_PointContents(p_l, model);
