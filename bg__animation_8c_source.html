320 long BG_StringHashValue(const char *fname)
321 {
322  int i;
323  long hash;
324 
325  if (!fname)
326  {
327  return -1;
328  }
329 
330  hash = 0;
331  i = 0;
332  while (fname[i] != '\0')
333  {
334  if (Q_isupper(fname[i]))
335  {
336  hash += (long)(fname[i] + ('a' - 'A')) * (i + 119);
337  }
338  else
339  {
340  hash += (long)(fname[i]) * (i + 119);
341  }
342 
343  i++;
344  }
345  if (hash == -1)
346  {
347  hash = 0; // never return -1
348  Com_Printf("BG_StringHash WARNING: fname with empty string returning 0");
349  }
350  return hash;
351 }

549 void BG_ParseConditionBits(char **text_pp, animStringItem_t *stringTable, int condIndex, int result[RESULT_SIZE])
550 {
551  qboolean endFlag = qfalse;
552  int indexFound;
553  int tempBits[2];
554  char currentString[MAX_QPATH];
555  qboolean minus = qfalse;
556  char *token;
557 
558  currentString[0] = '\0';
559  Com_Memset(result, 0, sizeof(result[0]) * RESULT_SIZE);
560  Com_Memset(tempBits, 0, sizeof(tempBits));
561 
562  while (!endFlag)
563  {
564  token = COM_ParseExt(text_pp, qfalse);
565  if (!token[0])
566  {
567  COM_RestoreParseSession(text_pp); // go back to the previous token
568  endFlag = qtrue; // done parsing indexes
569  if (!strlen(currentString))
570  {
571  break;
572  }
573  }
574 
575  if (!Q_stricmp(token, ","))
576  {
577  endFlag = qtrue; // end of indexes
578  }
579 
580  if (!Q_stricmp(token, "none")) // first bit is always the "unused" bit
581  {
582  COM_BitSet(result, 0);
583  continue;
584  }
585 
586  if (!Q_stricmp(token, "none,")) // first bit is always the "unused" bit
587  {
588  COM_BitSet(result, 0);
589  endFlag = qtrue; // end of indexes
590  continue;
591  }
592 
593  if (!Q_stricmp(token, "NOT"))
594  {
595  token = "MINUS"; // NOT is equivalent to MINUS
596  }
597 
598  if (!endFlag && Q_stricmp(token, "AND") && Q_stricmp(token, "MINUS")) // must be a index
599  { // check for a comma (end of indexes)
600  if (token[strlen(token) - 1] == ',')
601  {
602  endFlag = qtrue;
603  token[strlen(token) - 1] = '\0';
604  }
605  // append this to the currentString
606  if (strlen(currentString))
607  {
608  Q_strcat(currentString, sizeof(currentString), " ");
609  }
610  Q_strcat(currentString, sizeof(currentString), token);
611  }
612 
613  if (!Q_stricmp(token, "AND") || !Q_stricmp(token, "MINUS") || endFlag)
614  {
615  // process the currentString
616  if (!strlen(currentString))
617  {
618  if (endFlag)
619  {
620  BG_AnimParseError("BG_AnimParseAnimScript: unexpected end of condition");
621  }
622  else
623  {
624  // check for minus indexes to follow
625  if (!Q_stricmp(token, "MINUS"))
626  {
627  minus = qtrue;
628  continue;
629  }
630  BG_AnimParseError("BG_AnimParseAnimScript: unexpected '%s'", token);
631  }
632  }
633  if (!Q_stricmp(currentString, "all"))
634  {
635  tempBits[0] = ~0x0;
636  tempBits[1] = ~0x0;
637  }
638  else
639  {
640  // first check this string with our defines
641  indexFound = BG_IndexForString(currentString, defineStr[condIndex], qtrue);
642  if (indexFound >= 0)
643  {
644  // we have precalculated the bitflags for the defines
645  tempBits[0] = defineBits[condIndex][indexFound][0];
646  tempBits[1] = defineBits[condIndex][indexFound][1];
647  }
648  else
649  {
650  // convert the string into an index
651  indexFound = BG_IndexForString(currentString, stringTable, qfalse);
652  // convert the index into a bitflag
653  COM_BitSet(tempBits, indexFound);
654  }
655  }
656  // perform operation
657  if (minus) // subtract
658  {
659  result[0] &= ~tempBits[0];
660  result[1] &= ~tempBits[1];
661  }
662  else // add
663  {
664  result[0] |= tempBits[0];
665  result[1] |= tempBits[1];
666  }
667  // clear the currentString
668  currentString[0] = '\0';
669  // check for minus indexes to follow
670  if (!Q_stricmp(token, "MINUS"))
671  {
672  minus = qtrue;
673  }
674  }
675  }
676 }

684 qboolean BG_ParseConditions(char **text_pp, animScriptItem_t *scriptItem)
685 {
686  int conditionIndex, conditionValue[2];
687  char *token;
688  qboolean minus;
689 
690  conditionValue[0] = 0;
691  conditionValue[1] = 0;
692 
693  while (1)
694  {
695  token = COM_ParseExt(text_pp, qfalse);
696  if (!token[0])
697  {
698  break;
699  }
700 
701  // special case, "default" has no conditions
702  if (!Q_stricmp(token, "default"))
703  {
704  return qtrue;
705  }
706 
707  if (!Q_stricmp(token, "NOT") || !Q_stricmp(token, "MINUS"))
708  {
709  minus = qtrue;
710 
711  token = COM_ParseExt(text_pp, qfalse);
712  //Com_Printf("NOT: %s \n", token );
713  if (!token[0])
714  {
715  break;
716  }
717  }
718  else
719  {
720  minus = qfalse;
721  }
722 
723 
724  conditionIndex = BG_IndexForString(token, animConditionsStr, qfalse);
725 
726  switch (animConditionsTable[conditionIndex].type)
727  {
728  case ANIM_CONDTYPE_BITFLAGS:
729  BG_ParseConditionBits(text_pp, animConditionsTable[conditionIndex].values, conditionIndex, conditionValue);
730  break;
731  case ANIM_CONDTYPE_VALUE:
732  if (animConditionsTable[conditionIndex].values)
733  {
734  token = COM_ParseExt(text_pp, qfalse);
735  if (!token[0])
736  {
737  BG_AnimParseError("BG_AnimParseAnimScript: expected condition value, found end of line");
738  }
739  // check for a comma (condition divider)
740  if (token[strlen(token) - 1] == ',')
741  {
742  token[strlen(token) - 1] = '\0';
743  }
744  conditionValue[0] = BG_IndexForString(token, animConditionsTable[conditionIndex].values, qfalse);
745  }
746  else
747  {
748  conditionValue[0] = 1; // not used, just check for a positive condition
749  }
750  break;
751  default:
752  BG_AnimParseError("BG_AnimParseAnimScript: unknown condition type");
753  break;
754  }
755 
756  // now append this condition to the item
757  scriptItem->conditions[scriptItem->numConditions].index = conditionIndex;
758  scriptItem->conditions[scriptItem->numConditions].value[0] = conditionValue[0];
759  scriptItem->conditions[scriptItem->numConditions].value[1] = conditionValue[1];
760  scriptItem->conditions[scriptItem->numConditions].negative = minus;
761  scriptItem->numConditions++;
762  }
763 
764  if (scriptItem->numConditions == 0)
765  {
766  BG_AnimParseError("BG_ParseConditions: no conditions found");
767  }
768 
769  return qtrue;
770 }
