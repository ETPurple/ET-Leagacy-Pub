559 static void CL_ParseDemoSnapShotSimple(msg_t *msg)
560 {
561  int len;
562  clSnapshot_t *old;
563  clSnapshot_t newSnap;
564  int deltaNum;
565  int oldMessageNum;
566  int i, packetNum;
567 
568  Com_Memset(&newSnap, 0, sizeof(newSnap));
569  newSnap.serverCommandNum = clc.serverCommandSequence;
570  newSnap.serverTime = MSG_ReadLong(msg);
571  newSnap.messageNum = clc.serverMessageSequence;
572  deltaNum = MSG_ReadByte(msg);
573  if (!deltaNum)
574  {
575  newSnap.deltaNum = -1;
576  }
577  else
578  {
579  newSnap.deltaNum = newSnap.messageNum - deltaNum;
580  }
581  newSnap.snapFlags = MSG_ReadByte(msg);
582 
583  if (newSnap.deltaNum <= 0)
584  {
585  newSnap.valid = qtrue; // uncompressed frame
586  old = NULL;
587  }
588  else
589  {
590  old = &cl.snapshots[newSnap.deltaNum & PACKET_MASK];
591  if (!old->valid)
592  {
593  // should never happen
594  Com_FuncPrinf("Delta from invalid frame (not supposed to happen!).\n");
595  }
596  else if (old->messageNum != newSnap.deltaNum)
597  {
598  // The frame that the server did the delta from
599  // is too old, so we can't reconstruct it properly.
600  Com_FuncPrinf("Delta frame too old.\n");
601  }
602  else if (cl.parseEntitiesNum - old->parseEntitiesNum > MAX_PARSE_ENTITIES - 128)
603  {
604  Com_FuncPrinf("Delta parseEntitiesNum too old.\n");
605  }
606  else
607  {
608  newSnap.valid = qtrue; // valid delta parse
609  }
610  }
611 
612  // read areamask
613  len = MSG_ReadByte(msg);
614 
615  if (len > sizeof(newSnap.areamask))
616  {
617  Com_FuncDrop("Invalid size %d for areamask.", len);
618  }
619 
620  MSG_ReadData(msg, &newSnap.areamask, len);
621 
622  if (old)
623  {
624  MSG_ReadDeltaPlayerstate(msg, &old->ps, &newSnap.ps);
625  }
626  else
627  {
628  MSG_ReadDeltaPlayerstate(msg, NULL, &newSnap.ps);
629  }
630 
631  // read packet entities
632  CL_ParsePacketEntities(msg, old, &newSnap);
633 
634  // if not valid, dump the entire thing now that it has
635  // been properly read
636  if (!newSnap.valid)
637  {
638  return;
639  }
640 
641  // clear the valid flags of any snapshots between the last
642  // received and this one, so if there was a dropped packet
643  // it won't look like something valid to delta from next
644  // time we wrap around in the buffer
645  oldMessageNum = cl.snap.messageNum + 1;
646 
647  if (newSnap.messageNum - oldMessageNum >= PACKET_BACKUP)
648  {
649  oldMessageNum = newSnap.messageNum - (PACKET_BACKUP - 1);
650  }
651  for (; oldMessageNum < newSnap.messageNum; oldMessageNum++)
652  {
653  cl.snapshots[oldMessageNum & PACKET_MASK].valid = qfalse;
654  }
655 
656  // copy to the current good spot
657  cl.snap = newSnap;
658  cl.snap.ping = 999;
659  // calculate ping time
660  for (i = 0; i < PACKET_BACKUP; i++)
661  {
662  packetNum = (clc.netchan.outgoingSequence - 1 - i) & PACKET_MASK;
663  if (cl.snap.ps.commandTime >= cl.outPackets[packetNum].p_serverTime)
664  {
665  cl.snap.ping = cls.realtime - cl.outPackets[packetNum].p_realtime;
666  break;
667  }
668  }
669  // save the frame off in the backup array for later delta comparisons
670  cl.snapshots[cl.snap.messageNum & PACKET_MASK] = cl.snap;
671  cl.newSnapshots = qtrue;
672 }
