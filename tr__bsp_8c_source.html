54 static void HSVtoRGB(float h, float s, float v, float rgb[3])
55 {
56  int i;
57  float f;
58  float p, q, t;
59 
60  h *= 5;
61 
62  i = floor(h);
63  f = h - i;
64 
65  p = v * (1 - s);
66  q = v * (1 - s * f);
67  t = v * (1 - s * (1 - f));
68 
69  switch (i)
70  {
71  case 0:
72  rgb[0] = v;
73  rgb[1] = t;
74  rgb[2] = p;
75  break;
76  case 1:
77  rgb[0] = q;
78  rgb[1] = v;
79  rgb[2] = p;
80  break;
81  case 2:
82  rgb[0] = p;
83  rgb[1] = v;
84  rgb[2] = t;
85  break;
86  case 3:
87  rgb[0] = p;
88  rgb[1] = q;
89  rgb[2] = v;
90  break;
91  case 4:
92  rgb[0] = t;
93  rgb[1] = p;
94  rgb[2] = v;
95  break;
96  case 5:
97  rgb[0] = v;
98  rgb[1] = p;
99  rgb[2] = q;
100  break;
101  }
102 }

109 static void R_ColorShiftLightingBytes(byte in[4], byte out[4])
110 {
111  // shift the color data based on overbright range
112  int shift = r_mapOverBrightBits->integer - tr.overbrightBits;
113  // shift the data based on overbright range
114  int r = in[0] << shift;
115  int g = in[1] << shift;
116  int b = in[2] << shift;
117 
118  // normalize by color instead of saturating to white
119  if ((r | g | b) > 255)
120  {
121  int max = r > g ? r : g;
122 
123  max = max > b ? max : b;
124  r = r * 255 / max;
125  g = g * 255 / max;
126  b = b * 255 / max;
127  }
128 
129  out[0] = (byte)r;
130  out[1] = (byte)g;
131  out[2] = (byte)b;
132  out[3] = in[3];
133 }

144 float R_ProcessLightmap(byte *pic, int in_padding, int width, int height, byte *pic_out)
145 {
146  int j;
147  float maxIntensity = 0;
148 
149  if (r_lightMap->integer > 1) // color code by intensity as development tool (FIXME: check range)
150  {
151  //double sumIntensity = 0;
152  float r, g, b, intensity;
153  float out[3] = { 0, 0, 0 };
154 
155  for (j = 0; j < width * height; j++)
156  {
157  r = pic[j * in_padding + 0];
158  g = pic[j * in_padding + 1];
159  b = pic[j * in_padding + 2];
160 
161  intensity = 0.33f * r + 0.685f * g + 0.063f * b;
162 
163  if (intensity > 255)
164  {
165  intensity = 1.0f;
166  }
167  else
168  {
169  intensity /= 255.0f;
170  }
171 
172  if (intensity > maxIntensity)
173  {
174  maxIntensity = intensity;
175  }
176 
177  HSVtoRGB(intensity, 1.00, 0.50, out);
178 
179  if (r_lightMap->integer == 3)
180  {
181  // Arnout: artists wanted the colours to be inversed
182  pic_out[j * 4 + 0] = (byte)(out[2] * 255);
183  pic_out[j * 4 + 1] = (byte)(out[1] * 255);
184  pic_out[j * 4 + 2] = (byte)(out[0] * 255);
185  }
186  else
187  {
188  pic_out[j * 4 + 0] = (byte)(out[0] * 255);
189  pic_out[j * 4 + 1] = (byte)(out[1] * 255);
190  pic_out[j * 4 + 2] = (byte)(out[2] * 255);
191  }
192  pic_out[j * 4 + 3] = 255;
193 
194  //sumIntensity += intensity;
195  }
196  }
197  else
198  {
199  for (j = 0 ; j < width * height; j++)
200  {
201  R_ColorShiftLightingBytes(&pic[j * in_padding], &pic_out[j * 4]);
202  pic_out[j * 4 + 3] = 255;
203  }
204  }
205 
206  return maxIntensity;
207 }

437  int lightmapNum;
438  vec3_t bounds[2];
439  vec3_t tmpVec;
440 
441  lightmapNum = LittleLong(ds->lightmapNum);
442 
443  // get fog volume
444  surf->fogIndex = LittleLong(ds->fogNum) + 1;
445 
446  // get shader value

448  if (r_singleShader->integer && !surf->shader->isSky)
449  {

466  for (i = 0 ; i < numPoints ; i++)
467  {

469  {
470  points[i].xyz[j] = LittleFloat(verts[i].xyz[j]);

472  }
