95 void moveit(gentity_t *ent, float yaw, float dist)
96 {
97  vec3_t move;
98  vec3_t origin;
99  trace_t tr;
100  vec3_t mins, maxs;
101 
102  yaw = (float)(yaw * M_TAU_F / 360);
103 
104  move[0] = (float)cos((double)yaw) * dist;
105  move[1] = (float)sin((double)yaw) * dist;
106  move[2] = 0;
107 
108  VectorAdd(ent->r.currentOrigin, move, origin);
109 
110  mins[0] = ent->r.mins[0];
111  mins[1] = ent->r.mins[1];
112  mins[2] = ent->r.mins[2] + .01f;
113 
114  maxs[0] = ent->r.maxs[0];
115  maxs[1] = ent->r.maxs[1];
116  maxs[2] = ent->r.maxs[2] - .01f;
117 
118  trap_Trace(&tr, ent->r.currentOrigin, mins, maxs, origin, ent->s.number, MASK_SHOT);
119 
120  if ((tr.endpos[0] != origin[0]) || (tr.endpos[1] != origin[1]))
121  {
122  mins[0] = ent->r.mins[0] - 2.0f;
123  mins[1] = ent->r.mins[1] - 2.0f;
124  maxs[0] = ent->r.maxs[0] + 2.0f;
125  maxs[1] = ent->r.maxs[1] + 2.0f;
126 
127  trap_Trace(&tr, ent->r.currentOrigin, mins, maxs, origin, ent->s.number, MASK_SHOT);
128  }
129 
130  VectorCopy(tr.endpos, ent->r.currentOrigin);
131 
132  VectorCopy(ent->r.currentOrigin, ent->s.pos.trBase);
133 
134  trap_LinkEntity(ent);
135 }
