412  Cmd_TokenizeString(s);

415  return qtrue;
416  }
417

420  return qfalse;
421  }
422 
423  // the clientLevelShot command is used during development
424  // to generate 128*128 screenshots from the intermission

426  // we pass it along to the cgame to make apropriate adjustments,

447  return qtrue;

1098  token = COM_Parse(&buftrav); // read the size
