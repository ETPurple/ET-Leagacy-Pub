54 {
55  int i;
56 
57  if (!level.spawning)
58  {
59  *out = (char *)defaultString;

62  }
63 
64  for (i = 0 ; i < level.numSpawnVars ; i++)
65  {
66  if (!strcmp(key, level.spawnVars[i][0]))
67  {
68  *out = level.spawnVars[i][1];
69  return qtrue;
70  }
71  }
72 
73  *out = (char *)defaultString;
74  return qfalse;

917  {

919  }
920
