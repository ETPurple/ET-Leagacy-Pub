675 void S_Play_f(void)
676 {
677  static char tempBuffer[MAX_QPATH];
678  int i;
679  int c;
680  sfxHandle_t h;
681 
682 
683  if (!si.RegisterSound || !si.StartLocalSound)
684  {
685  return;
686  }
687 
688  c = Cmd_Argc();
689 
690  if (c < 2)
691  {
692  Com_Printf("Usage: play <sound filename> [sound filename] [sound filename] ...\n");
693  return;
694  }
695 
696  for (i = 1; i < c; i++)
697  {
698  Q_strncpyz(tempBuffer, Cmd_Argv(i), MAX_QPATH);
699  COM_DefaultExtension(tempBuffer, sizeof(tempBuffer), ".wav");
700 
701  h = si.RegisterSound(tempBuffer, qfalse); // TODO: detect compression via extension? ioq uses qfalse by default
702 
703  if (h)
704  {
705  si.StartLocalSound(h, CHAN_LOCAL_SOUND, 127);
706  }
707  else
708  {
709  Com_Printf("Warning: S_Play_f sound '%s' not played\n", tempBuffer);
710  }
711  }

810 void S_Init(void)
811 {
812  cvar_t *cv = Cvar_Get("s_initsound", "2", CVAR_ARCHIVE | CVAR_LATCH | CVAR_UNSAFE); // 0 = disabled, 1 = SDL2, 2 = OpenAL
813 
814  Com_Printf("----- Initializing Sound (%i) ---\n", cv->integer);
815 
816  s_volume = Cvar_Get("s_volume", "0.8", CVAR_ARCHIVE);
817  s_musicVolume = Cvar_Get("s_musicvolume", "0.25", CVAR_ARCHIVE);
818  s_muted = Cvar_Get("s_muted", "0", CVAR_ROM);
819  s_doppler = Cvar_Get("s_doppler", "1", CVAR_ARCHIVE);
820  s_backend = Cvar_Get("s_backend", "", CVAR_ROM);
821  s_muteWhenMinimized = Cvar_Get("s_muteWhenMinimized", "1", CVAR_ARCHIVE);
822  s_muteWhenUnfocused = Cvar_Get("s_muteWhenUnfocused", "0", CVAR_ARCHIVE);
823 
824  if (!cv->integer)
825  {
826  Com_Printf("Sound disabled\n");
827  }
828  else
829  {
830  qboolean started = qfalse;
831 
832  S_CodecInit();
833 
834  Cmd_AddCommand("play", S_Play_f, "Plays a given sound file.");
835  Cmd_AddCommand("music", S_Music_f, "Starts background track.");
836  Cmd_AddCommand("stopmusic", S_StopMusic_f, "Stops background track.");
837  Cmd_AddCommand("stream", S_Stream_f, "Starts streaming sound.");
838  Cmd_AddCommand("s_list", S_SoundList, "Prints a list of available sounds.");
839  Cmd_AddCommand("s_stop", S_StopAllSounds, "Stops all sounds.");
840  Cmd_AddCommand("s_info", S_SoundInfo, "Prints sound info.");
841 
842 #ifdef FEATURE_OPENAL
843  if (cv->integer == 2)
844  {
845  // OpenAL
846  started = S_AL_Init(&si);
847  Cvar_Set("s_backend", "OpenAL");
848  }
849 
850  if (!started)
851 #endif // FEATURE_OPENAL
852  {
853  if (cv->integer == 2)
854  {
855 #ifdef FEATURE_OPENAL
856  Com_Printf("Can't initialize OpenAL - reverting to SDL2 interface\n");
857 #else
858  Com_Printf("Can't initialize OpenAL - disabled on build-time\n");
859 #endif
860  }
861 
862  started = S_Base_Init(&si);
863  Cvar_Set("s_backend", "SDL2");
864  }
865 
866  if (started)
867  {
868  if (!S_ValidSoundInterface(&si))
869  {
870  Com_Error(ERR_FATAL, "Invalid sound interface");
871  }
872 
873  S_SoundInfo();
874  Com_Printf("Sound initialization successfully done\ns_backend set to %s\n", s_backend->string);
875  }
876  else
877  {
878  Com_Printf("Sound initialization failed\n");
879  }
880  }
881 
882  Com_Printf("--------------------------------\n");
883 }
