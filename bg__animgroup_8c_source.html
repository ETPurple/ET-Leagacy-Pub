174

178  }
179

181  {
182  animation->flags |= ANIMFL_REVERSED;
183  }
184 
185  // calculate the duration
186  animation->duration = animation->initialLerp
187  + animation->frameLerp * animation->numFrames
188  + animation->animBlend;
189

191  animation->nameHash = BG_StringHashValue(animation->name);
192 
193  // hacky-ish stuff
194  if (!Q_strncmp(animation->name, "climb", 5))

197  }
198 
199  if (strstr(animation->name, "firing"))

202  animation->initialLerp = 40;
203  }

218 static qboolean BG_RAG_ParseAnimFile(int handle, animModelInfo_t *animModelInfo)
219 {
220  pc_token_t token;
221 #ifdef USE_MDXFILE
222  qhandle_t mdxFile;
223 #else
224  char mdxFileName[MAX_QPATH];
225 #endif // USE_MDXFILE
226 
227  animation_t *animation;
228 
229  if (!trap_PC_ReadToken(handle, &token))
230  {
231  return BG_RAG_ParseError(handle, "expected mdx filename");
232  }
233 
234 #ifdef USE_MDXFILE
235  if (!(mdxFile = trap_R_RegisterModel(token.string)))
236  {
237  return BG_RAG_ParseError(handle, "failed to load %s", token.string);
238  }
239 #else
240  Q_strncpyz(mdxFileName, token.string, sizeof(mdxFileName));
241 #endif // USE_MDXFILE
242 
243  if (!trap_PC_ReadToken(handle, &token) || Q_stricmp(token.string, "{"))
244  {
245  return BG_RAG_ParseError(handle, "expected '{'");
246  }
247 
248  while (1)
249  {
250  if (!trap_PC_ReadToken(handle, &token))
251  {
252  return BG_RAG_ParseError(handle, "unexpected EOF");
253  }
254 
255  if (token.string[0] == '}')
256  {
257  break;
258  }
259 
260 #ifdef USE_MDXFILE
261  if (!(animation = BG_RAG_FindFreeAnimation(mdxFile, token.string)))
262  {
263 #else
264  if (!(animation = BG_RAG_FindFreeAnimation(mdxFileName, token.string)))
265  {
266 #endif // USE_MDXFILE
267  return BG_RAG_ParseError(handle, "out of animation storage space");
268  }
269 
270  Q_strncpyz(animation->name, token.string, sizeof(animation->name));
271  Q_strlwr(animation->name);
272 
273  if (!BG_RAG_ParseAnimation(handle, animation))
274  {
275  return qfalse;
276  }
277 
278  animModelInfo->animations[animModelInfo->numAnimations] = animation;
279  animModelInfo->numAnimations++;
280  }
281 
282  return qtrue;
283 }
