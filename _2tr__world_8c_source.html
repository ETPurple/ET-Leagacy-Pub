132 static qboolean R_CullSurface(surfaceType_t *surface, shader_t *shader, int *frontFace)
133 {
134  srfGeneric_t *gen;
135  int cull;
136 
137  // force to non-front facing
138  *frontFace = 0;
139 
140  // allow culling to be disabled
141  if (r_noCull->integer)
142  {
143  return qfalse;
144  }
145 
146  // made surface culling generic, inline with q3map2 surface classification
147  switch (*surface)
148  {
149  case SF_FACE:
150  case SF_TRIANGLES:
151  break;
152  case SF_GRID:
153  if (r_noCurves->integer)
154  {
155  return qtrue;
156  }
157  break;
158  case SF_FOLIAGE:
159  if (r_drawFoliage->value == 0.f)
160  {
161  return qtrue;
162  }
163  break;
164  default:
165  return qtrue;
166  }
167 
168  // get generic surface
169  gen = (srfGeneric_t *) surface;
170 
171  // plane cull
172  if (gen->plane.type != PLANE_NON_PLANAR && r_facePlaneCull->integer)
173  {
174  float d;
175 
176  d = DotProduct(tr.orientation.viewOrigin, gen->plane.normal) - gen->plane.dist;
177  if (d > 0.0f)
178  {
179  *frontFace = 1;
180  }
181 
182  // don't cull exactly on the plane, because there are levels of rounding
183  // through the BSP, ICD, and hardware that may cause pixel gaps if an
184  // epsilon isn't allowed here
185  if (shader->cullType == CT_FRONT_SIDED)
186  {
187  if (d < -8.0f)
188  {
189  tr.pc.c_plane_cull_out++;
190  return qtrue;
191  }
192  }
193  else if (shader->cullType == CT_BACK_SIDED)
194  {
195  if (d > 8.0f)
196  {
197  tr.pc.c_plane_cull_out++;
198  return qtrue;
199  }
200  }
201 
202  tr.pc.c_plane_cull_in++;
203  }
204 
205  // try sphere cull
206  if (tr.currentEntity != &tr.worldEntity)
207  {
208  cull = R_CullLocalPointAndRadius(gen->origin, gen->radius);
209  }
210  else
211  {
212  cull = R_CullPointAndRadius(gen->origin, gen->radius);
213  }
214 
215  if (cull == CULL_OUT)
216  {
217  tr.pc.c_sphere_cull_out++;
218  return qtrue;
219  }
220 
221  tr.pc.c_sphere_cull_in++;
222 
223  // must be visible
224  return qfalse;
225 }

606 static void R_RecursiveWorldNode(bspNode_t *node, int planeBits, int decalBits)
607 {
608  do
609  {
610  // if the node wasn't marked as potentially visible, exit
611  if (node->visCounts[tr.visIndex] != tr.visCounts[tr.visIndex])
612  {
613  return;
614  }
615 
616  if (node->contents != -1 && !node->numMarkSurfaces)
617  {
618  // don't waste time dealing with this empty leaf
619  return;
620  }
621 
622  // if the bounding volume is outside the frustum, nothing
623  // inside can be visible OPTIMIZE: don't do this all the way to leafs?
624  if (!r_noCull->integer)
625  {
626  int i;
627  int r;
628 
629  for (i = 0; i < FRUSTUM_PLANES; i++)
630  {
631  if (planeBits & (1 << i))
632  {
633  r = BoxOnPlaneSide(node->mins, node->maxs, &tr.viewParms.frustums[0][i]);
634  if (r == 2)
635  {
636  return; // culled
637  }
638  if (r == 1)
639  {
640  planeBits &= ~(1 << i); // all descendants will also be in front
641  }
642  }
643  }
644  }
645 
646  InsertLink(&node->visChain, &tr.traversalStack);
647 
648  // cull decals
649  if (decalBits)
650  {
651  int i;
652 
653  for (i = 0; i < tr.refdef.numDecalProjectors; i++)
654  {
655  if (decalBits & (1 << i))
656  {
657  // test decal bounds against node surface bounds
658  if (tr.refdef.decalProjectors[i].shader == NULL ||
659  !R_TestDecalBoundingBox(&tr.refdef.decalProjectors[i], node->surfMins, node->surfMaxs))
660  {
661  decalBits &= ~(1 << i);
662  }
663  }
664  }
665  }
666 
667  if (node->contents != -1)
668  {
669  break;
670  }
671 
672  // recurse down the children, front side first
673  R_RecursiveWorldNode(node->children[0], planeBits, decalBits);
674 
675  // tail recurse
676  node = node->children[1];
677  }
678  while (1);
679 
680  if (node->numMarkSurfaces)
681  {

683  R_AddLeafSurfaces(node, decalBits);
684  }
