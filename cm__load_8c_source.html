506 void CMod_LoadPatches(lump_t *surfs, lump_t *verts)
507 {
508  drawVert_t *dv, *dv_p;
509  dsurface_t *in;
510  int count;
511  int i, j;
512  int c;
513  cPatch_t *patch;
514  vec3_t points[MAX_PATCH_VERTS];
515  int width, height;
516  int shaderNum;
517 
518  in = ( void * )(cmod_base + surfs->fileofs);
519  if (surfs->filelen % sizeof(*in))
520  {
521  Com_Error(ERR_DROP, "CMod_LoadPatches: funny lump size");
522  }
523  cm.numSurfaces = count = surfs->filelen / sizeof(*in);
524  cm.surfaces = Hunk_Alloc(cm.numSurfaces * sizeof(cm.surfaces[0]), h_high);
525 
526  dv = ( void * )(cmod_base + verts->fileofs);
527  if (verts->filelen % sizeof(*dv))
528  {
529  Com_Error(ERR_DROP, "CMod_LoadPatches: funny lump size");
530  }
531 
532  // scan through all the surfaces, but only load patches,
533  // not planar faces
534  for (i = 0 ; i < count ; i++, in++)
535  {
536  if (LittleLong(in->surfaceType) != MST_PATCH)
537  {
538  continue; // ignore other surfaces
539  }
540  // FIXME: check for non-colliding patches
541 
542  cm.surfaces[i] = patch = Hunk_Alloc(sizeof(*patch), h_high);
543 
544  // load the full drawverts onto the stack
545  width = LittleLong(in->patchWidth);
546  height = LittleLong(in->patchHeight);
547  c = width * height;
548  if (c > MAX_PATCH_VERTS)
549  {
550  Com_Error(ERR_DROP, "CMod_LoadPatches: ParseMesh - MAX_PATCH_VERTS");
551  }
552 
553  dv_p = dv + LittleLong(in->firstVert);
554  for (j = 0 ; j < c ; j++, dv_p++)
555  {
556  points[j][0] = LittleFloat(dv_p->xyz[0]);
557  points[j][1] = LittleFloat(dv_p->xyz[1]);
558  points[j][2] = LittleFloat(dv_p->xyz[2]);
559  }
560 
561  shaderNum = LittleLong(in->shaderNum);
562  patch->contents = cm.shaders[shaderNum].contentFlags;
563  patch->surfaceFlags = cm.shaders[shaderNum].surfaceFlags;
564 
565  // create the internal facet structure
566  patch->pc = CM_GeneratePatchCollide(width, height, points, qtrue);
567  }
568 }

612 void CM_LoadMap(const char *name, qboolean clientload, unsigned int *checksum)
613 {
614  union
615  {
616  int *i;
617  void *v;
618  } buf;
619  unsigned int i;
620  dheader_t header;
621  int length;
622  static unsigned last_checksum;
623 
624  if (!name || !name[0])
625  {
626  Com_Error(ERR_DROP, "CM_LoadMap: NULL name");
627  }
628 
629  cm_noAreas = Cvar_Get("cm_noAreas", "0", CVAR_CHEAT);
630  cm_noCurves = Cvar_Get("cm_noCurves", "0", CVAR_CHEAT);
631  cm_playerCurveClip = Cvar_Get("cm_playerCurveClip", "1", CVAR_ARCHIVE | CVAR_CHEAT);
632  cm_optimize = Cvar_Get("cm_optimize", "1", CVAR_CHEAT);
633 
634  Com_DPrintf("CM_LoadMap( %s, %i )\n", name, clientload);
635 
636  if (!strcmp(cm.name, name) && clientload)
637  {
638  *checksum = last_checksum;
639  return;
640  }
641 
642  // free old stuff
643  Com_Memset(&cm, 0, sizeof(cm));
644  CM_ClearLevelPatches();
645 
646  if (!name[0])
647  {
648  cm.numLeafs = 1;
649  cm.numClusters = 1;
650  cm.numAreas = 1;
651  cm.cmodels = Hunk_Alloc(sizeof(*cm.cmodels), h_high);
652  *checksum = 0;
653  return;
654  }
655 
656  // load the file
657  length = FS_ReadFile(name, &buf.v);
658 
659  if (!buf.i || length <= 0)
660  {
661  Com_Error(ERR_DROP, "CM_LoadMap: Couldn't load %s", name);
662  }
663 
664  last_checksum = LittleLong(Com_BlockChecksum(buf.i, length));
665  *checksum = last_checksum;
666 
667  header = *(dheader_t *)buf.i;
668  for (i = 0 ; i < sizeof(dheader_t) / 4 ; i++)
669  {
670  ((int *)&header)[i] = LittleLong(((int *)&header)[i]);
671  }
672 
673  if (header.version != BSP_VERSION)
674  {
675  Com_Error(ERR_DROP, "CM_LoadMap: %s has wrong version number (%i should be %i)"
676  , name, header.version, BSP_VERSION);
677  }
678 
679  cmod_base = (byte *)buf.i;
680 
681  // load into heap
682  CMod_LoadShaders(&header.lumps[LUMP_SHADERS]);
683  CMod_LoadLeafs(&header.lumps[LUMP_LEAFS]);
684  CMod_LoadLeafBrushes(&header.lumps[LUMP_LEAFBRUSHES]);
685  CMod_LoadLeafSurfaces(&header.lumps[LUMP_LEAFSURFACES]);
686  CMod_LoadPlanes(&header.lumps[LUMP_PLANES]);
687  CMod_LoadBrushSides(&header.lumps[LUMP_BRUSHSIDES]);
688  CMod_LoadBrushes(&header.lumps[LUMP_BRUSHES]);
689  CMod_LoadSubmodels(&header.lumps[LUMP_MODELS]);
690  CMod_LoadNodes(&header.lumps[LUMP_NODES]);
691  CMod_LoadEntityString(&header.lumps[LUMP_ENTITIES]);
692  CMod_LoadVisibility(&header.lumps[LUMP_VISIBILITY]);
693  CMod_LoadPatches(&header.lumps[LUMP_SURFACES], &header.lumps[LUMP_DRAWVERTS]);
694 
695  // we are NOT freeing the file, because it is cached for the ref
696  FS_FreeFile(buf.v);
697 
698  CM_InitBoxHull();
699 
700  CM_FloodAreaConnections();
701 
702  // allow this to be cached if it is loaded by the server
703  if (!clientload)
704  {
705  Q_strncpyz(cm.name, name, sizeof(cm.name));
706  }
707 }
