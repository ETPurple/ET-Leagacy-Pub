179  }

182  hha[i].initialLerp = (int)(1000 / fps);
183 
184  token = COM_Parse(&text_p); // looping frames
185 
186  if (!token[0])
187  {
188  break;
189  }
190 
191  hha[i].loopFrames = atoi(token);
192 
193  if (hha[i].loopFrames > hha[i].numFrames)
194  {
195  hha[i].loopFrames = hha[i].numFrames;
196  }
197  else if (hha[i].loopFrames < 0)
198  {
199  hha[i].loopFrames = 0;
200  }
201  }
202 
203  if (i != MAX_HD_ANIMATIONS)
204  {
205  CG_Printf("Error parsing hud head animation file: %s", filename);
206  return qfalse;
207  }
208 
209  return qtrue;

217 static void CG_CalcMoveSpeeds(bg_character_t *character)
218 {
219  const char *tags[2] = { "tag_footleft", "tag_footright" };
220  vec3_t oldPos[2] = { { 0, 0 } };
221  refEntity_t refent;
222  animation_t *anim;
223  int i, j, k;
224  float totalSpeed;
225  int numSpeed;
226  int low;
227  orientation_t o[2];
228 
229  Com_Memset(&refent, 0, sizeof(refent));
230 
231  refent.hModel = character->mesh;
232 
233  for (i = 0; i < character->animModelInfo->numAnimations; i++)
234  {
235  anim = character->animModelInfo->animations[i];
236 
237  if (anim->moveSpeed >= 0)
238  {
239  continue;
240  }
241 
242  totalSpeed = 0;
243  numSpeed = 0;
244 
245  // for each frame
246  for (j = 0; j < anim->numFrames; j++)
247  {
248  refent.frame = anim->firstFrame + j;
249  refent.oldframe = refent.frame;
250  refent.torsoFrameModel = refent.oldTorsoFrameModel = refent.frameModel = refent.oldframeModel = anim->mdxFile;
251 
252  // for each foot
253  for (k = 0; k < 2; k++)
254  {
255  if (trap_R_LerpTag(&o[k], &refent, tags[k], 0) < 0)
256  {
257  CG_Error("CG_CalcMoveSpeeds: unable to find tag %s, cannot calculate movespeed\n", tags[k]);
258  }
259  }
260 
261  // find the contact foot
262  if (anim->flags & ANIMFL_LADDERANIM)
263  {
264  if (o[0].origin[0] > o[1].origin[0])
265  {
266  low = 0;
267  }
268  else
269  {
270  low = 1;
271  }
272  totalSpeed += Q_fabs(oldPos[low][2] - o[low].origin[2]);
273  }
274  else
275  {
276  if (o[0].origin[2] < o[1].origin[2])
277  {
278  low = 0;
279  }
280  else
281  {
282  low = 1;
283  }
284  totalSpeed += Q_fabs(oldPos[low][0] - o[low].origin[0]);
285  }
286 
287  numSpeed++;
288 
289  // save the positions
290  for (k = 0; k < 2; k++)
291  {
292  VectorCopy(o[k].origin, oldPos[k]);
293  }
294  }
295 
296  // record the speed
297  anim->moveSpeed = round(((totalSpeed / numSpeed) * 1000.0f / anim->frameLerp));
298  }
299 }
