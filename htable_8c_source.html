171 hashtable_t HT_Create(
172  size_t size,
173  unsigned int flags,
174  size_t item_size,
175  size_t key_offset,
176  size_t key_length
177  )
178 {
179  hashtable_t table;
180  size_t real_size;
181  struct listhead_t *t_item;
182 
183  // Allocate table
184  real_size = _HT_NextPrime(size);
185  table = Z_Malloc(sizeof(struct hashtable_s) + real_size * sizeof(struct listhead_t));
186  etl_assert(table);
187 
188  // Initialise main table fields
189  table->size = real_size;
190  table->flags = flags;
191  table->item_size = item_size;
192  table->key_offset = key_offset;
193  table->key_length = key_length;
194  RESET_LIST(&table->all_items);
195 
196  // Set functions
197  table->GetKey = (flags & HT_FLAG_CASE) ? _HT_GetKey : _HT_GetCIKey;
198  table->CompareKey = (flags & HT_FLAG_CASE) ? strcmp : Q_stricmp; // Q_strcasecmp
199  if ((flags & HT_FLAG_INTABLE) == 0)
200  {
201  table->KeyFromEntry = key_length ? _HT_KeyFromEntryPI : _HT_KeyFromEntryPP;
202  }
203  else
204  {
205  table->KeyFromEntry = key_length ? _HT_KeyFromEntryII : _HT_KeyFromEntryIP;
206  }
207 
208  // Initialise table entries
209  t_item = TABLE_START(table);
210  while (real_size > 0)
211  {
212  RESET_LIST(t_item);
213  t_item++;
214  real_size--;
215  }
216 
217  return table;
218 }

262 void *HT_GetItem(
263  hashtable_t table,
264  const char *key,
265  qboolean *created
266  )
267 {
268  unsigned int hash;
269  struct listhead_t *list_head;
270  struct listhead_t *list_entry;
271  struct tentry_t *t_entry;
272  void *data;
273 
274  etl_assert(table->key_length == 0 || table->key_length >= strlen(key));
275 
276  // Try finding the item
277  hash = table->GetKey(key);
278  list_head = (TABLE_START(table) + (hash % table->size));
279  list_entry = list_head->next;
280  while (list_entry != list_head)
281  {
282  t_entry = ( struct tentry_t * ) list_entry;
283 
284  if (t_entry->hash > hash)
285  {
286  break;
287  }
288 
289  if (t_entry->hash == hash)
290  {
291  char *item_key = table->KeyFromEntry(t_entry, table->key_offset);
292  if (!table->CompareKey(key, item_key))
293  {
294  data = (void *)(((char *)t_entry) + sizeof(struct tentry_t));
295  if (created != NULL)
296  {
297  *created = qfalse;
298  }
299  return (table->flags & HT_FLAG_INTABLE) ? data : (*(void **)data);
300  }
301  }
302 
303  list_entry = list_entry->next;
304  }
305 
306  // Check if we can create the entry
307  if (created == NULL)
308  {
309  return NULL;
310  }
311 
312  // Create entry
313  *created = qtrue;
314  t_entry = _HT_CreateEntry(table, hash, list_entry, key);
315 
316  // Initialise data
317  data = (void *)(((char *)t_entry) + sizeof(struct tentry_t));
318  if ((table->flags & HT_FLAG_INTABLE) == 0)
319  {
320  *(void **) data = Z_Malloc(table->item_size);
321  data = *(void **) data;
322  }
323  Com_Memset(data, 0, table->item_size);
324 
325  // Copy key
326  if (table->key_length == 0)
327  {
328  char **key_ptr = (char **)(((char *) data) + table->key_offset);
329  *key_ptr = Z_Malloc(strlen(key) + 1);
330  strcpy(*key_ptr, key);
331  }
332  else
333  {
334  char *key_ptr = ((char *) data) + table->key_offset;
335  strcpy(key_ptr, key);
336  }
337 
338  return data;
339 }

357 void *HT_PutItem(
358  hashtable_t table,
359  void *item,
360  qboolean allow_replacement
361  )
362 {
363  void *ret_val = NULL;
364  void *prev_entry = NULL;
365  const char *insert_key;
366  unsigned int hash;
367  struct listhead_t *list_head;
368  struct listhead_t *list_entry;
369  struct tentry_t *t_entry = NULL;
370 
371  // Extract item key
372  if (table->key_length)
373  {
374  insert_key = ((const char *) item) + table->key_offset;
375  }
376  else
377  {
378  insert_key = *(const char **)(((char *) item) + table->key_offset);
379  }
380 
381  // Try finding an item with that key, or the new item's location
382  hash = table->GetKey(insert_key);
383  list_head = (TABLE_START(table) + (hash % table->size));
384  list_entry = list_head->next;
385  while (list_entry != list_head)
386  {
387  t_entry = ( struct tentry_t * ) list_entry;
388 
389  if (t_entry->hash > hash)
390  {
391  break;
392  }
393 
394  if (t_entry->hash == hash)
395  {
396  const char *item_key = table->KeyFromEntry(t_entry, table->key_offset);
397  int cres = table->CompareKey(insert_key, item_key);
398  if (!cres)
399  {
400  prev_entry = (((char *)t_entry) + sizeof(struct tentry_t));
401  ret_val = (table->flags & HT_FLAG_INTABLE) ? prev_entry : (*(void **)prev_entry);
402  if (!allow_replacement)
403  {
404  return ret_val;
405  }
406  break;
407  }
408  else if (cres > 0)
409  {
410  break;
411  }
412  }
413 
414  list_entry = list_entry->next;
415  }
416 
417  if (ret_val != NULL)
418  {
419  // Delete previous item's key if it was a pointer and either
420  // items are in-table or should be freed automatically
421  if (table->key_length == 0 && (table->flags & (HT_FLAG_INTABLE | HT_FLAG_FREE)) != 0)
422  {
423  Z_Free(table->KeyFromEntry(t_entry, table->key_offset));
424  }
425 
426  if ((table->flags & HT_FLAG_INTABLE) != 0)
427  {
428  // Copy item data
429  Com_Memcpy(prev_entry, item, table->item_size);
430  ret_val = NULL;
431  }
432  else
433  {
434  if ((table->flags & HT_FLAG_FREE) != 0)
435  {
436  // Free previous item
437  Z_Free(ret_val);
438  ret_val = NULL;
439  }
440  *(void **) prev_entry = item;
441  }
442  }
443  else
444  {
445  void *data;
446 
447  t_entry = _HT_CreateEntry(table, hash, list_entry, insert_key);
448  data = (void *)(((char *)t_entry) + sizeof(struct tentry_t));
449  if ((table->flags & HT_FLAG_INTABLE) != 0)
450  {
451  Com_Memcpy(data, item, table->item_size);
452  }
453  else
454  {
455  *(void **) data = item; // FIXME: never used
456  }
457  }
458 
459  return ret_val;
460 }

475 qboolean HT_DeleteItem(
476  hashtable_t table,
477  const char *key,
478  void **found
479  )
480 {
481  unsigned int hash;
482  struct listhead_t *list_head;
483  struct listhead_t *list_entry;
484  struct tentry_t *t_entry = NULL;
485  void *data = NULL;
486 
487  // Try finding the item
488  hash = table->GetKey(key);
489  list_head = (TABLE_START(table) + (hash % table->size));
490  list_entry = list_head->next;
491  while (list_entry != list_head)
492  {
493  t_entry = ( struct tentry_t * ) list_entry;
494 
495  if (t_entry->hash > hash)
496  {
497  break;
498  }
499 
500  if (t_entry->hash == hash)
501  {
502  char *item_key = table->KeyFromEntry(t_entry, table->key_offset);
503  if (!table->CompareKey(key, item_key))
504  {
505  data = (void *)(((char *)t_entry) + sizeof(struct tentry_t));
506  data = (table->flags & HT_FLAG_INTABLE) ? data : (*(void **)data);
507  break;
508  }
509  }
510 
511  list_entry = list_entry->next;
512  }
513 
514  // Did we find it?
515  if (data == NULL)
516  {
517  if (found != NULL)
518  {
519  *found = NULL;
520  }
521  return qfalse;
522  }
523 
524  // Detach it from its lists
525  t_entry->loc_list.previous->next = t_entry->loc_list.next;
526  t_entry->loc_list.next->previous = t_entry->loc_list.previous;
527  t_entry->full_list.previous->next = t_entry->full_list.next;
528  t_entry->full_list.next->previous = t_entry->full_list.previous;
529 
530  // Delete key
531  if (table->key_length == 0 && (table->flags & (HT_FLAG_INTABLE | HT_FLAG_FREE)) != 0)
532  {
533  Z_Free(table->KeyFromEntry(t_entry, table->key_offset));
534  }
535 
536  // Delete item
537  if ((table->flags & (HT_FLAG_INTABLE | HT_FLAG_FREE)) == HT_FLAG_FREE)
538  {
539  Z_Free(data);
540  }
541 
542  // Delete entry
543  Z_Free(t_entry);
544 
545  // Set found pointer
546  if (found != NULL)
547  {
548  if ((table->flags & (HT_FLAG_INTABLE | HT_FLAG_FREE)) != 0)
549  {
550  data = NULL;
551  }
552  *found = data;
553  }
554 
555  return qtrue;
556 }

608 static qboolean _HT_IsPrime(size_t n)
609 {
610  size_t temp;
611  size_t nsq;
612  size_t inc;
613 
614  if (n == 0)
615  {
616  return qfalse;
617  }
618 
619  nsq = ceil(sqrt((double)n));
620  for (inc = 1, temp = 2 ; temp <= nsq ; temp += inc)
621  {
622  if (n % temp == 0)
623  {
624  return qfalse;
625  }
626  if (temp == 3)
627  {
628  inc = 2;
629  }
630  }
631  return qtrue;
632 }

656 static unsigned int _HT_GetCIKey(const char *key)
657 {
658  const char *current = key;
659  unsigned int hash = 111119;
660 
661  while (*current)
662  {
663  hash += (unsigned char)tolower(*current);
664  hash += (hash << 10);
665  hash ^= (hash >> 6);
666  current++;
667  }
668 
669  hash += (hash << 3);
670  hash ^= (hash >> 11);
671  hash += (hash << 15);
672 
673  return hash;
674 }

803 static void _HT_InsertInGlobalList(hashtable_t table, struct tentry_t *t_entry, const char *key)
804 {
805  if ((table->flags & HT_FLAG_SORTED) == 0)
806  {
807  // Append to global list
808  t_entry->full_list.previous = table->all_items.previous;
809  t_entry->full_list.next = &(table->all_items);
810  table->all_items.previous = t_entry->full_list.previous->next = &(t_entry->full_list);
811  }
812  else
813  {
814  // Global list must be kept sorted, find insert location
815  struct listhead_t *list_entry = table->all_items.next;
816 
817  while (list_entry != &(table->all_items))
818  {
819  struct tentry_t *ai_entry;
820  const char *ai_key;
821  int cres;
822 
823  ai_entry = (struct tentry_t *)(((char *) list_entry) - HT_OffsetOfField(struct tentry_t, full_list));
824  ai_key = table->KeyFromEntry(ai_entry, table->key_offset);
825  cres = table->CompareKey(ai_key, key);
826 
827  etl_assert(cres != 0);
828  if (cres > 0)
829  {
830  break;
831  }
832  list_entry = list_entry->next;
833  }
834  t_entry->full_list.previous = list_entry->previous;
