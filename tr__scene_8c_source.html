141 void RE_AddPolyToScene(qhandle_t hShader, int numVerts, const polyVert_t *verts)
142 {

176  // see if it is in a fog volume
177  if (tr.world->numfogs == 1)
178  {
179  fogIndex = 0;
180  }
181  else
182  {
183  int i;
184

186  VectorCopy(poly->verts[0].xyz, bounds[0]);
187  VectorCopy(poly->verts[0].xyz, bounds[1]);
188  for (i = 1 ; i < poly->numVerts ; i++)
189  {
190  AddPointToBounds(poly->verts[i].xyz, bounds[0], bounds[1]);
191  }
192  for (fogIndex = 1 ; fogIndex < tr.world->numfogs ; fogIndex++)
193  {
194  fog = &tr.world->fogs[fogIndex];
195  if (bounds[1][0] >= fog->bounds[0][0]
196  && bounds[1][1] >= fog->bounds[0][1]
197  && bounds[1][2] >= fog->bounds[0][2]
198  && bounds[0][0] <= fog->bounds[1][0]
199  && bounds[0][1] <= fog->bounds[1][1]
200  && bounds[0][2] <= fog->bounds[1][2])
201  {
202  break;
203  }
204  }
205  if (fogIndex == tr.world->numfogs)
206  {
207  fogIndex = 0;
208  }
209  }

587

591  tr.refdef.areamaskModified = qtrue;
592  }
593  }
594 
595  // derived info
596 
597  tr.refdef.floatTime = tr.refdef.time * 0.001;
598
