197 const char *String_Alloc(const char *p)
198 {
199  int len;
200  long hash;
201  stringDef_t *str, *last;
202  static const char *staticNULL = "";
203 
204  if (p == NULL)
205  {
206  return NULL;
207  }
208 
209  if (*p == 0)
210  {
211  return staticNULL;
212  }
213 
214  hash = hashForString(p);
215 
216  str = strHandle[hash];
217  while (str)
218  {
219  if (strcmp(p, str->str) == 0)
220  {
221  return str->str;
222  }
223 
224  str = str->next;
225  }
226 
227  len = strlen(p);
228  if (len + strPoolIndex + 1 < STRING_POOL_SIZE)
229  {
230  int ph = strPoolIndex;
231 
232  strcpy(&strPool[strPoolIndex], p);
233  strPoolIndex += len + 1;
234 
235  str = strHandle[hash];
236  last = str;
237  while (str && str->next)
238  {
239  str = str->next;
240  last = str;
241  }
242 
243  str = UI_Alloc(sizeof(stringDef_t));
244  str->next = NULL;
245  str->str = &strPool[ph];
246  if (last)
247  {
248  last->next = str;
249  }
250  else
251  {
252  strHandle[hash] = str;
253  }
254 
255  return &strPool[ph];
256  }
257 
258  return NULL;
259 }
