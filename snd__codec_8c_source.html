70 static snd_codec_t *S_FindCodecForFile(const char *filename)
71 {
72  char *ext = S_FileExtension(filename);
73  snd_codec_t *codec = codecs;
74 
75  if (!ext)
76  {
77  // No extension - auto-detect
78  while (codec)
79  {
80  char fn[MAX_QPATH];
81 
82  // there is no extension so we do not need to subtract 4 chars
83  Q_strncpyz(fn, filename, MAX_QPATH);
84  COM_DefaultExtension(fn, MAX_QPATH, codec->ext);
85 
86  // Check it exists
87  if (FS_ReadFile(fn, NULL) > 0)
88  {
89  return codec;
90  }
91 
92  // Nope. Next!
93  codec = codec->next;
94  }
95 
96  // Nothin'
97  return NULL;
98  }
99 
100  while (codec)
101  {
102  if (!Q_stricmp(ext, codec->ext))
103  {
104  return codec;
105  }
106  codec = codec->next;
107  }
108 
109  return NULL;
110 }
