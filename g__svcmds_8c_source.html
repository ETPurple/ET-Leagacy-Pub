47 /*
48 ==============================================================================
49 PACKET FILTERING
50 
51 You can add or remove addresses from the filter list with:
52 
53 addip <ip>
54 removeip <ip>
55 
56 The ip address is specified in dot format, and you can use '*' to match any value
57 so you can specify an entire class C network with "addip 192.246.40.*"
58 
59 Removeip will only remove an address specified exactly the same way. You cannot addip a subnet, then removeip a single host.
60 
61 listip
62 Prints the current list of filters.
63 
64 g_filterban <0 or 1>
65 
66 If 1 (the default), then ip addresses matching the current list will be prohibited from entering the game. This is the default setting.
67 
68 If 0, then only addresses matching the list will be allowed. This lets you easily set up a private game, or a game that only allows players from your local network.
69 
70 TTimo NOTE: GUID functions are copied over from the model of IP banning,
71 used to enforce max lives independently from server reconnect and team changes (Xian)
72 
73 TTimo NOTE: for persistence, bans are stored in g_banIPs cvar MAX_CVAR_VALUE_STRING
74 The size of the cvar string buffer is limiting the banning to around 20 masks
75 this could be improved by putting some g_banIPs2 g_banIps3 etc. maybe
76 still, you should rely on PB for banning instead
77 ==============================================================================
78 */

106 qboolean StringToFilter(const char *s, ipFilter_t *f)
107 {
108  char num[128];
109  int i, j;
110  byte b[4];
111  byte m[4];
112 
113  for (i = 0 ; i < 4 ; i++)
114  {
115  b[i] = 0;
116  m[i] = 0;
117  }
118 
119  for (i = 0 ; i < 4 ; i++)
120  {
121  if (*s < '0' || *s > '9')
122  {
123  if (*s == '*') // 'match any'
124  { // b[i] and m[i] to 0
125  s++;
126  if (!*s)
127  {
128  break;
129  }
130  s++;
131  continue;
132  }
133  G_Printf("Bad filter address: %s\n", s);
134  return qfalse;
135  }
136 
137  j = 0;
138  while (*s >= '0' && *s <= '9')
139  {
140  num[j++] = *s++;
141  }
142  num[j] = 0;
143  b[i] = atoi(num);
144  m[i] = 255;
145 
146  if (!*s)
147  {
148  break;
149  }
150  s++;
151  }
152 
153  f->mask = *(unsigned *)m;
154  f->compare = *(unsigned *)b;
155 
156  return qtrue;
157 }

163 static void UpdateIPBans(ipFilterList_t *ipFilterList)
164 {
165  byte b[4];
166  byte m[4];
167  int i, j;
168  char iplist_final[MAX_CVAR_VALUE_STRING];

173  {

176  continue;
177  }
178 
179  *(unsigned *)b = ipFilterList->ipFilters[i].compare;
