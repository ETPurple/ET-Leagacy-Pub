194 void MSG_WriteBits(msg_t *msg, int value, int bits)
195 {
196  oldsize += bits;
197 
198  msg->uncompsize += bits; // net debugging
199 
200  if (msg->overflowed)
201  {
202  return;
203  }
204 
205  if (bits == 0 || bits < -31 || bits > 32)
206  {
207  Com_Error(ERR_DROP, "MSG_WriteBits: bad bits %i", bits);
208  }
209 
210  if (bits < 0)
211  {
212  bits = -bits;
213  }
214 
215  if (msg->oob)
216  {
217  if (msg->cursize + (bits >> 3) > msg->maxsize)
218  {
219  msg->overflowed = qtrue;
220  return;
221  }
222 
223  switch (bits)
224  {
225  case 8:
226  msg->data[msg->cursize] = value;
227  msg->cursize += 1;
228  msg->bit += 8;
229  break;
230  case 16:
231  {
232  unsigned short *sp = (unsigned short *)&msg->data[msg->cursize];
233 
234  *sp = LittleShort(value);
235  msg->cursize += 2;
236  msg->bit += 16;
237  }
238  break;
239  case 32:
240  {
241  unsigned int *ip = (unsigned int *)&msg->data[msg->cursize];
242 
243  *ip = LittleLong(value);
244  msg->cursize += 4;
245  msg->bit += 32;
246  }
247  break;
248  default:
249  Com_Error(ERR_DROP, "MSG_WriteBits: can't read %d bits", bits);
250  break;
251  }
252  }
253  else
254  {
255  int i;
256 
257  value &= (0xffffffff >> (32 - bits));
258  if (bits & 7)
259  {
260  int nbits = bits & 7;
261 
262  if (msg->bit + nbits > msg->maxsize << 3)
263  {
264  msg->overflowed = qtrue;
265  return;
266  }
267 
268  for (i = 0; i < nbits; i++)
269  {
270  Huff_putBit((value & 1), msg->data, &msg->bit);
271  value = (value >> 1);
272  }
273  bits = bits - nbits;
274  }
275  if (bits)
276  {
277  for (i = 0; i < bits; i += 8)
278  {
279  Huff_offsetTransmit(&msgHuff.compressor, (value & 0xff), msg->data, &msg->bit, msg->maxsize << 3);
280  value = (value >> 8);
281 
282  if (msg->bit > msg->maxsize << 3)
283  {
284  msg->overflowed = qtrue;
285  return;
286  }
287  }
288  }
289  msg->cursize = (msg->bit >> 3) + 1;
290  }
291 }

299 int MSG_ReadBits(msg_t *msg, int bits)
300 {
301  int value = 0;
302  qboolean sgn;
303 
304  if (msg->readcount > msg->cursize)
305  {
306  return 0;
307  }
308 
309  if (bits < 0)
310  {
311  bits = -bits;
312  sgn = qtrue;
313  }
314  else
315  {
316  sgn = qfalse;
317  }
318 
319  if (msg->oob)
320  {
321  if (msg->readcount + (bits >> 3) > msg->cursize)
322  {
323  msg->readcount = msg->cursize + 1;
324  return 0;
325  }
326 
327  switch (bits)
328  {
329  case 8:
330  value = msg->data[msg->readcount];
331  msg->readcount += 1;
332  msg->bit += 8;
333  break;
334  case 16:
335  {
336  unsigned short *sp = (unsigned short *)&msg->data[msg->readcount];
337 
338  value = LittleShort(*sp);
339  msg->readcount += 2;
340  msg->bit += 16;
341  }
342  break;
343  case 32:
344  {
345  unsigned int *ip = (unsigned int *)&msg->data[msg->readcount];
346 
347  value = LittleLong(*ip);
348  msg->readcount += 4;
349  msg->bit += 32;
350  }
351  break;
352  default:
353  Com_Error(ERR_DROP, "MSG_ReadBits: can't read %d bits", bits);
354  break;
355  }
356  }
357  else
358  {
359  int i, nbits = 0;
360 
361  if (bits & 7)
362  {
363  nbits = bits & 7;
364 
365  if (msg->bit + nbits > msg->cursize << 3)
366  {
367  msg->readcount = msg->cursize + 1;
368  return 0;
369  }
370 
371  for (i = 0; i < nbits; i++)
372  {
373  value |= (Huff_getBit(msg->data, &msg->bit) << i);
374  }
375  bits = bits - nbits;
376  }
377  if (bits)
378  {
379  int get;
380 
381  for (i = 0; i < bits; i += 8)
382  {
383  Huff_offsetReceive(msgHuff.decompressor.tree, &get, msg->data, &msg->bit, msg->cursize << 3);
384  value = (unsigned int)value | ((unsigned int)get << (i + nbits));
385 
386  if (msg->bit > msg->cursize << 3)
387  {
388  msg->readcount = msg->cursize + 1;
389  return 0;
390  }
391  }
392  }
393  msg->readcount = (msg->bit >> 3) + 1;
394  }
395  if (sgn && bits > 0 && bits < 32)
396  {
397  if (value & (1 << (bits - 1)))
398  {
399  value |= -1 ^ ((1 << bits) - 1);
400  }
401  }
402 
403  return value;
404 }
