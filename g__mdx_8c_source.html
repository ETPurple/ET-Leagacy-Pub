224 static void mdx_quaternion_to_matrix(vec4_t q, vec3_t m[3])
225 {
226  m[0][0] = 1 - 2 * (q[1] * q[1] + q[2] * q[2]);

229

231  m[1][1] = 1 - 2 * (q[0] * q[0] + q[2] * q[2]);
232  m[1][2] = 2 * (q[1] * q[2] + q[0] * q[3]);
233 
234  m[2][0] = 2 * (q[0] * q[2] + q[1] * q[3]);
235  m[2][1] = 2 * (q[1] * q[2] - q[0] * q[3]);

246 static void mdx_quaternion_nlerp(const vec4_t q1, const vec4_t q2, vec4_t qout, float backlerp)
247 {
248  float fwdlerp = 1.0 - backlerp;
249  float len;
250 
251  qout[0] = q1[0] * backlerp + q2[0] * fwdlerp;
252  qout[1] = q1[1] * backlerp + q2[1] * fwdlerp;
253  qout[2] = q1[2] * backlerp + q2[2] * fwdlerp;
254  qout[3] = q1[3] * backlerp + q2[3] * fwdlerp;
255 
256  len = sqrt(qout[0] * qout[0] + qout[1] * qout[1] + qout[2] * qout[2] + qout[3] * qout[3]);
257  if (len)
258  {
259  qout[0] /= len;
260  qout[1] /= len;
261  qout[2] /= len;
262  qout[3] /= len;
263  }
264  else
265  {
266  // very rare -- quaternions pointing in opposite direction with backlerp 0.5
267  // since they're exactly opposite, they're the same rotation :x
268  Vector4Copy(q1, qout);
269  }
270 }
