159  {
160  return; // shouldn't happen, since we check the clip[] above, except for FP rounding
161  }
162 
163  // see if a flare with a matching surface, scene, and view exists
164  for (f = r_activeFlares ; f ; f = f->next)

171  }
172

174  if (!f)

192
