560 {
561  byte *ip = NULL;
562  size_t size = 0;

565

569  break;

600 static leakyBucket_t *SVC_BucketForAddress(netadr_t address, int burst, int period)
601 {
602  leakyBucket_t *bucket = NULL;
603  int i;
604  long hash = SVC_HashForAddress(address);
605  int now = Sys_Milliseconds();
606 
607  for (bucket = bucketHashes[hash]; bucket; bucket = bucket->next)
608  {
609  switch (bucket->type)
610  {
611  case NA_IP:
612  if (memcmp(bucket->ipv._4, address.ip, 4) == 0)
613  {
614  return bucket;
615  }
616  break;
617  case NA_IP6:
618  if (memcmp(bucket->ipv._6, address.ip6, 16) == 0)
619  {
620  return bucket;
621  }
622  break;
623  default:
624  break;
625  }
626  }
627 
628  // make sure we will never use time 0
629  now = now ? now : 1;
630 
631  for (i = 0; i < MAX_BUCKETS; i++)
632  {
633  int interval;
634 
635  bucket = &buckets[i];
636  interval = now - bucket->lastTime;
637 
638  // Reclaim expired buckets
639  if (bucket->lastTime > 0 && (unsigned) interval > (burst * period))
640  {
641  if (bucket->prev != NULL)
642  {
643  bucket->prev->next = bucket->next;
644  }
645  else
646  {
647  bucketHashes[bucket->hash] = bucket->next;
648  }
649 
650  if (bucket->next != NULL)
651  {
652  bucket->next->prev = bucket->prev;
653  }
654 
655  Com_Memset(bucket, 0, sizeof(leakyBucket_t));
656  }
657 
658  if (bucket->type == NA_BAD)
659  {
660  bucket->type = address.type;
661  switch (address.type)
662  {
663  case NA_IP: Com_Memcpy(bucket->ipv._4, address.ip, 4);
664  break;
665  case NA_IP6: Com_Memcpy(bucket->ipv._6, address.ip6, 16);
666  break;
667  default:
668  break;
669  }
670 
671  bucket->lastTime = now;
672  bucket->burst = 0;
673  bucket->hash = hash;
674 
675  // Add to the head of the relevant hash chain
676  bucket->next = bucketHashes[hash];
677  if (bucketHashes[hash] != NULL)
678  {
679  bucketHashes[hash]->prev = bucket;
680  }
681 
682  bucket->prev = NULL;
683  bucketHashes[hash] = bucket;
684 
685  return bucket;
686  }
687  }

692

706 {
