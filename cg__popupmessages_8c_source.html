163 void CG_UpdatePMLists(void)
164 {
165  pmListItem_t *listItem;
166  pmListItem_t *lastItem;
167  pmListItemBig_t *listItem2;
168 
169  if ((listItem = cg_pmWaitingList))
170  {
171  int t = listItem->time;
172 
173  if (cg.time > t)
174  {
175  if (listItem->next)
176  {
177  // there's another item waiting to come on, so move to old list
178  cg_pmWaitingList = listItem->next;
179  cg_pmWaitingList->time = cg.time; // set time we popped up at
180 
181  CG_AddToListFront(&cg_pmOldList, listItem);
182  }
183  else
184  {
185  if (cg.time > t + cg_popupStayTime.integer + cg_popupFadeTime.integer)
186  {
187  // we're gone completely
188  cg_pmWaitingList = NULL;
189  listItem->inuse = qfalse;
190  listItem->next = NULL;
191  }
192  //else
193  //{ // just sit where we are, no pressure to do anything...
194  //}
195  }
196  }
197  }
198 
199  listItem = cg_pmOldList;
200  lastItem = NULL;
201  while (listItem)
202  {
203  int t = listItem->time + cg_popupStayTime.integer + cg_popupFadeTime.integer;
204 
205  if (cg.time > t)
206  {
207  // nuke this, and everything below it (though there shouldn't BE anything below us anyway)
208  pmListItem_t *next;
209 
210  if (!lastItem)
211  {
212  // we're the top of the old list, so set to NULL
213  cg_pmOldList = NULL;
214  }
215  else
216  {
217  lastItem->next = NULL;
218  }
219 
220  do
221  {
222  next = listItem->next;
223 
224  listItem->next = NULL;
225  listItem->inuse = qfalse;
226  }
227  while ((listItem = next));
228 
229 
230  break;
231  }
232 
233  lastItem = listItem;
234  listItem = listItem->next;
235  }
236 
237  if ((listItem2 = cg_pmWaitingListBig))
238  {
239  int t = PM_BIGPOPUP_TIME + listItem2->time;
240 
241  if (cg.time > t)
242  {
243  if (listItem2->next)

343 static qboolean CG_CheckPMItemFilter(popupMessageType_t type)
344 {
345  switch (type)
346  {
347  case PM_CONNECT:
348  if (cg_popupFilter.integer & POPUP_FILTER_CONNECT)
349  {
350  return qtrue;
351  }
352  break;
353  case PM_TEAM:
354  if (cg_popupFilter.integer & POPUP_FILTER_TEAMJOIN)
355  {
356  return qtrue;
357  }
358  break;
359  case PM_MESSAGE:
360  case PM_DYNAMITE:
361  case PM_CONSTRUCTION:
362  case PM_MINES:
363  case PM_OBJECTIVE:
364  case PM_DESTRUCTION:
365  if (cg_popupFilter.integer & POPUP_FILTER_MISSION)
366  {
367  return qtrue;
368  }
369  break;
370  case PM_AMMOPICKUP:
371  case PM_HEALTHPICKUP:
372  case PM_WEAPONPICKUP:
373  if (cg_popupFilter.integer & POPUP_FILTER_PICKUP)
374  {
375  return qtrue;
376  }
377  break;
378  case PM_DEATH:
379  if (cg_popupFilter.integer & POPUP_FILTER_DEATH)
380  {
381  return qtrue;
382  }
383  break;
384  default:
385  break;
386  }
387  return qfalse;
388 }

400 void CG_AddPMItem(popupMessageType_t type, const char *message, const char *message2, qhandle_t shader, qhandle_t weaponShader, int scaleShader, vec3_t color)
401 {
402  pmListItem_t *listItem;
403  char *end;
404 
405  if (!message || !*message)
406  {
407  return;
408  }
409 
410  if (type >= PM_NUM_TYPES)
411  {
412  CG_Printf("Invalid popup type: %d\n", type);
413  return;
414  }
415 
416  if (CG_CheckPMItemFilter(type))
417  {
418  // log filtered pm to console. Death obituaries are logged in CG_Obituary().
419  if (type != PM_DEATH)
420  {
421  trap_Print(va("%s\n", message));
422  }
423  return;
424  }
425 
426  listItem = CG_FindFreePMItem();
427 
428  if (!listItem)
429  {
430  return;
431  }
432 
433  if (shader)
434  {
435  listItem->shader = shader;
436  }
437  else
438  {
439  listItem->shader = -1;
440  }
441 
442  if (message2)
443  {
444  listItem->weaponShader = weaponShader;
445  listItem->scaleShader = scaleShader;
446  }
447  else
448  {
449  listItem->weaponShader = -1;
450  }
451 
452  // colored obituaries
453  listItem->color[0] = listItem->color[1] = listItem->color[2] = 1.f;
454  if (color != NULL)
455  {
456  VectorCopy(color, listItem->color);
457  }
458 
459  listItem->inuse = qtrue;
460  listItem->type = type;
461  Q_strncpyz(listItem->message, message, sizeof(cg_pmStack[0].message));
462 
463  // print and THEN chop off the newline, as the console deals with newlines perfectly
464  if (listItem->message[strlen(listItem->message) - 1] == '\n')
465  {
466  listItem->message[strlen(listItem->message) - 1] = 0;
467  }
468 
469  // do not write obituary popups into console - we'll get double kill-messages otherwise
470  if (type != PM_DEATH)
471  {
472  trap_Print(va("%s\n", listItem->message)); // FIXME: translate this (does it makes sense?)
473  }
474 
475  // chop off the newline at the end if any
476  while ((end = strchr(listItem->message, '\n')))
477  {
478  *end = '\0';
479  }
480  // don't eat popups for empty lines
481  if (*listItem->message == '\0')
482  {
483  return;
484  }
485 
486  if (message2)
487  {
488  Q_strncpyz(listItem->message2, message2, sizeof(cg_pmStack[0].message2));
489 
490  if (listItem->message[strlen(listItem->message2) - 1] == '\n')
491  {
492  listItem->message[strlen(listItem->message2) - 1] = 0;
493  }
494 
495  while ((end = strchr(listItem->message2, '\n')))
496  {
497  *end = '\0';
498  }
499 
500  if (*listItem->message2 == '\0')
501  {
502  return;
503  }
504  }
505 
506  if (!cg_pmWaitingList)
507  {
508  cg_pmWaitingList = listItem;
509  listItem->time = cg.time;
510  }
511  else
512  {
513  pmListItem_t *loop = cg_pmWaitingList;
514 
515  while (loop->next)
516  {
517  loop = loop->next;
518  }
519 
520  loop->next = listItem;
521  }
522 }
