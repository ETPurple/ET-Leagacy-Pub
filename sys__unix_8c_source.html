96 void Sys_Chmod(const char *file, int mode)
97 {
98  struct stat stat_infoBefore;
99  struct stat fstat_infoAfter;
100  int perm;
101  int fd;
102 
103  // Get the state of the file
104  if (stat(file, &stat_infoBefore) == -1)
105  {
106  Com_Printf("Sys_Chmod: first stat('%s') failed: errno %d\n", file, errno);
107  return;
108  }
109 
110  perm = stat_infoBefore.st_mode | mode;
111 
112  // Try to get the file descriptor
113  if ((fd = open(file, O_RDONLY, S_IXUSR)) == -1)
114  {
115  Com_Printf("Sys_Chmod: open('%s', %d, %d) failed: errno %d\n", file, O_RDONLY, S_IXUSR, errno);
116  return;
117  }
118 
119  Com_DPrintf("chmod +%d '%s'\n", mode, file);
120 
121  // Get the state of the file after opening function
122  if (stat(file, &fstat_infoAfter) == -1)
123  {
124  Com_Printf("Sys_Chmod: second stat('%s') failed: errno %d\n", file, errno);
125  close(fd);
126  return;
127  }
128 
129  // Compare the state before and after opening
130  if (stat_infoBefore.st_mode != fstat_infoAfter.st_mode ||
131  stat_infoBefore.st_ino != fstat_infoAfter.st_ino)
132  {
133  Com_Printf("Sys_Chmod: stat before and after open are different. The file ('%s') may differ (TOCTOU Attacks ?)\n", file);
134  close(fd);
135  return;
136  }
137 
138  // Chmod the file by using the file descriptor
139  if (fchmod(fd, perm) != 0)
140  {
141  Com_Printf("Sys_Chmod: fchmod('%s', %d) failed: errno %d\n", file, perm, errno);
142  }
143 
144  close(fd);
145  Com_DPrintf("chmod +%d '%s'\n", mode, file);
146 }

261 FILE *Sys_FOpen(const char *ospath, const char *mode)
262 {
263  struct stat stat_infoBefore;
264  struct stat stat_infoAfter;
265  FILE *fp;
266  int fd;
267  int oflag = 0;
268 
269  // Retrive the oflag for open depending of mode parameter
270  if (*mode == 'w')
271  {
272  oflag |= O_WRONLY | O_CREAT | O_TRUNC;
273  }
274  else if (*mode == 'r')
275  {
276  oflag |= O_RDONLY;
277  }
278  else if (*mode == 'a')
279  {
280  oflag |= O_WRONLY | O_APPEND;
281  }
282  else
283  {
284  Com_Printf("Sys_FOpen: invalid mode parameter to open file ('%s')", mode);
285  return NULL;
286  }
287 
288  // Check the state (if path exists)
289  if (stat(ospath, &stat_infoBefore) == -1)
290  {
291  // Check the error in case the the file doesn't exist
292  if (errno != ENOENT)
293  {
294  Com_Printf("Sys_FOpen: first stat('%s') failed: errno %d\n", ospath, errno);
295  return NULL;
296  }
297  else if (*mode != 'w')
298  {
299  return NULL;
300  }
301  }
302  else if (S_ISDIR(stat_infoBefore.st_mode))
303  {
304  return NULL;
305  }
306 
307  // Try to open the file and get the file descriptor
308  if ((fd = open(ospath, oflag, S_IRWXU)) == -1)
309  {
310  Com_Printf("Sys_FOpen: open('%s', %d) failed: errno %d\n", ospath, oflag, errno);
311  return NULL;
312  }
313 
314  // Get the state of the current handle file only if the file wasn't created
315  if (*mode != 'w' && fstat(fd, &stat_infoAfter) == -1)
316  {
317  Com_Printf("Sys_FOpen: second stat('%s') failed: errno %d\n", ospath, errno);
318  close(fd);
319  return NULL;
320  }
321 
322  // Compare the state before and after opening only if the file wasn't created
323  if (*mode != 'w' && (stat_infoBefore.st_mode != stat_infoAfter.st_mode ||
324  stat_infoBefore.st_ino != stat_infoAfter.st_ino))
325  {
326  Com_Printf("Sys_FOpen: stat before and after chmod are different. The file ('%s') may differ (TOCTOU Attacks ?)\n", ospath);
327  close(fd);
328  return NULL;
329  }
330 
331  // Try to open the file with the file descriptor
332  if (!(fp = fdopen(fd, mode)))
333  {
334  Com_Printf("Sys_FOpen: fdopen('%s', %s) failed: errno %d\n", ospath, mode, errno);
335  close(fd);
336  unlink(ospath);
337  return NULL;
338  }
339 
340  return fp;
341 }

464 char **Sys_ListFiles(const char *directory, const char *extension, const char *filter, int *numfiles, qboolean wantsubs)
465 {
466  struct dirent *d;
467  DIR *fdir;
468  qboolean dironly = wantsubs;
469  char search[MAX_OSPATH];
470  int nfiles;
471  char **listCopy;
472  char *list[MAX_FOUND_FILES];
473  int i;
474  struct stat st;
475  int extLen;
476  qboolean invalid;
477 
478  if (filter)
479  {
480  nfiles = 0;
481  Sys_ListFilteredFiles(directory, "", filter, list, &nfiles);
482 
483  list[nfiles] = NULL;
484  *numfiles = nfiles;
485 
486  if (!nfiles)
487  {
488  return NULL;
489  }
490 
491  listCopy = Z_Malloc((nfiles + 1) * sizeof(*listCopy));
492  for (i = 0 ; i < nfiles ; i++)
493  {
494  listCopy[i] = list[i];
495  }
496 
497  listCopy[i] = NULL;
498 
499  return listCopy;
500  }
501 
502  if (!extension)
503  {
504  extension = "";
505  }
506 
507  if (extension[0] == '/' && extension[1] == 0)
508  {
509  extension = "";
510  dironly = qtrue;
511  }
512 
513  extLen = strlen(extension);
514 
515  // search
516  nfiles = 0;
517 
518  if ((fdir = opendir(directory)) == NULL)
519  {
520  *numfiles = 0;
521  return NULL;
522  }
523 
524  while ((d = readdir(fdir)) != NULL)
525  {
526  Com_sprintf(search, sizeof(search), "%s/%s", directory, d->d_name);
527  if (stat(search, &st) == -1)
528  {
529  continue;
530  }
531  if ((dironly && !(st.st_mode & S_IFDIR)) || (!dironly && (st.st_mode & S_IFDIR)))
532  {
533  continue;
534  }
535 
536  if (*extension)
537  {
538  if (strlen(d->d_name) < extLen || Q_stricmp(d->d_name + strlen(d->d_name) - extLen, extension))
539  {
540  continue; // didn't match
541  }
542  }
543 
544  // check for bad file names
545  invalid = qfalse;
546  // note: this isn't done in Sys_ListFilteredFiles()
547 
548  for (i = 0; i < strlen(d->d_name); i++)
549  {
550  if (d->d_name[i] <= 31 || d->d_name[i] >= 127)
551  {
552  Com_Printf(S_COLOR_RED "ERROR: invalid char in name of file '%s'.\n", d->d_name);
553  invalid = qtrue;
554  break;
555  }
556  }
557 
558  if (invalid)
559  {
560  remove(va("%s%c%s", directory, PATH_SEP, d->d_name));
561 #ifdef DEDICATED
562  Sys_Error("Invalid character in file name '%s'. The file has been removed. Start the server again.", d->d_name);
563 #else
564  Cvar_Set("com_missingFiles", "");
565  Com_Error(ERR_DROP, "Invalid file name detected & removed\nFile \"%s\" did contain an invalid character for ET: L file structure.\nSome admins take advantage of this to ensure their menu loads last.\nThe file has been removed.", d->d_name);
566 #endif
567  }
568 
569  if (nfiles == MAX_FOUND_FILES - 1)
570  {
571  break;
572  }
573 
574  list[nfiles] = CopyString(d->d_name);
575  nfiles++;
576  }
577 
578  list[nfiles] = NULL;
579 
580  closedir(fdir);
581 
582  // return a copy of the list
583  *numfiles = nfiles;
584 
585  if (!nfiles)
586  {
587  return NULL;
588  }
589 
590  listCopy = Z_Malloc((nfiles + 1) * sizeof(*listCopy));
591  for (i = 0 ; i < nfiles ; i++)
592  {
593  listCopy[i] = list[i];
594  }
595 
596  listCopy[i] = NULL;
597 
598  return listCopy;
599 }

622 void Sys_Sleep(int msec)
623 {
624  if (msec == 0)
625  {
626  return;
627  }
628 
629  if (stdinIsATTY)
630  {
631  fd_set fdset;
632 
633  FD_ZERO(&fdset);
634  FD_SET(STDIN_FILENO, &fdset);
635  if (msec < 0)
636  {
637  select(STDIN_FILENO + 1, &fdset, NULL, NULL, NULL);
638  }
639  else
640  {
641  struct timeval timeout;
642 
643  timeout.tv_sec = msec / 1000;
644  timeout.tv_usec = (msec % 1000) * 1000;
645  select(STDIN_FILENO + 1, &fdset, NULL, NULL, &timeout);
646  }
647  }
648  else
649  {
650  // With nothing to select() on, we can't wait indefinitely
651  if (msec < 0)
652  {
653  msec = 10;
654  }
655 
656  usleep(msec * 1000);
657  }
658 }

664 void Sys_ErrorDialog(const char *error)
665 {
666  char buffer[1024];
667  unsigned int size;
668  int f = -1;
669  const char *homepath = Cvar_VariableString("fs_homepath");
670  const char *gamedir = Cvar_VariableString("fs_gamedir");
671  const char *fileName = "crashlog.txt";
672  char *dirpath = FS_BuildOSPath(homepath, gamedir, "");
673  char *ospath = FS_BuildOSPath(homepath, gamedir, fileName);
674 
675  Sys_Print(va("%s\n", error));
676 
677 #ifndef DEDICATED
678  Sys_Dialog(DT_ERROR, va("%s\nSee \"%s\" for details.\n", error, ospath), "Error");
679 #endif
680 
681  // Make sure the write path for the crashlog exists...
682  // check homepath
683  if (!Sys_Mkdir(homepath))
684  {
685  Com_Printf("ERROR: couldn't create path '%s' to write file '%s'.\n", homepath, ospath);
686  return;
687  }
688  // check gamedir (inside homepath)
689  if (!Sys_Mkdir(dirpath))
690  {
691  Com_Printf("ERROR: couldn't create path '%s' to write file '%s'.\n", dirpath, ospath);
692  return;
693  }
694 
695  // We might be crashing because we maxed out the Quake MAX_FILE_HANDLES,
696  // which will come through here, so we don't want to recurse forever by
697  // calling FS_FOpenFileWrite()...use the Unix system APIs instead.
698  f = open(ospath, O_CREAT | O_TRUNC | O_WRONLY, 0640);
699  if (f == -1)
700  {
701  Com_Printf("ERROR: couldn't open '%s'\n", fileName);
702  return;
703  }
704 
705  // We're crashing, so we don't care much if write() or close() fails.
706  while ((size = CON_LogRead(buffer, sizeof(buffer))) > 0)
707  {
708  if (write(f, buffer, size) != size)
709  {
710  Com_Printf("ERROR: couldn't fully write to '%s'\n", fileName);
711  break;
712  }
713  }
714 
715  close(f);
716 }

748 {
