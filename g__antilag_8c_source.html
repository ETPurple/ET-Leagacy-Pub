44 static void TimeShiftLerp(vec3_t start, vec3_t end, float frac, vec3_t result)
45 {
46 // From CG_InterpolateEntityPosition in cg_ents.c:
47 /*
48  cent->lerpOrigin[0] = current[0] + f * ( next[0] - current[0] );
49  cent->lerpOrigin[1] = current[1] + f * ( next[1] - current[1] );
50  cent->lerpOrigin[2] = current[2] + f * ( next[2] - current[2] );
51 
52  Angles would be done in a different function, but here they are only the s.angles
53  lerpAngles is copied over from ent->s.angles
54  cent->lerpAngles[0] = LerpAngle( current[0], next[0], f );
55  cent->lerpAngles[1] = LerpAngle( current[1], next[1], f );
56  cent->lerpAngles[2] = LerpAngle( current[2], next[2], f );
57 
58 */
59 // Making these exactly the same should avoid floating-point error
60 
61  result[0] = start[0] + frac * (end[0] - start[0]);
62  result[1] = start[1] + frac * (end[1] - start[1]);
63  result[2] = start[2] + frac * (end[2] - start[2]);
64 }

71 static qboolean G_AntilagSafe(gentity_t *ent)
72 {
73  if (!ent)
74  {
75  return qfalse;
76  }
77 
78  if (!ent->inuse)
79  {
80  return qfalse;
81  }
82 
83  if (!ent->r.linked)
84  {
85  return qfalse;
86  }
87 
88  if (!ent->client)
89  {
90  return qfalse;
91  }
92 
93  if (ent->client->sess.sessionTeam != TEAM_AXIS && ent->client->sess.sessionTeam != TEAM_ALLIES)
94  {
95  return qfalse;
96  }
97 
98  if ((ent->client->ps.pm_flags & PMF_LIMBO))
99  {
100  return qfalse;
101  }
102 
103  // realhead support
104  // restore players who have just died to keep the corpse head box in sync.
105  if (ent->client->backupMarker.time == level.time && ent->client->ps.pm_type == PM_DEAD && g_realHead.integer)
106  {
107  return qtrue;
108  }
109 
110  if (ent->health <= 0)
111  {
112  return qfalse;
113  }
114 
115  // don't store clientMarkers for corpses, etc
116  if (!(ent->client->ps.pm_type == PM_NORMAL))
117  {
118  return qfalse;
119  }
120 
121  // don't store clientMarkers for the player on the tank
122  if ((ent->client->ps.eFlags & EF_MOUNTEDTANK))
123  {
124  return qfalse;
125  }
126 
127  return qtrue;
128 }
