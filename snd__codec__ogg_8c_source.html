73 size_t S_OGG_Callback_read(void *ptr, size_t size, size_t nmemb, void *datasource)
74 {
75  snd_stream_t *stream;
76  int byteSize = 0;
77  int bytesRead = 0;
78  size_t nMembRead = 0;
79 
80  // check if input is valid
81  if (!ptr)
82  {
83  errno = EFAULT;
84  return 0;
85  }
86 
87  if (!(size && nmemb))
88  {
89  // It's not an error, caller just wants zero bytes!
90  errno = 0;
91  return 0;
92  }
93 
94  if (!datasource)
95  {
96  errno = EBADF;
97  return 0;
98  }
99 
100  // we use a snd_stream_t in the generic pointer to pass around
101  stream = (snd_stream_t *) datasource;
102 
103  // FS_Read does not support multi-byte elements
104  byteSize = nmemb * size;
105 
106  // read it with the Q3 function FS_Read()
107  bytesRead = FS_Read(ptr, byteSize, stream->file);
108 
109  // update the file position
110  stream->pos += bytesRead;
111 
112  // this function returns the number of elements read not the number of bytes
113  nMembRead = bytesRead / size;
114 
115  // even if the last member is only read partially
116  // it is counted as a whole in the return value
117  if (bytesRead % size)
118  {
119  nMembRead++;
120  }
121 
122  return nMembRead;
123 }

134 int S_OGG_Callback_seek(void *datasource, ogg_int64_t offset, int whence)
135 {
136  snd_stream_t *stream;
137  int retVal = 0;
138 
139  // check if input is valid
140  if (!datasource)
141  {
142  errno = EBADF;
143  return -1;
144  }
145 
146  // snd_stream_t in the generic pointer
147  stream = (snd_stream_t *) datasource;
148 
149  // we must map the whence to its Q3 counterpart
150  switch (whence)
151  {
152  case SEEK_SET:
153  {
154  // set the file position in the actual file with the Q3 function
155  retVal = FS_Seek(stream->file, (long) offset, FS_SEEK_SET);
156 
157  // something has gone wrong, so we return here
158  if (retVal < 0)
159  {
160  return retVal;
161  }
162 
163  // keep track of file position
164  stream->pos = (int) offset;
165  break;
166  }
167 
168  case SEEK_CUR:
169  {
170  // set the file position in the actual file with the Q3 function
171  retVal = FS_Seek(stream->file, (long) offset, FS_SEEK_CUR);
172 
173  // something has gone wrong, so we return here
174  if (retVal < 0)
175  {
176  return retVal;
177  }
178 
179  // keep track of file position
180  stream->pos += (int) offset;
181  break;
182  }
183 
184  case SEEK_END:
185  {
186  // set the file position in the actual file with the Q3 function
187  retVal = FS_Seek(stream->file, (long) offset, FS_SEEK_END);
188 
189  // something has gone wrong, so we return here
190  if (retVal < 0)
191  {
192  return retVal;
193  }
194 
195  // keep track of file position
196  stream->pos = stream->length + (int) offset;
197  break;
198  }
199 
200  default:
201  {
202  // unknown whence, so we return an error
203  errno = EINVAL;
204  return -1;
205  }
206  }
207 
208  // stream->pos shouldn't be smaller than zero or bigger than the filesize
209  stream->pos = (stream->pos < 0) ? 0 : stream->pos;
210  stream->pos = (stream->pos > stream->length) ? stream->length : stream->pos;
211 
212  return 0;
213 }

272 
273  // some variables used to get informations about the OGG
274  vorbis_info *OGGInfo;
275  ogg_int64_t numSamples;
276 
277  // check if input is valid
278  if (!filename)

280  return NULL;
281  }
282

314  Z_Free(vf);
315 
316  S_CodecUtilClose(&stream);
317 
318  return NULL;
319  }
320 
321  // we only support OGGs with one substream
322  if (ov_streams(vf) != 1)
323  {
324  ov_clear(vf);
325 
326  Z_Free(vf);
327 
328  S_CodecUtilClose(&stream);
329 
330  return NULL;
331  }
332 
333  // get the info about channels and rate
334  OGGInfo = ov_info(vf, 0);
335  if (!OGGInfo)
336  {
337  ov_clear(vf);
338 
339  Z_Free(vf);
340 
341  S_CodecUtilClose(&stream);
342 
343  return NULL;
344  }
345 
346  // get the number of sample-frames in the OGG
347  numSamples = ov_pcm_total(vf, 0);
348 
349  // fill in the info-structure in the stream
350  stream->info.rate = OGGInfo->rate;
351  stream->info.width = OGG_SAMPLEWIDTH;
352  stream->info.channels = OGGInfo->channels;
353  stream->info.samples = numSamples;
354  stream->info.size = stream->info.samples * stream->info.channels * stream->info.width;
355  stream->info.dataofs = 0;
356 
357  // We use stream->pos for the file pointer in the compressed ogg file
358  stream->pos = 0;
359 
360  // We use the generic pointer in stream for the OGG codec control structure
361  stream->ptr = vf;
362 
363  return stream;
364 }

395 int S_OGG_CodecReadStream(snd_stream_t *stream, int bytes, void *buffer)
396 {
397  // buffer handling
398  int bytesRead, bytesLeft, c;
399  char *bufPtr;
400 
401  // Bitstream for the decoder
402  int BS = 0;
403 
404  // big endian machines want their samples in big endian order
405  int IsBigEndian = 0;
406 
407 # ifdef Q3_BIG_ENDIAN
408  IsBigEndian = 1;
409 # endif // Q3_BIG_ENDIAN
410 
411  // check if input is valid
412  if (!(stream && buffer))
413  {
414  return 0;
415  }
416 
417  if (bytes <= 0)
418  {
419  return 0;
420  }
421 
422  bytesRead = 0;
423  bytesLeft = bytes;
424  bufPtr = buffer;
425 
426  // cycle until we have the requested or all available bytes read
427  while (-1)
428  {
429  // read some bytes from the OGG codec
430  c = ov_read((OggVorbis_File *) stream->ptr, bufPtr, bytesLeft, IsBigEndian, OGG_SAMPLEWIDTH, 1, &BS);
431 
432  // no more bytes are left
433  if (c <= 0)
434  {
435  break;
436  }
437 
438  bytesRead += c;
439  bytesLeft -= c;
440  bufPtr += c;
441 
442  // we have enough bytes
443  if (bytesLeft <= 0)
444  {
445  break;
446  }
447  }
448 
449  return bytesRead;
450 }
