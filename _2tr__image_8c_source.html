698 static void R_HeightMapToNormalMap(byte *in, int width, int height, float scale)
699 {
700  int x, y;
701  float r, g, b;
702  float c, cx, cy;
703  float dcx, dcy;
704  float inv255 = 1.0f / 255.0f;
705  vec3_t n;
706  byte *out = in;
707 
708  for (y = 0; y < height; y++)
709  {
710  for (x = 0; x < width; x++)
711  {

713

715  r = in[4 * (y * width + x) + 0];
716  g = in[4 * (y * width + x) + 1];
717  b = in[4 * (y * width + x) + 2];
718 
719  c = (r + g + b) * inv255;
720 
721  // expand the texel to its right
722  if (x == width - 1)
723  {
724  r = in[4 * (y * width + x) + 0];
725  g = in[4 * (y * width + x) + 1];
726  b = in[4 * (y * width + x) + 2];
727  }

731  g = in[4 * (y * width + ((x + 1) % width)) + 1];
732  b = in[4 * (y * width + ((x + 1) % width)) + 2];
733  }
734 
735  cx = (r + g + b) * inv255;
736 
737  // expand the texel one up
738  if (y == height - 1)
739  {
740  r = in[4 * (y * width + x) + 0];
741  g = in[4 * (y * width + x) + 1];
742  b = in[4 * (y * width + x) + 2];
743  }

748  b = in[4 * (((y + 1) % height) * width + x) + 2];
