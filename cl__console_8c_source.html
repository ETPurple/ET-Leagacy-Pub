135  COM_DefaultExtension(filename, sizeof(filename), ".txt");
136 
137  if (!COM_CompareExtension(filename, ".txt"))

140  return;
141  }

144  if (!f)

146  Com_Printf("ERROR: couldn't open %s.\n", filename);
147  return;

151 
152  // skip empty lines
153  for (l = con.current - con.maxTotalLines + 1 ; l <= con.current ; l++)
154  {

156  for (x = 0 ; x < con.linewidth ; x++)
157  if (line[x] != ' ')
158  {
159  break;
160  }
161  if (x != con.linewidth)
162  {
163  break;
164  }
165  }
166

168  bufferlen = con.linewidth + 3 * (int)sizeof(char);
169 #else
170  bufferlen = con.linewidth + 2 * sizeof(char);
171 #endif
172

174 
175  // write the remaining lines
176  buffer[bufferlen - 1] = 0;
177  for ( ; l <= con.current ; l++)
178  {
179  line = con.text + (l % con.maxTotalLines) * con.linewidth;
180  for (i = 0; i < con.linewidth; i++)
181  buffer[i] = line[i] & 0xff;
182  for (x = con.linewidth - 1 ; x >= 0 ; x--)
183  {
184  if (buffer[x] == ' ')
185  {
186  buffer[x] = 0;
187  }
188  else
189  {
190  break;
191  }
192  }
193 #ifdef _WIN32
194  Q_strcat(buffer, bufferlen, "\r\n");
195 #else
196  Q_strcat(buffer, bufferlen, "\n");
197 #endif

199  }

202  FS_FCloseFile(f);

270  }
271

281  {
282  for (j = 0 ; j < numchars ; j++)
283  {

285  tbuf[((con.current - i + oldtotalLines) %
286  oldtotalLines) * oldwidth + j];
287  con.textColor[(con.maxTotalLines - 1 - i) * con.linewidth + j] =
288  tbuff[((con.current - i + oldtotalLines) %
289  oldtotalLines) * oldwidth + j];
290  }
291  }
292 
293  Con_ClearNotify();
294  }

297  con.bottomDisplayedLine = con.current;
298  con.scrollIndex = con.current;

399 void CL_ConsolePrint(char *txt)
400 {
401  int y;
402  int c, l;
403  int color;
404  qboolean skipnotify = qfalse;
405  int prev;
406 
407  // for some demos we don't want to ever show anything on the console
408  if (cl_noprint && cl_noprint->integer)
409  {
410  return;
411  }
412 
413  if (!con.initialized)
414  {
415  con.color[0] = con.color[1] = con.color[2] = con.color[3] = 1.0f;
416  con.linewidth = -1;
417  Con_CheckResize();
418  con.initialized = qtrue;
419  }
420 
421  // work around for text that shows up in console but not in notify
422  if (!Q_strncmp(txt, "[skipnotify]", 12))
423  {
424  txt += 12;
425  }
426 
427  color = ColorIndex(CONSOLE_COLOR);
428 
429  while (*txt != 0)
430  {
431  if (Q_IsColorString(txt))
432  {
433  if (*(txt + 1) == COLOR_NULL)
434  {
435  color = ColorIndex(CONSOLE_COLOR);
436  }
437  else
438  {
439  color = ColorIndex(*(txt + 1));
440  }
441  txt += 2;
442  continue;
443  }
444 
445  // count word length
446  for (l = 0 ; l < con.linewidth ; l++)
447  {
448  if (txt[l] <= ' ')
449  {
450  break;
451  }
452  }
453 
454  // word wrap
455  if (l != con.linewidth && (con.x + l >= con.linewidth))
456  {
457  Con_Linefeed(skipnotify);
458  }
459 
460  c = Q_UTF8_CodePoint(txt);
461 
462  switch (c)
463  {
464  case '\n':
465  Con_Linefeed(skipnotify);
466  break;
467  case '\r':
468  con.x = 0;
469  break;
470  default:
471  // display character and advance
472  y = con.current % con.maxTotalLines;
473 
474  // sign extension caused the character to carry over
475  // into the color info for high ascii chars; casting c to unsigned
476  con.text[y * con.linewidth + con.x] = c;
477  con.textColor[y * con.linewidth + con.x] = color;
478  con.x++;
479  if (con.x >= con.linewidth)
480  {
481  Con_Linefeed(skipnotify);
482  con.x = 0;
483  }
484  break;
485  }
486 
487  txt += Q_UTF8_Width(txt);
488  }
489 
490  // mark time for transparent overlay
491  if (con.current >= 0)
492  {
493  if (skipnotify)
494  {
495  prev = con.current % NUM_CON_TIMES - 1;
496 
497  if (prev < 0)
498  {
499  prev = NUM_CON_TIMES - 1;
500  }
501  con.times[prev] = 0;
502  }
503  else
504  {
505  con.times[con.current % NUM_CON_TIMES] = cls.realtime;
506  }
507  }
508 }
