709 static void VectorArrayNormalize(vec4_t *normals, unsigned int count)
710 {
711 // etl_assert(count);
712 
713 #if idppc
714  {
715  register float half = 0.5;
716  register float one = 1.0;
717  float *components = (float *)normals;
718 
719  // Vanilla PPC code, but since PPC has a reciprocal square root estimate instruction,
720  // runs *much* faster than calling sqrt(). We'll use a single Newton-Raphson
721  // refinement step to get a little more precision. This seems to yeild results
722  // that are correct to 3 decimal places and usually correct to at least 4 (sometimes 5).
723  // (That is, for the given input range of about 0.6 to 2.0).
724  do
725  {
726  float x, y, z;
727  float B, y0, y1;
728 
729  x = components[0];
730  y = components[1];
731  z = components[2];
732  components += 4;
733  B = x * x + y * y + z * z;
734 
735 #ifdef __GNUC__
736  asm ("frsqrte %0,%1" : "=f" (y0) : "f" (B));
737 #else
738  y0 = __frsqrte(B);
739 #endif
740  y1 = y0 + half * y0 * (one - B * y0 * y0);
741 
742  x = x * y1;
743  y = y * y1;
744  components[-4] = x;
745  z = z * y1;
746  components[-3] = y;
747  components[-2] = z;
748  }
749  while (count--);
750  }
751 #else // No assembly version for this architecture, or C_ONLY defined
752  // given the input, it's safe to call VectorNormalizeFast
753  while (count--)
754  {
755  VectorNormalizeFast(normals[0]);
756  normals++;
757  }
758 #endif
759 
760 }

848  VectorArrayNormalize((vec4_t *) tess.normals[tess.numVertexes], numVertexes);
849

851  }

854  tess.numVertexes += numVertexes;
