43 
44 KEY BUTTONS
45 
46 Continuous button event tracking is complicated by the fact that two different
47 input sources (say, mouse button 1 and the control key) can both press the
48 same button, but the button should only be released when both of the
49 pressing key have been released.
50 
51 When a key event issues a button command (+forward, +attack, etc), it appends
52 its key number as argv(1) so it can be matched up with the release.
53 
54 argv(2) will be set to the time the event happened, which allows exact
55 control even at low framerates when the down and up events may both get qued
56 at the same time.

147 void IN_KeyUp(kbutton_t *b)
148 {
149  int k;
150  char *c;
151  unsigned uptime;
152 
153  c = Cmd_Argv(1);
154  if (c[0])
155  {
156  k = atoi(c);
157  }
158  else
159  {
160  // typed manually at the console, assume for unsticking, so clear all
161  b->down[0] = b->down[1] = 0;
162  b->active = qfalse;
163  return;
164  }
165 
166  if (b->down[0] == k)
167  {
168  b->down[0] = 0;
169  }
170  else if (b->down[1] == k)
171  {
172  b->down[1] = 0;
173  }
174  else
175  {
176  return; // key up without coresponding down (menu pass through)
177  }
178  if (b->down[0] || b->down[1])
179  {
180  return; // some other key is still holding it down
181  }
182 
183  b->active = qfalse;
184 
185  // save timestamp for partial frame summing
186  c = Cmd_Argv(2);
187  uptime = atoi(c);
188  if (uptime)
189  {
190  b->msec += uptime - b->downtime;
191  }
192  else
193  {
194  b->msec += frame_msec / 2;
195  }
196 
197  b->active = qfalse;
198 }

722 void CL_KeyMove(usercmd_t *cmd)
723 {
724  int movespeed;
725  int forward = 0, side = 0, up = 0;
726 
727  // adjust for speed key / running
728  // the walking flag is to keep animations consistant
729  // even during acceleration and develeration
730 
731  if (kb[KB_SPEED].active ^ cl_run->integer)
732  {
733  movespeed = 127;
734  cmd->buttons &= ~BUTTON_WALKING;
735  }
736  else
737  {
738  cmd->buttons |= BUTTON_WALKING;
739  movespeed = 64;
740  }
741 
742  if (kb[KB_STRAFE].active)
743  {
744  side += movespeed * CL_KeyState(&kb[KB_RIGHT]);
745  side -= movespeed * CL_KeyState(&kb[KB_LEFT]);
746  }
747 
748  side += movespeed * CL_KeyState(&kb[KB_MOVERIGHT]);
749  side -= movespeed * CL_KeyState(&kb[KB_MOVELEFT]);
750 
751  if ((cmd->buttons & BUTTON_ACTIVATE) && cl_activatelean->integer)
752  {
753  if (side > 0)
754  {
755  cmd->wbuttons |= WBUTTON_LEANRIGHT;
756  }
757  else if (side < 0)
758  {
759  cmd->wbuttons |= WBUTTON_LEANLEFT;
760  }
761 
762  side = 0; // disallow the strafe when holding 'activate'
763  }
764 
765 
766  up += movespeed * CL_KeyState(&kb[KB_UP]);
767  up -= movespeed * CL_KeyState(&kb[KB_DOWN]);
768 
769  forward += movespeed * CL_KeyState(&kb[KB_FORWARD]);
770  forward -= movespeed * CL_KeyState(&kb[KB_BACK]);
771 
772  cmd->forwardmove = ClampChar(forward);
773  cmd->rightmove = ClampChar(side);
774  cmd->upmove = ClampChar(up);
775 
776  // double tap
777  cmd->doubleTap = DT_NONE; // reset
778  if (com_frameTime - cl.doubleTap.lastdoubleTap > cl_doubletapdelay->integer + 150 + cls.frametime) // double tap only once every 500 msecs (add
779  { // frametime for low(-ish) fps situations)
780  int i;
781  qboolean key_down;
782 
783  for (i = 1; i < DT_NUM; i++)
784  {
785  key_down = kb[dtmapping[i]].active || kb[dtmapping[i]].wasPressed;
786 
787  if (key_down && !cl.doubleTap.pressedTime[i])
788  {
789  cl.doubleTap.pressedTime[i] = com_frameTime;
790  }
791  else if (!key_down && !cl.doubleTap.releasedTime[i]
792  && (com_frameTime - cl.doubleTap.pressedTime[i]) < (cl_doubletapdelay->integer + cls.frametime))
793  {
794  cl.doubleTap.releasedTime[i] = com_frameTime;
795  }
796  else if (key_down && (com_frameTime - cl.doubleTap.pressedTime[i]) < (cl_doubletapdelay->integer + cls.frametime)
797  && (com_frameTime - cl.doubleTap.releasedTime[i]) < (cl_doubletapdelay->integer + cls.frametime))
798  {
799  cl.doubleTap.pressedTime[i] = cl.doubleTap.releasedTime[i] = 0;
800  cmd->doubleTap = i;
801  cl.doubleTap.lastdoubleTap = com_frameTime;
802  }
803  else if (!key_down && (cl.doubleTap.pressedTime[i] || cl.doubleTap.releasedTime[i]))
804  {
805  if (com_frameTime - cl.doubleTap.pressedTime[i] >= (cl_doubletapdelay->integer + cls.frametime))
806  {
807  cl.doubleTap.pressedTime[i] = cl.doubleTap.releasedTime[i] = 0;
808  }
809  }
810  }
811  }
812 }

916 void CL_MouseMove(usercmd_t *cmd)
917 {
918  float mx, my;
919  float accelSensitivity;
920  float rate;
921 
922  // allow mouse smoothing
923  if (m_filter->integer)
924  {
925  mx = (cl.mouseDx[0] + cl.mouseDx[1]) * 0.5f;
926  my = (cl.mouseDy[0] + cl.mouseDy[1]) * 0.5f;
927  }
928  else
929  {
930  mx = cl.mouseDx[cl.mouseIndex];
931  my = cl.mouseDy[cl.mouseIndex];
932  }
933  cl.mouseIndex ^= 1;
934  cl.mouseDx[cl.mouseIndex] = 0;
935  cl.mouseDy[cl.mouseIndex] = 0;
936 
937  rate = (float)(sqrt((double)(mx * mx + my * my)) / frame_msec);
938  accelSensitivity = cl_sensitivity->value + rate * cl_mouseAccel->value;
939 
940  // scale by FOV
941  accelSensitivity *= cl.cgameSensitivity;
942 
943  if (rate != 0.f && cl_showMouseRate->integer)
944  {
945  Com_Printf("%f : %f\n", (double)rate, (double)accelSensitivity);
946  }
947 
948  // mg42
949  if (cl.snap.ps.persistant[PERS_HWEAPON_USE])
950  {
951  mx *= 2.5; //(accelSensitivity * 0.1);
952  my *= 2; //(accelSensitivity * 0.075);
953  }
954  else
955  {
956  mx *= accelSensitivity;
957  my *= accelSensitivity;
958  }
959 
960  if (mx == 0.f && my == 0.f)
961  {
962  return;
963  }
964 
965  // add mouse X/Y movement to cmd
966  if (kb[KB_STRAFE].active)
967  {
968  cmd->rightmove = ClampChar(cmd->rightmove + m_side->value * mx);
969  }
970  else
971  {
972  cl.viewangles[YAW] -= m_yaw->value * mx;
973  }
974 
975  if ((kb[KB_MLOOK].active || cl_freelook->integer) && !kb[KB_STRAFE].active)
976  {
977  cl.viewangles[PITCH] += m_pitch->value * my;
978  }
979  else
980  {

982  }

1248 void CL_WritePacket(void)
1249 {

1251  byte data[MAX_MSGLEN];
1252  int i, j;
1253  usercmd_t *cmd, *oldcmd;
1254  usercmd_t nullcmd;
1255  int packetNum;
1256  int oldPacketNum;
1257  int count, key;
1258 
1259  // don't send anything if playing back a demo
1260  if (clc.demoplaying || cls.state == CA_CINEMATIC)
1261  {
1262  return;
1263  }
1264 
1265  Com_Memset(&nullcmd, 0, sizeof(nullcmd));
1266  oldcmd = &nullcmd;
1267 
1268  MSG_Init(&buf, data, sizeof(data));
1269 
1270  MSG_Bitstream(&buf);
1271  // write the current serverId so the server
1272  // can tell if this is from the current gameState
1273  MSG_WriteLong(&buf, cl.serverId);
1274 
1275  // write the last message we received, which can
1276  // be used for delta compression, and is also used
1277  // to tell if we dropped a gamestate
1278  MSG_WriteLong(&buf, clc.serverMessageSequence);
1279 
1280  // write the last reliable message we received
1281  MSG_WriteLong(&buf, clc.serverCommandSequence);
1282 
1283  // write any unacknowledged clientCommands
1284  // NOTE: if you verbose this, you will see that there are quite a few duplicates
1285  // typically several unacknowledged cp or userinfo commands stacked up
1286  for (i = clc.reliableAcknowledge + 1 ; i <= clc.reliableSequence ; i++)
1287  {
1288  MSG_WriteByte(&buf, clc_clientCommand);
1289  MSG_WriteLong(&buf, i);
1290  MSG_WriteString(&buf, clc.reliableCommands[i & (MAX_RELIABLE_COMMANDS - 1)]);
1291  }
1292 
1293  // we want to send all the usercmds that were generated in the last
1294  // few packet, so even if a couple packets are dropped in a row,
1295  // all the cmds will make it to the server
1296  if (cl_packetdup->integer < 0)
1297  {
1298  Cvar_Set("cl_packetdup", "0");
1299  }
1300  else if (cl_packetdup->integer > 5)
1301  {
1302  Cvar_Set("cl_packetdup", "5");
1303  }
1304  oldPacketNum = (clc.netchan.outgoingSequence - 1 - cl_packetdup->integer) & PACKET_MASK;
1305  count = cl.cmdNumber - cl.outPackets[oldPacketNum].p_cmdNumber;
1306  if (count > MAX_PACKET_USERCMDS)
1307  {
1308  count = MAX_PACKET_USERCMDS;
1309  Com_Printf("MAX_PACKET_USERCMDS\n");
1310  }
1311  if (count >= 1)
1312  {
1313  if (cl_showSend->integer)
1314  {
1315  Com_Printf("(%i)", count);
1316  }
1317 
1318  // begin a client move command
1319  if (cl_nodelta->integer || !cl.snap.valid || clc.demowaiting
1320  || clc.serverMessageSequence != cl.snap.messageNum)
1321  {
1322  MSG_WriteByte(&buf, clc_moveNoDelta);
1323  }
1324  else
1325  {
1326  MSG_WriteByte(&buf, clc_move);
1327  }
1328 
1329  // write the command count
1330  MSG_WriteByte(&buf, count);
1331 
1332  // use the checksum feed in the key
1333  key = clc.checksumFeed;
1334  // also use the message acknowledge
1335  key ^= clc.serverMessageSequence;
1336  // also use the last acknowledged server command in the key
1337  key ^= MSG_HashKey(clc.serverCommands[clc.serverCommandSequence & (MAX_RELIABLE_COMMANDS - 1)], 32);
1338 
1339  // write all the commands, including the predicted command
1340  for (i = 0 ; i < count ; i++)
1341  {
1342  j = (cl.cmdNumber - count + i + 1) & CMD_MASK;
1343  cmd = &cl.cmds[j];
1344  MSG_WriteDeltaUsercmdKey(&buf, key, oldcmd, cmd);
1345  oldcmd = cmd;
1346  }
1347  }
1348 
1349  // deliver the message
1350  packetNum = clc.netchan.outgoingSequence & PACKET_MASK;
1351  cl.outPackets[packetNum].p_realtime = cls.realtime;
1352  cl.outPackets[packetNum].p_serverTime = oldcmd->serverTime;
1353  cl.outPackets[packetNum].p_cmdNumber = cl.cmdNumber;
1354  clc.lastPacketSentTime = cls.realtime;
1355 
1356  if (cl_showSend->integer)
1357  {
1358  Com_Printf("%i ", buf.cursize);
1359  }
1360  CL_Netchan_Transmit(&clc.netchan, &buf);
1361 
1362  // clients never really should have messages large enough
1363  // to fragment, but in case they do, fire them all off
1364  // at once
1365  // - this causes a packet burst, which is bad karma for winsock
1366  // added a WARNING message, we'll see if there are legit situations where this happens
1367  while (clc.netchan.unsentFragments)
1368  {
1369  if (cl_showSend->integer)
1370  {
1371  Com_Printf("WARNING: unsent fragments (not supposed to happen!)\n");
1372  }
1373  CL_Netchan_TransmitNextFragment(&clc.netchan);
1374  }
1375 }
