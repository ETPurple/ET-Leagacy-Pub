67 static void AddSkyPolygon(int nump, vec3_t vecs)
68 {
69  int i, j;
70  vec3_t v, av;
71  float s, t, dv;
72  int axis;
73  float *vp;
74 
75  static int vec_to_st[6][3] =
76  {
77  { -2, 3, 1 },
78  { 2, 3, -1 },
79 
80  { 1, 3, 2 },
81  { -1, 3, -2 },
82 
83  { -2, -1, 3 },
84  { -2, 1, -3 }
85  };
86 
87  // decide which face it maps to
88  VectorCopy(vec3_origin, v);
89  for (i = 0, vp = vecs; i < nump; i++, vp += 3)
90  {
91  VectorAdd(vp, v, v);
92  }
93  av[0] = Q_fabs(v[0]);
94  av[1] = Q_fabs(v[1]);
95  av[2] = Q_fabs(v[2]);
96  if (av[0] > av[1] && av[0] > av[2])
97  {
98  if (v[0] < 0)
99  {
100  axis = 1;
101  }
102  else
103  {
104  axis = 0;
105  }
106  }
107  else if (av[1] > av[2] && av[1] > av[0])
108  {
109  if (v[1] < 0)
110  {
111  axis = 3;
112  }
113  else
114  {
115  axis = 2;
116  }
117  }
118  else
119  {
120  if (v[2] < 0)
121  {
122  axis = 5;
123  }
124  else
125  {
126  axis = 4;
127  }
128  }
129 
130  // project new texture coords
131  for (i = 0; i < nump; i++, vecs += 3)
132  {
133  j = vec_to_st[axis][2];
134  if (j > 0)
135  {
136  dv = vecs[j - 1];
137  }
138  else
139  {
140  dv = -vecs[-j - 1];
141  }
142  if (dv < 0.001f)
143  {
144  continue; // don't divide by zero
145  }
146  j = vec_to_st[axis][0];
147  if (j < 0)
148  {
149  s = -vecs[-j - 1] / dv;
150  }
151  else
152  {
153  s = vecs[j - 1] / dv;
154  }
155  j = vec_to_st[axis][1];
156  if (j < 0)
157  {
158  t = -vecs[-j - 1] / dv;
159  }
160  else
161  {
162  t = vecs[j - 1] / dv;
163  }
164 
165  if (s < sky_mins[0][axis])
166  {
167  sky_mins[0][axis] = s;
168  }
169  if (t < sky_mins[1][axis])
170  {
171  sky_mins[1][axis] = t;
172  }
173  if (s > sky_maxs[0][axis])
174  {
175  sky_maxs[0][axis] = s;
176  }
177  if (t > sky_maxs[1][axis])
178  {
179  sky_maxs[1][axis] = t;
180  }
181  }
182 }

193 static void ClipSkyPolygon(int nump, vec3_t vecs, int stage)
194 {
195  float *norm;
196  float *v;
197  qboolean front, back;
198  float d, e;
199  float dists[MAX_CLIP_VERTS];
200  int sides[MAX_CLIP_VERTS];
201  vec3_t newv[2][MAX_CLIP_VERTS];
202  int newc[2];
203  int i, j;
204 
205  if (nump > MAX_CLIP_VERTS - 2)
206  {
207  Ren_Drop("ClipSkyPolygon: MAX_CLIP_VERTS");
208  }
209  if (stage == 6) // fully clipped, so draw it
210  {
211  AddSkyPolygon(nump, vecs);
212  return;
213  }
214 
215  front = back = qfalse;
216  norm = sky_clip[stage];
217  for (i = 0, v = vecs; i < nump; i++, v += 3)
218  {
219  d = DotProduct(v, norm);
220  if (d > ON_EPSILON)
221  {
222  front = qtrue;
223  sides[i] = SIDE_FRONT;
224  }
225  else if (d < -ON_EPSILON)
226  {
227  back = qtrue;
228  sides[i] = SIDE_BACK;
229  }
230  else
231  {
232  sides[i] = SIDE_ON;
233  }
234  dists[i] = d;
235  }
236 
237  if (!front || !back) // not clipped
238  {
239  ClipSkyPolygon(nump, vecs, stage + 1);
240  return;
241  }
242 
243  // clip it
244  sides[i] = sides[0];
245  dists[i] = dists[0];
246  VectorCopy(vecs, (vecs + (i * 3)));
247  newc[0] = newc[1] = 0;
248 
249  for (i = 0, v = vecs; i < nump; i++, v += 3)
250  {
251  switch (sides[i])
252  {
253  case SIDE_FRONT:
254  VectorCopy(v, newv[0][newc[0]]);
255  newc[0]++;
256  break;
257  case SIDE_BACK:
258  VectorCopy(v, newv[1][newc[1]]);
259  newc[1]++;
260  break;
261  case SIDE_ON:
262  VectorCopy(v, newv[0][newc[0]]);
263  newc[0]++;
264  VectorCopy(v, newv[1][newc[1]]);
265  newc[1]++;
266  break;
267  }
268 
269  if (sides[i] == SIDE_ON || sides[i + 1] == SIDE_ON || sides[i + 1] == sides[i])
270  {
271  continue;
272  }
273 
274  d = dists[i] / (dists[i] - dists[i + 1]);
275  for (j = 0; j < 3; j++)
276  {
277  e = v[j] + d * (v[j + 3] - v[j]);
278  newv[0][newc[0]][j] = e;
279  newv[1][newc[1]][j] = e;
280  }
281  newc[0]++;
282  newc[1]++;
283  }
284 
285  // continue
286  ClipSkyPolygon(newc[0], newv[0][0], stage + 1);
287  ClipSkyPolygon(newc[1], newv[1][0], stage + 1);
288 }

537 static void DrawSkyBox(shader_t *shader)
538 {

554

558  continue;

563  sky_maxs_subd[0] = sky_maxs[0][i] * HALF_SKY_SUBDIVISIONS;

610  }
611  }
612 
613  DrawSkySide(shader->sky.outerbox[sky_texorder[i]],
614  sky_mins_subd,
615  sky_maxs_subd);
616  }
