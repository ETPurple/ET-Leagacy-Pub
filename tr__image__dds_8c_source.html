244 static ID_INLINE void UnpackRGB565(byte rgb[3], uint16_t cl)
245 {
246  byte r = (byte) ((cl >> 11) & 0x1F);
247  byte g = (byte) ((cl >> 5) & 0x3F);

330  rgba[2][2] = (byte) ((rgba[0][2] + rgba[1][2]) / 2);
331 
332  rgba[3][0] = 0;
333  rgba[3][1] = 0;
334  rgba[3][2] = 0;
335 
336  break;
337  }

340  case GL_COMPRESSED_RGBA_S3TC_DXT5_EXT:

342  rgba[2][1] = (byte) ((rgba[0][1] * 2 + rgba[1][1]) / 3);
343  rgba[2][2] = (byte) ((rgba[0][2] * 2 + rgba[1][2]) / 3);
344

377  }
378  }

384

386  a[1] = alpha_base[1];
387 
388  if (a[0] > a[1])
389  {
390  a[2] = (6 * a[0] + 1 * a[1]) / 7;
391  a[3] = (5 * a[0] + 2 * a[1]) / 7;
392  a[4] = (4 * a[0] + 3 * a[1]) / 7;
393  a[5] = (3 * a[0] + 4 * a[1]) / 7;
