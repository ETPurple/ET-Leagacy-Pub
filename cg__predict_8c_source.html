514 static void CG_InterpolatePlayerState(qboolean grabAngles)
515 {
516  float f;
517  int i;
518  playerState_t *out = &cg.predictedPlayerState;
519  snapshot_t *prev = cg.snap;
520  snapshot_t *next = cg.nextSnap;
521 
522  *out = cg.snap->ps;
523 
524  if (cg.showGameView)
525  {
526  return;
527  }
528 
529  // if we are still allowing local input, short circuit the view angles
530  if (grabAngles)
531  {
532  usercmd_t cmd;
533  int cmdNum;
534 
535  cmdNum = trap_GetCurrentCmdNumber();
536  trap_GetUserCmd(cmdNum, &cmd);
537 
538  // added tracemask
539  PM_UpdateViewAngles(out, &cg.pmext, &cmd, CG_Trace, MASK_PLAYERSOLID);
540  }
541 
542  // if the next frame is a teleport, we can't lerp to it
543  if (cg.nextFrameTeleport)
544  {
545  return;
546  }
547 
548  if (!next || next->serverTime <= prev->serverTime)
549  {
550  return;
551  }
552 
553  f = (float)(cg.time - prev->serverTime) / (next->serverTime - prev->serverTime);
554 
555  i = next->ps.bobCycle;
556  if (i < prev->ps.bobCycle)
557  {
558  i += 256; // handle wraparound
559  }
560  out->bobCycle = prev->ps.bobCycle + (int)(f * (i - prev->ps.bobCycle));
561 
562  for (i = 0 ; i < 3 ; i++)
563  {
564  out->origin[i] = prev->ps.origin[i] + f * (next->ps.origin[i] - prev->ps.origin[i]);
565  if (!grabAngles)
566  {
567  out->viewangles[i] = LerpAngle(
568  prev->ps.viewangles[i], next->ps.viewangles[i], f);
569  }
570  out->velocity[i] = prev->ps.velocity[i] +
571  f * (next->ps.velocity[i] - prev->ps.velocity[i]);
572  }
573 }
