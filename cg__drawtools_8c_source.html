146 void CG_FilledBar(float x, float y, float w, float h, float *startColor, float *endColor, const float *bgColor, float frac, int flags)
147 {
148  vec4_t backgroundcolor = { 1, 1, 1, 0.25f }, colorAtPos; // colorAtPos is the lerped color if necessary
149 
150  if (frac > 1)
151  {
152  frac = 1.f;
153  }
154  if (frac < 0)
155  {
156  frac = 0;
157  }
158 
159  if ((flags & BAR_BG) && bgColor) // BAR_BG set, and color specified, use specified bg color
160  {
161  Vector4Copy(bgColor, backgroundcolor);
162  }
163 
164  if (flags & BAR_LERP_COLOR)
165  {
166  Vector4Average(startColor, endColor, frac, colorAtPos);
167  }
168 
169  // background
170  if ((flags & BAR_BG))
171  {
172  int indent = BAR_BORDERSIZE;
173 
174  // draw background at full size and shrink the remaining box to fit inside with a border. (alternate border may be specified by a BAR_BGSPACING_xx)
175  CG_FillRect(x,
176  y,
177  w,
178  h,
179  backgroundcolor);
180 
181  if (flags & BAR_BGSPACING_X0Y0) // fill the whole box (no border)
182  {
183  }
184  else if (flags & BAR_BGSPACING_X0Y5) // spacing created for weapon heat
185  {
186  indent *= 3;
187  y += indent;
188  h -= (2 * indent);
189 
190  }
191  else // default spacing of 2 units on each side
192  {
193  x += indent;
194  y += indent;
195  w -= (2 * indent);
196  h -= (2 * indent);
197  }
198  }
199  else if ((flags & BAR_BORDER) || (flags & BAR_BORDER_SMALL))
200  {
201  int indent = (flags & BAR_BORDER_SMALL) ? 1 : BAR_BORDERSIZE;
202 
203  CG_DrawRect_FixedBorder(x, y, w, h, indent, bgColor);
204  x += indent;
205  y += indent;
206  w -= indent;
207  h -= (2 * indent);
208  }
209 
210  // adjust for horiz/vertical and draw the fractional box
211  if (flags & BAR_VERT)
212  {
213  if (flags & BAR_LEFT) // TODO: remember to swap colors on the ends here
214  {
215  y += (h * (1 - frac));
216  }
217  else if (flags & BAR_CENTER)
218  {
219  y += (h * (1 - frac) / 2);
220  }
221 
222  if (flags & BAR_LERP_COLOR)
223  {
224  CG_FillRect(x, y, w, h * frac, colorAtPos);
225  }
226  else
227  {
228  CG_FillRect(x, y, w, h * frac, startColor);
229  }
230  }
231  else
232  {
233  if (flags & BAR_LEFT) // TODO: remember to swap colors on the ends here
234  {
235  x += (w * (1 - frac));
236  }
237  else if (flags & BAR_CENTER)
238  {
239  x += (w * (1 - frac) / 2);
240  }
241

244  CG_FillRect(x, y, w * frac, h, colorAtPos);

529 float *CG_FadeColor(int startMsec, int totalMsec)
530 {
531  static vec4_t color;
532  int t;
533 
534  if (startMsec == 0)
535  {
536  return NULL;
537  }
538 
539  t = cg.time - startMsec;
540 
541  if (t >= totalMsec)
542  {
543  return NULL;
544  }
545 
546  // fade out
547  if (totalMsec - t < FADE_TIME)
548  {
549  color[3] = (totalMsec - t) * 1.0f / FADE_TIME;
550  }
551  else
552  {
553  color[3] = 1.0;
554  }
555  color[0] = color[1] = color[2] = 1.f;
556 
557  return color;

711 qboolean CG_WorldCoordToScreenCoordFloat(vec3_t point, float *x, float *y)
712 {
713  vec3_t trans;
714  float xc, yc;
715  float px, py;
716  float z;
717 
718  px = (float)tan(DEG2RAD((double)cg.refdef.fov_x) / 2);
719  py = (float)tan(DEG2RAD((double)cg.refdef.fov_y) / 2);
720 
721  VectorSubtract(point, cg.refdef.vieworg, trans);
722 
723  xc = 640.0f / 2.0f;
724  yc = 480.0f / 2.0f;
725 
726  z = DotProduct(trans, cg.refdef.viewaxis[0]);
727  if (z < 0.1f)
728  {
729  return qfalse;
730  }
731  px *= z;
732  py *= z;
733  if (px == 0.f || py == 0.f)
734  {
735  return qfalse;
736  }
737 
738  *x = xc - (DotProduct(trans, cg.refdef.viewaxis[1]) * xc) / px;
739  *y = yc - (DotProduct(trans, cg.refdef.viewaxis[2]) * yc) / py;
740  *x = Ccg_WideX(*x);
741 
742  return qtrue;
743 }
