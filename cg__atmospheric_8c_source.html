242 static qboolean CG_ParticleGenerate(cg_atmosphericParticle_t *particle, vec3_t currvec, float currweight, atmFXType_t atmFX)
243 {
244  float angle = random() * M_TAU_F;
245  float distance = 20 + MAX_ATMOSPHERIC_DISTANCE * random();
246  float groundHeight, skyHeight;
247 
248  particle->pos[0] = cg.refdef_current->vieworg[0] + sin(angle) * distance;
249  particle->pos[1] = cg.refdef_current->vieworg[1] + cos(angle) * distance;
250 
251  // choose a spawn point randomly between sky and ground
252  skyHeight = BG_GetSkyHeightAtPoint(particle->pos);
253  if (skyHeight >= MAX_ATMOSPHERIC_HEIGHT)
254  {
255  return qfalse;
256  }
257  groundHeight = BG_GetSkyGroundHeightAtPoint(particle->pos);
258  if (groundHeight + particle->height + ATMOSPHERIC_PARTICLE_OFFSET >= skyHeight)
259  {
260  return qfalse;
261  }
262  particle->pos[2] = groundHeight + random() * (skyHeight - groundHeight);
263 
264  // make sure it doesn't fall from too far cause it then will go over our heads ('lower the ceiling')
265  if (cg_atmFx.baseHeightOffset > 0)
266  {
267  if (particle->pos[2] - cg.refdef_current->vieworg[2] > cg_atmFx.baseHeightOffset)
268  {
269  particle->pos[2] = cg.refdef_current->vieworg[2] + cg_atmFx.baseHeightOffset;
270 
271  if (particle->pos[2] < groundHeight)
272  {
273  return qfalse;
274  }
275  }
276  }
277 
278  // rain goes in bursts - allow max raindrops every 10 seconds
279  if (atmFX == ATM_RAIN && (cg_atmFx.oldDropsActive > (0.5f * cg_atmFx.numDrops + 0.001f * cg_atmFx.numDrops * (10000 - (cg.time % 10000)))))
280  {
281  return qfalse;
282  }
283 
284  CG_SetParticleActive(particle, ACT_FALLING);
285 
286  VectorCopy(currvec, particle->delta);
287  if (atmFX == ATM_RAIN)
288  {
289  particle->delta[2] += crandom() * 100;
290  }
291  else
292  {
293  particle->delta[2] += crandom() * 25;
294  }
295 
296  VectorCopy(particle->delta, particle->deltaNormalized);
297  VectorNormalizeFast(particle->deltaNormalized);
298 
299  if (atmFX == ATM_RAIN)
300  {
301  particle->height = ATMOSPHERIC_RAIN_HEIGHT + crandom() * 100;
302  particle->weight = currweight;
303 
304  // special color
305  particle->colour[0] = 0.6f + 0.2f * random() * 0xFF;
306  particle->colour[1] = 0.6f + 0.2f * random() * 0xFF;
307  particle->colour[2] = 0.6f + 0.2f * random() * 0xFF;
308  }
309  else
310  {
311  particle->height = ATMOSPHERIC_SNOW_HEIGHT + random() * 2;
312  particle->weight = particle->height * 0.5f;
313 
314  particle->colour[0] = 255;
315  particle->colour[1] = 255;
316  particle->colour[2] = 255;
317  }
318 
319  //particle->effectshader = &cg_atmFx.effectshaders[ (int) (random() * ( 2 )) + 1 ];
320  particle->effectshader = &cg_atmFx.effectshaders[atmFX];
321 
322  particle->partFX = atmFX;
323 
324  return qtrue;
325 }

333 static qboolean CG_ParticleCheckVisible(cg_atmosphericParticle_t *particle)
334 {
335  float moved;
336  vec2_t distance;
337 
338  if (!particle || particle->active == ACT_NOT)
339  {
340  return qfalse;
341  }
342 
343  // units moved since last frame
344  moved = (cg.time - cg_atmFx.lastEffectTime) * 0.001f;
345  VectorMA(particle->pos, moved, particle->delta, particle->pos);
346 
347  if ((particle->partFX == ATM_RAIN ? (particle->pos[2] + particle->height) : particle->pos[2]) < BG_GetSkyGroundHeightAtPoint(particle->pos))
348  {
349  return CG_SetParticleActive(particle, ACT_NOT);
350  }
351 
352  distance[0] = particle->pos[0] - cg.refdef_current->vieworg[0];
353  distance[1] = particle->pos[1] - cg.refdef_current->vieworg[1];
354 
355  if ((distance[0] * distance[0] + distance[1] * distance[1]) > Square(MAX_ATMOSPHERIC_DISTANCE))
356  {
357  // just nuke this particle, let it respawn
358  return CG_SetParticleActive(particle, ACT_NOT);
359  }
360 
361  return qtrue;
362 }

373 static void CG_ParticleRender(cg_atmosphericParticle_t *particle)
374 {
375  vec3_t forward, right;
376  polyVert_t verts[3];
377  vec2_t line;
378  float len, sinTumbling, cosTumbling, particleWidth, dist = 0.0;
379  vec3_t start, finish;
380  float groundHeight;
381 
382  if (particle->active == ACT_NOT)
383  {
384  return;
385  }
386 
387  if (CG_CullPoint(particle->pos))
388  {
389  return;
390  }
391 
392  VectorCopy(particle->pos, start);
393 
394  if (particle->partFX == ATM_SNOW)
395  {
396  sinTumbling = sin(particle->pos[2] * 0.03125f * (0.5f * particle->weight));
397  cosTumbling = cos((particle->pos[2] + particle->pos[1]) * 0.03125f * (0.5f * particle->weight));
398  start[0] += 24 * (1 - particle->deltaNormalized[2]) * sinTumbling;
399  start[1] += 24 * (1 - particle->deltaNormalized[2]) * cosTumbling;
400  }
401  else // ATM_RAIN
402  {
403  dist = DistanceSquared(particle->pos, cg.refdef_current->vieworg);
404  }
405  // make sure it doesn't clip through surfaces
406  groundHeight = BG_GetSkyGroundHeightAtPoint(start);
407  len = particle->height;
408 
409  if (particle->partFX == ATM_SNOW)
410  {
411  if (start[2] - len - ATMOSPHERIC_PARTICLE_OFFSET <= groundHeight)
412  {
413  return;
414  // stop snow going through surfaces
415  //len = particle->height - groundHeight + start[2];
416  //VectorMA(start, len - particle->height, particle->deltaNormalized, start);
417  }
418  }
419  else // ATM_RAIN
420  {
421  if (start[2] - ATMOSPHERIC_PARTICLE_OFFSET <= groundHeight)
422  {
423  // stop rain going through surfaces
424  len = particle->height - groundHeight + start[2];
425  VectorMA(start, len - particle->height, particle->deltaNormalized, start);
426  }
427  }
428 
429  if (len <= 0)
430  {
431  return;
432  }
433 
434  if (particle->partFX == ATM_SNOW)
435  {
436  line[0] = particle->pos[0] - cg.refdef_current->vieworg[0];
437  line[1] = particle->pos[1] - cg.refdef_current->vieworg[1];
438  dist = DistanceSquared(particle->pos, cg.refdef_current->vieworg);
439 
440  // dist becomes scale
441  if (dist > Square(500.f))
442  {
443  dist = 1.f + ((dist - Square(500.f)) * (10.f / Square(2000.f)));
444  }
445  else
446  {
447  dist = 1.f;
448  }
449 
450  len *= dist;
451  }
452  else // ATM_RAIN
453  {
454  // fade nearby rain particles
455  if (dist < Square(128.f))
456  {
457  dist = .25f + .75f * (dist / Square(128.f));
458  }
459  else
460  {
461  dist = 1.0f;
462  }
463 
464  }
465 
466  VectorCopy(particle->deltaNormalized, forward);
467  VectorMA(start, -len, forward, finish);
468 
469  line[0] = DotProduct(forward, cg.refdef_current->viewaxis[1]);
470  line[1] = DotProduct(forward, cg.refdef_current->viewaxis[2]);
471 
472  VectorScale(cg.refdef_current->viewaxis[1], line[1], right);
473  VectorMA(right, -line[0], cg.refdef_current->viewaxis[2], right);
474  VectorNormalize(right);
475 
476  if (particle->partFX == ATM_SNOW)
477  {
478  particleWidth = dist * (particle->weight);
479 
480  VectorMA(finish, -particleWidth, right, verts[0].xyz);
481  verts[0].st[0] = 0;
482  verts[0].st[1] = 0;
483  verts[0].modulate[0] = particle->colour[0];
484  verts[0].modulate[1] = particle->colour[1];
485  verts[0].modulate[2] = particle->colour[2];
486  verts[0].modulate[3] = 255;
487 
488  VectorMA(start, -particleWidth, right, verts[1].xyz);
489  verts[1].st[0] = 0;
490  verts[1].st[1] = 1;
491  verts[1].modulate[0] = particle->colour[0];
492  verts[1].modulate[1] = particle->colour[1];
493  verts[1].modulate[2] = particle->colour[2];
494  verts[1].modulate[3] = 255;
495 
496  VectorMA(start, particleWidth, right, verts[2].xyz);
497  verts[2].st[0] = 1;
498  verts[2].st[1] = 1;
499  verts[2].modulate[0] = particle->colour[0];
500  verts[2].modulate[1] = particle->colour[1];
501  verts[2].modulate[2] = particle->colour[2];
502  verts[2].modulate[3] = 255;
503  }
504  else // ATM_RAIN
505  {
506  VectorCopy(finish, verts[0].xyz);
507  verts[0].st[0] = 0.5f;
508  verts[0].st[1] = 0;
509  verts[0].modulate[0] = particle->colour[0];
510  verts[0].modulate[1] = particle->colour[1];
511  verts[0].modulate[2] = particle->colour[2];
512  verts[0].modulate[3] = 100 * dist;
513 
514  VectorMA(start, -particle->weight, right, verts[1].xyz);
515  verts[1].st[0] = 0;
516  verts[1].st[1] = 1;
517  verts[1].modulate[0] = particle->colour[0];
518  verts[1].modulate[1] = particle->colour[1];
519  verts[1].modulate[2] = particle->colour[2];
520  verts[1].modulate[3] = 200 * dist;
521 
522  VectorMA(start, particle->weight, right, verts[2].xyz);
523  verts[2].st[0] = 1;
524  verts[2].st[1] = 1;
525  verts[2].modulate[0] = particle->colour[0];
526  verts[2].modulate[1] = particle->colour[1];
527  verts[2].modulate[2] = particle->colour[2];
528  verts[2].modulate[3] = 200 * dist;
529  }
530  CG_AddPolyToPool(*particle->effectshader, verts);
531 }

562 static qboolean CG_EffectGustCurrent(vec3_t curr, float *weight, int *num)
563 {
564  vec3_t temp;
565 
566  if (cg.time < cg_atmFx.baseEndTime)
567  {
568  VectorCopy(cg_atmFx.baseVec, curr);
569  *weight = cg_atmFx.baseWeight;
570  *num = cg_atmFx.baseDrops;
571  }
572  else
573  {
574  float frac;
575 
576  VectorSubtract(cg_atmFx.gustVec, cg_atmFx.baseVec, temp);
577 
578  if (cg.time < cg_atmFx.gustStartTime)
579  {
580  frac = ((float)(cg.time - cg_atmFx.baseEndTime)) / ((float)(cg_atmFx.gustStartTime - cg_atmFx.baseEndTime));
581  VectorMA(cg_atmFx.baseVec, frac, temp, curr);
582  *weight = cg_atmFx.baseWeight + (cg_atmFx.gustWeight - cg_atmFx.baseWeight) * frac;
583  *num = cg_atmFx.baseDrops + ((float)(cg_atmFx.gustDrops - cg_atmFx.baseDrops)) * frac;
584  }
585  else if (cg.time < cg_atmFx.gustEndTime)
586  {
587  VectorCopy(cg_atmFx.gustVec, curr);
588  *weight = cg_atmFx.gustWeight;
589  *num = cg_atmFx.gustDrops;
590  }
591  else
592  {
593  frac = 1.0f - ((float)(cg.time - cg_atmFx.gustEndTime)) / ((float)(cg_atmFx.baseStartTime - cg_atmFx.gustEndTime));
594  VectorMA(cg_atmFx.baseVec, frac, temp, curr);
595  *weight = cg_atmFx.baseWeight + (cg_atmFx.gustWeight - cg_atmFx.baseWeight) * frac;
596  *num = cg_atmFx.baseDrops + ((float)(cg_atmFx.gustDrops - cg_atmFx.baseDrops)) * frac;
597 
598  if (cg.time >= cg_atmFx.baseStartTime)
599  {
600  return qtrue;
601  }
602  }
603  }
604  return qfalse;
605 }

667 void CG_EffectParse(const char *effectstr)
668 {
669  float bmin, bmax, cmin, cmax, gmin, gmax, bdrop, gdrop;
670  int bheight;
671  char *startptr, *eqptr, *endptr;
672  char workbuff[128];
673  atmFXType_t atmFXType = ATM_NONE;
674 
675  if (CG_AtmosphericKludge())
676  {
677  return;
678  }
679 
680  // set up some default values
681  cg_atmFx.baseVec[0] = cg_atmFx.baseVec[1] = 0;
682  cg_atmFx.gustVec[0] = cg_atmFx.gustVec[1] = 100;
683  bmin = 5;
684  bmax = 10;
685  cmin = 1;
686  cmax = 1;
687  gmin = 0;
688  gmax = 2;
689  bdrop = gdrop = 300;
690  cg_atmFx.baseWeight = 0.7f;
691  cg_atmFx.gustWeight = 1.5f;
692  bheight = 0;
693 
694  // parse the parameter string
695  Q_strncpyz(workbuff, effectstr, sizeof(workbuff));
696 
697  for (startptr = workbuff; *startptr; )
698  {
699  for (eqptr = startptr; *eqptr && *eqptr != '=' && *eqptr != ','; eqptr++)
700  ;
701  if (!*eqptr)
702  {
703  break; // No more string
704  }
705  if (*eqptr == ',')
706  {
707  startptr = eqptr + 1; // Bad argument, continue
708  continue;
709  }
710 
711  *eqptr++ = 0;
712 
713  for (endptr = eqptr; *endptr && *endptr != ','; endptr++)
714  ;
715  if (*endptr)
716  {
717  *endptr++ = 0;
718  }
719 
720  if (atmFXType == ATM_NONE) // first roll
721  {
722  if (Q_stricmp(startptr, "T"))
723  {
724  cg_atmFx.numDrops = 0;
725  cg_atmFx.currentFX = ATM_NONE;
726  CG_Printf("Atmospheric effect must start with a type.\n");
727  return;
728  }
729 
730  if (!Q_stricmp(eqptr, "RAIN"))
731  {
732  atmFXType = ATM_RAIN;
733  cg_atmFx.currentFX = ATM_RAIN;
734  cg_atmFx.ParticleCheckVisible = &CG_ParticleCheckVisible;
735  cg_atmFx.ParticleGenerate = &CG_ParticleGenerate;
736  cg_atmFx.ParticleRender = &CG_ParticleRender;
737 
738  cg_atmFx.baseVec[2] = cg_atmFx.gustVec[2] = -ATMOSPHERIC_RAIN_SPEED;
739  }
740  else if (!Q_stricmp(eqptr, "SNOW"))
741  {
742  atmFXType = ATM_SNOW;
743  cg_atmFx.currentFX = ATM_SNOW;
744  cg_atmFx.ParticleCheckVisible = &CG_ParticleCheckVisible;
745  cg_atmFx.ParticleGenerate = &CG_ParticleGenerate;
746  cg_atmFx.ParticleRender = &CG_ParticleRender;
747 
748  cg_atmFx.baseVec[2] = cg_atmFx.gustVec[2] = -ATMOSPHERIC_SNOW_SPEED;
749  }
750  else
751  {
752  cg_atmFx.numDrops = 0;
753  cg_atmFx.currentFX = ATM_NONE;
754  CG_Printf("Only effect type 'rain' and 'snow' are supported.\n");
755  return;
756  }
757  }
758  else
759  {
760  if (!Q_stricmp(startptr, "B"))
761  {
762  CG_EP_ParseFloats(eqptr, &bmin, &bmax);
763  }
764  else if (!Q_stricmp(startptr, "C"))
765  {
766  CG_EP_ParseFloats(eqptr, &cmin, &cmax);
767  }
768  else if (!Q_stricmp(startptr, "G"))
769  {
770  CG_EP_ParseFloats(eqptr, &gmin, &gmax);
771  }
772  else if (!Q_stricmp(startptr, "BV"))
773  {
774  CG_EP_ParseFloats(eqptr, &cg_atmFx.baseVec[0], &cg_atmFx.baseVec[1]);
775  }
776  else if (!Q_stricmp(startptr, "GV"))
777  {
778  CG_EP_ParseFloats(eqptr, &cg_atmFx.gustVec[0], &cg_atmFx.gustVec[1]);
779  }
780  else if (!Q_stricmp(startptr, "W"))
781  {
782  CG_EP_ParseFloats(eqptr, &cg_atmFx.baseWeight, &cg_atmFx.gustWeight);
783  }
784  else if (!Q_stricmp(startptr, "D"))
785  {
786  CG_EP_ParseFloats(eqptr, &bdrop, &gdrop);
787  }
788  else if (!Q_stricmp(startptr, "H"))
789  {
790  CG_EP_ParseInts(eqptr, &bheight, &bheight);
791  }
792  else
793  {
794  CG_Printf("Unknown effect key '%s'.\n", startptr);
795  }
796  }
797  startptr = endptr;
798  }
799 
800  if (atmFXType == ATM_NONE || !BG_LoadTraceMap(cgs.rawmapname, cg.mapcoordsMins, cg.mapcoordsMaxs))
801  {
802  // no effects
803  cg_atmFx.numDrops = 0;
804  cg_atmFx.currentFX = ATM_NONE;
805  return;
806  }
807 
808  cg_atmFx.baseHeightOffset = bheight;
809 
810  if (cg_atmFx.baseHeightOffset < 0)
811  {
812  cg_atmFx.baseHeightOffset = 0;
813  }
814 
815  cg_atmFx.baseMinTime = 1000 * bmin;
816  cg_atmFx.baseMaxTime = 1000 * bmax;
817  cg_atmFx.changeMinTime = 1000 * cmin;
818  cg_atmFx.changeMaxTime = 1000 * cmax;
819  cg_atmFx.gustMinTime = 1000 * gmin;
820  cg_atmFx.gustMaxTime = 1000 * gmax;
821  cg_atmFx.baseDrops = bdrop;
822  cg_atmFx.gustDrops = gdrop;
823 
824  cg_atmFx.numDrops = (cg_atmFx.baseDrops > cg_atmFx.gustDrops) ? cg_atmFx.baseDrops : cg_atmFx.gustDrops;
825 
826  if (cg_atmFx.numDrops > MAX_ATMOSPHERIC_PARTICLES)
827  {
828  cg_atmFx.numDrops = MAX_ATMOSPHERIC_PARTICLES;
829  }
830 
831  cg_atmFx.effectshaders[ATM_NONE] = 0;
832  cg_atmFx.effectshaders[ATM_RAIN] = trap_R_RegisterShader("gfx/misc/raindrop");
833  cg_atmFx.effectshaders[ATM_SNOW] = trap_R_RegisterShader("gfx/misc/snow");
834 
835  CG_EffectGust();
836 }
