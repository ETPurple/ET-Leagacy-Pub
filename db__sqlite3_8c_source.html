267 static qboolean DB_CreateOrUpdateSchema(int startSchemaVersion)
268 {
269  int result, i;
270  char *err_msg = 0;
271  char *sql;
272 
273  for (i = startSchemaVersion; i < SQL_DBMS_SCHEMA_VERSION; i++)
274  {
275  result = sqlite3_exec(db, sql_Version_Statements[i], 0, 0, &err_msg);
276 
277  if (result != SQLITE_OK)
278  {
279  Com_Printf(S_COLOR_YELLOW "WARNING: SQLite3 failed to create schema version %i: %s\n", i + 1, err_msg);
280  sqlite3_free(err_msg);
281  return qfalse;
282  }
283 
284  if (i + 1 == SQL_DBMS_SCHEMA_VERSION)
285  {
286  sql = va("INSERT INTO etl_version VALUES (%i, 'ET: L DBMS schema V%i for %s', '%s', CURRENT_TIMESTAMP);", i + 1, i + 1, ET_VERSION, sql_Version_Statements[i]);
287  }
288  else
289  {
290  sql = va("INSERT INTO etl_version VALUES (%i, 'ET: L DBMS schema V%i', '%s', CURRENT_TIMESTAMP);", i + 1 , i + 1, sql_Version_Statements[i]);
291  }
292 
293  result = sqlite3_exec(db, sql, 0, 0, &err_msg);
294 
295  if (result != SQLITE_OK)
296  {
297  Com_Printf(S_COLOR_YELLOW "WARNING: SQLite3 failed to write ETL version %i: %s\n", i + 1, err_msg);
298  sqlite3_free(err_msg);
299  return qfalse;
300  }
301 
302  Com_Printf("... DB schema version: %i created\n", i + 1);
303  }
304 
305  return qtrue;
306 }

329 qboolean DB_CheckUpdates(void)
330 {
331  int version = 0;
332  char *sql;
333  int result;
334  sqlite3_stmt *res;
335 
336  // read version from version table
337  sql = va("SELECT id from etl_version ORDER BY Id DESC LIMIT 1;");
338  result = sqlite3_prepare_v2(db, sql, -1, &res, 0);
339 
340  if (result == SQLITE_OK)
341  {
342  result = sqlite3_step(res);
343 
344  if (result == SQLITE_ROW)
345  {
346  version = sqlite3_column_int(res, 0);
347  }
348  }
349 
350  if (version == SQL_DBMS_SCHEMA_VERSION) // we are done
351  {
352  Com_Printf("SQLite3 ETL: DB schema version #%i is up to date\n", version);
353  sqlite3_finalize(res);
354  return qtrue;
355  }
356  else if (version < SQL_DBMS_SCHEMA_VERSION)
357  {
358  char *to_ospath;
359 
360  Com_Printf("SQLite3 ETL: Old DB schema #%i detected - performing backup and update ...\n", version);
361 
362  sqlite3_finalize(res);
363 
364  // backup old database file etl.dbX.old
365  // Make sure that we actually have the homepath available so we dont try to create a database file into a nonexisting path
366  to_ospath = FS_BuildOSPath(Cvar_VariableString("fs_homepath"), "", "");
367  if (FS_CreatePath(to_ospath))
368  {
369  Com_Printf("... DB_CheckUpdates failed - can't create path for backup file\n");
370  return qfalse;
371  }
372 
373  to_ospath = FS_BuildOSPath(Cvar_VariableString("fs_homepath"), va("%s%i.old", db_uri->string, version), "");
374  to_ospath[strlen(to_ospath) - 1] = '\0';
375 
376  result = DB_LoadOrSaveDb(db, to_ospath, 1);
377 
378  if (result != SQLITE_OK)
379  {
380  Com_Printf(S_COLOR_YELLOW "WARNING: can't save database backup file [%i]\n", result);
381  return qfalse;
382  }
383 
384  Com_Printf("SQLite3 ETL: backup '%s' created ...\n", to_ospath);
385 
386  if (!DB_CreateOrUpdateSchema(version))
387  {
388  return qfalse;
389  }
390 
391  Com_Printf("SQLite3 ETL: Old database schema has been updated to version #%i...\n", SQL_DBMS_SCHEMA_VERSION);
392 
393  return qtrue;
394  }
395 
396  if (version == 0)
397  {
398  Com_Printf(S_COLOR_YELLOW "WARNING: database update can't find a valid schema. Game and database are not in sync!\n");
399  }
400  else // downgrade case ... we can't ensure a working system
401  {
402  Com_Printf(S_COLOR_YELLOW "WARNING: database update has detected an unknown schema #%i. Game and database are not in sync!\n", version);
403  }
404 
405  sqlite3_finalize(res);
406 
407  return qfalse;
408 }

613 int DB_BackupDB(const char *zFilename, void (*xProgress)(int, int))
614 {
615  int rc; // Function return code
616  sqlite3 *pFile; // Database connection opened on zFilename
617  sqlite3_backup *pBackup; // Backup handle used to copy data
618 
619  // Open the database file identified by zFilename.
620  rc = sqlite3_open(zFilename, &pFile);
621  // FIXME
622  //rc = sqlite3_open_v2(zFilename, &pFile, (SQLITE_OPEN_READWRITE | "SQLITE_OPEN_CREATE"), NULL);
623 
624  if (rc == SQLITE_OK)
625  {
626  // Open the sqlite3_backup object used to accomplish the transfer
627  pBackup = sqlite3_backup_init(pFile, "main", db, "main");
628  if (pBackup)
629  {
630  // Each iteration of this loop copies 5 database pages from database
631  // pDb to the backup database. If the return value of backup_step()
632  // indicates that there are still further pages to copy, sleep for
633  // 250 ms before repeating.
634  do
635  {
636  rc = sqlite3_backup_step(pBackup, 5);
637  xProgress(sqlite3_backup_remaining(pBackup), sqlite3_backup_pagecount(pBackup));
638  if (rc == SQLITE_OK || rc == SQLITE_BUSY || rc == SQLITE_LOCKED)
639  {
640  sqlite3_sleep(250);
641  }
642  }
643  while (rc == SQLITE_OK || rc == SQLITE_BUSY || rc == SQLITE_LOCKED);
644 
645  // Release resources allocated by backup_init().
646  (void) sqlite3_backup_finish(pBackup);
647  }
648  rc = sqlite3_errcode(pFile);
649  }
650 
651  // Close the database connection opened on database file zFilename and return the result of this function.
652  (void) sqlite3_close(pFile);
653  return rc;
654 }

679 int DB_LoadOrSaveDb(sqlite3 *pInMemory, const char *zFilename, int isSave)
680 {
681  int rc; // Function return code
682  sqlite3 *pFile; // Database connection opened on zFilename
683  sqlite3_backup *pBackup; // Backup object used to copy data
684  sqlite3 *pTo; // Database to copy to (pFile or pInMemory)
685  sqlite3 *pFrom; // Database to copy from (pFile or pInMemory)
686 
687  // Open the database file identified by zFilename. Exit early if this fails for any reason.
688  rc = sqlite3_open(zFilename, &pFile);
689  // FIXME
690  //rc = sqlite3_open_v2(zFilename, &pFile, (SQLITE_OPEN_READWRITE | "SQLITE_OPEN_CREATE"), NULL);
691 
692  if (rc == SQLITE_OK)
693  {
694  // If this is a 'load' operation (isSave==0), then data is copied
695  // from the database file just opened to database pInMemory.
696  // Otherwise, if this is a 'save' operation (isSave==1), then data
697  // is copied from pInMemory to pFile. Set the variables pFrom and
698  // pTo accordingly.
699  pFrom = (isSave ? pInMemory : pFile);
700  pTo = (isSave ? pFile : pInMemory);
701 
702  // Set up the backup procedure to copy from the "main" database of
703  // connection pFile to the main database of connection pInMemory.
704  // If something goes wrong, pBackup will be set to NULL and an error
705  // code and message left in connection pTo.
706  //
707  // If the backup object is successfully created, call backup_step()
708  // to copy data from pFile to pInMemory. Then call backup_finish()
709  // to release resources associated with the pBackup object. If an
710  // error occurred, then an error code and message will be left in
711  // connection pTo. If no error occurred, then the error code belonging
712  // to pTo is set to SQLITE_OK.
713  pBackup = sqlite3_backup_init(pTo, "main", pFrom, "main");
714  if (pBackup)
715  {
716  (void) sqlite3_backup_step(pBackup, -1);
717  (void) sqlite3_backup_finish(pBackup);
718  }
719  rc = sqlite3_errcode(pTo);
720  }
721 
722  // Close the database connection opened on database file zFilename and return the result of this function.
723  (void) sqlite3_close(pFile);
724  return rc;
725 }
