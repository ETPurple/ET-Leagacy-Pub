358 qboolean Com_InitUpdateDownloads(void)
359 {
360 #ifdef FEATURE_AUTOUPDATE
361  if (autoupdate.updateStarted
362 #ifndef DEDICATED
363  && NET_CompareAdr(autoupdate.autoupdateServer, clc.serverAddress)
364 #endif
365  )
366  {
367  if (strlen(com_updatefiles->string) > MIN_PACK_LEN)
368  {
369  char *updateFile;
370  char updateFilesRemaining[MAX_TOKEN_CHARS] = "";
371 
372  upd.bWWWDl = qtrue;
373  upd.bWWWDlDisconnected = qtrue;
374 
375  updateFile = strtok(com_updatefiles->string, ";");
376 
377  if (updateFile == NULL)
378  {
379  Com_Error(ERR_AUTOUPDATE, "Could not parse update string.");
380  }
381  else
382  {
383  // download format: @remotename@localname
384  Q_strncpyz(upd.downloadList, va("@%s@%s", updateFile, updateFile), MAX_INFO_STRING);
385  Q_strncpyz(upd.originalDownloadName, va("%s/%s", AUTOUPDATE_DIR, updateFile), sizeof(upd.originalDownloadName));
386 
387  if (!Q_stricmp(updateFile, UPDATE_PACKAGE))
388  {
389  Q_strncpyz(upd.downloadName, va("%s/updater/%s-%s-%s", UPDATE_SERVER_NAME, ETLEGACY_VERSION_SHORT, CPUSTRING, updateFile), sizeof(upd.downloadName));
390  }
391  else
392  {
393  Q_strncpyz(upd.downloadName, va("%s/packages/%s", UPDATE_SERVER_NAME, updateFile), sizeof(upd.downloadName));
394  }
395 
396  Q_strncpyz(upd.downloadTempName, FS_BuildOSPath(Cvar_VariableString("fs_homepath"), AUTOUPDATE_DIR, va("%s.tmp", updateFile)), sizeof(upd.downloadTempName));
397  // TODO: add file size, so UI can show progress bar
398  //Cvar_SetValue("cl_downloadSize", clc.downloadSize);
399 
400  if (!DL_BeginDownload(upd.downloadTempName, upd.downloadName))
401  {
402  Com_Error(ERR_AUTOUPDATE, "Could not download an update file: \"%s\"", upd.downloadName);
403  upd.bWWWDlAborting = qtrue; //TODO: never executed
404  }
405 
406  while (1)
407  {
408  updateFile = strtok(NULL, ";");
409 
410  if (updateFile == NULL)
411  {
412  break;
413  }
414 
415  Q_strcat(updateFilesRemaining, sizeof(updateFilesRemaining), va("%s;", updateFile));
416  }
417 
418  if (strlen(updateFilesRemaining) > MIN_PACK_LEN)
419  {
420  Cvar_Set("com_updatefiles", updateFilesRemaining);
421  }
422  else
423  {
424  Cvar_Set("com_updatefiles", "");
425  }
426  }
427  }
428  return qtrue;
429  }
430 #endif // FEATURE_AUTOUPDATE
431 
432  return qfalse;
433 }
