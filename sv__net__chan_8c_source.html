48 static void SV_Netchan_Encode(client_t *client, msg_t *msg, char *commandString)
49 {
50  long i, index;
51  byte key, *string;
52  int srdc, sbit;
53  qboolean soob;
54 
55  if (msg->cursize < SV_ENCODE_START)
56  {
57  return;
58  }
59 
60  // NOTE: saving pos, reading reliableAck, restoring, not using it .. useless?
61  srdc = msg->readcount;
62  sbit = msg->bit;
63  soob = msg->oob;
64 
65  msg->bit = 0;
66  msg->readcount = 0;
67  msg->oob = qfalse;
68 
69  MSG_ReadLong(msg);
70 
71  msg->oob = soob;
72  msg->bit = sbit;
73  msg->readcount = srdc;
74 
75  string = (byte *)commandString;
76  index = 0;
77  // xor the client challenge with the netchan sequence number
78  key = client->challenge ^ client->netchan.outgoingSequence;
79  for (i = SV_ENCODE_START; i < msg->cursize; i++)
80  {
81  // modify the key with the last received and with this message acknowledged client command
82  if (!string[index])
83  {
84  index = 0;
85  }
86  if ((!IS_LEGACY_MOD && (byte)string[index] > 127) || string[index] == '%')
87  {
88  string[index] = '.';
89  }
90  key ^= string[index] << (i & 1);
91 
92  index++;
93  // encode the data with this key
94  *(msg->data + i) = *(msg->data + i) ^ key;
95  }
96 }

108 static void SV_Netchan_Decode(client_t *client, msg_t *msg)
109 {
110  int serverId, messageAcknowledge, reliableAcknowledge;
111  int i;
112  int index = 0;
113  int srdc = msg->readcount;
114  int sbit = msg->bit;
115  qboolean soob = msg->oob;
116  byte key, *string;
117 
118  msg->oob = qfalse;
119 
120  serverId = MSG_ReadLong(msg);
121  messageAcknowledge = MSG_ReadLong(msg);
122  reliableAcknowledge = MSG_ReadLong(msg);
123 
124  msg->oob = soob;
125  msg->bit = sbit;
126  msg->readcount = srdc;
127 
128  string = (byte *)client->reliableCommands[reliableAcknowledge & (MAX_RELIABLE_COMMANDS - 1)];
129 
130  key = client->challenge ^ serverId ^ messageAcknowledge;
131  for (i = msg->readcount + SV_DECODE_START; i < msg->cursize; i++)
132  {
133  // modify the key with the last sent and acknowledged server command
134  if (!string[index])
135  {
136  index = 0;
137  }
138 
139  if ((!IS_LEGACY_MOD && (byte)string[index] > 127) || string[index] == '%')
140  {
141  string[index] = '.';
142  }
143  key ^= string[index] << (i & 1);
144 
145  index++;
146  // decode the data with this key
147  *(msg->data + i) = *(msg->data + i) ^ key;
148  }
149 }

255 void SV_Netchan_Transmit(client_t *client, msg_t *msg)
256 {
257  MSG_WriteByte(msg, svc_EOF);
258  SV_WriteBinaryMessage(msg, client);
259 
260  if (client->netchan.unsentFragments || client->netchan_start_queue)
261  {
262  netchan_buffer_t *netbuf;
263  Com_DPrintf("SV_Netchan_Transmit: unsent fragments, stacked\n");
264  netbuf = (netchan_buffer_t *)Z_Malloc(sizeof(netchan_buffer_t));
265  // store the msg, we can't store it encoded, as the encoding depends on stuff we still have to finish sending
266  MSG_Copy(&netbuf->msg, netbuf->msgBuffer, sizeof(netbuf->msgBuffer), msg);
267  Q_strncpyz(netbuf->lastClientCommandString, client->lastClientCommandString, sizeof(netbuf->lastClientCommandString));
268  netbuf->next = NULL;
269  // insert it in the queue, the message will be encoded and sent later
270  *client->netchan_end_queue = netbuf;
271  client->netchan_end_queue = &(*client->netchan_end_queue)->next;
272  }
273  else
274  {
275  SV_Netchan_Encode(client, msg, client->lastClientCommandString);
276  Netchan_Transmit(&client->netchan, msg->cursize, msg->data);
277  }
278 }
