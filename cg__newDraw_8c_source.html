47 int CG_TrimLeftPixels(char *instr, float scale, float w, int size)
48 {
49  char buffer[1024];
50  char *p, *s;
51  int tw;
52  int i;
53 
54  Q_strncpyz(buffer, instr, 1024);
55  Com_Memset(instr, 0, size);
56 
57  for (i = 0, p = buffer; *p; p++, i++)
58  {
59  instr[i] = *p;
60  tw = CG_Text_Width(instr, scale, 0);
61  if (tw >= w)
62  {
63  Com_Memset(instr, 0, size);
64  for (s = instr, p = &buffer[i + 1]; *p && ((s - instr) < size); p++, s++)
65  {
66  *s = *p;
67  }
68  return tw - w;
69  }
70  }
71 
72  return -1;
73 }

83 void CG_FitTextToWidth_Ext(char *instr, float scale, float w, int size, fontHelper_t *font)
84 {
85  char buffer[1024];
86  char *s, *p, *c, *ls = NULL;
87 
88  Q_strncpyz(buffer, instr, 1024);
89  Com_Memset(instr, 0, size);
90 
91  c = s = instr;
92  p = buffer;
93 
94  while (*p)
95  {
96  *c = *p++;
97 
98  if (*c == ' ')
99  {
100  ls = c;
101  } // store last space, to try not to break mid word
102 
103  c++;
104 
105  if (*p == '\n')
106  {
107  s = c + 1;
108  }
109  else if (CG_Text_Width_Ext(s, scale, 0, font) > w)
110  {
111  if (ls)
112  {
113  *ls = '\n';
114  s = ls + 1;
115  }
116  else
117  {
118  *c = *(c - 1);
119  *(c - 1) = '\n';
120  s = c++;
121  }
122 
123  ls = NULL;
124  }
125  }
126 
127  if (c != buffer && (*(c - 1) != '\n'))
128  {
129  *c++ = '\n';
130  }
131 
132  *c = '\0';
133 }
