43 qboolean R_CheckFBO(const FBO_t *fbo)
44 {
45  int code;
46  int id;
47 
48  glGetIntegerv(GL_FRAMEBUFFER_BINDING, &id);
49  glBindFramebuffer(GL_FRAMEBUFFER, fbo->frameBuffer);
50 
51  code = glCheckFramebufferStatus(GL_FRAMEBUFFER);
52 
53  switch (code)
54  {
55  case GL_FRAMEBUFFER_COMPLETE:
56  glBindFramebuffer(GL_FRAMEBUFFER_EXT, id); // ok
57  return qtrue;
58  // an error occured
59  case GL_FRAMEBUFFER_UNSUPPORTED:
60  Ren_Warning("R_CheckFBO: (%s) Unsupported framebuffer format\n", fbo->name);
61  break;
62  case GL_FRAMEBUFFER_INCOMPLETE_ATTACHMENT:
63  Ren_Warning("R_CheckFBO: (%s) Framebuffer incomplete attachment\n", fbo->name);
64  break;
65  case GL_FRAMEBUFFER_INCOMPLETE_MISSING_ATTACHMENT:
66  Ren_Warning("R_CheckFBO: (%s) Framebuffer incomplete, missing attachment\n", fbo->name);
67  break;
68  //case GL_FRAMEBUFFER_INCOMPLETE_DUPLICATE_ATTACHMENT_EXT:
69  // Ren_Warning( "R_CheckFBO: (%s) Framebuffer incomplete, duplicate attachment\n", fbo->name);
70  // break;
71  case GL_FRAMEBUFFER_INCOMPLETE_DIMENSIONS_EXT:
72  Ren_Warning("R_CheckFBO: (%s) Framebuffer incomplete, attached images must have same dimensions\n", fbo->name);
73  break;
74  case GL_FRAMEBUFFER_INCOMPLETE_FORMATS_EXT:
75  Ren_Warning("R_CheckFBO: (%s) Framebuffer incomplete, attached images must have same format\n", fbo->name);
76  break;
77  case GL_FRAMEBUFFER_INCOMPLETE_DRAW_BUFFER:
78  Ren_Warning("R_CheckFBO: (%s) Framebuffer incomplete, missing draw buffer\n", fbo->name);
79  break;
80  case GL_FRAMEBUFFER_INCOMPLETE_READ_BUFFER:
81  Ren_Warning("R_CheckFBO: (%s) Framebuffer incomplete, missing read buffer\n", fbo->name);
82  break;
83  default:
84  Ren_Warning("R_CheckFBO: (%s) unknown error 0x%X\n", fbo->name, code);
85  //Ren_Fatal( "R_CheckFBO: (%s) unknown error 0x%X", fbo->name, code);
86  //etl_assert(0);
87  break;
88  }
89 
90  glBindFramebuffer(GL_FRAMEBUFFER, id);
91 
92  return qfalse;
93 }
