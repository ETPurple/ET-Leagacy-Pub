79 qboolean isEntVisible(entityState_t *ent)
80 {
81  trace_t tr;
82  vec3_t start, end, temp;
83  vec3_t forward, up, right, right2;
84  float view_height;
85 
86  VectorCopy(cl.cgameClientLerpOrigin, start);
87  start[2] += (cl.snap.ps.viewheight - 1);
88  if (cl.snap.ps.leanf != 0.f)
89  {
90  vec3_t lright, v3ViewAngles;
91  VectorCopy(cl.snap.ps.viewangles, v3ViewAngles);
92  v3ViewAngles[2] += cl.snap.ps.leanf / 2.0f;
93  angles_vectors(v3ViewAngles, NULL, lright, NULL);
94  VectorMA(start, cl.snap.ps.leanf, lright, start);
95  }
96 
97  VectorCopy(ent->pos.trBase, end);
98 
99  // Compute vector perpindicular to view to ent
100  VectorSubtract(end, start, forward);
101  vec3_norm_fast(forward);
102  VectorSet(up, 0, 0, 1);
103  vec3_cross(forward, up, right);
104  vec3_norm_fast(right);
105  VectorScale(right, 10, right2);
106  VectorScale(right, 18, right);
107 
108  // Set viewheight
109  if (ent->animMovetype)
110  {
111  view_height = 16;
112  }
113  else
114  {
115  view_height = 40;
116  }
117 
118  // First, viewpoint to viewpoint
119  end[2] += view_height;
120  CM_BoxTrace(&tr, start, end, NULL, NULL, 0, CONTENTS_SOLID, qfalse);
121  if (tr.fraction == 1.f)
122  {
123  return qtrue;
124  }
125 
126  // First-b, viewpoint to top of head
127  end[2] += 16;
128  CM_BoxTrace(&tr, start, end, NULL, NULL, 0, CONTENTS_SOLID, qfalse);
129  if (tr.fraction == 1.f)
130  {
131  return qtrue;
132  }
133  end[2] -= 16;
134 
135  // Second, viewpoint to ent's origin
136  end[2] -= view_height;
137  CM_BoxTrace(&tr, start, end, NULL, NULL, 0, CONTENTS_SOLID, qfalse);
138  if (tr.fraction == 1.f)
139  {
140  return qtrue;
141  }
142 
143  // Third, to ent's right knee
144  VectorAdd(end, right, temp);
145  temp[2] += 8;
146  CM_BoxTrace(&tr, start, temp, NULL, NULL, 0, CONTENTS_SOLID, qfalse);
147  if (tr.fraction == 1.f)
148  {
149  return qtrue;
150  }
151 
152  // Fourth, to ent's right shoulder
153  VectorAdd(end, right2, temp);
154  if (ent->animMovetype)
155  {
156  temp[2] += 28;
157  }
158  else
159  {
160  temp[2] += 52;
161  }
162  CM_BoxTrace(&tr, start, temp, NULL, NULL, 0, CONTENTS_SOLID, qfalse);
163  if (tr.fraction == 1.f)
164  {
165  return qtrue;
166  }
167 
168  // Fifth, to ent's left knee
169  VectorScale(right, -1, right);
170  VectorScale(right2, -1, right2);
171  VectorAdd(end, right2, temp);
172  temp[2] += 2;
173  CM_BoxTrace(&tr, start, temp, NULL, NULL, 0, CONTENTS_SOLID, qfalse);
174  if (tr.fraction == 1.f)
175  {
176  return qtrue;
177  }
178 
179  // Sixth, to ent's left shoulder
180  VectorAdd(end, right, temp);
181  if (ent->animMovetype)
182  {
183  temp[2] += 16;
184  }
185  else
186  {
187  temp[2] += 36;
188  }
189  CM_BoxTrace(&tr, start, temp, NULL, NULL, 0, CONTENTS_SOLID, qfalse);
190  if (tr.fraction == 1.f)
191  {
192  return qtrue;
193  }
194 
195  return qfalse;
196 }

207 void CL_DeltaEntity(msg_t *msg, clSnapshot_t *frame, int newnum, entityState_t *old, qboolean unchanged)
208 {
209  // save the parsed entity state into the big circular buffer so
210  // it can be used as the source for a later delta
211  entityState_t *state = &cl.parseEntities[cl.parseEntitiesNum & (MAX_PARSE_ENTITIES - 1)];
212 
213  if (unchanged)
214  {
215  *state = *old;
216  }
217  else
218  {
219  MSG_ReadDeltaEntity(msg, old, state, newnum);
220  }
221 
222  if (state->number == (MAX_GENTITIES - 1))
223  {
224  return; // entity was delta removed
225  }
226 
227  // Only draw clients if visible
228  if (clc.onlyVisibleClients)
229  {
230  if (state->number < MAX_CLIENTS)
231  {
232  if (isEntVisible(state))
233  {
234  entLastVisible[state->number] = frame->serverTime;
235  state->eFlags &= ~EF_NODRAW;
236  }
237  else
238  {
239  if (entLastVisible[state->number] < (frame->serverTime - 600))
240  {
241  state->eFlags |= EF_NODRAW;
242  }
243  }
244  }
245  }
246 
247  cl.parseEntitiesNum++;
248  frame->numEntities++;

260 void CL_ParsePacketEntities(msg_t *msg, clSnapshot_t *oldframe, clSnapshot_t *newframe)
261 {
262  entityState_t *oldstate;
263  int oldindex, newnum, oldnum;
264 
265  oldstate = NULL;
266  oldindex = 0;
267 
268  newframe->parseEntitiesNum = cl.parseEntitiesNum;
269  newframe->numEntities = 0;
270 
271  // delta from the entities present in oldframe
272 
273  if (!oldframe)
274  {
275  oldnum = MAX_GENTITIES;
276  }
277  else
278  {
279  if (oldindex >= oldframe->numEntities)
280  {
281  oldnum = MAX_GENTITIES;
282  }
283  else
284  {
285  oldstate = &cl.parseEntities[
286  (oldframe->parseEntitiesNum + oldindex) & (MAX_PARSE_ENTITIES - 1)];
287  oldnum = oldstate->number;
288  }
289  }
290 
291  while (1)
292  {
293  // read the entity index number
294  newnum = MSG_ReadBits(msg, GENTITYNUM_BITS);
295 
296  if (newnum >= (MAX_GENTITIES - 1))
297  {
298  break;
299  }
300 
301  if (msg->readcount > msg->cursize)
302  {
303  Com_Error(ERR_DROP, "CL_ParsePacketEntities: end of message");
304  }
305 
306  while (oldnum < newnum)
307  {
308  // one or more entities from the old packet are unchanged
309  if (cl_shownet->integer == 3)
310  {
311  Com_Printf("%3i: unchanged: %i\n", msg->readcount, oldnum);
312  }
313  CL_DeltaEntity(msg, newframe, oldnum, oldstate, qtrue);
314 
315  oldindex++;
316 
317  if (!oldframe || oldindex >= oldframe->numEntities)
318  {
319  oldnum = MAX_GENTITIES;
320  }
321  else
322  {
323  oldstate = &cl.parseEntities[
324  (oldframe->parseEntitiesNum + oldindex) & (MAX_PARSE_ENTITIES - 1)];
325  oldnum = oldstate->number;
326  }
327  }
328 
329  if (oldnum == newnum)
330  {
331  // delta from previous state
332  if (cl_shownet->integer == 3)
333  {
334  Com_Printf("%3i: delta: %i\n", msg->readcount, newnum);
335  }
336  CL_DeltaEntity(msg, newframe, newnum, oldstate, qfalse);
337 
338  oldindex++;
339 
340  if (oldindex >= oldframe->numEntities)
341  {
342  oldnum = MAX_GENTITIES;
343  }
344  else
345  {
346  oldstate = &cl.parseEntities[
347  (oldframe->parseEntitiesNum + oldindex) & (MAX_PARSE_ENTITIES - 1)];
348  oldnum = oldstate->number;
349  }
350  }
351  else if (oldnum > newnum)
352  {
353  // delta from baseline
354  if (cl_shownet->integer == 3)
355  {
356  Com_Printf("%3i: baseline: %i\n", msg->readcount, newnum);
357  }
358  CL_DeltaEntity(msg, newframe, newnum, &cl.entityBaselines[newnum], qfalse);
359  }
360  }
361 
362  // any remaining entities in the old frame are copied over
363  while (oldnum != MAX_GENTITIES)
364  {
365  // one or more entities from the old packet are unchanged
366  if (cl_shownet->integer == 3)
367  {
368  Com_Printf("%3i: unchanged: %i\n", msg->readcount, oldnum);
369  }
370  CL_DeltaEntity(msg, newframe, oldnum, oldstate, qtrue);
371 
372  oldindex++;
373 
374  if (oldindex >= oldframe->numEntities)
375  {
376  oldnum = MAX_GENTITIES;
377  }
378  else
379  {
380  oldstate = &cl.parseEntities[(oldframe->parseEntitiesNum + oldindex) & (MAX_PARSE_ENTITIES - 1)];
381  oldnum = oldstate->number;
382  }
383  }
384 
385  if (cl_shownuments->integer)
386  {
387  Com_Printf("Entities in packet: %i\n", newframe->numEntities);
388  }
389 }

398 void CL_ParseSnapshot(msg_t *msg)
399 {
400  int len;
401  clSnapshot_t *old;
402  clSnapshot_t newSnap;
403  int deltaNum;
404  int oldMessageNum;
405  int i, packetNum;
406 
407  // get the reliable sequence acknowledge number
408  // NOTE: now sent with all server to client messages
409  //clc.reliableAcknowledge = MSG_ReadLong( msg );
410 
411  // read in the new snapshot to a temporary buffer
412  // we will only copy to cl.snap if it is valid
413  Com_Memset(&newSnap, 0, sizeof(newSnap));
414 
415  // we will have read any new server commands in this
416  // message before we got to svc_snapshot
417  newSnap.serverCommandNum = clc.serverCommandSequence;
418 
419  newSnap.serverTime = MSG_ReadLong(msg);
420 
421  newSnap.messageNum = clc.serverMessageSequence;
422 
423  deltaNum = MSG_ReadByte(msg);
424  if (!deltaNum)
425  {
426  newSnap.deltaNum = -1;
427  }
428  else
429  {
430  newSnap.deltaNum = newSnap.messageNum - deltaNum;
431  }
432  newSnap.snapFlags = MSG_ReadByte(msg);
433 
434  // If the frame is delta compressed from data that we
435  // no longer have available, we must suck up the rest of
436  // the frame, but not use it, then ask for a non-compressed
437  // message
438  if (newSnap.deltaNum <= 0)
439  {
440  newSnap.valid = qtrue; // uncompressed frame
441  old = NULL;
442  if (clc.demorecording)
443  {
444  clc.demowaiting = qfalse; // we can start recording now
445  //if(cl_autorecord->integer) {
446  // Cvar_Set( "g_synchronousClients", "0" );
447  //}
448  }
449  else
450  {
451  if (cl_autorecord->integer /*&& Cvar_VariableValue( "g_synchronousClients")*/)
452  {
453  char name[256];
454  char mapname[MAX_QPATH];
455  char *period;
456  qtime_t time;
457 
458  Com_RealTime(&time);
459 
460  Q_strncpyz(mapname, cl.mapname, MAX_QPATH);
461  for (period = mapname; *period; period++)
462  {
463  if (*period == '.')
464  {
465  *period = '\0';
466  break;
467  }
468  }
469 
470  for (period = mapname; *period; period++)
471  {
472  if (*period == '/')
473  {
474  break;
475  }
476  }
477  if (*period)
478  {
479  period++;
480  }
481 
482  Com_sprintf(name, sizeof(name), "demos/%s_%i_%i.dm_%d", period, time.tm_mday, time.tm_mon + 1, PROTOCOL_VERSION);
483 
484  CL_Record(name);
485  }
486  }
487  }
488  else
489  {
490  old = &cl.snapshots[newSnap.deltaNum & PACKET_MASK];
491  if (!old->valid)
492  {
493  // should never happen
494  Com_Printf("Delta from invalid frame (not supposed to happen!).\n");
495  }
496  else if (old->messageNum != newSnap.deltaNum)
497  {
498  // The frame that the server did the delta from
499  // is too old, so we can't reconstruct it properly.
500  Com_DPrintf("Delta frame too old.\n");
501  }
502  else if (cl.parseEntitiesNum - old->parseEntitiesNum > MAX_PARSE_ENTITIES - 128)
503  {
504  Com_DPrintf("Delta parseEntitiesNum too old.\n");
505  }
506  else
507  {
508  newSnap.valid = qtrue; // valid delta parse
509  }
510  }
511 
512  // read areamask
513  len = MSG_ReadByte(msg);
514 
515  if (len < 0 || len > sizeof(newSnap.areamask))
516  {
517  Com_Error(ERR_DROP, "CL_ParseSnapshot: Invalid size %d for areamask.", len);
518  return;
519  }
520 
521  MSG_ReadData(msg, &newSnap.areamask, len);
522 
523  // read playerinfo
524  SHOWNET(msg, "playerstate");
525  if (old)
526  {
527  MSG_ReadDeltaPlayerstate(msg, &old->ps, &newSnap.ps);
528  }
529  else
530  {
531  MSG_ReadDeltaPlayerstate(msg, NULL, &newSnap.ps);
532  }
533 
534  // read packet entities
535  SHOWNET(msg, "packet entities");
536  CL_ParsePacketEntities(msg, old, &newSnap);
537 
538  // if not valid, dump the entire thing now that it has
539  // been properly read
540  if (!newSnap.valid)
541  {
542  return;
543  }
544 
545  // clear the valid flags of any snapshots between the last
546  // received and this one, so if there was a dropped packet
547  // it won't look like something valid to delta from next
548  // time we wrap around in the buffer
549  oldMessageNum = cl.snap.messageNum + 1;
550 
551  if (newSnap.messageNum - oldMessageNum >= PACKET_BACKUP)
552  {
553  oldMessageNum = newSnap.messageNum - (PACKET_BACKUP - 1);
554  }
555  for ( ; oldMessageNum < newSnap.messageNum ; oldMessageNum++)
556  {
557  cl.snapshots[oldMessageNum & PACKET_MASK].valid = qfalse;
558  }
559 
560  // copy to the current good spot
561  cl.snap = newSnap;
562  cl.snap.ping = 999;
563  // calculate ping time
564  for (i = 0 ; i < PACKET_BACKUP ; i++)
565  {
566  packetNum = (clc.netchan.outgoingSequence - 1 - i) & PACKET_MASK;
567  if (cl.snap.ps.commandTime >= cl.outPackets[packetNum].p_serverTime)
568  {
569  cl.snap.ping = cls.realtime - cl.outPackets[packetNum].p_realtime;
570  break;
571  }
572  }
573  // save the frame off in the backup array for later delta comparisons
574  cl.snapshots[cl.snap.messageNum & PACKET_MASK] = cl.snap;
575 
576  if (cl_shownet->integer == 3)
577  {
578  Com_Printf(" snapshot:%i delta:%i ping:%i\n", cl.snap.messageNum,
579  cl.snap.deltaNum, cl.snap.ping);
580  }
581 
582  cl.newSnapshots = qtrue;
583 }

724 void CL_ParseGamestate(msg_t *msg)
725 {
726  int i;
727  entityState_t *es;
728  int newnum;
729  entityState_t nullstate;
730  int cmd;
731  char *s;
732 
733  // Con_Close();
734 
735  clc.connectPacketCount = 0;
736 
737  // wipe local client state
738  CL_ClearState();
739 
740  // a gamestate always marks a server command sequence
741  clc.serverCommandSequence = MSG_ReadLong(msg);
742 
743  // parse all the configstrings and baselines
744  cl.gameState.dataCount = 1; // leave a 0 at the beginning for uninitialized configstrings
745  while (1)
746  {
747  cmd = MSG_ReadByte(msg);
748 
749  if (cmd == svc_EOF)
750  {
751  break;
752  }
753 
754  if (cmd == svc_configstring)
755  {
756  int len;
757 
758  i = MSG_ReadShort(msg);
759  if (i < 0 || i >= MAX_CONFIGSTRINGS)
760  {
761  Com_Error(ERR_DROP, "configstring < 0 or configstring >= MAX_CONFIGSTRINGS");
762  }
763  s = MSG_ReadBigString(msg);
764  len = strlen(s);
765 
766  if (len + 1 + cl.gameState.dataCount > MAX_GAMESTATE_CHARS)
767  {
768  Com_Error(ERR_DROP, "MAX_GAMESTATE_CHARS exceeded");
769  }
770 
771  // append it to the gameState string buffer
772  cl.gameState.stringOffsets[i] = cl.gameState.dataCount;
773  Com_Memcpy(cl.gameState.stringData + cl.gameState.dataCount, s, len + 1);
774  cl.gameState.dataCount += len + 1;
775  }
776  else if (cmd == svc_baseline)
777  {
778  newnum = MSG_ReadBits(msg, GENTITYNUM_BITS);
779  if (newnum < 0 || newnum >= MAX_GENTITIES)
780  {
781  Com_Error(ERR_DROP, "Baseline number out of range: %i", newnum);
782  }
783  Com_Memset(&nullstate, 0, sizeof(nullstate));
