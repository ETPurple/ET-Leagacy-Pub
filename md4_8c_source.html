116 /* F, G and H are basic MD4 functions. */
117 #define F(x, y, z) (((x) & (y)) | ((~x) & (z)))
118 #define G(x, y, z) (((x) & (y)) | ((x) & (z)) | ((y) & (z)))
119 #define H(x, y, z) ((x) ^ (y) ^ (z))

121 /* ROTATE_LEFT rotates x left n bits. */
122 #define ROTATE_LEFT(x, n) (((x) << (n)) | ((x) >> (32 - (n))))

124 /* FF, GG and HH are transformations for rounds 1, 2 and 3 */
125 /* Rotation is separate from addition to prevent recomputation */
126 #define FF(a, b, c, d, x, s) { (a) += F((b), (c), (d)) + (x); (a) = ROTATE_LEFT((a), (s)); }

145 /* MD4 block update operation. Continues an MD4 message-digest operation, processing another message block, and updating the context. */
146 void MD4Update(MD4_CTX *context, const unsigned char *input, unsigned int inputLen)
147 {
148  unsigned int i, index, partLen;
149 
150  /* Compute number of bytes mod 64 */
151  index = ( unsigned int )((context->count[0] >> 3) & 0x3F);
152 
153  /* Update number of bits */
154  if ((context->count[0] += ((UINT4)inputLen << 3)) < ((UINT4)inputLen << 3))
155  {
156  context->count[1]++;
157  }
158 
159  context->count[1] += ((UINT4)inputLen >> 29);
160 
161  partLen = 64 - index;
162 
163  /* Transform as many times as possible.*/
164  if (inputLen >= partLen)
165  {
166  Com_Memcpy((POINTER)&context->buffer[index], (POINTER)input, partLen);
167  MD4Transform(context->state, context->buffer);
168 
169  for (i = partLen; i + 63 < inputLen; i += 64)
170  MD4Transform(context->state, &input[i]);
171 
172  index = 0;
173  }
174  else
175  {
176  i = 0;
177  }
178 
179  /* Buffer remaining input */
180  Com_Memcpy((POINTER)&context->buffer[index], (POINTER)&input[i], inputLen - i);
181 }

184 /* MD4 finalization. Ends an MD4 message-digest operation, writing the the message digest and zeroizing the context. */
185 void MD4Final(unsigned char digest[16], MD4_CTX *context)
186 {
187  unsigned char bits[8];
188  unsigned int index, padLen;
189 
190  /* Save number of bits */
191  Encode(bits, context->count, 8);
192 
193  /* Pad out to 56 mod 64.*/
194  index = ( unsigned int )((context->count[0] >> 3) & 0x3f);
195  padLen = (index < 56) ? (56 - index) : (120 - index);
196  MD4Update(context, PADDING, padLen);
197 
198  /* Append length (before padding) */
199  MD4Update(context, bits, 8);
200 
201  /* Store state in digest */
202  Encode(digest, context->state, 16);
203 
204  /* Zeroize sensitive information.*/
205  Com_Memset((POINTER)context, 0, sizeof(*context));
206 }

281 static void Encode(unsigned char *output, UINT4 *input, unsigned int len)
282 {
283  unsigned int i, j;
284 
285  for (i = 0, j = 0; j < len; i++, j += 4)
286  {
287  output[j] = ( unsigned char )(input[i] & 0xff);
288  output[j + 1] = ( unsigned char )((input[i] >> 8) & 0xff);
289  output[j + 2] = ( unsigned char )((input[i] >> 16) & 0xff);
290  output[j + 3] = ( unsigned char )((input[i] >> 24) & 0xff);
291  }
