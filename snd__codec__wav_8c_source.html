100 static int S_FindRIFFChunk(fileHandle_t f, const char *chunk)
101 {
102  char name[5];
103  int len;
104 
105  while ((len = S_ReadChunkInfo(f, name)) >= 0)
106  {

129 static void S_ByteSwapRawSamples(int samples, int width, int s_channels, byte *data)
130 {
131  int i;
132 
133  if (width != 2)
134  {
135  return;
136  }
137  if (LittleShort(256) == 256)
138  {
139  return;
140  }
141 
142  if (s_channels == 2)
143  {
144  samples <<= 1;
145  }
146  for (i = 0 ; i < samples ; i++)
147  {
148  ((short *)data)[i] = LittleShort(((short *)data)[i]);

158 static qboolean S_ReadRIFFHeader(fileHandle_t file, snd_info_t *info)
159 {
160  char dump[16];
161  //int wav_format;
162  int bits;
163  int fmtlen = 0;
164 
165  // skip the riff wav header
166  FS_Read(dump, 12, file);
167 
168  // Scan for the format chunk
169  if ((fmtlen = S_FindRIFFChunk(file, "fmt ")) < 0)
170  {
171  Com_Printf(S_COLOR_RED "ERROR: Couldn't find \"fmt\" chunk\n");
172  return qfalse;
173  }
174 
175  // Save the parameters
176  //wav_format = FGetLittleShort(file);
177  FGetLittleShort(file);
178  info->channels = FGetLittleShort(file);
179  info->rate = FGetLittleLong(file);
180  FGetLittleLong(file);
181  FGetLittleShort(file);
182  bits = FGetLittleShort(file);
183 
184  if (bits < 8)
185  {
186  Com_Printf(S_COLOR_RED "ERROR: Less than 8 bit sound is not supported\n");
187  return qfalse;
188  }
189 
190  info->width = bits / 8;
191  info->dataofs = 0;
192 
193  // Skip the rest of the format chunk if required
194  if (fmtlen > 16)
195  {
196  fmtlen -= 16;
197  (void) FS_Seek(file, fmtlen, FS_SEEK_CUR);
198  }
199 
200  // Scan for the data chunk
201  if ((info->size = S_FindRIFFChunk(file, "data")) < 0)
202  {
203  Com_Printf(S_COLOR_RED "ERROR: Couldn't find \"data\" chunk\n");
204  return qfalse;
205  }
206  info->samples = (info->size / info->width) / info->channels;
207 
208  return qtrue;
209 }

228 void *S_WAV_CodecLoad(const char *filename, snd_info_t *info)
229 {
230  fileHandle_t file;
231  void *buffer;
232 
233  // Try to open the file
234  FS_FOpenFileRead(filename, &file, qtrue);
235  if (!file)
236  {
237  if (com_developer->integer)
238  {
239  Com_Printf(S_COLOR_RED "ERROR: Could not open \"%s\"\n", filename);
240  }
241  return NULL;
242  }
243 
244  // Read the RIFF header
245  if (!S_ReadRIFFHeader(file, info))
246  {
247  FS_FCloseFile(file);
248  if (com_developer->integer)
249  {
250  Com_Printf(S_COLOR_RED "ERROR: Incorrect/unsupported format in \"%s\"\n", filename);
251  }
252  return NULL;
253  }
254 
255  // Allocate some memory
256  buffer = Hunk_AllocateTempMemory(info->size);
257  if (!buffer)
258  {
259  FS_FCloseFile(file);
260  Com_Printf(S_COLOR_RED "ERROR: Out of memory reading \"%s\"\n", filename);
261  return NULL;
262  }
263 
264  // Read, byteswap
265  FS_Read(buffer, info->size, file);
266  S_ByteSwapRawSamples(info->samples, info->width, info->channels, (byte *)buffer);
267 
268  // Close and return
269  FS_FCloseFile(file);
270  return buffer;
271 }
