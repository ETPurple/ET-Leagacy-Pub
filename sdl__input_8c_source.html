210 static qboolean IN_IsConsoleKey(keyNum_t key, int character)
211 {
212  typedef struct consoleKey_s
213  {
214  enum
215  {
216  KEY,
217  CHARACTER
218  } type;
219 
220  union
221  {
222  keyNum_t key;
223  int character;
224  } u;
225  } consoleKey_t;
226 
227  static consoleKey_t consoleKeys[MAX_CONSOLE_KEYS];
228  static int numConsoleKeys = 0;
229  int i;
230 
231  if (key == K_GRAVE
232 #ifdef __APPLE__
233  || key == 60 // Same as console key
234 #endif
235  )
236  {
237  return qtrue;
238  }
239 
240  // Only parse the variable when it changes
241  if (cl_consoleKeys->modified)
242  {
243  char *text_p, *token;
244 
245  cl_consoleKeys->modified = qfalse;
246  text_p = cl_consoleKeys->string;
247  numConsoleKeys = 0;
248 
249  while (numConsoleKeys < MAX_CONSOLE_KEYS)
250  {
251  consoleKey_t *c = &consoleKeys[numConsoleKeys];
252  int charCode = 0;
253 
254  token = COM_Parse(&text_p);
255  if (!token[0])
256  {
257  break;
258  }
259 
260  if (strlen(token) == 4)
261  {
262  charCode = Com_HexStrToInt(token);
263  }
264 
265  if (charCode > 0)
266  {
267  c->type = CHARACTER;
268  c->u.character = charCode;
269  }
270  else
271  {
272  c->type = KEY;
273  c->u.key = Key_StringToKeynum(token);
274 
275  // 0 isn't a key
276  if (c->u.key <= 0)
277  {
278  continue;
279  }
280  }
281 
282  numConsoleKeys++;
283  }
284  }
285 
286  // If the character is the same as the key, prefer the character
287  if (key == character)
288  {
289  key = 0;
290  }
291 
292  for (i = 0; i < numConsoleKeys; i++)
293  {
294  consoleKey_t *c = &consoleKeys[i];
295 
296  switch (c->type)
297  {
298  case KEY:
299  if (key && c->u.key == key)
300  {
301  return qtrue;
302  }
303  break;
304  case CHARACTER:
305  if (c->u.character == character)
306  {
307  return qtrue;
308  }
309  break;
310  }
311  }
312 
313  return qfalse;
314 }
