292 qboolean PC_Script_Parse(int handle, const char **out)
293 {
294  char script[4096];
295  pc_token_t token;
296 
297  Com_Memset(script, 0, sizeof(script));
298  // scripts start with { and have ; separated command lists.. commands are command, arg..
299  // basically we want everything between the { } as it will be interpreted at run time
300 
301  if (!trap_PC_ReadToken(handle, &token))
302  {
303  return qfalse;
304  }
305 
306  if (Q_stricmp(token.string, "{") != 0)
307  {
308  return qfalse;
309  }
310 
311  while (1)
312  {
313  if (!trap_PC_ReadToken(handle, &token))
314  {
315  return qfalse;
316  }
317 
318  if (Q_stricmp(token.string, "}") == 0)
319  {
320  *out = String_Alloc(script);
321  return qtrue;
322  }
323 
324  if (token.string[1] != '\0')
325  {
326  Q_strcat(script, 4096, va("\"%s\"", token.string));
327  }
328  else
329  {
330  Q_strcat(script, 4096, token.string);
331  }
332 
333  Q_strcat(script, 4096, " ");
334  }

356 int KeywordHash_Key(const char *keyword)
357 {
358  int hash = 0, i;
359 
360  for (i = 0; keyword[i] != '\0'; i++)
361  {
362  if (keyword[i] >= 'A' && keyword[i] <= 'Z')
363  {
364  hash += (keyword[i] + ('a' - 'A')) * (119 + i);
365  }
366  else
367  {
368  hash += keyword[i] * (119 + i);
369  }
370  }
371 
372  hash = (hash ^ (hash >> 10) ^ (hash >> 20)) & (KEYWORDHASH_SIZE - 1);
373  return hash;
374 }
