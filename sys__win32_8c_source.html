313

315  {
316  return;
317  }
318 
319  if (strlen(subdirs))

376 static qboolean strgtr(const char *s0, const char *s1)
377 {
378  size_t l0, l1, i;
379 
380  l0 = strlen(s0);
381  l1 = strlen(s1);
382 
383  if (l1 < l0)
384  {
385  l0 = l1;
386  }
387 
388  for (i = 0; i < l0; i++)
389  {
390  if (s1[i] > s0[i])

392  return qtrue;

395  {
396  return qfalse;
397  }
398  }
399 
400  return qfalse;

412 char **Sys_ListFiles(const char *directory, const char *extension, const char *filter, int *numfiles, qboolean wantsubs)
413 {
414  char search[MAX_OSPATH];
415  int nfiles;
416  char **listCopy;
417  char *list[MAX_FOUND_FILES];
418  struct _finddata_t findinfo;
419  intptr_t findhandle;
420  int flag;
421  int i;
422  size_t extLen;
423  qboolean invalid;
424 
425  if (filter)
426  {
427  nfiles = 0;
428  Sys_ListFilteredFiles(directory, "", filter, list, &nfiles);
429 
430  list[nfiles] = 0;
431  *numfiles = nfiles;
432 
433  if (!nfiles)
434  {
435  return NULL;
436  }
437 
438  listCopy = Z_Malloc((nfiles + 1) * sizeof(*listCopy));
439  for (i = 0 ; i < nfiles ; i++)
440  {
441  listCopy[i] = list[i];
442  }
443 
444  listCopy[i] = NULL;
445 
446  return listCopy;
447  }
448 
449  if (!extension)
450  {
451  extension = "";
452  }
453 
454  // passing a slash as extension will find directories
455  if (extension[0] == '/' && extension[1] == 0)
456  {
457  extension = "";
458  flag = 0;
459  }
460  else
461  {
462  flag = _A_SUBDIR;
463  }
464 
465  extLen = strlen(extension);
466 
467  Com_sprintf(search, sizeof(search), "%s\\*%s", directory, extension);
468 
469  // search
470  nfiles = 0;
471 
472  findhandle = _findfirst(search, &findinfo);
473  if (findhandle == -1)
474  {
475  *numfiles = 0;
476  return NULL;
477  }
478 
479  do
480  {
481  if ((!wantsubs && (flag ^ (findinfo.attrib & _A_SUBDIR))) || (wantsubs && (findinfo.attrib & _A_SUBDIR)))
482  {
483  if (*extension)
484  {
485  if (strlen(findinfo.name) < extLen || Q_stricmp(findinfo.name + strlen(findinfo.name) - extLen, extension))
486  {
487  continue; // didn't match
488  }
489  }
490 
491  // check for bad file names and don't add these
492  invalid = qfalse;
493  // note: this isn't done in Sys_ListFilteredFiles()
494 
495  for (i = 0; i < strlen(findinfo.name); i++)
496  {
497  if (findinfo.name[i] <= 31 || findinfo.name[i] >= 127)
498  {
499  Com_Printf(S_COLOR_RED "ERROR: invalid char in name of file '%s'.\n", findinfo.name);
500  invalid = qtrue;
501  break;
502  }
503  }
504 
505  if (invalid)
506  {
507  remove(va("%s%c%s", directory, PATH_SEP, findinfo.name));
508 #ifdef DEDICATED
509  Sys_Error("Invalid character in file name '%s'. The file has been removed. Start the server again.", findinfo.name);
510 #else
511  Cvar_Set("com_missingFiles", "");
512  Com_Error(ERR_DROP, "Invalid file name detected & removed\nFile \"%s\" did contain an invalid character for ET: L file structure.\nSome admins take advantage of this to ensure their menu loads last.\nThe file has been removed.", findinfo.name);
513 #endif
514  }
515 
516  if (nfiles == MAX_FOUND_FILES - 1)
517  {
518  break;
519  }
520 
521  list[nfiles] = CopyString(findinfo.name);
522  nfiles++;
523  }
524  }
525  while (_findnext(findhandle, &findinfo) != -1);
526 
527  list[nfiles] = 0;
528 
529  _findclose(findhandle);
530 
531  // return a copy of the list
532  *numfiles = nfiles;
533 
534  if (!nfiles)
535  {
536  return NULL;
537  }
538 
539  listCopy = Z_Malloc((nfiles + 1) * sizeof(*listCopy));
540  for (i = 0 ; i < nfiles ; i++)
541  {
542  listCopy[i] = list[i];
543  }
544  listCopy[i] = NULL;
545 
546  do
547  {
548  flag = 0;
549  for (i = 1; i < nfiles; i++)
550  {
551  if (strgtr(listCopy[i - 1], listCopy[i]))
552  {
553  char *temp = listCopy[i];
554  listCopy[i] = listCopy[i - 1];
555  listCopy[i - 1] = temp;
556  flag = 1;
557  }
558  }
559  }
560  while (flag);
561 
562  return listCopy;
563 }
