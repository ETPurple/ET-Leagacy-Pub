54 static void R_ChopPolyBehindPlane(int numInPoints, vec3_t inPoints[MAX_VERTS_ON_POLY],
55  int *numOutPoints, vec3_t outPoints[MAX_VERTS_ON_POLY],
56  vec3_t normal, vec_t dist, vec_t epsilon)
57 {
58  float dists[MAX_VERTS_ON_POLY + 4];
59  int sides[MAX_VERTS_ON_POLY + 4];
60  int counts[3];
61  float dot;
62  int i, j;
63  float *p1, *p2, *clip;
64  float d;
65 
66  Com_Memset(dists, 0, MAX_VERTS_ON_POLY + 4);
67  Com_Memset(sides, 0, MAX_VERTS_ON_POLY + 4);
68 
69  // don't clip if it might overflow
70  if (numInPoints >= MAX_VERTS_ON_POLY - 2)
71  {
72  *numOutPoints = 0;
73  return;
74  }
75 
76  counts[0] = counts[1] = counts[2] = 0;
77 
78  // determine sides for each point
79  for (i = 0; i < numInPoints; i++)
80  {
81  dot = DotProduct(inPoints[i], normal);
82  dot -= dist;
83  dists[i] = dot;
84  if (dot > epsilon)
85  {
86  sides[i] = SIDE_FRONT;
87  }
88  else if (dot < -epsilon)
89  {
90  sides[i] = SIDE_BACK;
91  }
92  else
93  {
94  sides[i] = SIDE_ON;
95  }
96  counts[sides[i]]++;
97  }
98  sides[i] = sides[0];
99  dists[i] = dists[0];
100 
101  *numOutPoints = 0;
102 
103  if (!counts[0])
104  {
105  return;
106  }
107  if (!counts[1])
108  {
109  *numOutPoints = numInPoints;
110  Com_Memcpy(outPoints, inPoints, numInPoints * sizeof(vec3_t));
111  return;
112  }
113 
114  for (i = 0; i < numInPoints; i++)
115  {
116  p1 = inPoints[i];
117  clip = outPoints[*numOutPoints];
118 
119  if (sides[i] == SIDE_ON)
120  {
121  VectorCopy(p1, clip);
122  (*numOutPoints)++;
123  continue;
124  }
125 
126  if (sides[i] == SIDE_FRONT)
127  {
128  VectorCopy(p1, clip);
129  (*numOutPoints)++;
130  clip = outPoints[*numOutPoints];
131  }
132 
133  if (sides[i + 1] == SIDE_ON || sides[i + 1] == sides[i])
134  {
135  continue;
136  }
137 
138  // generate a split point
139  p2 = inPoints[(i + 1) % numInPoints];
140 
141  d = dists[i] - dists[i + 1];
142  if (d == 0.f)
143  {
144  dot = 0;
145  }
146  else
147  {
148  dot = dists[i] / d;
149  }
150 
151  // clip xyz
152 
153  for (j = 0; j < 3; j++)
154  {
155  clip[j] = p1[j] + dot * (p2[j] - p1[j]);
156  }
157 
158  (*numOutPoints)++;
159  }
160 }

172 void R_BoxSurfaces_r(bspNode_t *node, vec3_t mins, vec3_t maxs, surfaceType_t **list, int listsize, int *listlength, vec3_t dir)
173 {
174  int s, c;
175  bspSurface_t *surf, **mark;
176 
177  // do the tail recursion in a loop
178  while (node->contents == -1)
179  {
180  s = BoxOnPlaneSide(mins, maxs, node->plane);
181  if (s == 1)
182  {
183  node = node->children[0];
184  }
185  else if (s == 2)
186  {
187  node = node->children[1];
188  }
189  else
190  {
191  R_BoxSurfaces_r(node->children[0], mins, maxs, list, listsize, listlength, dir);
192  node = node->children[1];
193  }
194  }
195 
196  // add the individual surfaces
197  mark = node->markSurfaces;
198  c = node->numMarkSurfaces;
199  while (c--)
200  {
201  if (*listlength >= listsize)
202  {
203  break;
204  }
205 
206  surf = *mark;
207 
208  // check if the surface has NOIMPACT or NOMARKS set
209  if ((surf->shader->surfaceFlags & (SURF_NOIMPACT | SURF_NOMARKS)) || (surf->shader->contentFlags & CONTENTS_FOG))
210  {
211  surf->viewCount = tr.viewCountNoReset;
212  }
213  // extra check for surfaces to avoid list overflows
214  else if (*(surf->data) == SF_FACE)
215  {
216  // the face plane should go through the box
217  s = BoxOnPlaneSide(mins, maxs, &((srfSurfaceFace_t *) surf->data)->plane);
218  if (s == 1 || s == 2)
219  {
220  surf->viewCount = tr.viewCountNoReset;
221  }
222  else if (DotProduct(((srfSurfaceFace_t *) surf->data)->plane.normal, dir) > -0.5f)
223  {
224  // don't add faces that make sharp angles with the projection direction
225  surf->viewCount = tr.viewCountNoReset;
226  }
227  }
228  else if (*(surfaceType_t *) (surf->data) != SF_GRID
229  && *(surfaceType_t *) (surf->data) != SF_TRIANGLES)
230  {
231  surf->viewCount = tr.viewCountNoReset;
232  }
233  // check the viewCount because the surface may have
234  // already been added if it spans multiple leafs
235  if (surf->viewCount != tr.viewCountNoReset)
236  {
237  surf->viewCount = tr.viewCountNoReset;
238  list[*listlength] = (surfaceType_t *) surf->data;
239  (*listlength)++;
240  }
241  mark++;
242  }

333 int R_MarkFragments(int numPoints, const vec3_t *points, const vec3_t projection,
334  int maxPoints, vec3_t pointBuffer, int maxFragments, markFragment_t *fragmentBuffer)
335 {
336  int numsurfaces, numPlanes;
337  int i, j, k, m, n;
338  surfaceType_t *surfaces[64];
339  vec3_t mins, maxs;
340  int returnedFragments;
341  int returnedPoints;
342  vec3_t normals[MAX_VERTS_ON_POLY + 2];
343  float dists[MAX_VERTS_ON_POLY + 2];
344  vec3_t clipPoints[2][MAX_VERTS_ON_POLY];
345  int numClipPoints;
346  float *v;
347  srfSurfaceFace_t *face;
348  srfGridMesh_t *cv;
349  srfTriangles_t *trisurf;
350  srfVert_t *dv;
351  srfTriangle_t *tri;
352  vec3_t normal;
353  vec3_t projectionDir;
354  vec3_t v1, v2;
355 
356  //increment view count for double check prevention
357  tr.viewCountNoReset++;
358 
359  VectorNormalize2(projection, projectionDir);
360  // find all the brushes that are to be considered
361  ClearBounds(mins, maxs);
362  for (i = 0; i < numPoints; i++)
363  {
364  vec3_t temp;
365 
366  AddPointToBounds(points[i], mins, maxs);
367  VectorAdd(points[i], projection, temp);
368  AddPointToBounds(temp, mins, maxs);
369  // make sure we get all the leafs (also the one(s) in front of the hit surface)
370  VectorMA(points[i], -20, projectionDir, temp);
371  AddPointToBounds(temp, mins, maxs);
372  }
373 
374  if (numPoints > MAX_VERTS_ON_POLY)
375  {
376  numPoints = MAX_VERTS_ON_POLY;
377  }
378  // create the bounding planes for the to be projected polygon
379  for (i = 0; i < numPoints; i++)
380  {
381  VectorSubtract(points[(i + 1) % numPoints], points[i], v1);
382  VectorAdd(points[i], projection, v2);
383  VectorSubtract(points[i], v2, v2);
384  CrossProduct(v1, v2, normals[i]);
385  VectorNormalizeFast(normals[i]);
386  dists[i] = DotProduct(normals[i], points[i]);
387  }
388  // add near and far clipping planes for projection
389  VectorCopy(projectionDir, normals[numPoints]);
390  dists[numPoints] = DotProduct(normals[numPoints], points[0]) - 32;
391  VectorCopy(projectionDir, normals[numPoints + 1]);
392  VectorInverse(normals[numPoints + 1]);
393  dists[numPoints + 1] = DotProduct(normals[numPoints + 1], points[0]) - 20;
394  numPlanes = numPoints + 2;
395 
396  numsurfaces = 0;
397  R_BoxSurfaces_r(tr.world->nodes, mins, maxs, surfaces, 64, &numsurfaces, projectionDir);
398  //etl_assert(numsurfaces <= 64);
399  //etl_assert(numsurfaces != 64);
400 
401  returnedPoints = 0;
402  returnedFragments = 0;
403 
404  for (i = 0; i < numsurfaces; i++)
405  {
406  if (*surfaces[i] == SF_GRID)
407  {
408  cv = (srfGridMesh_t *) surfaces[i];
409  for (m = 0; m < cv->height - 1; m++)
410  {
411  for (n = 0; n < cv->width - 1; n++)
412  {
413  // We triangulate the grid and chop all triangles within
414  // the bounding planes of the to be projected polygon.
415  // LOD is not taken into account, not such a big deal though.
416  //
417  // It's probably much nicer to chop the grid itself and deal
418  // with this grid as a normal SF_GRID surface so LOD will
419  // be applied. However the LOD of that chopped grid must
420  // be synced with the LOD of the original curve.
421  // One way to do this; the chopped grid shares vertices with
422  // the original curve. When LOD is applied to the original
423  // curve the unused vertices are flagged. Now the chopped curve
424  // should skip the flagged vertices. This still leaves the
425  // problems with the vertices at the chopped grid edges.
426  //
427  // To avoid issues when LOD applied to "hollow curves" (like
428  // the ones around many jump pads) we now just add a 2 unit
429  // offset to the triangle vertices.
430  // The offset is added in the vertex normal vector direction
431  // so all triangles will still fit together.
432  // The 2 unit offset should avoid pretty much all LOD problems.
433 
434  numClipPoints = 3;
435 
436  dv = cv->verts + m * cv->width + n;
437 
438  VectorCopy(dv[0].xyz, clipPoints[0][0]);
439  VectorMA(clipPoints[0][0], MARKER_OFFSET, dv[0].normal, clipPoints[0][0]);
440  VectorCopy(dv[cv->width].xyz, clipPoints[0][1]);
441  VectorMA(clipPoints[0][1], MARKER_OFFSET, dv[cv->width].normal, clipPoints[0][1]);
442  VectorCopy(dv[1].xyz, clipPoints[0][2]);
443  VectorMA(clipPoints[0][2], MARKER_OFFSET, dv[1].normal, clipPoints[0][2]);
444  // check the normal of this triangle
445  VectorSubtract(clipPoints[0][0], clipPoints[0][1], v1);
446  VectorSubtract(clipPoints[0][2], clipPoints[0][1], v2);
447  CrossProduct(v1, v2, normal);
448  VectorNormalizeFast(normal);
449  if (DotProduct(normal, projectionDir) < -0.1f)
450  {
451  // add the fragments of this triangle
452  R_AddMarkFragments(numClipPoints, clipPoints,
453  numPlanes, normals, dists,
454  maxPoints, pointBuffer,
455  maxFragments, fragmentBuffer,
456  &returnedPoints, &returnedFragments, mins, maxs);
457 
458  if (returnedFragments == maxFragments)
459  {
460  return returnedFragments; // not enough space for more fragments
461  }
462  }
463 
464  VectorCopy(dv[1].xyz, clipPoints[0][0]);
465  VectorMA(clipPoints[0][0], MARKER_OFFSET, dv[1].normal, clipPoints[0][0]);
466  VectorCopy(dv[cv->width].xyz, clipPoints[0][1]);
467  VectorMA(clipPoints[0][1], MARKER_OFFSET, dv[cv->width].normal, clipPoints[0][1]);
468  VectorCopy(dv[cv->width + 1].xyz, clipPoints[0][2]);
469  VectorMA(clipPoints[0][2], MARKER_OFFSET, dv[cv->width + 1].normal, clipPoints[0][2]);
470  // check the normal of this triangle
471  VectorSubtract(clipPoints[0][0], clipPoints[0][1], v1);
472  VectorSubtract(clipPoints[0][2], clipPoints[0][1], v2);
473  CrossProduct(v1, v2, normal);
474  VectorNormalizeFast(normal);
475  if (DotProduct(normal, projectionDir) < -0.05f)
476  {
477  // add the fragments of this triangle
478  R_AddMarkFragments(numClipPoints, clipPoints,
479  numPlanes, normals, dists,
480  maxPoints, pointBuffer,
481  maxFragments, fragmentBuffer,
482  &returnedPoints, &returnedFragments, mins, maxs);
483 
484  if (returnedFragments == maxFragments)
485  {
486  return returnedFragments; // not enough space for more fragments
487  }
488  }
489  }
490  }
491  }
492  else if (*surfaces[i] == SF_FACE)
493  {
494  face = (srfSurfaceFace_t *) surfaces[i];
495 
496  // check the normal of this face
497  if (DotProduct(face->plane.normal, projectionDir) > -0.5f)
498  {
499  continue;
500  }
501 
502  for (k = 0, tri = face->triangles; k < face->numTriangles; k++, tri++)
503  {
504  for (j = 0; j < 3; j++)
505  {
506  v = face->verts[tri->indexes[j]].xyz;
507  VectorMA(v, MARKER_OFFSET, face->plane.normal, clipPoints[0][j]);
508  }
509 
510  /*
511  VectorSubtract(clipPoints[0][0], clipPoints[0][1], v1);
512  VectorSubtract(clipPoints[0][2], clipPoints[0][1], v2);
513  CrossProduct(v1, v2, normal);
514  VectorNormalize(normal);
515  if (DotProduct(normal, projectionDir) > -0.5) continue;
516  */
517 
518  // add the fragments of this face
519  R_AddMarkFragments(3, clipPoints,
520  numPlanes, normals, dists,
521  maxPoints, pointBuffer,
522  maxFragments, fragmentBuffer, &returnedPoints, &returnedFragments, mins, maxs);
523  if (returnedFragments == maxFragments)
524  {
525  return returnedFragments; // not enough space for more fragments
526  }
527  }
528  }
529  else if (*surfaces[i] == SF_TRIANGLES && !r_noMarksOnTrisurfs->integer)
530  {
531  trisurf = (srfTriangles_t *) surfaces[i];
532 
533  for (k = 0, tri = trisurf->triangles; k < trisurf->numTriangles; k++, tri++)
534  {
535  for (j = 0; j < 3; j++)
536  {
537  VectorCopy(trisurf->verts[tri->indexes[j]].xyz, clipPoints[0][j]);
538  }
539 
540  /*
541  VectorSubtract(clipPoints[0][0], clipPoints[0][1], v1);
542  VectorSubtract(clipPoints[0][2], clipPoints[0][1], v2);
543  CrossProduct(v1, v2, normal);
544  VectorNormalize(normal);
545  if (DotProduct(normal, projectionDir) > -0.5) continue;
546  */
547 
548  // add the fragments of this face
549  R_AddMarkFragments(3, clipPoints,
550  numPlanes, normals, dists,
551  maxPoints, pointBuffer,
552  maxFragments, fragmentBuffer, &returnedPoints, &returnedFragments, mins, maxs);
553  if (returnedFragments == maxFragments)
554  {
555  return returnedFragments; // not enough space for more fragments
556  }
557  }
558  }
559  }
560  return returnedFragments;
561 }
