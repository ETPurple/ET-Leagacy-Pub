49 void Huff_putBit(int bit, byte *fout, int *offset)
50 {
51  int x, y;
52 
53  bloc = *offset;
54  x = bloc >> 3;
55  y = bloc & 7;
56  if (!y)
57  {
58  fout[x] = 0;
59  }
60  fout[x] |= bit << y;
61  bloc++;
62  *offset = bloc;

240 static void increment(huff_t *huff, node_t *node)
241 {
242  if (!node)
243  {
244  return;
245  }
246 
247  if (node->next != NULL && node->next->weight == node->weight)
248  {
249  node_t *lnode = *node->head;
250 
251  if (lnode != node->parent)
252  {
253  swap(huff, lnode, node);
254  }
255  swaplist(lnode, node);
256  }
257  if (node->prev && node->prev->weight == node->weight)
258  {
259  *node->head = node->prev;
260  }
261  else
262  {
263  *node->head = NULL;
264  free_ppnode(huff, node->head);
265  }
266  node->weight++;
267  if (node->next && node->next->weight == node->weight)
268  {
269  node->head = node->next->head;
270  }
271  else
272  {
273  node->head = get_ppnode(huff);
274  *node->head = node;
275  }
276  if (node->parent)
277  {
278  increment(huff, node->parent);
279  if (node->prev == node->parent)
280  {
281  swaplist(node, node->parent);
282  if (*node->head == node)
283  {
284  *node->head = node->parent;
285  }
286  }

528 void Huff_Decompress(msg_t *mbuf, int offset)
529 {
530  int ch, cch, i, j, size;
531  byte seq[65536];
532  byte *buffer;
533  huff_t huff;
534 
535  size = mbuf->cursize - offset;
536  buffer = mbuf->data + offset;
537 
538  if (size <= 0)
539  {
540  return;
541  }
542 
543  Com_Memset(&huff, 0, sizeof(huff_t));
544  // Initialize the tree & list with the NYT node
545  huff.tree = huff.lhead = huff.ltail = huff.loc[NYT] = &(huff.nodeList[huff.blocNode++]);
546  huff.tree->symbol = NYT;
547  huff.tree->weight = 0;
548  huff.lhead->next = huff.lhead->prev = NULL;
549  huff.tree->parent = huff.tree->left = huff.tree->right = NULL;
550 
551  cch = buffer[0] * 256 + buffer[1];
552  // don't overflow with bad messages
553  if (cch > mbuf->maxsize - offset)
554  {
555  cch = mbuf->maxsize - offset;
556  }
557  bloc = 16;
558 
559  for (j = 0; j < cch; j++)
560  {
561  ch = 0;
562  // don't overflow reading from the messages
563  // FIXME: would it be better to have a overflow check in get_bit ?
564  if ((bloc >> 3) > size)
565  {
566  seq[j] = 0;
567  break;
568  }
569  Huff_Receive(huff.tree, &ch, buffer); // Get a character
570  if (ch == NYT) // We got a NYT, get the symbol associated with it
571  {
572  ch = 0;
573  for (i = 0; i < 8; i++)
574  {
575  ch = (ch << 1) + get_bit(buffer);
576  }
577  }
578 
579  seq[j] = ch; // Write symbol
580 
581  Huff_addRef(&huff, (byte)ch); // Increment node
582  }
583  mbuf->cursize = cch + offset;
584  Com_Memcpy(mbuf->data + offset, seq, cch);
585 }

594 void Huff_Compress(msg_t *mbuf, int offset)
595 {
596  int i, ch, size;
597  byte seq[65536];
598  byte *buffer;
599  huff_t huff;
600 
601  size = mbuf->cursize - offset;
602  buffer = mbuf->data + offset;
603 
604  if (size <= 0)
605  {
606  return;
607  }
608 
609  Com_Memset(&huff, 0, sizeof(huff_t));
610  // Add the NYT (not yet transmitted) node into the tree/list
611  huff.tree = huff.lhead = huff.loc[NYT] = &(huff.nodeList[huff.blocNode++]);
612  huff.tree->symbol = NYT;
613  huff.tree->weight = 0;
614  huff.lhead->next = huff.lhead->prev = NULL;
615  huff.tree->parent = huff.tree->left = huff.tree->right = NULL;
616  huff.loc[NYT] = huff.tree;
617 
618  seq[0] = (size >> 8);
619  seq[1] = size & 0xff;
620 
621  bloc = 16;
622 
623  for (i = 0; i < size; i++)
624  {
625  ch = buffer[i];
626  Huff_transmit(&huff, ch, seq, size<<3); // Transmit symbol
627  Huff_addRef(&huff, (byte)ch); // Do update
628  }
629 
630  bloc += 8; // next byte
631 
632  mbuf->cursize = (bloc >> 3) + offset;
633  Com_Memcpy(mbuf->data + offset, seq, (bloc >> 3));
634 }
