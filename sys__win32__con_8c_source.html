335 static void Win_FindIndexMatch(const char *s)
336 {
337  if (Q_stricmpn(s, win_completionString, strlen(win_completionString)))
338  {
339  return;
340  }

342  {
343  Q_strncpyz(win_currentMatch, s, sizeof(win_currentMatch));

346  win_findMatchIndex++;

353 static void Win_FindMatches(const char *s)
354 {
355  int i;
356 
357  if (Q_stricmpn(s, win_completionString, strlen(win_completionString)))
358  {
359  return;
360  }
361  win_matchCount++;
362  if (win_matchCount == 1)
363  {
364  Q_strncpyz(win_currentMatch, s, sizeof(win_currentMatch));
365  return;
366  }
367 
368  // cut currentMatch to the amount common with s
369  for (i = 0; s[i]; i++)
370  {
371  if (tolower(win_currentMatch[i]) != tolower(s[i]))
372  {
373  win_currentMatch[i] = 0;
374  }
375  }
376 
377  win_currentMatch[i] = 0;
378 }

383 static void Win_KeyConcatArgs(void)
384 {
385  int i;

388  for (i = 1; i < Cmd_Argc(); i++)
389  {
390  Q_strcat(win_consoleField.buffer, sizeof(win_consoleField.buffer), " ");

392  while (*arg)
393  {
394  if (*arg == ' ')

396  Q_strcat(win_consoleField.buffer, sizeof(win_consoleField.buffer), "\"");
397  break;
398  }
399 
400  arg++;
401  }
402  Q_strcat(win_consoleField.buffer, sizeof(win_consoleField.buffer), Cmd_Argv(i));
403  if (*arg == ' ')
404  {
405  Q_strcat(win_consoleField.buffer, sizeof(win_consoleField.buffer), "\"");
406  }
407  }

458 static void Win_CompleteCommand(qboolean showMatches)
459 {
460  field_t *edit = &win_consoleField;
461  field_t temp;
462 
463  if (win_acLength == 0)
464  {
465  // only look at the first token for completion purposes
466  Cmd_TokenizeString(edit->buffer);
467 
468  Q_strncpyz(win_completionString, Cmd_Argv(0), sizeof(win_completionString));
469  if (win_completionString[0] == '\\' || win_completionString[0] == '/')
470  {
471  Q_strncpyz(win_completionString, win_completionString + 1, sizeof(win_completionString));
472  }
473 
474  win_matchCount = 0;
475  win_matchIndex = 0;
476  win_currentMatch[0] = 0;
477 
478  if (strlen(win_completionString) == 0)
479  {
480  return;
481  }
482 
483  Cmd_CommandCompletion(Win_FindMatches);
484  Cvar_CommandCompletion(Win_FindMatches);
485 
486  if (win_matchCount == 0)
487  {
488  return; // no matches
489  }
490 
491  Com_Memcpy(&temp, edit, sizeof(field_t));
492 
493  if (win_matchCount == 1)
494  {
495  Com_sprintf(edit->buffer, sizeof(edit->buffer), "%s", win_currentMatch);
496  if (Cmd_Argc() == 1)
497  {
498  Q_strcat(win_consoleField.buffer, sizeof(win_consoleField.buffer), " ");
499  }
500  else
501  {
502  Win_ConcatRemaining(temp.buffer, win_completionString);
503  }
504 
505  edit->cursor = strlen(edit->buffer);
506  }
507  else
508  {
509  // multiple matches, complete to shortest
510  Com_sprintf(edit->buffer, sizeof(edit->buffer), "%s", win_currentMatch);
511  win_acLength = edit->cursor = strlen(edit->buffer);
512  Win_ConcatRemaining(temp.buffer, win_completionString);
513  showMatches = qtrue;
514  }
515  }
516  else if (win_matchCount != 1)
517  {
518  // get the next match and show instead
519  char lastMatch[MAX_TOKEN_CHARS];
520 
521  Q_strncpyz(lastMatch, win_currentMatch, sizeof(lastMatch));
522 
523  win_matchIndex++;
524  if (win_matchIndex == win_matchCount)
525  {
526  win_matchIndex = 0;
527  }
528 
529  win_findMatchIndex = 0;
530  Cmd_CommandCompletion(Win_FindIndexMatch);
531  Cvar_CommandCompletion(Win_FindIndexMatch);
532 
533  Com_Memcpy(&temp, edit, sizeof(field_t));
534 
535  // and print it
536  Com_sprintf(edit->buffer, sizeof(edit->buffer), "%s", win_currentMatch);
537  edit->cursor = strlen(edit->buffer);
538  Win_ConcatRemaining(temp.buffer, lastMatch);
539  }
540 
541  // hijack it
542  if (win_matchCount == 1)
543  {
544  win_acLength = strlen(win_currentMatch);
545  }
546 
547  // run through again, printing matches
548  if (showMatches && win_matchCount > 0)
549  {
550  Com_Memcpy(&temp, edit, sizeof(*edit));
551  temp.buffer[win_acLength] = '\0';
552  Sys_Print(va("] %s\n", temp.buffer));
553  Cmd_CommandCompletion(Win_PrintMatches);
554  Cvar_CommandCompletion(Win_PrintCvarMatches);
555  }
556 }

920 void Conbuf_AppendText(const char *pMsg)
921 {
922  char buffer[CONSOLE_BUFFER_SIZE * 2], *b = buffer;
923  const char *msg;
924  int bufLen, i = 0;
925  static unsigned long s_totalChars;
926 
927  // if the message is REALLY long, use just the last portion of it
928  if (strlen(pMsg) > CONSOLE_BUFFER_SIZE - 1)
929  {
930  msg = pMsg + strlen(pMsg) - CONSOLE_BUFFER_SIZE + 1;
931  }
932  else
933  {
934  msg = pMsg;
935  }
936 
937  // copy into an intermediate buffer
938  while (msg[i] && ((b - buffer) < sizeof(buffer) - 1))
939  {
940  if (msg[i] == '\n' && msg[i + 1] == '\r')
941  {
942  b[0] = '\r';
943  b[1] = '\n';
944  b += 2;
945  i++;
946  }
947  else if (msg[i] == '\r')
948  {
949  b[0] = '\r';
950  b[1] = '\n';
951  b += 2;
952  }
953  else if (msg[i] == '\n')
954  {
955  b[0] = '\r';
956  b[1] = '\n';
957  b += 2;
958  }
959  else if (Q_IsColorString(&msg[i]))
960  {
961  i++;
962  }
963  else
964  {
965  *b = msg[i];
966  b++;
967  }
968 
969  i++;
970  }
971  *b = 0;
972  bufLen = b - buffer;
973 
974  s_totalChars += bufLen;
975 
976  // replace selection instead of appending if we're overflowing
977  if (s_totalChars > CONSOLE_BUFFER_SIZE)
978  {
979  SendMessage(s_wcd.hwndBuffer, EM_SETSEL, 0, -1);
980  s_totalChars = bufLen;
981  }
982  else
983  {
984  // always append at the bottom of the textbox
985  SendMessage(s_wcd.hwndBuffer, EM_SETSEL, 0xFFFF, 0xFFFF);
986  }
987 
988  // put this text into the windows console
989  SendMessage(s_wcd.hwndBuffer, EM_LINESCROLL, 0, 0xffff);
990  SendMessage(s_wcd.hwndBuffer, EM_SCROLLCARET, 0, 0);
991  SendMessage(s_wcd.hwndBuffer, EM_REPLACESEL, 0, (LPARAM) buffer);
992 }
