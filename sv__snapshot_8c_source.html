65 static void SV_EmitPacketEntities(clientSnapshot_t *from, clientSnapshot_t *to, msg_t *msg)
66 {
67  entityState_t *oldent = NULL, *newent = NULL;
68  int oldindex = 0, newindex = 0;
69  int oldnum, newnum;
70  int from_num_entities;
71 
72  // generate the delta update
73  if (!from)
74  {
75  from_num_entities = 0;
76  }
77  else
78  {
79  from_num_entities = from->num_entities;
80  }
81 
82  while (newindex < to->num_entities || oldindex < from_num_entities)
83  {
84  if (newindex >= to->num_entities)
85  {
86  newnum = 9999;
87  }
88  else
89  {
90  newent = &svs.snapshotEntities[(to->first_entity + newindex) % svs.numSnapshotEntities];
91  newnum = newent->number;
92  }
93 
94  if (oldindex >= from_num_entities)
95  {
96  oldnum = 9999;
97  }
98  else
99  {
100  oldent = &svs.snapshotEntities[(from->first_entity + oldindex) % svs.numSnapshotEntities];
101  oldnum = oldent->number;
102  }
103 
104  if (newnum == oldnum)
105  {
106  // delta update from old position
107  // because the force parm is qfalse, this will not result
108  // in any bytes being emited if the entity has not changed at all
109  MSG_WriteDeltaEntity(msg, oldent, newent, qfalse);
110  oldindex++;
111  newindex++;
112  continue;
113  }
114 
115  if (newnum < oldnum)
116  {
117  if (newnum >= MAX_GENTITIES)
118  {
119  Com_Error(ERR_FATAL, "SV_EmitPacketEntities: MAX_GENTITIES exceeded");
120  }
121 
122  // this is a new entity, send it from the baseline
123  MSG_WriteDeltaEntity(msg, &sv.svEntities[newnum].baseline, newent, qtrue);
124  newindex++;
125  continue;
126  }
127 
128  if (newnum > oldnum)
129  {
130  // the old entity isn't present in the new message
131  MSG_WriteDeltaEntity(msg, oldent, NULL, qtrue);
132  oldindex++;
133  continue;
134  }
135  }
136 
137  MSG_WriteBits(msg, (MAX_GENTITIES - 1), GENTITYNUM_BITS); // end of packetentities
138 }

145 static void SV_WriteSnapshotToClient(client_t *client, msg_t *msg)
146 {
147  clientSnapshot_t *frame, *oldframe;
148  int lastframe;
149  int snapFlags;
150 
151  // this is the snapshot we are creating
152  frame = &client->frames[client->netchan.outgoingSequence & PACKET_MASK];
153 
154  // try to use a previous frame as the source for delta compressing the snapshot
155  if (client->deltaMessage <= 0 || client->state != CS_ACTIVE)
156  {
157  // client is asking for a retransmit
158  oldframe = NULL;
159  lastframe = 0;
160  }
161  else if (client->netchan.outgoingSequence - client->deltaMessage >= (PACKET_BACKUP - 3))
162  {
163  // client hasn't gotten a good message through in a long time
164  Com_DPrintf("%s: Delta request from out of date packet.\n", client->name);
165  oldframe = NULL;
166  lastframe = 0;
167  }
168  else
169  {
170  // we have a valid snapshot to delta from
171  oldframe = &client->frames[client->deltaMessage & PACKET_MASK];
172  lastframe = client->netchan.outgoingSequence - client->deltaMessage;
173 
174  // the snapshot's entities may still have rolled off the buffer, though
175  if (oldframe->first_entity <= svs.nextSnapshotEntities - svs.numSnapshotEntities)
176  {
177  Com_DPrintf("%s: Delta request from out of date entities.\n", client->name);
178  oldframe = NULL;
179  lastframe = 0;
180  }
181  }
182 
183  MSG_WriteByte(msg, svc_snapshot);
184 
185  // NOTE, MRE: now sent at the start of every message from server to client
186  // let the client know which reliable clientCommands we have received
187  //MSG_WriteLong( msg, client->lastClientCommand );
188 
189  // send over the current server time so the client can drift
190  // its view of time to try to match
191  MSG_WriteLong(msg, svs.time);
192 
193  // what we are delta'ing from
194  MSG_WriteByte(msg, lastframe);
195 
196  snapFlags = svs.snapFlagServerBit;
197  if (client->rateDelayed)
198  {
199  snapFlags |= SNAPFLAG_RATE_DELAYED;
200  }
201  if (client->state != CS_ACTIVE)
202  {
203  snapFlags |= SNAPFLAG_NOT_ACTIVE;
204  }
205 
206  MSG_WriteByte(msg, snapFlags);
207 
208  // send over the areabits
209  MSG_WriteByte(msg, frame->areabytes);
210  MSG_WriteData(msg, frame->areabits, frame->areabytes);
211 
212  //{
213  //int sz = msg->cursize;
214  //int usz = msg->uncompsize;
215 
216  // delta encode the playerstate
217  if (oldframe)
218  {
219  MSG_WriteDeltaPlayerstate(msg, &oldframe->ps, &frame->ps);
220  }
221  else
222  {
223  MSG_WriteDeltaPlayerstate(msg, NULL, &frame->ps);
224  }
225 
226  //Com_Printf( "Playerstate delta size: %f\n", ((msg->cursize - sz) * sv_fps->integer) / 8.f );
227  //}
228 
229  // delta encode the entities
230  SV_EmitPacketEntities(oldframe, frame, msg);
231 
232  // padding for rate debugging
233  if (sv_padPackets->integer)
234  {
235  int i;
236 
237  for (i = 0 ; i < sv_padPackets->integer ; i++)
238  {
239  MSG_WriteByte(msg, svc_nop);
240  }
241  }
242 }

643 static void SV_BuildClientSnapshot(client_t *client)
644 {
645  vec3_t org;
646  clientSnapshot_t *frame;
647  snapshotEntityNumbers_t entityNumbers;
648  int i;
649  sharedEntity_t *ent;
650  entityState_t *state;
651  svEntity_t *svEnt;
652  sharedEntity_t *clent;
653  int clientNum;
654  playerState_t *ps;
655 
656  // bump the counter used to prevent double adding
657  sv.snapshotCounter++;
658 
659  // this is the frame we are creating
660  frame = &client->frames[client->netchan.outgoingSequence & PACKET_MASK];
661 
662  // clear everything in this snapshot
663  entityNumbers.numSnapshotEntities = 0;
664  Com_Memset(frame->areabits, 0, sizeof(frame->areabits));
665 
666  frame->num_entities = 0;
667 
668  clent = client->gentity;
669  if (!clent || client->state == CS_ZOMBIE)
670  {
671  return;
672  }
673 
674  // grab the current playerState_t
675  ps = SV_GameClientNum(client - svs.clients);
676  frame->ps = *ps;
677 
678  // never send client's own entity, because it can
679  // be regenerated from the playerstate
680  clientNum = frame->ps.clientNum;
681  if (clientNum < 0 || clientNum >= MAX_GENTITIES)
682  {
683  Com_Error(ERR_DROP, "SV_BuildClientSnapshot: bad gEnt");
684  }
685  svEnt = &sv.svEntities[clientNum];
686 
687  svEnt->snapshotCounter = sv.snapshotCounter;
688 
689  if (clent->r.svFlags & SVF_SELF_PORTAL_EXCLUSIVE)
690  {
691  // find the client's viewpoint
692  VectorCopy(clent->s.origin2, org);
693  }
694  else
695  {
696  VectorCopy(ps->origin, org);
697  }
698  org[2] += ps->viewheight;
699 
700  // added for 'lean'
701  // need to account for lean, so areaportal doors draw properly
702  if (frame->ps.leanf != 0.f)
703  {
704  vec3_t right, v3ViewAngles;
705  VectorCopy(ps->viewangles, v3ViewAngles);
706  v3ViewAngles[2] += frame->ps.leanf / 2.0f;
707  angles_vectors(v3ViewAngles, NULL, right, NULL);
708  VectorMA(org, frame->ps.leanf, right, org);
709  }
710 
711  // add all the entities directly visible to the eye, which
712  // may include portal entities that merge other viewpoints
713 #ifdef FEATURE_ANTICHEAT
714  SV_AddEntitiesVisibleFromPoint(org, frame, &entityNumbers, qfalse /*client->netchan.remoteAddress.type == NA_LOOPBACK*/);
715 #else
716  SV_AddEntitiesVisibleFromPoint(org, frame, &entityNumbers /*, qfalse, client->netchan.remoteAddress.type == NA_LOOPBACK*/);
717 #endif
718 
719  // if there were portals visible, there may be out of order entities
720  // in the list which will need to be resorted for the delta compression
721  // to work correctly. This also catches the error condition
722  // of an entity being included twice.
723  qsort(entityNumbers.snapshotEntities, entityNumbers.numSnapshotEntities,
724  sizeof(entityNumbers.snapshotEntities[0]), SV_QsortEntityNumbers);
725 
726  // now that all viewpoint's areabits have been OR'd together, invert
727  // all of them to make it a mask vector, which is what the renderer wants
728  for (i = 0 ; i < MAX_MAP_AREA_BYTES / 4 ; i++)
729  {
730  ((int *)frame->areabits)[i] = ((int *)frame->areabits)[i] ^ -1;
731  }
732 
733  // copy the entity states out
734  frame->num_entities = 0;
735  frame->first_entity = svs.nextSnapshotEntities;
736  for (i = 0 ; i < entityNumbers.numSnapshotEntities ; i++)
737  {
738  ent = SV_GentityNum(entityNumbers.snapshotEntities[i]);
739  state = &svs.snapshotEntities[svs.nextSnapshotEntities % svs.numSnapshotEntities];
740  *state = ent->s;
741 
742 #ifdef FEATURE_ANTICHEAT
743  if (sv_wh_active->integer && entityNumbers.snapshotEntities[i] < sv_maxclients->integer)
744  {
745  if (SV_PositionChanged(entityNumbers.snapshotEntities[i]))
746  {
747  SV_RestorePos(entityNumbers.snapshotEntities[i]);
748  }
749  }
750 #endif
751 
752  svs.nextSnapshotEntities++;
753  // this should never hit, map should always be restarted first in SV_Frame
754  if (svs.nextSnapshotEntities >= 0x7FFFFFFE)
755  {
756  Com_Error(ERR_FATAL, "SV_BuildClientSnapshot: svs.nextSnapshotEntities wrapped");
757  }
758 
759  frame->num_entities++;
760  }
761 }
