213 qboolean CG_SpawnpointsCheckExecKey(int key, qboolean doaction)
214 {
215  if (key == K_ESCAPE)
216  {
217  return qtrue;
218  }
219 
220  if ((key & K_CHAR_FLAG))
221  {
222  return qfalse;
223  }
224 
225  key &= ~K_CHAR_FLAG;
226 
227  if (key >= '0' && key <= '9')
228  {
229  int i = (key - '0') % 10;
230 
231  if (i > cg.spawnCount)
232  {
233  return qfalse;
234  }
235 
236  if (doaction)
237  {
238  trap_SendClientCommand(va("setspawnpt %i", i));
239  CG_EventHandling(CGAME_EVENT_NONE, qfalse);
240 
241  if (i == 0)
242  {
243  CG_PriorityCenterPrint(CG_TranslateString("Your spawn point will be auto-picked."), 400, cg_fontScaleCP.value, -1);
244  }
245  else
246  {
247  CG_PriorityCenterPrint(va(CG_TranslateString("You will spawn at %s."), cg.spawnPoints[i]), 400, cg_fontScaleCP.value, -1);
248  }
249  }
250 
251  return qtrue;
252  }
253 
254  return qfalse;
