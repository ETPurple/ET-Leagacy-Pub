395 void RE_SaveTGA(const char *filename, byte *data, int width, int height, qboolean withAlpha)
396 {
397  byte *buffer;
398  int i, c;
399  int row;
400  unsigned char *flip;
401  unsigned char *src, *dst;
402 
403  buffer = (byte *)ri.Z_Malloc(width * height * 4 + 18);
404  Com_Memset(buffer, 0, 18);
405  buffer[2] = 2; // uncompressed type
406  buffer[12] = width & 255;
407  buffer[13] = width >> 8;
408  buffer[14] = height & 255;
409  buffer[15] = height >> 8;
410  buffer[16] = 32; // pixel size
411 
412  // swap rgb to bgr
413  c = 18 + width * height * 4;
414  for (i = 18 ; i < c ; i += 4)
415  {
416  buffer[i] = data[i - 18 + 2]; // blue
417  buffer[i + 1] = data[i - 18 + 1]; // green
418  buffer[i + 2] = data[i - 18 + 0]; // red
419 
420  if (withAlpha)
421  {
422  buffer[i + 3] = data[i - 18 + 3]; // alpha
423  }
424  else
425  {
426  buffer[i + 3] = 255; //data[i - 18 + 3]; // alpha
427  }
428  }
429 
430  // flip upside down
431  flip = (unsigned char *)ri.Z_Malloc(width * 4);
432  for (row = 0; row < height / 2; row++)
433  {
434  src = buffer + 18 + row * 4 * width;
435  dst = buffer + 18 + (height - row - 1) * 4 * width;
436 
437  Com_Memcpy(flip, src, width * 4);
438  Com_Memcpy(src, dst, width * 4);
439  Com_Memcpy(dst, flip, width * 4);
440  }

443  ri.FS_WriteFile(filename, buffer, c);
444  ri.Free(buffer);
