48 void CG_DamageFeedback(int yawByte, int pitchByte, int damage)
49 {
50  float kick;
51  int health = cg.snap->ps.stats[STAT_HEALTH];
52  float scale;
53  vec3_t dir;
54  vec3_t angles;
55  int slot;
56  viewDamage_t *vd;
57 
58  // show the attacking player's head and name in corner
59  cg.attackerTime = cg.time;
60 
61  // the lower on health you are, the greater the view kick will be
62  if (health < 40)
63  {
64  scale = 1;
65  }
66  else
67  {
68  scale = 40.0f / health;
69  }
70  kick = damage * scale;
71 
72  if (kick < 5)
73  {
74  kick = 5;
75  }
76  if (kick > 10)
77  {
78  kick = 10;
79  }
80 
81  // find a free slot
82  for (slot = 0; slot < MAX_VIEWDAMAGE; slot++)
83  {
84  if (cg.viewDamage[slot].damageTime + cg.viewDamage[slot].damageDuration < cg.time)
85  {
86  break;
87  }
88  }
89 
90  if (slot == MAX_VIEWDAMAGE)
91  {
92  return; // no free slots, never override or splats will suddenly disappear
93 
94  }
95  vd = &cg.viewDamage[slot];
96 
97  // if yaw and pitch are both 255, make the damage always centered (falling, etc)
98  if (yawByte == 255 && pitchByte == 255)
99  {
100  vd->damageX = 0;
101  vd->damageY = 0;
102  cg.v_dmg_roll = 0;
103  cg.v_dmg_pitch = -kick;
104  }
105  else
106  {
107  float left, front, up, dist;
108  // positional
109  float pitch = pitchByte / 255.0f * 360;
110  float yaw = yawByte / 255.0f * 360;
111 
112  angles[PITCH] = pitch;
113  angles[YAW] = yaw;
114  angles[ROLL] = 0;
115 
116  AngleVectors(angles, dir, NULL, NULL);
117  VectorSubtract(vec3_origin, dir, dir);
118 
119  front = DotProduct(dir, cg.refdef.viewaxis[0]);
120  left = DotProduct(dir, cg.refdef.viewaxis[1]);
121  up = DotProduct(dir, cg.refdef.viewaxis[2]);
122 
123  dir[0] = front;
124  dir[1] = left;
125  dir[2] = 0;
126  dist = VectorLength(dir);
127  if (dist < 0.1f)
128  {
129  dist = 0.1f;
130  }
131 
132  cg.v_dmg_roll = kick * left;
133 
134  cg.v_dmg_pitch = -kick * front;
135 
136  if (front <= 0.1f)
137  {
138  front = 0.1f;
139  }
140  vd->damageX = crandom() * 0.3f + -left / front;
141  vd->damageY = crandom() * 0.3f + up / dist;
142  }
143 
144  // clamp the position
145  if (vd->damageX > 1.0f)
146  {
147  vd->damageX = 1.0f;
148  }
149  if (vd->damageX < -1.0f)
150  {
151  vd->damageX = -1.0f;
152  }
153 
154  if (vd->damageY > 1.0f)
155  {
156  vd->damageY = 1.0f;
157  }
158  if (vd->damageY < -1.0f)
159  {
160  vd->damageY = -1.0f;
161  }
162 
163  // don't let the screen flashes vary as much
164  if (kick > 10)
165  {
166  kick = 10;
167  }
168  vd->damageValue = kick;
169  cg.v_dmg_time = cg.time + DAMAGE_TIME;
170  vd->damageTime = cg.snap->serverTime;
171  vd->damageDuration = (int)(kick * 50 * (1 + 2 * (!vd->damageX && !vd->damageY)));
172  cg.damageTime = cg.snap->serverTime;
173  cg.damageIndex = slot;
174 }
