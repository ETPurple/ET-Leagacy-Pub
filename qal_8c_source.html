136 static void *GPA(char *str)
137 {
138  void *rv;
139 
140  rv = Sys_LoadFunction(OpenALLib, str);
141  if (!rv)
142  {
143  Com_Printf(" Can't load symbol %s\n", str);
144  alinit_fail = qtrue;
145  return NULL;
146  }
147  else
148  {
149  Com_DPrintf(" Loaded symbol %s (%p)\n", str, rv);
150  return rv;
151  }
152 }
