52 static void G_CalcMoveSpeeds(bg_character_t *character)
53 {
54  char *tags[2] = { "tag_footleft", "tag_footright" };
55  vec3_t oldPos[2] = { { 0, 0, 0 }, { 0, 0, 0 } };
56  grefEntity_t refent;
57  animation_t *anim;
58  int i, j, k;
59  float totalSpeed;
60  int numSpeed;
61  int low;
62  orientation_t o[2];
63 
64  Com_Memset(&refent, 0, sizeof(refent));
65 
66  refent.hModel = character->mesh;
67 
68  for (i = 0; i < character->animModelInfo->numAnimations; i++)
69  {
70  anim = character->animModelInfo->animations[i];
71 
72  if (anim->moveSpeed >= 0)
73  {
74  continue;
75  }
76 
77  totalSpeed = 0;
78  numSpeed = 0;
79 
80  // for each frame
81  for (j = 0; j < anim->numFrames; j++)
82  {
83  refent.frame = anim->firstFrame + j;
84  refent.oldframe = refent.frame;
85  refent.torsoFrameModel = refent.oldTorsoFrameModel = refent.frameModel = refent.oldframeModel = anim->mdxFile;
86 
87  // for each foot
88  for (k = 0; k < 2; k++)
89  {
90  if (trap_R_LerpTag(&o[k], &refent, tags[k], 0) < 0)
91  {
92  G_Error("G_CalcMoveSpeeds: unable to find tag %s, cannot calculate movespeed", tags[k]);
93  }
94  }
95 
96  // find the contact foot
97  if (anim->flags & ANIMFL_LADDERANIM)
98  {
99  if (o[0].origin[0] > o[1].origin[0])
100  {
101  low = 0;
102  }
103  else
104  {
105  low = 1;
106  }
107  totalSpeed += Q_fabs(oldPos[low][2] - o[low].origin[2]);
108  }
109  else
110  {
111  if (o[0].origin[2] < o[1].origin[2])
112  {
113  low = 0;
114  }
115  else
116  {
117  low = 1;
118  }
119  totalSpeed += Q_fabs(oldPos[low][0] - o[low].origin[0]);
120  }
121 
122  numSpeed++;
123 
124  // save the positions
125  for (k = 0; k < 2; k++)
126  {
127  VectorCopy(o[k].origin, oldPos[k]);
128  }
129  }
130  // record the speed
131  anim->moveSpeed = round((totalSpeed / numSpeed) * 1000.0 / anim->frameLerp);
132  }
133 }

143 static qboolean G_ParseAnimationFiles(bg_character_t *character, const char *animationGroup, const char *animationScript)
144 {
145  fileHandle_t f;

298 void G_UpdateCharacter(gclient_t *client)
299 {
300  char infostring[MAX_INFO_STRING];
301  char *s;
302  int characterIndex;
303  bg_character_t *character;
304 
305  trap_GetUserinfo(client->ps.clientNum, infostring, sizeof(infostring));
306  s = Info_ValueForKey(infostring, "ch");
307  if (*s)
308  {
309  characterIndex = atoi(s);
310  if (characterIndex < 0 || characterIndex >= MAX_CHARACTERS)
311  {
312  goto set_default_character;
313  }
314 
315  if (client->pers.characterIndex != characterIndex)
316  {
317  client->pers.characterIndex = characterIndex;
318  trap_GetConfigstring(CS_CHARACTERS + characterIndex, infostring, MAX_INFO_STRING);
319 
320  if (!(client->pers.character = BG_FindCharacter(infostring)))
321  {
322  // not found - create it (this should never happen as we should have everything precached)
323  client->pers.character = BG_FindFreeCharacter(infostring);
324 
325  if (!client->pers.character)
326  {
327  goto set_default_character;
328  }
329 
330  Q_strncpyz(client->pers.character->characterFile, infostring, sizeof(client->pers.character->characterFile));
331 
332  if (!G_RegisterCharacter(infostring, client->pers.character))
333  {
334  G_Printf(S_COLOR_YELLOW "WARNING: G_UpdateCharacter: failed to load character file '%s' for %s\n", infostring,
335  client->pers.netname);
336 
337  goto set_default_character;
338  }
339  }
340  // reset anims so client's dont freak out
341 
342  // this can only be done if the model really changed - otherwise, the animation may get
343  // screwed up if we are in the middle of some animation and we come into this function;
344  // plus, also reset the timer so we can properly start the next animation
345  client->ps.legsAnim = 0;
346  client->ps.torsoAnim = 0;
347  client->ps.legsTimer = 0;
348  client->ps.torsoTimer = 0;
349  }
350  return;
351  }
352 
353 set_default_character:
354  // set default character
355  character = BG_GetCharacter(client->sess.sessionTeam, client->sess.playerType);
356  if (client->pers.character != character)
357  {
358  client->pers.characterIndex = -1;
359  client->pers.character = character;
360 
361  client->ps.legsAnim = 0;
362  client->ps.torsoAnim = 0;
363  client->ps.legsTimer = 0;
364  client->ps.torsoTimer = 0;
365  }
