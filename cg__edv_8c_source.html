170 void CG_DrawLine(vec3_t start, vec3_t end, vec4_t color, qhandle_t shader)
171 {
172  polyBuffer_t *pb;
173  int vert;
174  byte bcolor[4];
175 
176  vec3_t dir, diff, up;
177 
178  pb = CG_PB_FindFreePolyBuffer(shader, 4, 6);
179  if (!pb)
180  {
181  return;
182  }
183 
184  bcolor[0] = (byte)(color[0] * 255.f);
185  bcolor[1] = (byte)(color[1] * 255.f);

187  bcolor[3] = (byte)(color[3] * 255.f);
188 
189  vert = pb->numVerts;
190 
191  VectorSubtract(start, end, dir);

199 
200  VectorAdd(start, up, pb->xyz[vert + 0]);
201  Vector2Set(pb->st[vert + 0], 0.0, 0.0);
202  Com_Memcpy(pb->color[vert + 0], bcolor, sizeof(*pb->color));
203 
204  VectorSubtract(start, up, pb->xyz[vert + 1]);
205  Vector2Set(pb->st[vert + 1], 0.0, 1.0);
206  Com_Memcpy(pb->color[vert + 1], bcolor, sizeof(*pb->color));
207 
208  // end points
209  VectorSubtract(end, cg.refdef_current->vieworg, diff);
210  CrossProduct(dir, diff, up);
211  VectorNormalizeFast(up);
212  VectorScale(up, ADDLINE_WIDTH * .5f, up);
213 
214  VectorAdd(end, up, pb->xyz[vert + 2]);
215  Vector2Set(pb->st[vert + 2], 1.0, 0.0);
216  Com_Memcpy(pb->color[vert + 2], bcolor, sizeof(*pb->color));
217 
218  VectorSubtract(end, up, pb->xyz[vert + 3]);
219  Vector2Set(pb->st[vert + 3], 1.0, 1.0);
220  Com_Memcpy(pb->color[vert + 3], bcolor, sizeof(*pb->color));
221 
222  pb->numVerts = vert + 4;
223 
224  pb->indicies[pb->numIndicies++] = (unsigned int)(vert + 2);
225  pb->indicies[pb->numIndicies++] = (unsigned int)(vert + 0);
226  pb->indicies[pb->numIndicies++] = (unsigned int)(vert + 1);
227 
228  pb->indicies[pb->numIndicies++] = (unsigned int)(vert + 1);
229  pb->indicies[pb->numIndicies++] = (unsigned int)(vert + 3);
230  pb->indicies[pb->numIndicies++] = (unsigned int)(vert + 2);
231 }
