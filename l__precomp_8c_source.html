280 {

282  int i, done, lastcomma, numparms, indent;
283  define_t *newdefine;
284 
285  if (!PC_ReadSourceToken(source, &token))
286  {
287  SourceError(source, "define %s missing parms", define->name);
288  return qfalse;
289  }
290 
291  if (define->numparms > maxparms)
292  {
293  SourceError(source, "define with more than %d parameters", maxparms);
294  return qfalse;
295  }
296 
297  for (i = 0; i < define->numparms; i++)
298  {
299  parms[i] = NULL;
300  }
301  // if no leading "("
302  if (strcmp(token.string, "("))
303  {

306  return qfalse;
307  }

310  {
311  if (numparms >= maxparms)
312  {
313  SourceError(source, "define %s with too many parms", define->name);

317  {
318  SourceWarning(source, "define %s has too many parms", define->name);
319  return qfalse;
320  }
321  parms[numparms] = NULL;
322  lastcomma = 1;
323  last = NULL;

330  }
331

333  {
334  if (indent <= 1)
335  {
336  if (lastcomma)
337  {
338  SourceWarning(source, "too many comma's");
339  }
340  // lastcomma = 1; // FIXME: lastcomma is never read !
341  break;
342  }
343  }
344  lastcomma = 0;
345 
346  if (!strcmp(token.string, "("))
347  {
348  indent++;
349  }
350  else if (!strcmp(token.string, ")"))
351  {
352  if (--indent <= 0)
353  {

355  {
356  SourceWarning(source, "too few define parms");
357  }
358  done = 1;
359  break;
360  }
361  }
362  else if (token.type == TT_NAME)

364  newdefine = PC_FindHashedDefine(source->definehash, token.string);
365  if (newdefine)

367  if (!PC_ExpandDefineIntoSource(source, &token, newdefine))

369  return qfalse;
370  }
371  continue;
372  }
373  }
374 
375  if (numparms < define->numparms)
376  {
377  t = PC_CopyToken(&token);
378  t->next = NULL;
379  if (last)
380  {
381  last->next = t;
382  }
383  else
384  {
385  parms[numparms] = t;
386  }
387  last = t;
388  }
389  }
390  numparms++;
391  }
392  return qtrue;
393 }

743 int PC_ExpandDefine(source_t *source, token_t *deftoken, define_t *define,
744  token_t **firsttoken, token_t **lasttoken)
745 {
746  token_t *parms[MAX_DEFINEPARMS], *dt, *pt, *t;
747  token_t *t1, *t2, *first, *last, *nextpt, token;
748  int parmnum, i;
749 
750  // if it is a builtin define
751  if (define->builtin)
752  {
753  return PC_ExpandBuiltinDefine(source, deftoken, define, firsttoken, lasttoken);
754  }
755  // if the define has parameters
756  if (define->numparms)
757  {
758  if (!PC_ReadDefineParms(source, define, parms, MAX_DEFINEPARMS))
759  {
760  return qfalse;
761  }
762 #ifdef DEBUG_EVAL
763  for (i = 0; i < define->numparms; i++)
764  {
765  Com_Printf("define parms %d:", i);
766  for (pt = parms[i]; pt; pt = pt->next)
767  {
768  Com_Printf("%s", pt->string);
769  }
770  }
771 #endif //DEBUG_EVAL
772  }
773  // empty list at first
774  first = NULL;
775  last = NULL;
776  // create a list with tokens of the expanded define
777  for (dt = define->tokens; dt; dt = dt->next)
778  {
779  parmnum = -1;
780  // if the token is a name, it could be a define parameter
781  if (dt->type == TT_NAME)
782  {
783  parmnum = PC_FindDefineParm(define, dt->string);
784  }
785  // if it is a define parameter
786  if (parmnum >= 0)
787  {
788  for (pt = parms[parmnum]; pt; pt = pt->next)
789  {
790  t = PC_CopyToken(pt);
791  // add the token to the list
792  t->next = NULL;
793  if (last)
794  {
795  last->next = t;
796  }
797  else
798  {
799  first = t;
800  }
801  last = t;
802  }
803  }
804  else
805  {
806  // if stringizing operator
807  if (dt->string[0] == '#' && dt->string[1] == '\0')
808  {
809  // the stringizing operator must be followed by a define parameter
810  if (dt->next)
811  {
812  parmnum = PC_FindDefineParm(define, dt->next->string);
813  }
814  else
815  {
816  parmnum = -1;
817  }
818 
819  if (parmnum >= 0)
820  {
821  // step over the stringizing operator
822  dt = dt->next;
823  // stringize the define parameter tokens
824  if (!PC_StringizeTokens(parms[parmnum], &token))
825  {
826  SourceError(source, "can't stringize tokens");
827  return qfalse;
828  }
829  t = PC_CopyToken(&token);
830  }
831  else
832  {
833  SourceWarning(source, "stringizing operator without define parameter");
834  continue;
835  }
836  }
837  else
838  {
839  t = PC_CopyToken(dt);
840  }
841  // add the token to the list
842  t->next = NULL;
843  if (last)
844  {
845  last->next = t;
846  }
847  else
848  {
849  first = t;
850  }
851  last = t;
852  }
853  }
854  // check for the merging operator
855  for (t = first; t; )
856  {
857  if (t->next)
858  {
859  // if the merging operator
860  if (t->next->string[0] == '#' && t->next->string[1] == '#')
861  {
862  t1 = t;
863  t2 = t->next->next;
864  if (t2)
865  {
866  if (!PC_MergeTokens(t1, t2))
867  {
868  SourceError(source, "can't merge %s with %s", t1->string, t2->string);
869  return qfalse;
870  }
871  PC_FreeToken(t1->next);
872  t1->next = t2->next;
873  if (t2 == last)
874  {
875  last = t1;
876  }
877  PC_FreeToken(t2);
878  continue;
879  }
880  }
881  }
882  t = t->next;
883  }
884  // store the first and last token of the list
885  *firsttoken = first;
886  *lasttoken = last;
887  // free all the parameter tokens
888  for (i = 0; i < define->numparms; i++)
889  {
890  for (pt = parms[i]; pt; pt = nextpt)
891  {
892  nextpt = pt->next;
893  PC_FreeToken(pt);
894  }
895  }
896 
897  return qtrue;
