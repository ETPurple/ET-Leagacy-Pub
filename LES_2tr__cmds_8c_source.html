99 void R_IssueRenderCommands(qboolean runPerformanceCounters)
100 {
101  renderCommandList_t *cmdList = &backEndData->commands;
102 
103  etl_assert(cmdList != NULL);
104  // add an end-of-list command
105  *(int *)(cmdList->cmds + cmdList->used) = RC_END_OF_LIST;
106 
107  // clear it out, in case this is a sync and not a buffer flip
108  cmdList->used = 0;
109 
110  // at this point, the back end thread is idle, so it is ok
111  // to look at it's performance counters
112  if (runPerformanceCounters)
113  {
114  R_PerformanceCounters();
115  }
116 
117  // actually start the commands going
118  if (!r_skipBackEnd->integer)
119  {
120  // let it start on the new batch
121  RB_ExecuteRenderCommands(cmdList->cmds);
122  }
123 }

147 void *R_GetCommandBuffer(unsigned int bytes)
148 {
149  renderCommandList_t *cmdList = &backEndData->commands;
150 
151  // always leave room for the swap buffers and end of list commands
152  if (cmdList->used + bytes + (sizeof(swapBuffersCommand_t) + sizeof(int)) > MAX_RENDER_COMMANDS)
153  {
154  if (bytes > MAX_RENDER_COMMANDS - (sizeof(swapBuffersCommand_t) + sizeof(int)))
155  {
156  Ren_Fatal("R_GetCommandBuffer: bad size %u", bytes);
157  }
158  // if we run out of room, just start dropping commands
159  return NULL;
160  }
161 
162  cmdList->used += bytes;
163 
164  return cmdList->cmds + cmdList->used - bytes;
165 }

439 void RE_BeginFrame(void)
440 {
441  drawBufferCommand_t *cmd;
442 
443  if (!tr.registered)
444  {
445  return;
446  }
447  glState.finishCalled = qfalse;
448 
449  tr.frameCount++;
450  tr.frameSceneNum = 0;
451 
452  // do overdraw measurement
453  if (r_measureOverdraw->integer)
454  {
455  if (glConfig.stencilBits < 4)
456  {
457  Ren_Print("Warning: not enough stencil bits to measure overdraw: %d\n", glConfig.stencilBits);
458  ri.Cvar_Set("r_measureOverdraw", "0");
459  }
460  else if (r_shadows->integer == 2)
461  {
462  Ren_Print("Warning: stencil shadows and overdraw measurement are mutually exclusive\n");
463  ri.Cvar_Set("r_measureOverdraw", "0");
464  }
465  else
466  {
467  R_IssuePendingRenderCommands();
468  qglEnable(GL_STENCIL_TEST);
469  qglStencilMask(~0U);
470  qglClearStencil(0U);
471  qglStencilFunc(GL_ALWAYS, 0U, ~0U);
472  qglStencilOp(GL_KEEP, GL_INCR, GL_INCR);
473  }
474  r_measureOverdraw->modified = qfalse;
475  }
476  else
477  {
478  // this is only reached if it was on and is now off
479  if (r_measureOverdraw->modified)
480  {
481  R_IssuePendingRenderCommands();
482  qglDisable(GL_STENCIL_TEST);
483  }
484  r_measureOverdraw->modified = qfalse;
485  }
486 
487  // texturemode stuff
488  if (r_textureMode->modified)
489  {
490  R_IssuePendingRenderCommands();
491  GL_TextureMode(r_textureMode->string);
492  r_textureMode->modified = qfalse;
493  }
494 
495  // gamma stuff
496  if (r_gamma->modified)
497  {
498  r_gamma->modified = qfalse;
499 
500  R_IssuePendingRenderCommands();
501  R_SetColorMappings();
502  }
503 
504  // check for errors
505  if (!r_ignoreGLErrors->integer)
506  {
507  unsigned int err;
508 
509  R_IssuePendingRenderCommands();
510  if ((err = qglGetError()) != GL_NO_ERROR)
511  {
512  Ren_Fatal("RE_BeginFrame() - glGetError() failed (0x%x)!\n", err);
513  }
514  }
515 
516  // draw buffer stuff
517  cmd = R_GetCommandBuffer(sizeof(*cmd));
518  if (!cmd)
519  {
520  return;
521  }
522  cmd->commandId = RC_DRAW_BUFFER;
523 
524  if (!Q_stricmp(r_drawBuffer->string, "GL_FRONT"))
525  {
526  cmd->buffer = (int)GL_FRONT;
527  }
528  else
529  {
530  cmd->buffer = (int)GL_BACK;
531  }
532 }

539 void RE_EndFrame(int *frontEndMsec, int *backEndMsec)
540 {
541  renderCommandList_t *cmdList;
542 
543  if (!tr.registered)
544  {
545  return;
546  }
547 
548  // Needs to use reserved space, so no R_GetCommandBuffer.
549  cmdList = &backEndData->commands;
550  etl_assert(cmdList != NULL);
551  // add swap-buffers command
552  *( int * )(cmdList->cmds + cmdList->used) = RC_SWAP_BUFFERS;
553  cmdList->used += sizeof(swapBuffersCommand_t);
554 
555  R_IssueRenderCommands(qtrue);
556 
557  // use the other buffers next frame, because another CPU

564  }
565  tr.frontEndMsec = 0;
566  if (backEndMsec)
567  {

569  }
