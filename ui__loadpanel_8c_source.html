230 const char *UI_DownloadInfo(const char *downloadName)
231 {
232  static char dlText[] = "Downloading:";
233  static char etaText[] = "Estimated time left:";
234  static char xferText[] = "Transfer rate:";
235  static int tleEstimates[ESTIMATES] = { 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60,
236  60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60,
237  60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60,
238  60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60, 60 };
239  static int tleIndex = 0;
240  char dlSizeBuf[64], totalSizeBuf[64], xferRateBuf[64], dlTimeBuf[64];
241  int downloadSize, downloadCount, downloadTime;
242  const char *s, *ds;
243 
244  downloadSize = trap_Cvar_VariableValue("cl_downloadSize");
245  downloadCount = trap_Cvar_VariableValue("cl_downloadCount");
246  downloadTime = trap_Cvar_VariableValue("cl_downloadTime");
247 
248  if (downloadSize > 0)
249  {
250  ds = va("%s (%d%%)", downloadName, (int)((float)downloadCount * 100.0f / (float)downloadSize));
251  }
252  else
253  {
254  ds = downloadName;
255  }
256 
257  UI_ReadableSize(dlSizeBuf, sizeof dlSizeBuf, downloadCount);
258  UI_ReadableSize(totalSizeBuf, sizeof totalSizeBuf, downloadSize);
259 
260  if (downloadCount < 4096 || !downloadTime)
261  {
262  s = va("%s\n %s\n%s\n\n%s\n estimating...\n\n%s\n\n%s copied", dlText, ds, totalSizeBuf,
263  etaText,
264  xferText,
265  dlSizeBuf);
266  }
267  else
268  {
269  int xferRate;
270 
271  if ((uiInfo.uiDC.realTime - downloadTime) / 1000)
272  {
273  xferRate = downloadCount / ((uiInfo.uiDC.realTime - downloadTime) / 1000);
274  }
275  else
276  {
277  xferRate = 0;
278  }
279  UI_ReadableSize(xferRateBuf, sizeof xferRateBuf, xferRate);
280 
281  // Extrapolate estimated completion time
282  if (downloadSize && xferRate)
283  {
284  int n = downloadSize / xferRate; // estimated time for entire d/l in secs
285  int timeleft = 0, i;
286 
287  // We do it in K (/1024) because we'd overflow around 4MB
288  tleEstimates[tleIndex] = (n - (((downloadCount / 1024) * n) / (downloadSize / 1024)));
289  tleIndex++;
290  if (tleIndex >= ESTIMATES)
291  {
292  tleIndex = 0;
293  }
294 
295  for (i = 0; i < ESTIMATES; i++)
296  timeleft += tleEstimates[i];
297 
298  timeleft /= ESTIMATES;
299 
300  UI_PrintTime(dlTimeBuf, sizeof dlTimeBuf, timeleft);
301  }
302  else
303  {
304  dlTimeBuf[0] = '\0';
305  }
306 
307  if (xferRate)
308  {
309  s = va("%s\n %s\n%s\n\n%s\n %s\n\n%s\n %s/sec\n\n%s copied", dlText, ds, totalSizeBuf,
310  etaText, dlTimeBuf,
311  xferText, xferRateBuf,
312  dlSizeBuf);
313  }
314  else
315  {
316  if (downloadSize)
317  {
318  s = va("%s\n %s\n%s\n\n%s\n estimating...\n\n%s\n\n%s copied", dlText, ds, totalSizeBuf,
319  etaText,
320  xferText,
321  dlSizeBuf);
322  }
323  else
324  {
325  s = va("%s\n %s\n\n%s\n estimating...\n\n%s\n\n%s copied", dlText, ds,
326  etaText,
327  xferText,
328  dlSizeBuf);
329  }
330  }
331  }
332 
333  return s;
334 }

393  }
394
