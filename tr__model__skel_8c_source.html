142 qboolean AddTriangleToVBOTriangleList(growList_t *vboTriangles, skelTriangle_t *tri, int *numBoneReferences, int boneReferences[MAX_BONES])
143 {
144  int i, j, k;
145  md5Vertex_t *v;
146  int boneIndex;
147  int numNewReferences = 0;
148  int newReferences[MAX_WEIGHTS * 3]; // a single triangle can have up to 12 new bone references !
149  qboolean hasWeights = qfalse;
150 
151  Com_Memset(newReferences, -1, sizeof(newReferences));
152 
153  for (i = 0; i < 3; i++)
154  {
155  v = tri->vertexes[i];
156 
157  // can the bones be referenced?
158  for (j = 0; j < MAX_WEIGHTS; j++)
159  {
160  if (j < v->numWeights)
161  {
162  boneIndex = v->weights[j]->boneIndex;
163  hasWeights = qtrue;
164 
165  // is the bone already referenced?
166  if (!boneReferences[boneIndex])
167  {
168  // the bone isn't yet and we have to test if we can give the mesh this bone at all
169  if ((*numBoneReferences + numNewReferences) >= glConfig2.maxVertexSkinningBones)
170  {
171  return qfalse;
172  }
173  else
174  {
175  for (k = 0; k < (MAX_WEIGHTS * 3); k++)
176  {
177  if (newReferences[k] == boneIndex)
178  {
179  break;
180  }
181  }
182 
183  if (k == (MAX_WEIGHTS * 3))
184  {
185  newReferences[numNewReferences] = boneIndex;
186  numNewReferences++;
187  }
188  }
189  }
190  }
191  }
192  }
193 
194  // reference them!
195  for (j = 0; j < numNewReferences; j++)
196  {
197  boneIndex = newReferences[j];
198 
199  boneReferences[boneIndex]++;
200 
201  *numBoneReferences = *numBoneReferences + 1;
202  }
203 
204 #if 0
205  if (numNewReferences)

213  newReferences[5],

217  newReferences[9],

221 #endif
222
