599 void trap_GetUsercmd(int clientNum, usercmd_t *cmd)
600 {
601  syscall(G_GET_USERCMD, clientNum, cmd);
602 
603 #ifdef FAKELAG
604  {
605  char s[MAX_STRING_CHARS];
606  int fakeLag;
607 
608  trap_Cvar_VariableStringBuffer("g_fakelag", s, sizeof(s));
609  fakeLag = atoi(s);
610  if (fakeLag < 0)
611  {
612  fakeLag = 0;
613  }
614 
615  if (fakeLag)
616  {
617  int i;
618  int realcmdtime, thiscmdtime;
619 
620  // store our newest usercmd
621  cmdNumber[clientNum]++;
622  Com_Memcpy(&cmds[clientNum][cmdNumber[clientNum] & MAX_USERCMD_MASK], cmd, sizeof(usercmd_t));
623 
624  // find a usercmd that is fakeLag msec behind
625  i = cmdNumber[clientNum] & MAX_USERCMD_MASK;
626  realcmdtime = cmds[clientNum][i].serverTime;
627  i--;
628  do
629  {
630  thiscmdtime = cmds[clientNum][i & MAX_USERCMD_MASK].serverTime;
631 
632  if (realcmdtime - thiscmdtime > fakeLag)
633  {
634  // found the right one
635  cmd = &cmds[clientNum][i & MAX_USERCMD_MASK];
636  return;
637  }
638 
639  i--;
640  }
641  while ((i & MAX_USERCMD_MASK) != (cmdNumber[clientNum] & MAX_USERCMD_MASK));
642 
643  // didn't find a proper one, just use the oldest one we have
644  cmd = &cmds[clientNum][(cmdNumber[clientNum] - 1) & MAX_USERCMD_MASK];
645  return;
646  }
647  }
648 #endif // FAKELAG
649 }
