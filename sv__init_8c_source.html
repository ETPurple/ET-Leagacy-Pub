490 void SV_DemoChangeMaxClients(void)
491 {
492  int oldMaxClients, oldDemoClients;
493  int i, j, k;
494  client_t *oldClients = NULL;
495  int count;
496  //qboolean firstTime = svs.clients == NULL;
497 
498  // == Checking the prerequisites
499  // Note: we check here that we have enough slots to fit all clients, and that it doesn't overflow the MAX_CLIENTS the engine can support. Also, we save the oldMaxClients and oldDemoClients values.
500 
501  // -- Get the highest client number in use
502  count = 0;
503  for (i = 0 ; i < sv_maxclients->integer ; i++)
504  {
505  if (svs.clients[i].state >= CS_CONNECTED)
506  {
507  if (i > count)
508  {
509  count = i;
510  }
511  }
512  }
513  count++;
514 
515  // -- Save the previous oldMaxClients and oldDemoClients values, and update
516 
517  // Save the previous sv_maxclients value before updating it
518  oldMaxClients = sv_maxclients->integer;
519  // update the cvars
520  Cvar_Get("sv_maxclients", "8", 0);
521  Cvar_Get("sv_democlients", "0", 0); // unnecessary now that sv_democlients is not latched anymore?
522  // Save the previous sv_democlients (since it's updated instantly, we cannot get it directly), we use a trick by computing the difference between the new and previous sv_maxclients (the difference should indeed be the exact value of sv_democlients)
523  oldDemoClients = (oldMaxClients - sv_maxclients->integer);
524  if (oldDemoClients < 0) // if the difference is negative, this means that before it was set to 0 (because the newer sv_maxclients is greater than the old)
525  {
526  oldDemoClients = 0;
527  }
528 
529  // -- Check limits
530  // never go below the highest client number in use (make sure we have enough room for all players)
531  SV_BoundMaxClients(count);
532 
533  // -- Change check: if still the same, we just quit, there's nothing to do
534  if (sv_maxclients->integer == oldMaxClients)
535  {
536  return;
537  }
538 
539  // == Memorizing clients
540  // Note: we save in a temporary variables the clients, because after we will wipe completely the svs.clients struct
541 
542  // copy the clients to hunk memory
543  oldClients = Hunk_AllocateTempMemory((sv_maxclients->integer - sv_democlients->integer) * sizeof(client_t)); // we allocate just enough memory for the real clients (not counting in the democlients)
544  // For all previous clients slots, we copy the entire client into a temporary var
545  for (i = 0, j = 0, k = sv_privateClients->integer ; i < oldMaxClients ; i++) // for all the previously connected clients, we copy them to a temporary var
546  { // If there is a real client in this slot
547  if (svs.clients[i].state >= CS_CONNECTED)
548  {
549  // if the client is in a privateClient reserved slot, we move him on the reserved slots
550  if (i >= oldDemoClients && i < oldDemoClients + sv_privateClients->integer)
551  {
552  oldClients[j++] = svs.clients[i];
553  // else the client is not a privateClient, and we move him to the first available slot after the privateClients slots
554  }
555  else
556  {
557  oldClients[k++] = svs.clients[i];
558  }
559  }
560  }
561 
562  // Fill in the remaining clients slots with empty clients (else the engine crash when copying into memory svs.clients)
563  for (i = j; i < sv_privateClients->integer; i++) // Fill the privateClients empty slots
564  {
565  Com_Memset(&oldClients[i], 0, sizeof(client_t));
566  }
567  for (i = k; i < (sv_maxclients->integer - sv_democlients->integer); i++) // Fill the other normal clients slots
568  {
569  Com_Memset(&oldClients[i], 0, sizeof(client_t));
570  }
571 
572  // free old clients arrays
573  Z_Free(svs.clients);
574 
575  // == Allocating the new svs.clients and moving the saved clients over from the temporary var
576 
577  // allocate new svs.clients
578  svs.clients = Z_Malloc(sv_maxclients->integer * sizeof(client_t));
579  Com_Memset(svs.clients, 0, sv_maxclients->integer * sizeof(client_t));
580 
581  // copy the clients over (and move them depending on sv_democlients: if >0, move them upwards, if == 0, move them to their original slots)
582  Com_Memcpy(svs.clients + sv_democlients->integer, oldClients, (sv_maxclients->integer - sv_democlients->integer) * sizeof(client_t));
583 
584  // free the old clients on the hunk
585  Hunk_FreeTempMemory(oldClients);
586 
587  // == Allocating snapshot entities
588

591

593 
594  // set demostate to none if it was just waiting to set maxclients and move real clients slots
595  if (sv.demoState == DS_WAITINGSTOP)
596  {
597  sv.demoState = DS_NONE;
598  Cvar_SetValue("sv_demoState", DS_NONE);
599  }
