79  {

81  PM_ClipVelocity(pm->ps->velocity, pml.groundTrace.plane.normal,

88  }

93  if (pml.groundPlane)
94  {

96  VectorCopy(pml.groundTrace.plane.normal, planes[0]);
97  }
98  else

101  }
102 
103  // never turn against original velocity
104  VectorNormalize2(pm->ps->velocity, planes[numplanes]);
105  numplanes++;
106 
107  for (bumpcount = 0 ; bumpcount < numbumps ; bumpcount++)
108  {
109  // calculate position we are trying to move to
110  VectorMA(pm->ps->origin, time_left, pm->ps->velocity, end);
111 
112  // see if we can make it there
113  PM_TraceAll(&trace, pm->ps->origin, end);
114  if (pm->debugLevel > 1)
115  {
116  Com_Printf("%i:%d %d (%f %f %f)\n",
117  c_pmove, trace.allsolid, trace.startsolid,
118  (double)trace.endpos[0],
119  (double)trace.endpos[1],
120  (double)trace.endpos[2]
121  );
122  }
123 
124  if (trace.allsolid)
125  {
126  // entity is completely trapped in another solid
127  pm->ps->velocity[2] = 0; // don't build up falling damage, but allow sideways acceleration
128  return qtrue;
129  }
130 
131  if (trace.fraction > 0)
132  {
133  // actually covered some distance
134  VectorCopy(trace.endpos, pm->ps->origin);
135  }
136 
137  if (trace.fraction == 1.f)
138  {
139  break; // moved the entire distance
140  }
141 
142  // save entity for contact
143  PM_AddTouchEnt(trace.entityNum);
144 
145  time_left -= time_left * trace.fraction;
146 
147  if (numplanes >= MAX_CLIP_PLANES)
148  {
149  // this shouldn't really happen
150  VectorClear(pm->ps->velocity);
151  return qtrue;
152  }
153 
154  // if this is the same plane we hit before, nudge velocity
155  // out along it, which fixes some epsilon issues with
156  // non-axial planes
157 
158  for (i = 0 ; i < numplanes ; i++)
159  {
160  if (DotProduct(trace.plane.normal, planes[i]) > 0.99f)
161  {
162  if (extrabumps <= 0)

165  extrabumps++;

167

171  }
172  }
173  else

176  // and trace it, to make sure we don't end up in a solid
177

181

184  Com_Printf("%i:planeoriginnudge\n", c_pmove);
185  }
186  }
187  break;
188  }
189  }

191  {
192  continue;
193  }
194  VectorCopy(trace.plane.normal, planes[numplanes]);
195  numplanes++;
196 
197  // modify velocity so it parallels all of the clip planes
198 
199  // find a plane that it enters
200  for (i = 0 ; i < numplanes ; i++)
201  {
202  into = DotProduct(pm->ps->velocity, planes[i]);
203  if (into >= 0.1f)
204  {
205  continue; // move doesn't interact with the plane
206  }
207 
208  // see how hard we are hitting things
209  if (-into > pml.impactSpeed)
210  {
211  pml.impactSpeed = -into;
212  }
213 
214  // slide along the plane
215  PM_ClipVelocity(pm->ps->velocity, planes[i], clipVelocity, OVERCLIP);
216 
217  // slide along the plane
218  PM_ClipVelocity(endVelocity, planes[i], endClipVelocity, OVERCLIP);
219 
220  // see if there is a second plane that the new move enters
221  for (j = 0 ; j < numplanes ; j++)
222  {
223  if (j == i)
224  {
225  continue;
226  }
227  if (DotProduct(clipVelocity, planes[j]) >= 0.1f)
228  {
229  continue; // move doesn't interact with the plane
230  }
231 
232  // try clipping the move to the plane
233  PM_ClipVelocity(clipVelocity, planes[j], clipVelocity, OVERCLIP);
234  PM_ClipVelocity(endClipVelocity, planes[j], endClipVelocity, OVERCLIP);
235 
236  // see if it goes back into the first clip plane
237  if (DotProduct(clipVelocity, planes[i]) >= 0)
238  {
239  continue;
240  }
241 
242  // slide the original velocity along the crease
243  CrossProduct(planes[i], planes[j], dir);
244  VectorNormalize(dir);
245  d = DotProduct(dir, pm->ps->velocity);
246  VectorScale(dir, d, clipVelocity);
247 
248  //CrossProduct(planes[i], planes[j], dir); // this has just been done
249  //VectorNormalize(dir); // and this too.. no need to do it again
250  d = DotProduct(dir, endVelocity);
251  VectorScale(dir, d, endClipVelocity);
252 
253  // see if there is a third plane the the new move enters
254  for (k = 0 ; k < numplanes ; k++)
255  {
256  if (k == i || k == j)
257  {
258  continue;
259  }
260  if (DotProduct(clipVelocity, planes[k]) >= 0.1f)
261  {
262  continue; // move doesn't interact with the plane
263  }
264 
265  // stop dead at a tripple plane interaction
266  VectorClear(pm->ps->velocity);
267  return qtrue;
268  }
269  }
270 
271  // if we have fixed all interactions, try another move
272  VectorCopy(clipVelocity, pm->ps->velocity);
273  VectorCopy(endClipVelocity, endVelocity);
274  break;
275  }
276  }
277 
278  if (gravity)
279  {
280  VectorCopy(endVelocity, pm->ps->velocity);
281  }
282 
283  // don't change velocity if in a timer (FIXME: is this correct?)
284  if (pm->ps->pm_time)
285  {
286  VectorCopy(primal_velocity, pm->ps->velocity);
287  }
288 
289  return (bumpcount != 0);
290 }
