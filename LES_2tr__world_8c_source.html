48 static qboolean R_CullSurface(surfaceType_t *surface, shader_t *shader, int *frontFace)
49 {
50  srfGeneric_t *gen;
51 
52  // force to non-front facing
53  *frontFace = 0;
54 
55  // allow culling to be disabled
56  if (r_noCull->integer)
57  {
58  return qfalse;
59  }
60 
61  // made surface culling generic, inline with q3map2 surface classification
62  switch (*surface)
63  {
64  case SF_FACE:
65  case SF_TRIANGLES:
66  break;
67  case SF_GRID:
68  if (r_noCurves->integer)
69  {
70  return qtrue;
71  }
72  break;
73  case SF_FOLIAGE:
74  if (r_drawFoliage->value == 0.f)
75  {
76  return qtrue;
77  }
78  break;
79 
80  default:
81  return qtrue;
82  }
83 
84  // get generic surface
85  gen = (srfGeneric_t *) surface;
86 
87  // plane cull
88  if (gen->plane.type != PLANE_NON_PLANAR && r_facePlaneCull->integer)
89  {
90  float d = DotProduct(tr.orientation.viewOrigin, gen->plane.normal) - gen->plane.dist;
91 
92  if (d > 0.0f)
93  {
94  *frontFace = 1;
95  }
96 
97  // don't cull exactly on the plane, because there are levels of rounding
98  // through the BSP, ICD, and hardware that may cause pixel gaps if an
99  // epsilon isn't allowed here
100  if (shader->cullType == CT_FRONT_SIDED)
101  {
102  if (d < -8.0f)
103  {
104  tr.pc.c_plane_cull_out++;
105  return qtrue;
106  }
107  }
108  else if (shader->cullType == CT_BACK_SIDED)
109  {
110  if (d > 8.0f)
111  {
112  tr.pc.c_plane_cull_out++;
113  return qtrue;
114  }
115  }
116 
117  tr.pc.c_plane_cull_in++;
118  }
119 
120  {
121  int cull;
122 
123  // try sphere cull
124  if (tr.currentEntityNum != ENTITYNUM_WORLD)
125  {
126  cull = R_CullLocalPointAndRadius(gen->origin, gen->radius);
127  }
128  else
129  {
130  cull = R_CullPointAndRadius(gen->origin, gen->radius);
131  }
132 
133  if (cull == CULL_OUT)
134  {
135  tr.pc.c_sphere_cull_out++;
136  return qtrue;
137  }
138 
139  tr.pc.c_sphere_cull_in++;
140  }
141 
142  // must be visible
143  return qfalse;
144 }

157 static int R_DlightSurface(msurface_t *surface, int dlightBits)
158 {
159  int i;
160  vec3_t origin;
161  float radius;
162  srfGeneric_t *gen;
163 
164  // made surface dlighting generic, inline with q3map2 surface classification
165  switch ((surfaceType_t) *surface->data)
166  {
167  case SF_FACE:
168  case SF_TRIANGLES:
169  case SF_GRID:
170  case SF_FOLIAGE:
171  break;
172 
173  default:
174  return 0;
175  }
176 
177  // get generic surface
178  gen = (srfGeneric_t *) surface->data;
179 
180  // debug code
181  // gen->dlightBits = dlightBits;
182  // return dlightBits;
183 
184  // try to cull out dlights
185  for (i = 0; i < tr.refdef.num_dlights; i++)
186  {
187  if (!(dlightBits & (1 << i)))
188  {
189  continue;
190  }
191 
192  // junior dlights don't affect world surfaces
193  if (tr.refdef.dlights[i].flags & REF_JUNIOR_DLIGHT)
194  {
195  dlightBits &= ~(1 << i);
196  continue;
197  }
198 
199  // lightning dlights affect all surfaces
200  if (tr.refdef.dlights[i].flags & REF_DIRECTED_DLIGHT)
201  {
202  continue;
203  }
204 
205  // test surface bounding sphere against dlight bounding sphere
206  VectorCopy(tr.refdef.dlights[i].transformed, origin);
207  radius = tr.refdef.dlights[i].radius;
208 
209  if ((gen->origin[0] + gen->radius) < (origin[0] - radius) ||
210  (gen->origin[0] - gen->radius) > (origin[0] + radius) ||
211  (gen->origin[1] + gen->radius) < (origin[1] - radius) ||
212  (gen->origin[1] - gen->radius) > (origin[1] + radius) ||
213  (gen->origin[2] + gen->radius) < (origin[2] - radius) ||
214  (gen->origin[2] - gen->radius) > (origin[2] + radius))
215  {
216  dlightBits &= ~(1 << i);
217  }
218  }
219 
220  // Ren_Print("Surf: 0x%08X dlightBits: 0x%08X\n", srf, dlightBits );
221 
222  // set counters
223  if (dlightBits == 0)
224  {
225  tr.pc.c_dlightSurfacesCulled++;
226  }
227  else
228  {
229  tr.pc.c_dlightSurfaces++;
230  }
231 
232  // set surface dlight bits and return
233  gen->dlightBits = dlightBits;
234  return dlightBits;
235 }

493 static void R_RecursiveWorldNode(mnode_t *node, int planeBits, int dlightBits, int decalBits)
494 {
495  int i, r;
496  dlight_t *dl;
497 
498  do
499  {
500  // if the node wasn't marked as potentially visible, exit
501  if (node->visframe != tr.visCount)
502  {
503  return;
504  }
505 
506  // if the bounding volume is outside the frustum, nothing
507  // inside can be visible OPTIMIZE: don't do this all the way to leafs?
508 
509  if (!r_noCull->integer)
510  {
511  if (planeBits & 1)
512  {
513  r = BoxOnPlaneSide(node->mins, node->maxs, &tr.viewParms.frustum[0]);
514  if (r == 2)
515  {
516  return; // culled
517  }
518  if (r == 1)
519  {
520  planeBits &= ~1; // all descendants will also be in front
521  }
522  }
523 
524  if (planeBits & 2)
525  {
526  r = BoxOnPlaneSide(node->mins, node->maxs, &tr.viewParms.frustum[1]);
527  if (r == 2)
528  {
529  return; // culled
530  }
531  if (r == 1)
532  {
533  planeBits &= ~2; // all descendants will also be in front
534  }
535  }
536 
537  if (planeBits & 4)
538  {
539  r = BoxOnPlaneSide(node->mins, node->maxs, &tr.viewParms.frustum[2]);
540  if (r == 2)
541  {
542  return; // culled
543  }
544  if (r == 1)
545  {
546  planeBits &= ~4; // all descendants will also be in front
547  }
548  }
549 
550  if (planeBits & 8)
551  {
552  r = BoxOnPlaneSide(node->mins, node->maxs, &tr.viewParms.frustum[3]);
553  if (r == 2)
554  {
555  return; // culled
556  }
557  if (r == 1)
558  {
559  planeBits &= ~8; // all descendants will also be in front
560  }
561  }
562 
563  // farplane culling
564  if (planeBits & 16)
565  {
566  r = BoxOnPlaneSide(node->mins, node->maxs, &tr.viewParms.frustum[4]);
567  if (r == 2)
568  {
569  return; // culled
570  }
571  if (r == 1)
572  {
573  planeBits &= ~8; // all descendants will also be in front
574  }
575  }
576 
577  }
578 
579  // cull dlights
580  if (dlightBits) //% && node->contents != -1 )
581  {
582  for (i = 0; i < tr.refdef.num_dlights; i++)
583  {
584  if (dlightBits & (1 << i))
585  {
586  // directional dlights don't get culled
587  if (tr.refdef.dlights[i].flags & REF_DIRECTED_DLIGHT)
588  {
589  continue;
590  }
591 
592  // test dlight bounds against node surface bounds
593  dl = &tr.refdef.dlights[i];
594  if (node->surfMins[0] >= (dl->origin[0] + dl->radius) || node->surfMaxs[0] <= (dl->origin[0] - dl->radius) ||
595  node->surfMins[1] >= (dl->origin[1] + dl->radius) || node->surfMaxs[1] <= (dl->origin[1] - dl->radius) ||
596  node->surfMins[2] >= (dl->origin[2] + dl->radius) || node->surfMaxs[2] <= (dl->origin[2] - dl->radius))
597  {
598  dlightBits &= ~(1 << i);
599  }
600  }
601  }
602  }
603 
604  // cull decals
605  if (decalBits)
606  {
607  for (i = 0; i < tr.refdef.numDecalProjectors; i++)
608  {
609  if (decalBits & (1 << i))
610  {
611  // test decal bounds against node surface bounds
612  if (tr.refdef.decalProjectors[i].shader == NULL ||
613  !R_TestDecalBoundingBox(&tr.refdef.decalProjectors[i], node->surfMins, node->surfMaxs))
614  {
615  decalBits &= ~(1 << i);
616  }
617  }
618  }
619  }
620 
621  // handle leaf nodes
622  if (node->contents != -1)
623  {
624  break;
625  }
626 
627  // recurse down the children, front side first
628  R_RecursiveWorldNode(node->children[0], planeBits, dlightBits, decalBits);
629 
630  // tail recurse
631  node = node->children[1];
632  }
633  while (1);
634 
635  // short circuit
636  if (node->nummarksurfaces == 0)
637  {
638  return;
639  }
640 
641  // moved off to separate function
642  R_AddLeafSurfaces(node, dlightBits, decalBits);
643 }

729 static void R_MarkLeaves(void)
730 {
731  const byte *vis;
732  mnode_t *leaf, *parent;
733  int i;
734  int cluster;
735 
736  // lockpvs lets designers walk around to determine the
737  // extent of the current pvs
738  if (r_lockPvs->integer)
739  {
740  return;
741  }
742 
743  // current viewcluster
744  leaf = R_PointInLeaf(tr.viewParms.pvsOrigin);
745  cluster = leaf->cluster;
746 
747  // if the cluster is the same and the area visibility matrix
748  // hasn't changed, we don't need to mark everything again
749 
750  // if r_showcluster was just turned on, remark everything
751  if (tr.viewCluster == cluster && !tr.refdef.areamaskModified
752  && !r_showCluster->modified)
753  {
754  return;
755  }
756 
757  if (r_showCluster->modified || r_showCluster->integer)
758  {
759  r_showCluster->modified = qfalse;
760  if (r_showCluster->integer)
761  {
762  Ren_Print("cluster:%i area:%i\n", cluster, leaf->area);
763  }
764  }
765 
766  tr.visCount++;
767  tr.viewCluster = cluster;
768 
769  if (r_noVis->integer || tr.viewCluster == -1)
770  {
771  for (i = 0 ; i < tr.world->numnodes ; i++)
772  {
773  if (tr.world->nodes[i].contents != CONTENTS_SOLID)
774  {
775  tr.world->nodes[i].visframe = tr.visCount;
776  }
777  }
778  return;
779  }
780 
781  vis = R_ClusterPVS(tr.viewCluster);
782 
783  for (i = 0, leaf = tr.world->nodes ; i < tr.world->numnodes ; i++, leaf++)
784  {
785  cluster = leaf->cluster;
786  if (cluster < 0 || cluster >= tr.world->numClusters)
787  {
788  continue;
789  }
790 
791  // check general pvs
792  if (!(vis[cluster >> 3] & (1 << (cluster & 7))))
793  {
794  continue;
795  }
796 
797  // check for door connection
798  if ((tr.refdef.areamask[leaf->area >> 3] & (1 << (leaf->area & 7))))
799  {
800  continue; // not visible
801  }
802 
803  // don't want to walk the entire bsp to add skybox surfaces
804  if (tr.refdef.rdflags & RDF_SKYBOXPORTAL)
805  {
806  // this only happens once, as game/cgame know the origin of the skybox
807  // this also means the skybox portal cannot move, as this list is calculated once and never again
808  if (tr.world->numSkyNodes < WORLD_MAX_SKY_NODES)
809  {
810  tr.world->skyNodes[tr.world->numSkyNodes++] = leaf;
811  }
812  R_AddLeafSurfaces(leaf, 0, 0);
813  continue;
814  }
815 
816  parent = leaf;
817  do
818  {
819  if (parent->visframe == tr.visCount)
820  {
821  break;
822  }
823  parent->visframe = tr.visCount;
824  parent = parent->parent;
825  }
826  while (parent);
827  }
828 }
