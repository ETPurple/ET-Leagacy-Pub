275 static void CG_SoundParseSounds(const char *filename, char *buffer)
276 {
277  char *token, **text = &buffer;
278  long hash;
279  soundScript_t sound; // the current sound being read
280  soundScriptSound_t *scriptSound = NULL;
281  qboolean inSound = qfalse, wantSoundName = qtrue;
282 
283  while (1)
284  {
285  token = COM_ParseExt(text, qtrue);
286  if (!token[0])
287  {
288  if (inSound)
289  {
290  CG_Error(S_COLOR_RED "CG_SoundParseSounds: no concluding '}' in sound %s, file %s\n", sound.name, filename);
291  }
292  if (scriptSound)
293  {
294  CG_Printf("...'%s' parsed - total script sounds: %i\n", filename, numSoundScripts);
295  }
296  return;
297  }
298 
299  if (!Q_stricmp(token, "{"))
300  {
301  if (inSound)
302  {
303  CG_Error(S_COLOR_RED "CG_SoundParseSounds: no concluding '}' in sound %s, file %s\n", sound.name, filename);
304  }
305  if (wantSoundName)
306  {
307  CG_Error(S_COLOR_RED "CG_SoundParseSounds:'{' found but not expected, after %s, file %s\n", sound.name, filename);
308  }
309  inSound = qtrue;
310 
311  // grab a free scriptSound
312  scriptSound = &soundScriptSounds[numSoundScriptSounds++];
313 
314  if (numSoundScriptSounds >= MAX_SOUND_SCRIPT_SOUNDS)
315  {
316  CG_Error(S_COLOR_RED "CG_SoundParseSounds: MAX_SOUND_SCRIPT_SOUNDS exceeded.\nReduce number of sound scripts.\n");
317  }
318 
319  scriptSound->lastPlayed = 0;
320  scriptSound->next = sound.soundList;
321  scriptSound->numsounds = 0;
322  sound.soundList = scriptSound;
323 
324  continue;
325  }
326 
327  if (!Q_stricmp(token, "}"))
328  {
329  if (!inSound)
330  {
331  CG_Error(S_COLOR_RED "CG_SoundParseSounds: '}' unexpected after sound %s, file %s\n", sound.name, filename);
332  }
333 
334  // end of a sound, copy it to the global list and stick it in the hashTable
335  hash = generateHashValue(sound.name);
336  sound.nextHash = hashTable[hash];
337  soundScripts[numSoundScripts] = sound;
338  hashTable[hash] = &soundScripts[numSoundScripts++];
339 
340  if (numSoundScripts >= MAX_SOUND_SCRIPTS)
341  {
342  CG_Error(S_COLOR_RED "CG_SoundParseSounds: MAX_SOUND_SCRIPTS exceeded. Reduce number of sound scripts.\n");
343  }
344 
345  inSound = qfalse;
346  wantSoundName = qtrue;
347 
348  CG_SoundScriptPrecache(sound.name);
349 
350  continue;
351  }
352 
353  if (!inSound)
354  {
355  // this is the identifier for a new sound
356  if (!wantSoundName)
357  {
358  CG_Error(S_COLOR_RED "CG_SoundParseSounds: '%s' unexpected after sound %s, file %s\n", token, sound.name, filename);
359  }
360  Com_Memset(&sound, 0, sizeof(sound));
361  Q_strncpyz(sound.name, token, sizeof(sound.name));
362  wantSoundName = qfalse;
363  sound.index = numSoundScripts;
364  // setup the new sound defaults
365  sound.channel = CHAN_AUTO;
366  sound.attenuation = 1; // default to fade away with distance (for streaming sounds)
367  continue;
368  }
369 
370  // we are inside a sound script
371 
372  if (!Q_stricmp(token, "channel"))
373  {
374  // ignore this now, just look for the channel identifiers explicitly
375  continue;
376  }
377  if (!Q_stricmp(token, "local"))
378  {
379  sound.channel = CHAN_LOCAL;
380  continue;
381  }
382  else if (!Q_stricmp(token, "announcer"))
383  {
384  sound.channel = CHAN_ANNOUNCER;
385  continue;
386  }
387  else if (!Q_stricmp(token, "body"))
388  {
389  sound.channel = CHAN_BODY;
390  continue;
391  }
392  else if (!Q_stricmp(token, "voice"))
393  {
394  sound.channel = CHAN_VOICE;
395  continue;
396  }
397  else if (!Q_stricmp(token, "weapon"))
398  {
399  sound.channel = CHAN_WEAPON;
400  continue;
401  }
402  else if (!Q_stricmp(token, "item"))
403  {
404  sound.channel = CHAN_ITEM;
405  continue;
406  }
407  else if (!Q_stricmp(token, "auto"))
408  {
409  sound.channel = CHAN_AUTO;
410  continue;
411  }
412  if (!Q_stricmp(token, "global"))
413  {
414  sound.attenuation = 0;
415  continue;
416  }
417  if (!Q_stricmp(token, "streaming"))
418  {
419  sound.streaming = qtrue;
420  continue;
421  }
422  if (!Q_stricmp(token, "looping"))
423  {
424  sound.looping = qtrue;
425  continue;
426  }
427  if (!Q_stricmp(token, "sound"))
428  {
429  if (scriptSound->numsounds >= MAX_SOUNDSCRIPT_SOUNDS)
430  {
431  CG_Error(S_COLOR_RED "CG_SoundParseSounds: Too many sounds for soundscript\n");
432  }
433 
434  token = COM_ParseExt(text, qtrue);
435 
436  if (!token[0])
437  {
438  CG_Error(S_COLOR_RED "CG_SoundParseSounds: empty token of sound file name\n");
439  }
440 
441  Q_strncpyz(scriptSound->sounds[scriptSound->numsounds].filename, token, sizeof(scriptSound->sounds[0].filename));
442  scriptSound->numsounds++;
443 
444  continue;
445  }
446  }

502  Com_sprintf(soundFiles[numSounds++], MAX_QPATH, "%s.sounds", cgs.rawmapname);
503 
504  // load and parse sound files
505  for (i = 0; i < numSounds; i++)
506  {
507  Com_sprintf(filename, sizeof(filename), "sound/scripts/%s", soundFiles[i]);
508  CG_Printf("...loading '%s'\n", filename);
509  len = trap_FS_FOpenFile(filename, &f, FS_READ);
510  if (len <= 0)
511  {
512  CG_Printf(S_COLOR_YELLOW "INFO: script sound file '%s' not found\n", filename);
513  sfalse++;
514  continue;
515  }
516  if (len > sizeof(bigTextBuffer))
517  {
518  trap_FS_FCloseFile(f);
519  CG_Printf(S_COLOR_RED "CG_SoundLoadSoundFiles: %s is too big, make it smaller (max = %i bytes)\n", filename, 100000);
520  sfalse++;
521  continue;
522  }
523  Com_Memset(bigTextBuffer, 0, sizeof(bigTextBuffer));
524  trap_FS_Read(bigTextBuffer, len, f);
525  trap_FS_FCloseFile(f);
526  CG_SoundParseSounds(filename, bigTextBuffer);
527  }
528  CG_Printf("...%i of %i sound scripts files loaded. Total sounds: %i\n", numSounds - sfalse, numSounds, numSoundScripts);
529  }
