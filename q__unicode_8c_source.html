47 int Q_UTF8_Width(const char *str)
48 {
49  int ewidth;
50  const unsigned char *s = (const unsigned char *)str;
51 
52  if (!str)
53  {
54  return 0;
55  }
56 
57  if (*s <= 0x7F)
58  {
59  ewidth = 0;
60  }
61  else if (0xC2 <= *s && *s <= 0xDF)
62  {
63  ewidth = 1;
64  }
65  else if (0xE0 <= *s && *s <= 0xEF)
66  {
67  ewidth = 2;
68  }
69  else if (0xF0 <= *s && *s <= 0xF4)
70  {
71  ewidth = 3;
72  }
73  else
74  {
75  ewidth = 0;
76  }
77 
78  for ( ; *s && ewidth > 0; s++, ewidth--)
79  ;
80 
81  return s - (const unsigned char *)str + 1;
82 }

318 static void shiftbitsright(unsigned char *p, unsigned long num, unsigned long by)
319 {
320  int step, off;
321  unsigned char *e;
322 
323  if (by >= num)
324  {
325  for (; num > 8; p++, num -= 8)
326  {
327  *p = 0;
328  }
329 
330  *p &= (~0x00) >> num;
331 
332  return;
333  }
334 
335  step = by / 8;
336  off = by % 8;
337 
338  for (e = p + (num + 7) / 8 - 1; e > p + step; e--)
339  {
340  *e = (*(e - step) >> off) | (*(e - step - 1) << (8 - off));
341  }
342 
343  *e = *(e - step) >> off;
344 
345  for (e = p; e < p + step; e++)
346  {
347  *e = 0;
348  }
349 }

414 char *Q_UTF8_Encode(unsigned long codepoint)
415 {
416  static char sbuf[2][5];
417  static int index = 0;
418  char *buf = sbuf[index++ & 1];
419 
420  if (codepoint <= 0x007F)
421  {
422  buf[0] = codepoint;
423  buf[1] = 0;
424  }
425  else if (0x0080 <= codepoint && codepoint <= 0x07FF)
426  {
427  buf[0] = 0xC0 | ((codepoint & 0x07C0) >> 6);
428  buf[1] = 0x80 | (codepoint & 0x003F);
429  buf[2] = 0;
430  }
431  else if (0x0800 <= codepoint && codepoint <= 0xFFFF)
432  {
433  buf[0] = 0xE0 | ((codepoint & 0xF000) >> 12);
434  buf[1] = 0x80 | ((codepoint & 0x0FC0) >> 6);
435  buf[2] = 0x80 | (codepoint & 0x003F);
436  buf[3] = 0;
437  }
438  else if (0x010000 <= codepoint && codepoint <= 0x10FFFF)
439  {
440  buf[0] = 0xF0 | ((codepoint & 0x1C0000) >> 18);
441  buf[1] = 0x80 | ((codepoint & 0x03F000) >> 12);
442  buf[2] = 0x80 | ((codepoint & 0x000FC0) >> 6);
443  buf[3] = 0x80 | (codepoint & 0x00003F);
444  buf[4] = 0;
445  }
446  else
447  {
448  buf[0] = 0;
449  }
450 
451  return buf;
452 }
