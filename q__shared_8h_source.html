229 // Vanilla PPC code, but since PPC has a reciprocal square root estimate instruction,
230 // runs *much* faster than calling sqrt(). We'll use two Newton-Raphson
231 // refinement steps to get bunch more precision in the 1/sqrt() value for very little cost.
232 // We'll then multiply 1/sqrt times the original value to get the sqrt.
233 // This is about 12.4 times faster than sqrt() and according to my testing (not exhaustive)
234 // it returns fairly accurate results (error below 1.0e-5 up to 100000.0 in 0.1 increments).
235 
236 // Note ET:L uses BSD Library functions for this
237 
238 static ID_INLINE float idSqrt(float x)
239 {
240  const float half = 0.5;
241  const float one = 1.0;
242  float B, y0, y1;
243 
244  // This'll NaN if it hits frsqrte. Handle both +0.0 and -0.0
245  if (Q_fabs(x) == 0.0)
246  {
247  return x;
248  }
249  B = x;
250 
251 #ifdef __GNUC__
252  asm ("frsqrte %0,%1" : "=f" (y0) : "f" (B));
253 #else
254  y0 = __frsqrte(B);
255 #endif

257 
258  y1 = y0 + half * y0 * (one - B * y0 * y0);
259 
260  // Second refinement step -- copy the output of the last step to the input of this step
261 
262  y0 = y1;
263  y1 = y0 + half * y0 * (one - B * y0 * y0);
264
