57 static qboolean MakeTextureMatrix(vec4_t texMat[2], vec4_t projection, decalVert_t *a, decalVert_t *b, decalVert_t *c)
58 {
59  int i, j;
60  float bb, s, t, d;
61  vec3_t pa, pb, pc;
62  vec3_t bary, origin, xyz;
63  vec3_t vecs[3], axis[3], lengths;
64 
65  // project triangle onto plane of projection
66  d = DotProduct(a->xyz, projection) - projection[3];
67  VectorMA(a->xyz, -d, projection, pa);
68  d = DotProduct(b->xyz, projection) - projection[3];
69  VectorMA(b->xyz, -d, projection, pb);
70  d = DotProduct(c->xyz, projection) - projection[3];
71  VectorMA(c->xyz, -d, projection, pc);
72 
73  // calculate barycentric basis for the triangle
74  bb = (b->st[0] - a->st[0]) * (c->st[1] - a->st[1]) - (c->st[0] - a->st[0]) * (b->st[1] - a->st[1]);
75  if (Q_fabs(bb) < 0.00000001f)
76  {
77  return qfalse;
78  }
79 
80  // calculate texture origin
81  s = 0.0;
82  t = 0.0;
83  bary[0] = ((b->st[0] - s) * (c->st[1] - t) - (c->st[0] - s) * (b->st[1] - t)) / bb;
84  bary[1] = ((c->st[0] - s) * (a->st[1] - t) - (a->st[0] - s) * (c->st[1] - t)) / bb;
85  bary[2] = ((a->st[0] - s) * (b->st[1] - t) - (b->st[0] - s) * (a->st[1] - t)) / bb;
86 
87  origin[0] = bary[0] * pa[0] + bary[1] * pb[0] + bary[2] * pc[0];
88  origin[1] = bary[0] * pa[1] + bary[1] * pb[1] + bary[2] * pc[1];
89  origin[2] = bary[0] * pa[2] + bary[1] * pb[2] + bary[2] * pc[2];
90 
91  // calculate s vector
92  s = 1.0;
93  t = 0.0;
94  bary[0] = ((b->st[0] - s) * (c->st[1] - t) - (c->st[0] - s) * (b->st[1] - t)) / bb;
95  bary[1] = ((c->st[0] - s) * (a->st[1] - t) - (a->st[0] - s) * (c->st[1] - t)) / bb;
96  bary[2] = ((a->st[0] - s) * (b->st[1] - t) - (b->st[0] - s) * (a->st[1] - t)) / bb;
97 
98  xyz[0] = bary[0] * pa[0] + bary[1] * pb[0] + bary[2] * pc[0];
99  xyz[1] = bary[0] * pa[1] + bary[1] * pb[1] + bary[2] * pc[1];
100  xyz[2] = bary[0] * pa[2] + bary[1] * pb[2] + bary[2] * pc[2];
101 
102  VectorSubtract(xyz, origin, vecs[0]);
103 
104  // calculate t vector
105  s = 0.0;
106  t = 1.0;
107  bary[0] = ((b->st[0] - s) * (c->st[1] - t) - (c->st[0] - s) * (b->st[1] - t)) / bb;
108  bary[1] = ((c->st[0] - s) * (a->st[1] - t) - (a->st[0] - s) * (c->st[1] - t)) / bb;
109  bary[2] = ((a->st[0] - s) * (b->st[1] - t) - (b->st[0] - s) * (a->st[1] - t)) / bb;
110 
111  xyz[0] = bary[0] * pa[0] + bary[1] * pb[0] + bary[2] * pc[0];
112  xyz[1] = bary[0] * pa[1] + bary[1] * pb[1] + bary[2] * pc[1];
113  xyz[2] = bary[0] * pa[2] + bary[1] * pb[2] + bary[2] * pc[2];
114 
115  VectorSubtract(xyz, origin, vecs[1]);
116 
117  // calcuate r vector
118  VectorScale(projection, -1.0f, vecs[2]);
119 
120  // calculate transform axis
121  for (i = 0; i < 3; i++)
122  lengths[i] = VectorNormalize2(vecs[i], axis[i]);
123 
124  for (i = 0; i < 2; i++)
125  for (j = 0; j < 3; j++)
126  texMat[i][j] = lengths[i] > 0.0f ? (axis[i][j] / lengths[i]) : 0.0f;
127  texMat[0][3] = a->st[0] - DotProduct(pa, texMat[0]);
128  texMat[1][3] = a->st[1] - DotProduct(pa, texMat[1]);
129 
130  // disco
131  return qtrue;
132 }

506 static void ChopWindingBehindPlane(int numInPoints, vec3_t inPoints[MAX_DECAL_VERTS],
507  int *numOutPoints, vec3_t outPoints[MAX_DECAL_VERTS], vec4_t plane, vec_t epsilon)
508 {
509  int i, j;
510  float dot;
511  float *p1, *p2, *clip;
512  float d;
513  float dists[MAX_DECAL_VERTS + 4];
514  int sides[MAX_DECAL_VERTS + 4];
515  int counts[3];
516 
517  // set initial count
518  *numOutPoints = 0;
519 
520  // don't clip if it might overflow
521  if (numInPoints >= MAX_DECAL_VERTS - 1)
522  {
523  return;
524  }
525 
526  // determine sides for each point
527  counts[SIDE_FRONT] = 0;
528  counts[SIDE_BACK] = 0;
529  counts[SIDE_ON] = 0;
530  for (i = 0; i < numInPoints; i++)
531  {
532  dists[i] = DotProduct(inPoints[i], plane) - plane[3];
533  if (dists[i] > epsilon)
534  {
535  sides[i] = SIDE_FRONT;
536  }
537  else if (dists[i] < -epsilon)
538  {
539  sides[i] = SIDE_BACK;
540  }
541  else
542  {
543  sides[i] = SIDE_ON;
544  }
545  counts[sides[i]]++;
546  }
547  sides[i] = sides[0];
548  dists[i] = dists[0];
549 
550  // all points on front
551  if (counts[SIDE_BACK] == 0)
552  {
553  return;
554  }
555 
556  // all points on back
557  if (counts[SIDE_FRONT] == 0)
558  {
559  *numOutPoints = numInPoints;
560  Com_Memcpy(outPoints, inPoints, numInPoints * sizeof(vec3_t));
561  return;
562  }
563 
564  // clip the winding
565  for (i = 0; i < numInPoints; i++)
566  {
567  p1 = inPoints[i];
568  clip = outPoints[*numOutPoints];
569 
570  if (sides[i] == SIDE_ON || sides[i] == SIDE_BACK)
571  {
572  VectorCopy(p1, clip);
573  (*numOutPoints)++;
574  }
575 
576  if (sides[i + 1] == SIDE_ON || sides[i + 1] == sides[i])
577  {
578  continue;
579  }
580 
581  // generate a split point
582  p2 = inPoints[(i + 1) % numInPoints];
583 
584  d = dists[i] - dists[i + 1];
585  if (d == 0.f)
586  {
587  dot = 0;
588  }
589  else
590  {
591  dot = dists[i] / d;
592  }
593 
594  // clip xyz
595  clip = outPoints[*numOutPoints];
596  for (j = 0; j < 3; j++)
597  clip[j] = p1[j] + dot * (p2[j] - p1[j]);
598 
599  (*numOutPoints)++;
600  }
601 }

612  bmodel_t *bmodel)
613 {

624  }
