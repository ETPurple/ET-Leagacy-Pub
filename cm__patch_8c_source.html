193 static qboolean CM_NeedsSubdivision(vec3_t a, vec3_t b, vec3_t c)
194 {
195  vec3_t cmid;
196  vec3_t lmid;
197  vec3_t delta;
198  float dist;
199  int i;
200 
201  // calculate the linear midpoint
202  for (i = 0 ; i < 3 ; i++)
203  {
204  lmid[i] = 0.5f * (a[i] + c[i]);
205  }
206 
207  // calculate the exact curve midpoint
208  for (i = 0 ; i < 3 ; i++)
209  {
210  cmid[i] = 0.5f * (0.5f * (a[i] + b[i]) + 0.5f * (b[i] + c[i]));
211  }
212 
213  // see if the curve is far enough away from the linear mid
214  VectorSubtract(cmid, lmid, delta);
215  dist = vec3_length(delta);
216 
217  return dist >= SUBDIVIDE_DISTANCE;

345 static void CM_SubdivideGridColumns(cGrid_t *grid)
346 {
347  int i, j, k;
348 
349  for (i = 0 ; i < grid->width - 2 ; )
350  {
351  // grid->points[i][x] is an interpolating control point
352  // grid->points[i+1][x] is an aproximating control point
353  // grid->points[i+2][x] is an interpolating control point
354 
355  // first see if we can collapse the aproximating collumn away
356  for (j = 0 ; j < grid->height ; j++)
357  {
358  if (CM_NeedsSubdivision(grid->points[i][j], grid->points[i + 1][j], grid->points[i + 2][j]))
359  {
360  break;
361  }
362  }
363  if (j == grid->height)
364  {
365  // all of the points were close enough to the linear midpoints
366  // that we can collapse the entire column away
367  for (j = 0 ; j < grid->height ; j++)
368  {
369  // remove the column
370  for (k = i + 2 ; k < grid->width ; k++)
371  {
372  VectorCopy(grid->points[k][j], grid->points[k - 1][j]);
373  }
374  }
375 
376  grid->width--;
377 
378  // go to the next curve segment
379  i++;
380  continue;
381  }
382 
383  // we need to subdivide the curve
384  for (j = 0 ; j < grid->height ; j++)
385  {
386  vec3_t prev, mid, next;
387 
388  // save the control points now
389  VectorCopy(grid->points[i][j], prev);
390  VectorCopy(grid->points[i + 1][j], mid);
391  VectorCopy(grid->points[i + 2][j], next);
392 
393  // make room for two additional columns in the grid
394  // columns i+1 will be replaced, column i+2 will become i+4
395  // i+1, i+2, and i+3 will be generated
396  for (k = grid->width - 1 ; k > i + 1 ; k--)
397  {
398  VectorCopy(grid->points[k][j], grid->points[k + 2][j]);
399  }
400 
401  // generate the subdivided points
402  CM_Subdivide(prev, mid, next, grid->points[i + 1][j], grid->points[i + 2][j], grid->points[i + 3][j]);
403  }
404 
405  grid->width += 2;
406 
407  // the new aproximating point at i+1 may need to be removed
408  // or subdivided farther, so don't advance i
409  }
410 }

980  {

1004  {
