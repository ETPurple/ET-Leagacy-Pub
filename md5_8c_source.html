169 static void MD5Update(struct MD5Context *ctx, unsigned char const *buf,
170  unsigned len)
171 {
172  uint32_t t;
173 
174  /* Update bitcount */
175 
176  t = ctx->bits[0];
177  if ((ctx->bits[0] = t + (( uint32_t ) len << 3)) < t)
178  {
179  ctx->bits[1]++; /* Carry from low to high */
180  }
181  ctx->bits[1] += len >> 29;
182 
183  t = (t >> 3) & 0x3f; /* Bytes already in shsInfo->data */
184 
185  /* Handle any leading odd-sized chunks */
186 
187  if (t)
188  {
189  unsigned char *p = ( unsigned char * ) ctx->in + t;
190 
191  t = 64 - t;
192  if (len < t)
193  {
194  Com_Memcpy(p, buf, len);
195  return;
196  }
197  Com_Memcpy(p, buf, t);
198  byteReverse(ctx->in, 16);
199  MD5Transform(ctx->buf, ( uint32_t * ) ctx->in);
200  buf += t;
201  len -= t;
202  }
203  /* Process data in 64-byte chunks */
204 
205  while (len >= 64)
206  {
207  Com_Memcpy(ctx->in, buf, 64);
208  byteReverse(ctx->in, 16);
209  MD5Transform(ctx->buf, ( uint32_t * ) ctx->in);
210  buf += 64;
211  len -= 64;
212  }
213 
214  /* Handle any remaining bytes of data. */
215 
216  Com_Memcpy(ctx->in, buf, len);
217 }

221  * Final wrapup - pad to 64-byte boundary with the bit pattern
222  * 1 0* (64-bit count of bits processed, MSB-first)
223  */
224 static void MD5Final(struct MD5Context *ctx, unsigned char *digest)
225 {
226  unsigned count;
227  unsigned char *p;
228 
229  /* Compute number of bytes mod 64 */
230  count = (ctx->bits[0] >> 3) & 0x3F;
231 
232  /* Set the first char of padding to 0x80. This is safe since there is
233  always at least one byte free */
234  p = ctx->in + count;
235  *p++ = 0x80;
236 
237  /* Bytes of padding needed to make 64 bytes */
238  count = 64 - 1 - count;
239 
240  /* Pad out to 56 mod 64 */
241  if (count < 8)
242  {
243  /* Two lots of padding: Pad the first block to 64 bytes */
244  Com_Memset(p, 0, count);
245  byteReverse(ctx->in, 16);
246  MD5Transform(ctx->buf, ( uint32_t * ) ctx->in);
247 
248  /* Now fill the next block with 56 bytes */
249  Com_Memset(ctx->in, 0, 56);
250  }
251  else
252  {
253  /* Pad block to 56 bytes */
254  Com_Memset(p, 0, count - 8);
255  }
256  byteReverse(ctx->in, 14);
257 
258  /* Append length in bits and transform */
259  (( uint32_t * ) ctx->in)[14] = ctx->bits[0];
260  (( uint32_t * ) ctx->in)[15] = ctx->bits[1];
261 
262  MD5Transform(ctx->buf, ( uint32_t * ) ctx->in);
263  byteReverse(( unsigned char * ) ctx->buf, 4);
264 
265  if (digest != NULL)
266  {
267  Com_Memcpy(digest, ctx->buf, 16);
268  }
269  Com_Memset(ctx, 0, sizeof(*ctx)); /* In case it's sensitive */
270 }
