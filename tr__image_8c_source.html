303 static void ResampleTexture(unsigned *in, int inwidth, int inheight, unsigned *out,
304  int outwidth, int outheight)
305 {
306  int i, j;
307  unsigned *inrow, *inrow2;
308  unsigned frac, fracstep;
309  unsigned p1[2048], p2[2048];
310  byte *pix1, *pix2, *pix3, *pix4;
311 
312  if (outwidth > 2048)
313  {
314  Ren_Drop("ResampleTexture: max width");
315  }
316 
317  if (outwidth < 1)
318  {
319  outwidth = 1;
320  }
321 
322  fracstep = inwidth * 0x10000 / outwidth;
323 
324  frac = fracstep >> 2;
325  for (i = 0 ; i < outwidth ; i++)
326  {
327  p1[i] = 4 * (frac >> 16);
328  frac += fracstep;
329  }
330  frac = 3 * (fracstep >> 2);
331  for (i = 0 ; i < outwidth ; i++)
332  {
333  p2[i] = 4 * (frac >> 16);
334  frac += fracstep;
335  }
336 
337  for (i = 0 ; i < outheight ; i++, out += outwidth)
338  {
339  inrow = in + inwidth * (int)((i + 0.25) * inheight / outheight);
340  inrow2 = in + inwidth * (int)((i + 0.75) * inheight / outheight);
341  //frac = fracstep >> 1; // FIXME: never read
342  for (j = 0 ; j < outwidth ; j++)
343  {
344  pix1 = (byte *)inrow + p1[j];
345  pix2 = (byte *)inrow + p2[j];
346  pix3 = (byte *)inrow2 + p1[j];
347  pix4 = (byte *)inrow2 + p2[j];
348  ((byte *)(out + j))[0] = (pix1[0] + pix2[0] + pix3[0] + pix4[0]) >> 2;
349  ((byte *)(out + j))[1] = (pix1[1] + pix2[1] + pix3[1] + pix4[1]) >> 2;
350  ((byte *)(out + j))[2] = (pix1[2] + pix2[2] + pix3[2] + pix4[2]) >> 2;
351  ((byte *)(out + j))[3] = (pix1[3] + pix2[3] + pix3[3] + pix4[3]) >> 2;
352  }
