393 qboolean G_smvRunCamera(gentity_t *ent)
394 {
395  int id = ent->TargetFlag;
396  int chargeTime, sprintTime, hintTime, weapHeat;
397  playerState_t *tps, *ps;
398 
399  // Opt out if not a real MV portal
400  if (ent->tagParent == NULL || ent->tagParent->client == NULL)
401  {
402  return qfalse;
403  }
404 
405  if ((ps = &ent->tagParent->client->ps) == NULL)
406  {
407  return qfalse;
408  }
409 
410  // If viewing client is no longer connected, delete this camera
411  if (ent->tagParent->client->pers.connected != CON_CONNECTED)
412  {
413  G_FreeEntity(ent);
414  return qtrue;
415  }
416 
417  // Also remove if the target player is no longer in the game playing
418  if (ent->target_ent->client->pers.connected != CON_CONNECTED ||
419  ent->target_ent->client->sess.sessionTeam == TEAM_SPECTATOR)
420  {
421  G_smvLocateEntityInMVList(ent->tagParent, ent->target_ent - g_entities, qtrue);
422  return qtrue;
423  }
424 
425  // Seems that depending on player's state, we pull from either r.currentOrigin, or s.origin
426  // if(!spec) then use: r.currentOrigin
427  // if(spec) then use: s.origin
428  //
429  // This is true for both the portal origin and its target (origin2)
430  //
431  VectorCopy(ent->tagParent->s.origin, ent->s.origin);
432  G_SetOrigin(ent, ent->s.origin);
433  VectorCopy(ent->target_ent->r.currentOrigin, ent->s.origin2);
434  trap_LinkEntity(ent);
435 
436  // Only allow client ids 0 to (MAX_MVCLIENTS-1) to be updated with extra info
437  if (id >= MAX_MVCLIENTS)
438  {
439  return qtrue;
440  }
441 
442  tps = &ent->target_ent->client->ps;
443 
444  if (tps->stats[STAT_PLAYER_CLASS] == PC_ENGINEER)
445  {
446  chargeTime = g_engineerChargeTime.value;
447  }
448  else if (tps->stats[STAT_PLAYER_CLASS] == PC_MEDIC)
449  {
450  chargeTime = g_medicChargeTime.value;
451  }
452  else if (tps->stats[STAT_PLAYER_CLASS] == PC_FIELDOPS)
453  {
454  chargeTime = g_fieldopsChargeTime.value;
455  }
456  else if (tps->stats[STAT_PLAYER_CLASS] == PC_COVERTOPS)
457  {
458  chargeTime = g_covertopsChargeTime.value;
459  }
460  else
461  {
462  chargeTime = g_soldierChargeTime.value;
463  }
464 
465  chargeTime = (level.time - tps->classWeaponTime >= (int)chargeTime) ? 0 : (1 + floor(15.0f * (float)(level.time - tps->classWeaponTime) / chargeTime));
466  sprintTime = (ent->target_ent->client->pmext.sprintTime >= 20000) ? 0.0f : (1 + floor(7.0f * (float)ent->target_ent->client->pmext.sprintTime / 20000.0f));
467  weapHeat = floor((float)tps->curWeapHeat * 15.0f / 255.0f);
468  hintTime = (tps->serverCursorHint != HINT_BUILD && (tps->serverCursorHintVal >= 255 || tps->serverCursorHintVal == 0)) ?
469  0 : (1 + floor(15.0f * (float)tps->serverCursorHintVal / 255.0f));
470 
471  // (Remaining bits)
472  // ammo : 0
473  // ammo-1 : 0
474  // ammiclip : 0
475  // ammoclip-1: 16
476  id = MAX_WEAPONS - 1 - (id * 2);
477 
478  if (tps->pm_flags & PMF_LIMBO)
479  {
480  ps->ammo[id] = 0;
481  ps->ammo[id - 1] = 0;
482  ps->ammoclip[id - 1] = 0;
483  }
484  else
485  {
486  ps->ammo[id] = (((ent->target_ent->health > 0) ? ent->target_ent->health : 0) & 0xFF); // Meds up to 140 :(
487  ps->ammo[id] |= (hintTime & 0x0F) << 8; // 4 bits for work on current item (dynamite, weapon repair, etc.)
488  ps->ammo[id] |= (weapHeat & 0x0F) << 12; // 4 bits for weapon heat info
489 
490  ps->ammo[id - 1] = tps->ammo[GetWeaponTableData(tps->weapon)->ammoIndex] & 0x3FF; // 11 bits needed to cover 1500 Venom ammo
491  ps->ammo[id - 1] |= (BG_simpleWeaponState(tps->weaponstate) & 0x03) << 11; // 2 bits for current weapon state
492  ps->ammo[id - 1] |= ((tps->persistant[PERS_HWEAPON_USE]) ? 1 : 0) << 13; // 1 bit for mg42 use
493  ps->ammo[id - 1] |= (BG_simpleHintsCollapse(tps->serverCursorHint, hintTime) & 0x03) << 14; // 2 bits for cursor hints
494 
495  //G_Printf("tps->hint: %d, dr: %d, collapse: %d\n", tps->serverCursorHint, HINT_DOOR_ROTATING, G_simpleHintsCollapse(tps->serverCursorHint, hintTime));
496 
497  ps->ammoclip[id - 1] = tps->ammoclip[GetWeaponTableData(tps->weapon)->clipIndex] & 0x1FF; // 9 bits to cover 500 Venom ammo clip
498  ps->ammoclip[id - 1] |= (chargeTime & 0x0F) << 9; // 4 bits for weapon charge time
499  ps->ammoclip[id - 1] |= (sprintTime & 0x07) << 13; // 3 bits for fatigue
500  }
501 
502  return qtrue;
503 }
