80 class QInputbuf : public std::streambuf
81 {
82 private:
83  static const size_t BUFFER_SIZE = 8192;
84  fileHandle_t fileHandle;
85  char buffer[BUFFER_SIZE];
86  size_t putBack;
87 public:
88  QInputbuf(const std::string& filename) : putBack(1)
89  {
90  char *end = buffer + BUFFER_SIZE - putBack;
91 
92  setg(end, end, end);
93 
94  if (FS_FOpenFileRead(filename.c_str(), &fileHandle, qfalse) <= 0)
95  {
96  Com_Printf("Warning: can't open or read file '%s' \n", filename.c_str());
97  }
98  }
99 
100  ~QInputbuf()
101  {
102  if (fileHandle)
103  {
104  FS_FCloseFile(fileHandle);
105  }
106  }
107 
108  int underflow()
109  {
110  if (gptr() < egptr()) // buffer not exhausted
111  {
112  return traits_type::to_int_type(*gptr());
113  }
114 
115  if (!fileHandle)
116  {
117  return traits_type::eof();
118  }
119 
120  char *base = buffer;
121  char *start = base;
122 
123  if (eback() == base)
124  {
125  // Make arrangements for putback characters
126  memmove(base, egptr() - putBack, putBack);
127  start += putBack;
128  }
129 
130  size_t n = FS_Read(start, BUFFER_SIZE - (start - base), fileHandle);
131 
132  if (n == 0)
133  {
134  return traits_type::eof();
135  }
136 
137  // Set buffer pointers
138  setg(base, start, start + n);
139 
140  return traits_type::to_int_type(*gptr());
141  }
142 };

217  FL_FindLocale(&locale);

219  // Do not change the language if it is already set
220  if (!cl_lang->string[0])

239  dictionary_mod.set_filesystem(std::auto_ptr<tinygettext::FileSystem>(new QFileSystem));
240 #endif
241  dictionary.add_directory("locale/client");
242  dictionary_mod.add_directory("locale/mod");
243 
244  languages = dictionary.get_languages();
245  Com_Printf("Available client translations:");
246  for (std::set<tinygettext::Language>::iterator p = languages.begin(); p != languages.end(); p++)
247  {

251 
252  languages_mod = dictionary_mod.get_languages();

258  Com_Printf("\n");
259 
260  I18N_SetLanguage(cl_lang->string);

262 }

302 {
