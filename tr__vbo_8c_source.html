118 VBO_t *R_CreateVBO2(const char *name, int numVertexes, srfVert_t *verts, unsigned int stateBits, vboUsage_t usage)
119 {
120  VBO_t *vbo;
121  int i, j;
122  byte *data;
123  int dataSize;
124  int dataOfs;
125  int glUsage;
126  unsigned int bits;
127 
128  switch (usage)
129  {
130  case VBO_USAGE_STATIC:
131  glUsage = GL_STATIC_DRAW;
132  break;
133  case VBO_USAGE_DYNAMIC:
134  glUsage = GL_DYNAMIC_DRAW;
135  break;
136  default:
137  glUsage = 0;
138  Ren_Fatal("bad vboUsage_t given: %i", usage);
139  //break;
140  }
141 
142  if (!numVertexes)
143  {
144  return NULL;
145  }
146 
147  if (strlen(name) >= MAX_QPATH)
148  {
149  Ren_Drop("R_CreateVBO2: \"%s\" is too long\n", name);
150  }
151 
152  // make sure the render thread is stopped
153  R_IssuePendingRenderCommands();
154 
155  vbo = (VBO_t *)ri.Hunk_Alloc(sizeof(*vbo), h_low);
156  Com_AddToGrowList(&tr.vbos, vbo);
157 
158  Q_strncpyz(vbo->name, name, sizeof(vbo->name));
159 
160  vbo->ofsXYZ = 0;
161  vbo->ofsTexCoords = 0;
162  vbo->ofsLightCoords = 0;
163  vbo->ofsBinormals = 0;
164  vbo->ofsTangents = 0;
165  vbo->ofsNormals = 0;
166  vbo->ofsColors = 0;
167  //vbo->ofsPaintColors = 0;
168  //vbo->ofsLightDirections = 0;
169  vbo->ofsBoneIndexes = 0;
170  vbo->ofsBoneWeights = 0;
171 
172  vbo->sizeXYZ = 0;
173  vbo->sizeTangents = 0;
174  vbo->sizeBinormals = 0;
175  vbo->sizeNormals = 0;
176 
177  // size VBO
178  dataSize = 0;
179  bits = stateBits;
180  while (bits)
181  {
182  if (bits & 1)
183  {
184  dataSize += sizeof(vec4_t);
185  }
186  bits >>= 1;
187  }
188  dataSize *= numVertexes;
189  data = (byte *)ri.Hunk_AllocateTempMemory(dataSize);
190  dataOfs = 0;
191 
192  // since this is all float, point tmp directly into data
193  // 2-entry -> { memb[0], memb[1], 0, 1 }
194  // 3-entry -> { memb[0], memb[1], memb[2], 1 }
195 #define VERTEXSIZE(memb) (sizeof(verts->memb) / sizeof(verts->memb[0]))
196 #define VERTEXCOPY(memb) \
197  do { \
198  vec_t *tmp = (vec_t *) (data + dataOfs); \
199  for (i = 0; i < numVertexes; i++) \
200  { \
201  for (j = 0; j < VERTEXSIZE(memb); j++) { *tmp++ = verts[i].memb[j]; } \
202  if (VERTEXSIZE(memb) < 3) { *tmp++ = 0; } \
203  if (VERTEXSIZE(memb) < 4) { *tmp++ = 1; } \
204  } \
205  dataOfs += i * sizeof(vec4_t); \
206  } while (0)
207 
208  if (stateBits & ATTR_POSITION)
209  {
210  vbo->ofsXYZ = dataOfs;
211  VERTEXCOPY(xyz);
212  }
213 
214  // feed vertex texcoords
215  if (stateBits & ATTR_TEXCOORD)
216  {
217  vbo->ofsTexCoords = dataOfs;
218  VERTEXCOPY(st);
219  }
220 
221  // feed vertex lightmap texcoords
222  if (stateBits & ATTR_LIGHTCOORD)
223  {
224  vbo->ofsLightCoords = dataOfs;
225  VERTEXCOPY(lightmap);
226  }
227 
228  // feed vertex tangents
229  if (stateBits & ATTR_TANGENT)
230  {
231  vbo->ofsTangents = dataOfs;
232  VERTEXCOPY(tangent);
233  }
234 
235  // feed vertex binormals
236  if (stateBits & ATTR_BINORMAL)
237  {
238  vbo->ofsBinormals = dataOfs;
239  VERTEXCOPY(binormal);
240  }
241 
242  // feed vertex normals
243  if (stateBits & ATTR_NORMAL)
244  {
245  vbo->ofsNormals = dataOfs;
246  VERTEXCOPY(normal);
247  }
248 
249  // feed vertex colors
250  if (stateBits & ATTR_COLOR)
251  {
252  vbo->ofsColors = dataOfs;
253  VERTEXCOPY(lightColor);
254  }
255 
256  vbo->vertexesSize = dataSize;
257  vbo->vertexesNum = numVertexes;
258 
259  glGenBuffers(1, &vbo->vertexesVBO);
260 
261  glBindBuffer(GL_ARRAY_BUFFER, vbo->vertexesVBO);
262  glBufferData(GL_ARRAY_BUFFER, dataSize, data, glUsage);
263 
264  glBindBuffer(GL_ARRAY_BUFFER, 0);
265 
266  GL_CheckErrors();
267 
268  ri.Hunk_FreeTempMemory(data);
269 
270  return vbo;
271 }
