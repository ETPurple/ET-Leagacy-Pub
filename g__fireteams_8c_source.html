518 void G_RemoveClientFromFireteams(int entityNum, qboolean update, qboolean print)
519 {
520  fireteamData_t *ft;
521  int i;
522 
523  if ((entityNum < 0 || entityNum >= MAX_CLIENTS) || !g_entities[entityNum].client)
524  {
525  G_Error("G_RemoveClientFromFireteams: invalid client\n");
526  }
527 
528  if (G_IsOnFireteam(entityNum, &ft))
529  {
530  int j, firstHuman;
531 
532  for (i = 0; i < MAX_FIRETEAM_MEMBERS && i < g_maxclients.integer; ++i)
533  {
534  if (ft->joinOrder[i] == entityNum)
535  {
536  if (i == 0)
537  {
538  if (ft->joinOrder[1] == -1)
539  {
540  ft->inuse = qfalse;
541  ft->ident = -1;
542  }
543  else
544  {
545  // disband the fireteam if only bots left in
546  if (G_OnlyBotsInFireteam(ft, entityNum, &firstHuman))
547  {
548  // empty the fireteam

553 #endif
554  ft->joinOrder[j] = -1;
555  }
556  ft->inuse = qfalse;
557  ft->ident = -1;
558  G_UpdateFireteamConfigString(ft);
559  return;
560  }
561  else
562  {
563  // try to pick a bot as fireteam leader
564  if (g_entities[(int)(ft->joinOrder[1])].r.svFlags & SVF_BOT)
565  {
566  if (firstHuman != -1)
567  {
568  // swap first human with first bot
569  int tmp = ft->joinOrder[1];
570 
571  ft->joinOrder[1] = ft->joinOrder[firstHuman];
572  ft->joinOrder[firstHuman] = tmp;
573  }
574  }
575  else
576  {
577  firstHuman = 1;
578  }
579  // inform client of promotion to leader
580  if (firstHuman != -1)
581  {
582  trap_SendServerCommand(ft->joinOrder[firstHuman], "cpm \"You are now the leader of your fireteam\"");
583  }
584  }
585  }
586  }
587  for (j = i; j < g_maxclients.integer - 1; j++)
588  {
589  ft->joinOrder[j] = ft->joinOrder[j + 1];
590  }
591  ft->joinOrder[g_maxclients.integer - 1] = -1;
592 
593  break;
594  }
595  }
596  }
597  else
598  {
599  return;
600  }
601 
602 #ifdef FEATURE_OMNIBOT
603  Bot_Event_LeftFireTeam(entityNum);
604 #endif
605 
606  if (print)
607  {
608  for (i = 0; i < MAX_CLIENTS; i++)
609  {
610  if (ft->joinOrder[i] == -1)
611  {
612  break;
613  }
614 
615  trap_SendServerCommand(ft->joinOrder[i], va("cpm \"%s ^7has left the fireteam\"", level.clients[entityNum].pers.netname));
616  }
617  }
618 
619  if (update)
620  {
621  G_UpdateFireteamConfigString(ft);
622  }
623 }
