82 void R_RenderShadowEdges(void)
83 {
84  int i;
85  int c, c2;
86  int j, k;
87  int i2;
88  // int c_edges = 0, c_rejected = 0; // TODO: remove ?
89  int hit[2];
90 
91  // an edge is NOT a silhouette edge if its face doesn't face the light,
92  // or if it has a reverse paired edge that also faces the light.
93  // A well behaved polyhedron would have exactly two faces for each edge,
94  // but lots of models have dangling edges or overfanned edges
95 
96  for (i = 0 ; i < tess.numVertexes ; i++)
97  {
98  c = numEdgeDefs[i];
99  for (j = 0 ; j < c ; j++)
100  {
101  if (!edgeDefs[i][j].facing)
102  {
103  continue;
104  }
105 
106  hit[0] = 0;
107  hit[1] = 0;
108 
109  i2 = edgeDefs[i][j].i2;
110  c2 = numEdgeDefs[i2];
111  for (k = 0 ; k < c2 ; k++)
112  {
113  if (edgeDefs[i2][k].i2 == i)
114  {
115  hit[edgeDefs[i2][k].facing]++;
116  }
117  }
118 
119  // if it doesn't share the edge with another front facing
120  // triangle, it is a sil edge
121  if (hit[1] == 0)
122  {
123  qglBegin(GL_TRIANGLE_STRIP);
124  qglVertex3fv(tess.xyz[i]);
125  qglVertex3fv(tess.xyz[i + tess.numVertexes]);
126  qglVertex3fv(tess.xyz[i2]);
127  qglVertex3fv(tess.xyz[i2 + tess.numVertexes]);
128  qglEnd();
129  // c_edges++;
130  }
131  /*
132  else
133  {
134  c_rejected++;
135  }
136  */
137  }
138  }

150 void RB_ShadowTessEnd(void)
151 {
152  int i;
153  int numTris;
154  vec3_t lightDir;
155 
156  // we can only do this if we have enough space in the vertex buffers
157  if (tess.numVertexes >= SHADER_MAX_VERTEXES / 2)
158  {
159  return;
160  }
161 
162  if (glConfig.stencilBits < 4)
163  {
164  return;
165  }
166 
167  VectorCopy(backEnd.currentEntity->lightDir, lightDir);
168 
169  // project vertexes away from light direction
170  for (i = 0 ; i < tess.numVertexes ; i++)
171  {
172  VectorMA(tess.xyz[i], -512, lightDir, tess.xyz[i + tess.numVertexes]);
173  }
174 
175  // decide which triangles face the light
176  Com_Memset(numEdgeDefs, 0, 4 * tess.numVertexes);
177 
178  numTris = tess.numIndexes / 3;
179 
180  {
181  int i1, i2, i3;
182  vec3_t d1, d2, normal;
183  float *v1, *v2, *v3;
184  float d;
185 
186  for (i = 0 ; i < numTris ; i++)
187  {
188  i1 = tess.indexes[i * 3 + 0];
189  i2 = tess.indexes[i * 3 + 1];
190  i3 = tess.indexes[i * 3 + 2];
191 
192  v1 = tess.xyz[i1];
193  v2 = tess.xyz[i2];
194  v3 = tess.xyz[i3];
195 
196  VectorSubtract(v2, v1, d1);
197  VectorSubtract(v3, v1, d2);
198  CrossProduct(d1, d2, normal);
199 
200  d = DotProduct(normal, lightDir);
201  if (d > 0)
202  {
203  facing[i] = 1;
204  }
205  else
206  {
207  facing[i] = 0;
208  }
209 
210  // create the edges
211  R_AddEdgeDef(i1, i2, facing[i]);
212  R_AddEdgeDef(i2, i3, facing[i]);
213  R_AddEdgeDef(i3, i1, facing[i]);
214  }
215  }
216 
217 
218  // draw the silhouette edges
219 
220  GL_Bind(tr.whiteImage);
221  qglEnable(GL_CULL_FACE);
222  GL_State(GLS_SRCBLEND_ONE | GLS_DSTBLEND_ZERO);
223  qglColor3f(0.2f, 0.2f, 0.2f);
224 
225  // don't write to the color buffer
226  qglColorMask(GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE);
227 
228  qglEnable(GL_STENCIL_TEST);
229  qglStencilFunc(GL_ALWAYS, 1, 255);
230 
231  // mirrors have the culling order reversed
232  if (backEnd.viewParms.isMirror)
233  {
234  qglCullFace(GL_FRONT);
235  qglStencilOp(GL_KEEP, GL_KEEP, GL_INCR);
236 
237  R_RenderShadowEdges();
238 
239  qglCullFace(GL_BACK);
240  qglStencilOp(GL_KEEP, GL_KEEP, GL_DECR);
241 
242  R_RenderShadowEdges();
243  }
244  else
245  {
246  qglCullFace(GL_BACK);
247  qglStencilOp(GL_KEEP, GL_KEEP, GL_INCR);
248 
249  R_RenderShadowEdges();
250 
251  qglCullFace(GL_FRONT);
252  qglStencilOp(GL_KEEP, GL_KEEP, GL_DECR);
253 
254  R_RenderShadowEdges();
255  }
256 
257  // reenable writing to the color buffer
258  qglColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
259 }

305 void RB_ProjectionShadowDeform(void)
306 {
307  float *xyz = (float *)tess.xyz;
308  int i;
309  float h;
310  vec3_t ground;
311  vec3_t light;
312  float groundDist;
313  float d;
314  vec3_t lightDir;
315 
316  ground[0] = backEnd.orientation.axis[0][2];
317  ground[1] = backEnd.orientation.axis[1][2];
318  ground[2] = backEnd.orientation.axis[2][2];
319 
320  groundDist = backEnd.orientation.origin[2] - backEnd.currentEntity->e.shadowPlane;
321 
322  VectorCopy(backEnd.currentEntity->lightDir, lightDir);
323  d = DotProduct(lightDir, ground);
324  // don't let the shadows get too long or go negative
325  if (d < 0.5f)
326  {
327  VectorMA(lightDir, (0.5f - d), ground, lightDir);
328  d = DotProduct(lightDir, ground);
329  }
330  d = 1.0f / d;
331 
332  light[0] = lightDir[0] * d;
333  light[1] = lightDir[1] * d;
334  light[2] = lightDir[2] * d;
335 
336  for (i = 0; i < tess.numVertexes; i++, xyz += 4)
337  {
338  h = DotProduct(xyz, ground) + groundDist;
339 
340  xyz[0] -= light[0] * h;
341  xyz[1] -= light[1] * h;
342  xyz[2] -= light[2] * h;
343  }
344 }
