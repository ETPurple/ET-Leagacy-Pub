66 qboolean PlanesGetIntersectionPoint(const vec4_t plane1, const vec4_t plane2, const vec4_t plane3, vec3_t out)
67 {
68  // http://www.cgafaq.info/wiki/Intersection_of_three_planes
69 
70  vec3_t n1, n2, n3;
71  vec3_t n1n2, n2n3, n3n1;
72  vec_t denom;
73 
74  VectorNormalize2(plane1, n1);
75  VectorNormalize2(plane2, n2);
76  VectorNormalize2(plane3, n3);
77 
78  CrossProduct(n1, n2, n1n2);
79  CrossProduct(n2, n3, n2n3);
80  CrossProduct(n3, n1, n3n1);
81 
82  denom = DotProduct(n1, n2n3);
83 
84  // check if the denominator is zero (which would mean that no intersection is to be found
85  if (denom == 0.f)
86  {
87  // no intersection could be found, return <0,0,0>
88  VectorClear(out);
89  return qfalse;
90  }
91 
92  VectorClear(out);
93 
94  VectorMA(out, plane1[3], n2n3, out);
95  VectorMA(out, plane2[3], n3n1, out);
96  VectorMA(out, plane3[3], n1n2, out);
97 
98  VectorScale(out, 1.0f / denom, out);
99 
100  return qtrue;
101 }
