153 
154  if (cSeconds > 1000)
155  {
156  AP(va("cp \"^3Match resuming in ^1%d^3 seconds!\n\"", cSeconds / 1000));
157  think_next = level.time + 15000;
158  fFree = qfalse;
159  }
160  else
161  {
162  level.match_pause = PAUSE_UNPAUSING;
163  AP("print \"^3Match resuming in 10 seconds!\n\"");

166  }
167  }
168  break;
169  case DP_UNPAUSING:
170  if (level.match_pause == PAUSE_UNPAUSING)
171  {
172  int cSeconds = 11 * 1000 - (level.time - dpent->timestamp);
173 
174  if (cSeconds > 1000)
175  {
176  AP(va("cp \"^3Match resuming in ^1%d^3 seconds!\n\"", cSeconds / 1000));
177  think_next = level.time + 1000;
178  fFree = qfalse;
179  }
180  else
181  {
182  level.match_pause = PAUSE_NONE;
183  G_globalSound("sound/osp/fight.wav");

185  trap_SetConfigstring(CS_LEVEL_START_TIME, va("%i", level.startTime + level.timeDelta));
186  level.server_settings &= ~CV_SVS_PAUSE;
187  trap_SetConfigstring(CS_SERVERTOGGLES, va("%d", level.server_settings));
188  }
189  }
190  break;
191 #ifdef FEATURE_MULTIVIEW
192  case DP_MVSPAWN:
193  {
194  int i;
195  gentity_t *ent;
196 
197  for (i = 0; i < level.numConnectedClients; i++)

272 void G_addStats(gentity_t *targ, gentity_t *attacker, int damage, meansOfDeath_t mod)
273 {
274  if (!targ || !targ->client)
275  {
276  return;
277  }
278 
279  // Keep track of only active player-to-player interactions in a real game
280  if (
281 #ifndef DEBUG_STATS
282  g_gamestate.integer != GS_PLAYING ||
283 #endif
284  mod == MOD_SWITCHTEAM || (targ->client->ps.pm_flags & PMF_LIMBO))
285  {
286  return;
287  }
288 
289  // Special hack for intentional gibbage
290  if (targ->health <= 0 && targ->client->ps.pm_type == PM_DEAD)
291  {
292  if (attacker && attacker->client)
293  {
294  int x;
295 
296  x = attacker->client->sess.aWeaponStats[GetMODTableData(mod)->indexWeaponStat].atts--;
297 
298  if (x < 1)
299  {
300  attacker->client->sess.aWeaponStats[GetMODTableData(mod)->indexWeaponStat].atts = 1;
301  }
302 
303  if (targ->health <= FORCE_LIMBO_HEALTH)
304  {
305  if (targ->client->sess.sessionTeam != attacker->client->sess.sessionTeam)
306  {
307  attacker->client->sess.gibs++;
308  }
309  else
310  {
311  attacker->client->sess.team_gibs++;
312  }
313  }
314  }
315 
316  return;
317  }
318 
319  // G_Printf("mod: %d, Index: %d, dmg: %d\n", mod, G_weapStatIndex_MOD(mod), dmg_ref);
320 
321  // Selfkills only affect the player specifically
322  if (targ == attacker || !attacker || !attacker->client || mod == MOD_SUICIDE) // FIXME: inspect world kills count as selfkill?
323  {
324  if (targ->health <= 0)
325  {
326  targ->client->sess.self_kills++;
327  }
328 #ifdef DEBUG_STATS
329  if (!attacker || !attacker->client)
330 #endif
331  return;
332  }
333 
334  // Player team stats
335  if (targ->client->sess.sessionTeam == attacker->client->sess.sessionTeam)
336  {
337  attacker->client->sess.team_damage_given += damage;
338  targ->client->sess.team_damage_received += damage;
339 
340  if (targ->health <= 0)
341  {
342  attacker->client->sess.team_kills++;
343  }
344 #ifndef DEBUG_STATS
345  return;
346 #endif
347  }
348 
349  // General player stats
350  if (mod != MOD_SYRINGE)
351  {
352  attacker->client->sess.damage_given += damage;
353  targ->client->sess.damage_received += damage;
354 
355  if (targ->health <= 0)
356  {
357  attacker->client->sess.kills++;
358  targ->client->sess.deaths++;
359  }
360  }
361 
362  // Player weapon stats
363  if (damage > 0)
364  {
365  attacker->client->sess.aWeaponStats[GetMODTableData(mod)->indexWeaponStat].hits++;
366  }
367  if (targ->health <= 0)
368  {
369  attacker->client->sess.aWeaponStats[GetMODTableData(mod)->indexWeaponStat].kills++;
370  targ->client->sess.aWeaponStats[GetMODTableData(mod)->indexWeaponStat].deaths++;
371  }
372 }

486  strWeapInfo,

490  ,
491  strSkillRatingInfo
492 #endif
493  ));
