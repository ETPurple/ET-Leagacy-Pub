44 void alarmbox_updateparts(gentity_t *ent, qboolean matestoo)
45 {
46  gentity_t *t = NULL;
47 
48  if (!ent)
49  {
50  return;
51  }
52 
53  // update teammates
54  if (matestoo)
55  {
56  gentity_t *mate;
57 
58  for (mate = ent->teammaster; mate; mate = mate->teamchain)
59  {
60  if (mate == ent)
61  {
62  continue;
63  }
64 
65  if (!mate->active) // don't update dead alarm boxes, they stay dead
66  {
67  continue;
68  }
69 
70  if (!ent->active) // destroyed, so just turn teammates off
71  {
72  mate->s.frame = 0;
73  }
74  else
75  {
76  mate->s.frame = ent->s.frame;
77  }
78 
79  alarmbox_updateparts(mate, qfalse);
80  }
81  }
82 
83  // update lights
84  if (!ent->target)
85  {
86  return;
87  }
88 
89  while ((t = G_FindByTargetname(t, ent->target)) != NULL)
90  {
91  if (t == ent)
92  {
93  G_Printf("WARNING: Entity used itself.\n");
94  continue;
95  }
96 
97  if (!Q_stricmp(t->classname, "dlight")) // give the dlight the sound
98  {
99  t->soundLoop = ent->soundLoop;
100 
101  if (ent->s.frame == 1)
102  {
103  if (!t->r.linked)
104  {
105  G_UseEntity(t, ent, 0);
106  }
107  }
108  else if (t->r.linked)
109  {
110  G_UseEntity(t, ent, 0);
111  }
112  }
113  // alarmbox can tell script_trigger about activation
114  // (but don't trigger if dying, only activation)
115  else if (!Q_stricmp(t->classname, "target_script_trigger"))
116  {
117  if (ent->active) // not dead
118  {
119  G_UseEntity(t, ent, 0);
120  }
121  }
122  }
123 }
