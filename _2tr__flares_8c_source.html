129 {

132  vec3_t local;
133  vec4_t eye, clip, normalized, window;
134  float distBias = 512.0;
135  float distLerp = 0.5;
136  float d1 = 0.0f, d2 = 0.0f;
137 
138  backEnd.pc.c_flareAdds++;
139 
140  // if the point is off the screen, don't bother adding it
141  // calculate screen coordinates and depth
142  R_TransformModelToClip(point, backEnd.orientation.modelViewMatrix, backEnd.viewParms.projectionMatrix, eye, clip);
143 
144  // check to see if the point is completely off screen
145  for (i = 0; i < 3; i++)
146  {
147  if (clip[i] >= clip[3] || clip[i] <= -clip[3])
148  {
149  return;
150  }
151  }
152 
153  R_TransformClipToWindow(clip, &backEnd.viewParms, normalized, window);
154

156  window[1] < 0 || window[1] >= backEnd.viewParms.viewportHeight)
157  {
158  return; // shouldn't happen, since we check the clip[] above, except for FP rounding
159  }

162  //oldest = r_flareStructs;
163  for (f = r_activeFlares; f; f = f->next)
164  {
165  if (f->surface == surface && f->frameSceneNum == backEnd.viewParms.frameSceneNum &&
166  f->inPortal == backEnd.viewParms.isPortal)
167  {
168  break;
169  }
170  }
171 
172  // allocate a new one
173  if (!f)
174  {
175  if (!r_inactiveFlares)
176  {
177  // the list is completely full
178  return;
179  }
180 
181  f = r_inactiveFlares;
182  r_inactiveFlares = r_inactiveFlares->next;
183  f->next = r_activeFlares;
184  r_activeFlares = f;
185

187  f->frameSceneNum = backEnd.viewParms.frameSceneNum;
188  f->inPortal = backEnd.viewParms.isPortal;

191  }
192 
193  f->cgvisible = visible;
194 
195  if (f->addedFrame != backEnd.viewParms.frameCount - 1)
196  {
197  f->visible = qfalse;
198  f->fadeTime = backEnd.refdef.time - 2000;
199  }
200 
201  f->addedFrame = backEnd.viewParms.frameCount;
202  f->fogNum = fogNum;
203 
204  VectorCopy(color, f->color);
205 
206  d2 = -eye[2];
207  if (d2 > distBias)
208  {
209  if (d2 > (distBias * 2.0f))
210  {
211  d2 = (distBias * 2.0f);
212  }
213  d2 -= distBias;
214  d2 = 1.0f - (d2 * (1.0f / distBias));
215  d2 *= distLerp;
216  }
217  else
218  {
219  d2 = distLerp;
220  }
221 
222  // fade the intensity of the flare down as the
223  // light surface turns away from the viewer
224  if (normal)
225  {
226  VectorSubtract(backEnd.viewParms.orientation.origin, point, local);
227  VectorNormalize(local);
228  d1 = DotProduct(local, normal);
229  d1 *= (1.0f - distLerp);
230  d1 += d2;
231  }
232 
233  VectorScale(f->color, d1, f->color);
234 
235  // save info needed to test
236  f->windowX = backEnd.viewParms.viewportX + window[0];
237  f->windowY = backEnd.viewParms.viewportY + window[1];
238 
239  f->eyeZ = eye[2];
240 }

408 void RB_RenderFlare(flare_t *f)
409 {
410  float size, distance, intensity, factor;
411  vec3_t color;
412  int iColor[3];
413 
414  backEnd.pc.c_flareRenders++;
415 
416  if (f->isCorona) // corona case
417  {
418  VectorScale(colorWhite, f->drawIntensity, color);
419  iColor[0] = color[0];
420  iColor[1] = color[1];
421  iColor[2] = color[2];
422 
423  size = backEnd.viewParms.viewportWidth * (r_flareSize->value / 640.0f + 8 / -f->eyeZ);
424  }
425  else
426  {
427  /*
428  As flare sizes stay nearly constant with increasing distance we must decrease the intensity
429  to achieve a reasonable visual result. The intensity is ~ (size^2 / distance^2) which can be
430  got by considering the ratio of (flaresurface on screen) : (Surface of sphere defined by flare origin and distance from flare)
431  An important requirement is: intensity <= 1 for all distances.
432 
433  The formula used here to compute the intensity is as follows:
434  intensity = flareCoeff * size^2 / (distance + size*sqrt(flareCoeff))^2.
435  As you can see, the intensity will have a max. of 1 when the distance is 0.
436 
437  The coefficient flareCoeff will determine the falloff speed with increasing distance.
438  */
439 
440  // We don't want too big values anyways when dividing by distance
441  if (f->eyeZ > -1.0f)
442  {
443  distance = 1.0f;
444  }
445  else
446  {
447  distance = -f->eyeZ;
448  }
449 
450  // calculate the flare size
451  size = backEnd.viewParms.viewportWidth * (r_flareSize->value / 640.0f + 8 / distance);
452 
453  factor = distance + size * sqrt(flareCoeff);
454  intensity = flareCoeff * size * size / (factor * factor);
455 
456  VectorScale(f->color, f->drawIntensity * intensity, color);
457  iColor[0] = color[0] * 255;
458  iColor[1] = color[1] * 255;
459  iColor[2] = color[2] * 255;
460  }

465  tess.xyz[tess.numVertexes][1] = f->windowY - size;
466  tess.xyz[tess.numVertexes][2] = 0;
467  tess.xyz[tess.numVertexes][3] = 1;
