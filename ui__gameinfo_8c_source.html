630 {
631  int numdirs;
632  char filename[128];
633  char dirlist[2048];
634  char *dirptr;
635  int i, j;
636  size_t dirlen;
637  long hash;
638  const char *ch;
639 
640  uiInfo.campaignCount = 0;
641  Com_Memset(&uiInfo.campaignList, 0, sizeof(uiInfo.campaignList));
642 
643  // get all campaigns from .campaign files
644  numdirs = trap_FS_GetFileList("scripts", ".campaign", dirlist, 2048);
645  dirptr = dirlist;
646  for (i = 0; i < numdirs && uiInfo.campaignCount < MAX_CAMPAIGNS; i++, dirptr += dirlen + 1)
647  {
648  dirlen = strlen(dirptr);
649  strcpy(filename, "scripts/");
650  strcat(filename, dirptr);
651  UI_LoadCampaignsFromFile(filename);
652  }
653 
654  if (UI_OutOfMemory())
655  {
656  trap_Print(S_COLOR_YELLOW "WARNING: not anough memory in pool to load all campaigns\n");
657  }
658 
659  // Sort the campaigns for single player
660 
661  // first, find the initial campaign
662  for (i = 0; i < uiInfo.campaignCount; i++)
663  {
664  if (!(uiInfo.campaignList[i].typeBits & (1 << GT_SINGLE_PLAYER)))
665  {
666  continue;
667  }
668 
669  if (uiInfo.campaignList[i].initial)
670  {
671  uiInfo.campaignList[i].order = 0;
672  break;
673  }
674  }
675 
676  // now use the initial nextCampaignShortName to find the next one, etc etc for single player campaigns
677  // rain - don't let i go above the maximum number of campaigns
678  while (i < MAX_CAMPAIGNS)
679  {
680  j = UI_FindCampaignInCampaignList(uiInfo.campaignList[i].nextCampaignShortName);
681 
682  if (j == -1)
683  {
684  break;
685  }
686 
687  uiInfo.campaignList[j].order = uiInfo.campaignList[i].order + 1;
688  i = j;
689  }
690 
691  for (i = 0; i < uiInfo.campaignCount; i++)
692  {
693  // generate hash for campaign shortname
694  for (hash = 0, ch = uiInfo.campaignList[i].campaignShortName; *ch != '\0'; ch++)
695  {
696  hash += (long)(tolower(*ch)) * ((ch - uiInfo.campaignList[i].campaignShortName) + 119);
697  }
698 
699  // find the entry in the campaignsave
700  for (j = 0; j < uiInfo.campaignStatus.header.numCampaigns; j++)
701  {
702  if (hash == uiInfo.campaignStatus.campaigns[j].shortnameHash)
703  {
704  uiInfo.campaignList[i].unlocked = qtrue;
705  uiInfo.campaignList[i].progress = uiInfo.campaignStatus.campaigns[j].progress;
706  uiInfo.campaignList[i].cpsCampaign = &uiInfo.campaignStatus.campaigns[j];
707  }
708  }
709  }
710 
711  // Sorting the campaign list
712  qsort(uiInfo.campaignList, uiInfo.campaignCount, sizeof(uiInfo.campaignList[0]), UI_SortCampaigns);
713 }
