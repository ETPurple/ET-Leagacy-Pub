163 void R_IssueRenderCommands(qboolean runPerformanceCounters)
164 {
165  renderCommandList_t *cmdList = &backEndData->commands;
166 
167  etl_assert(cmdList != NULL);
168  // add an end-of-list command
169  *(int *)(cmdList->cmds + cmdList->used) = RC_END_OF_LIST;
170 
171  // clear it out, in case this is a sync and not a buffer flip
172  cmdList->used = 0;
173 
174  // at this point, the back end thread is idle, so it is ok

176  if (runPerformanceCounters)

178  R_PerformanceCounters();
179  }
180 
181  // actually start the commands going
182  if (!r_skipBackEnd->integer)
183  {
184  // let it start on the new batch

186  }

207 void *R_GetCommandBuffer(unsigned int bytes)
208 {
209  renderCommandList_t *cmdList = &backEndData->commands;
210 
211  // always leave room for the swap buffers and end of list commands
212  // - added swapBuffers_t from ET
213  if (cmdList->used + bytes + (sizeof(swapBuffersCommand_t) + sizeof(int)) > MAX_RENDER_COMMANDS)
214  {
215  if (bytes > MAX_RENDER_COMMANDS - (sizeof(swapBuffersCommand_t) + sizeof(int)))
216  {
217  Ren_Fatal("R_GetCommandBuffer: bad size %u", bytes);
218  }
219  // if we run out of room, just start dropping commands
220  return NULL;
221  }
222 
223  cmdList->used += bytes;
224 
225  return cmdList->cmds + cmdList->used - bytes;
226 }

289 static qboolean R_ClipRegion(float *x, float *y, float *w, float *h, float *s1, float *t1, float *s2, float *t2)
290 {
291  float left, top, right, bottom;
292  float _s1, _t1, _s2, _t2;
293  float clipLeft, clipTop, clipRight, clipBottom;
294 
295  if (tr.clipRegion[2] <= tr.clipRegion[0] ||
296  tr.clipRegion[3] <= tr.clipRegion[1])
297  {
298  return qfalse;
299  }
300 
301  left = *x;
302  top = *y;
303  right = *x + *w;
304  bottom = *y + *h;
305 
306  _s1 = *s1;
307  _t1 = *t1;
308  _s2 = *s2;
309  _t2 = *t2;
310 
311  clipLeft = tr.clipRegion[0];
312  clipTop = tr.clipRegion[1];
313  clipRight = tr.clipRegion[2];
314  clipBottom = tr.clipRegion[3];
315 
316  // Completely clipped away
317  if (right <= clipLeft || left >= clipRight ||
318  bottom <= clipTop || top >= clipBottom)
319  {
320  return qtrue;
321  }
322 
323  // Clip left edge
324  if (left < clipLeft)
325  {
326  float f = (clipLeft - left) / (right - left);
327  *s1 = (f * (_s2 - _s1)) + _s1;
328  *x = clipLeft;
329  *w -= (clipLeft - left);
330  }
331 
332  // Clip right edge
333  if (right > clipRight)
334  {
335  float f = (clipRight - right) / (left - right);
336  *s2 = (f * (_s1 - _s2)) + _s2;
337  *w = clipRight - *x;
338  }
339 
340  // Clip top edge
341  if (top < clipTop)
342  {
343  float f = (clipTop - top) / (bottom - top);
344  *t1 = (f * (_t2 - _t1)) + _t1;
345  *y = clipTop;
346  *h -= (clipTop - top);
347  }
348 
349  // Clip bottom edge
350  if (bottom > clipBottom)
351  {
352  float f = (clipBottom - bottom) / (top - bottom);
353  *t2 = (f * (_t1 - _t2)) + _t2;
354  *h = clipBottom - *y;
355  }
356 
357  return qfalse;
358 }

543 {
544  drawBufferCommand_t *cmd;
545

548  return;
549  }
550 
551  Ren_LogComment("--- RE_BeginFrame ---\n");
552 
553  glState.finishCalled = qfalse;
554 
555  tr.frameCount++;
556  tr.frameSceneNum = 0;
557  tr.viewCount = 0;
558 
559  // do overdraw measurement
560  if (r_measureOverdraw->integer)
561  {
562  if (glConfig.stencilBits < 4)
563  {
564  Ren_Print("Warning: not enough stencil bits to measure overdraw: %d\n", glConfig.stencilBits);
565  ri.Cvar_Set("r_measureOverdraw", "0");
566  }
567  else if (r_shadows->integer == 2)
568  {
569  Ren_Print("Warning: stencil shadows and overdraw measurement are mutually exclusive\n");
570  ri.Cvar_Set("r_measureOverdraw", "0");

572  else
573  {
574  R_IssuePendingRenderCommands();
575  glEnable(GL_STENCIL_TEST);
576  glStencilMask(~0U);
577  GL_ClearStencil(0U);
578  glStencilFunc(GL_ALWAYS, 0U, ~0U);
579  glStencilOp(GL_KEEP, GL_INCR, GL_INCR);
580  }
581  r_measureOverdraw->modified = qfalse;
582  }
583  else
584  {
585  // this is only reached if it was on and is now off
586  if (r_measureOverdraw->modified)
587  {
588  R_IssuePendingRenderCommands();
589  glDisable(GL_STENCIL_TEST);
590  }
591  r_measureOverdraw->modified = qfalse;
592  }
593 
594  // texturemode stuff
595  if (r_textureMode->modified)
596  {
597  R_IssuePendingRenderCommands();
598  GL_TextureMode(r_textureMode->string);
599  r_textureMode->modified = qfalse;
600  }
601 
602  // gamma stuff
603  if (r_gamma->modified)
604  {
605  r_gamma->modified = qfalse;
606  R_IssuePendingRenderCommands();
607  R_SetColorMappings();
608  }
609 
610  // check for errors
611  if (!r_ignoreGLErrors->integer)
612  {
613  int err;
614  char s[128];
615 
616  R_IssuePendingRenderCommands();
617 
618  if ((err = glGetError()) != GL_NO_ERROR)
619  {
620  switch (err)
621  {
622  case GL_INVALID_ENUM:
623  Q_strcpy(s, "GL_INVALID_ENUM");
624  break;
625  case GL_INVALID_VALUE:
626  Q_strcpy(s, "GL_INVALID_VALUE");
627  break;
628  case GL_INVALID_OPERATION:
629  Q_strcpy(s, "GL_INVALID_OPERATION");
630  break;
631  case GL_STACK_OVERFLOW:
632  Q_strcpy(s, "GL_STACK_OVERFLOW");
633  break;
634  case GL_STACK_UNDERFLOW:
635  Q_strcpy(s, "GL_STACK_UNDERFLOW");
636  break;
637  case GL_OUT_OF_MEMORY:
638  Q_strcpy(s, "GL_OUT_OF_MEMORY");
639  break;
640  case GL_TABLE_TOO_LARGE:
641  Q_strcpy(s, "GL_TABLE_TOO_LARGE");
642  break;
643  case GL_INVALID_FRAMEBUFFER_OPERATION_EXT:
644  Q_strcpy(s, "GL_INVALID_FRAMEBUFFER_OPERATION_EXT");
645  break;
646  default:
647  Com_sprintf(s, sizeof(s), "0x%X", err);
648  break;
649  }
650 
651  //Ren_Fatal( "caught OpenGL error: %s in file %s line %i", s, filename, line);
652  Ren_Fatal("RE_BeginFrame() - glGetError() failed (%s)!\n", s);
653  }
654  }
655 
656  // draw buffer stuff
657  cmd = (drawBufferCommand_t *)R_GetCommandBuffer(sizeof(*cmd));
658  if (!cmd)
659  {
660  return;
661  }
662 
663  cmd->commandId = RC_DRAW_BUFFER;
664 
665  if (!Q_stricmp(r_drawBuffer->string, "GL_FRONT"))
666  {
667  cmd->buffer = (int)GL_FRONT;
668  }
669  else
670  {
671  cmd->buffer = (int)GL_BACK;
672  }
673 }
