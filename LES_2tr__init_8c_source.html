425 {
426  byte *allbuf, *buffer;
427  byte *srcptr, *destptr;
428  byte *endline, *endmem;
429  byte temp;
430  int linelen, padlen;
431  size_t offset = 18, memcount;
432 
433  allbuf = RB_ReadPixels(x, y, width, height, &offset, &padlen);
434  buffer = allbuf + offset - 18;
435 
436  Com_Memset(buffer, 0, 18);
437  buffer[2] = 2; // uncompressed type
438  buffer[12] = width & 255;
439  buffer[13] = width >> 8;
440  buffer[14] = height & 255;
441  buffer[15] = height >> 8;
442  buffer[16] = 24; // pixel size
443 
444  // swap rgb to bgr and remove padding from line endings
445  linelen = width * 3;
446 
447  srcptr = destptr = allbuf + offset;
448  endmem = srcptr + (linelen + padlen) * height;
449 
450  while (srcptr < endmem)
451  {
452  endline = srcptr + linelen;
453 
454  while (srcptr < endline)
455  {
456  temp = srcptr[0];
457  *destptr++ = srcptr[2];
458  *destptr++ = srcptr[1];
459  *destptr++ = temp;
460 
461  srcptr += 3;
462  }
463 
464  // Skip the pad
465  srcptr += padlen;
466  }
467 
468  memcount = linelen * height;
469 
470  // gamma correct
471  if (glConfig.deviceSupportsGamma)
472  {
473  R_GammaCorrect(allbuf + offset, memcount);
474  }
475

680 {

682  byte *buffer;
683  byte *source, *allsource;
684  byte *src, *dst;
685  size_t offset = 0;
686  int padlen;
687  int x, y;

691 
692  Com_sprintf(checkname, sizeof(checkname), "levelshots/%s.tga", tr.world->baseName);
693 
694  allsource = RB_ReadPixels(0, 0, glConfig.vidWidth, glConfig.vidHeight, &offset, &padlen);
695  source = allsource + offset;
696 
697  buffer = ri.Hunk_AllocateTempMemory(128 * 128 * 3 + 18);
698  Com_Memset(buffer, 0, 18);
699  buffer[2] = 2; // uncompressed type
700  buffer[12] = 128;
701  buffer[14] = 128;
702  buffer[16] = 24; // pixel size
703 
704  // resample from source
705  xScale = glConfig.vidWidth / 512.0f;
706  yScale = glConfig.vidHeight / 384.0f;
707  for (y = 0 ; y < 128 ; y++)
708  {
709  for (x = 0 ; x < 128 ; x++)
710  {
711  r = g = b = 0;
712  for (yy = 0 ; yy < 3 ; yy++)
713  {
714  for (xx = 0 ; xx < 4 ; xx++)
715  {
716  src = source + (3 * glConfig.vidWidth + padlen) * ( int ) ((y * 3 + yy) * yScale) +
717  3 * ( int ) ((x * 4 + xx) * xScale);
718  r += src[0];
719  g += src[1];
720  b += src[2];
721  }
722  }
723  dst = buffer + 18 + 3 * (y * 128 + x);
724  dst[0] = (byte)(b / 12);
725  dst[1] = (byte)(g / 12);
726  dst[2] = (byte)(r / 12);
727  }
728  }
729 
730  // gamma correct
731  if (glConfig.deviceSupportsGamma)
732  {
733  R_GammaCorrect(buffer + 18, 128 * 128 * 3);
734  }
735 
736  ri.FS_WriteFile(checkname, buffer, 128 * 128 * 3 + 18);
737 
738  ri.Hunk_FreeTempMemory(buffer);
739  ri.Hunk_FreeTempMemory(allsource);
740 
741  Ren_Print("Wrote %s\n", checkname);
742 }
