118  int m_MaxPlayers;
119  int m_AvailableTeams;
120 
121  int GetNumPlayers(int t = OB_TEAM_ALL, Controller c = Both) const
122  {
123  int n = 0;
124  for (int i = 0; i < MaxPlayers; ++i)
125  {
126  if (m_Players[i].m_Team == 0)
127  {
128  continue;
129  }
130  if (t != OB_TEAM_ALL && m_Players[i].m_Team != t)
131  {
132  continue;
133  }
134  if (c != Both && m_Players[i].m_Controller != c)
135  {
136  continue;
137  }
138  ++n;
139  }
140  return n;
141  }
142  int GetMaxPlayers() const
143  {
144  return m_MaxPlayers;
145  }
146  int GetAvailableTeams() const
147  {
148  return m_AvailableTeams;
149  }
150  obPlayerInfo()
151  {
152  for (int i = 0; i < MaxPlayers; ++i)
153  m_Players[i] = PInfo();
154  m_AvailableTeams = 0;
155  m_MaxPlayers = 0;
156  }
157 };
