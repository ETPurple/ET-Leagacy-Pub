185 static void R_SetupEntityLightingGrid(trRefEntity_t *ent)
186 {
187  vec3_t lightOrigin;
188  int pos[3];
189  int i, j;
190  byte *gridData;
191  float frac[3];
192  int gridStep[3];
193  vec3_t direction;
194  float totalFactor;
195  float factor;
196  byte *data;
197  int lat, lng;
198  vec3_t normal;
199  float v;
200 
201  if (ent->e.renderfx & RF_LIGHTING_ORIGIN)
202  {
203  // seperate lightOrigins are needed so an object that is
204  // sinking into the ground can still be lit, and so
205  // multi-part models can be lit identically
206  VectorCopy(ent->e.lightingOrigin, lightOrigin);
207  }
208  else
209  {
210  VectorCopy(ent->e.origin, lightOrigin);
211  }
212 
213  VectorSubtract(lightOrigin, tr.world->lightGridOrigin, lightOrigin);
214  for (i = 0 ; i < 3 ; i++)
215  {
216  v = lightOrigin[i] * tr.world->lightGridInverseSize[i];
217  pos[i] = floor(v);
218  frac[i] = v - pos[i];
219  if (pos[i] < 0)
220  {
221  pos[i] = 0;
222  }
223  else if (pos[i] > tr.world->lightGridBounds[i] - 1)
224  {
225  pos[i] = tr.world->lightGridBounds[i] - 1;
226  }
227  }
228 
229  VectorClear(ent->ambientLight);
230  VectorClear(ent->directedLight);
231  VectorClear(direction);
232 
233  etl_assert(tr.world->lightGridData); // bk010103 - NULL with -nolight maps
234 
235  // trilerp the light value
236  gridStep[0] = 8;
237  gridStep[1] = 8 * tr.world->lightGridBounds[0];
238  gridStep[2] = 8 * tr.world->lightGridBounds[0] * tr.world->lightGridBounds[1];
239  gridData = tr.world->lightGridData + pos[0] * gridStep[0]
240  + pos[1] * gridStep[1] + pos[2] * gridStep[2];
241 
242  totalFactor = 0;
243  for (i = 0 ; i < 8 ; i++)
244  {
245  factor = 1.0;
246  data = gridData;
247  for (j = 0 ; j < 3 ; j++)
248  {
249  if (i & (1 << j))
250  {
251  factor *= frac[j];
252  data += gridStep[j];
253  }
254  else
255  {
256  factor *= (1.0f - frac[j]);
257  }
258  }
259 
260  if (!(data[0] + data[1] + data[2]))
261  {
262  continue; // ignore samples in walls
263  }
264  totalFactor += factor;
265 
266  ent->ambientLight[0] += factor * data[0];
267  ent->ambientLight[1] += factor * data[1];
268  ent->ambientLight[2] += factor * data[2];
269 
270  ent->directedLight[0] += factor * data[3];
271  ent->directedLight[1] += factor * data[4];
272  ent->directedLight[2] += factor * data[5];
273 
274  lat = data[7];
275  lng = data[6];
276  lat *= (FUNCTABLE_SIZE / 256);
277  lng *= (FUNCTABLE_SIZE / 256);
278 
279  // decode X as cos( lat ) * sin( long )
280  // decode Y as sin( lat ) * sin( long )
281  // decode Z as cos( long )
282 
283  normal[0] = tr.sinTable[(lat + (FUNCTABLE_SIZE / 4)) & FUNCTABLE_MASK] * tr.sinTable[lng];
284  normal[1] = tr.sinTable[lat] * tr.sinTable[lng];
285  normal[2] = tr.sinTable[(lng + (FUNCTABLE_SIZE / 4)) & FUNCTABLE_MASK];
286 
287  VectorMA(direction, factor, normal, direction);
288 
289  // test code
290  //if( strstr( tr.models[ ent->e.hModel ]->name, ".mdm" ) && i == 0 )
291  // ri.Printf( PRINT_ALL, "lat: %3d lng: %3d dir: %2.3f %2.3f %2.3f\n",
292  // data[ 7 ], data[ 8 ], normal[ 0 ], normal[ 1 ], normal[ 2 ] );
293  }
294 
295  if (totalFactor > 0 && totalFactor < 0.99f)
296  {
297  totalFactor = 1.0f / totalFactor;
298  VectorScale(ent->ambientLight, totalFactor, ent->ambientLight);
299  VectorScale(ent->directedLight, totalFactor, ent->directedLight);
300  }
301 
302  VectorScale(ent->ambientLight, r_ambientScale->value, ent->ambientLight);
303  VectorScale(ent->directedLight, r_directedScale->value, ent->directedLight);
304 
305  // cheats? check for single player?
306  if (tr.lightGridMulDirected != 0.f)
307  {
308  VectorScale(ent->directedLight, tr.lightGridMulDirected, ent->directedLight);
309  }
310  if (tr.lightGridMulAmbient != 0.f)
311  {
312  VectorScale(ent->ambientLight, tr.lightGridMulAmbient, ent->ambientLight);
313  }
314 
315  VectorNormalize2(direction, ent->lightDir);
316 
317  // debug hack
318  //VectorSubtract( vec3_origin, direction, ent->lightDir );
319 }
