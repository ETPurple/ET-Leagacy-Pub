546 gentity_t *SelectRandomTeamSpawnPoint(int teamstate, team_t team, int spawnObjective)
547 {
548  gentity_t *spot;
549  gentity_t *spots[MAX_TEAM_SPAWN_POINTS];
550  int count, closest;
551  int i = 0;
552  char *classname;
553  float shortest, tmp;
554  vec3_t target;
555  vec3_t farthest;
556 
557  if (team == TEAM_AXIS)
558  {
559  classname = "team_CTF_redspawn";
560  }
561  else if (team == TEAM_ALLIES)
562  {
563  classname = "team_CTF_bluespawn";
564  }
565  else
566  {
567  return NULL;
568  }
569 
570  count = 0;
571 
572  spot = NULL;
573 
574  while ((spot = G_Find(spot, FOFS(classname), classname)) != NULL)
575  {
576  if (SpotWouldTelefrag(spot))
577  {
578  continue;
579  }
580 
581  // modified to allow initial spawnpoints to be disabled at gamestart
582  if (!(spot->spawnflags & 2))
583  {
584  continue;
585  }
586 
587  // invisible entities can't be used for spawning
588  if (spot->entstate == STATE_INVISIBLE || spot->entstate == STATE_UNDERCONSTRUCTION)
589  {
590  continue;
591  }
592 
593  spots[count] = spot;
594  if (++count == MAX_TEAM_SPAWN_POINTS)
595  {
596  break;
597  }
598  }
599 
600  if (!count) // no spots that won't telefrag
601  {
602  spot = NULL;
603  while ((spot = G_Find(spot, FOFS(classname), classname)) != NULL)
604  {
605  // modified to allow initial spawnpoints to be disabled at gamestart
606  if (!(spot->spawnflags & 2))
607  {
608  continue;
609  }
610 
611  // invisible entities can't be used for spawning
612  if (spot->entstate == STATE_INVISIBLE || spot->entstate == STATE_UNDERCONSTRUCTION)
613  {
614  continue;
615  }
616 
617  return spot;
618  }
619 
620  return G_Find(NULL, FOFS(classname), classname);
621  }
622 
623  if ((!level.numSpawnPoints))
624  {
625  G_Error("No spawnpoints found\n");
626  return NULL;
627  }
628  else
629  {
630  // adding ability to set autospawn
631  if (!spawnObjective)
632  {
633  switch (team)
634  {
635  case TEAM_AXIS:
636  spawnObjective = level.axisAutoSpawn + 1;
637  break;
638  case TEAM_ALLIES:
639  spawnObjective = level.alliesAutoSpawn + 1;
640  break;
641  default:
642  break;
643  }
644  }
645 
646  i = spawnObjective - 1;
647 
648  VectorCopy(level.spawnPointStates[i].origin, farthest);
649 
650  // now that we've got farthest vector, figure closest spawnpoint to it
651  VectorSubtract(farthest, spots[0]->s.origin, target);
652  shortest = VectorLength(target);
653  closest = 0;
654  for (i = 0; i < count; i++)
655  {
656  VectorSubtract(farthest, spots[i]->s.origin, target);
657  tmp = VectorLength(target);
658 
659  if (tmp < shortest)
660  {
661  shortest = tmp;
662  closest = i;
663  }
664  }
665  return spots[closest];
666  }
667 }
