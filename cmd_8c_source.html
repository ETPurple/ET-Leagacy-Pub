185 void Cbuf_Execute(void)
186 {
187  unsigned int i;
188  char *text;
189  char line[MAX_CMD_LINE];
190  int quotes;
191  // This will keep // style comments all on one line by not breaking on
192  // a semicolon. It will keep /* ... */ style comments all on one line by not
193  // breaking it for semicolon or newline.
194  qboolean in_star_comment = qfalse;
195  qboolean in_slash_comment = qfalse;
196 
197  while (cmd_text.cursize)
198  {
199  if (cmd_wait > 0)
200  {
201  // skip out while text still remains in buffer, leaving it
202  // for next frame
203  cmd_wait--;
204  break;
205  }
206 
207  // find a \n or ; line break or comment: // or /* */
208  text = (char *)cmd_text.data;
209 
210  quotes = 0;
211  for (i = 0 ; i < cmd_text.cursize ; i++)
212  {
213  // FIXME: ignore quoted text
214 
215  if (text[i] == '"')
216  {
217  quotes++;
218  }
219 
220  if (!(quotes & 1))
221  {
222  if (i < cmd_text.cursize - 1)
223  {
224  if (!in_star_comment && text[i] == '/' && text[i + 1] == '/')
225  {
226  in_slash_comment = qtrue;
227  }
228  else if (!in_slash_comment && text[i] == '/' && text[i + 1] == '*')
229  {
230  in_star_comment = qtrue;
231  }
232  else if (in_star_comment && text[i] == '*' && text[i + 1] == '/')
233  {
234  in_star_comment = qfalse;
235  // If we are in a star comment, then the part after it is valid
236  // Note: This will cause it to NUL out the terminating '/'
237  // but ExecuteString doesn't require it anyway.
238  i++;
239  break;
240  }
241  }
242  if (!in_slash_comment && !in_star_comment && text[i] == ';')
243  {
244  break;
245  }
246  }
247  if (!in_star_comment && (text[i] == '\n' || text[i] == '\r'))
248  {
249  in_slash_comment = qfalse;
250  break;
251  }
252  }
253 
254  if (i >= (MAX_CMD_LINE - 1))
255  {
256  i = MAX_CMD_LINE - 1;
257  }
258 
259  Com_Memcpy(line, text, i);
260  line[i] = 0;
261 
262  // delete the text from the command buffer and move remaining commands down
263  // this is necessary because commands (exec) can insert data at the
264  // beginning of the text buffer
265 
266  if (i == cmd_text.cursize)
267  {
268  cmd_text.cursize = 0;
269  }
270  else
271  {
272  i++;
273  cmd_text.cursize -= i;
274  memmove(text, text + i, cmd_text.cursize);
275  }
276 
277  // execute the command line
278  Cmd_ExecuteString(line);
279  }
280 }

588 static void Cmd_TokenizeString2(const char *text_in, qboolean ignoreQuotes)
589 {
590  const char *text;
591  char *textOut;
592 
593  // clear previous args
594  cmd_argc = 0;
595 
596  if (!text_in)
597  {
598  return;
599  }
600 
601  Q_strncpyz(cmd_cmd, text_in, sizeof(cmd_cmd));
602 
603  text = text_in;
604  textOut = cmd_tokenized;
605 
606  while (1)
607  {
608  if (cmd_argc == MAX_STRING_TOKENS)
609  {
610  return; // this is usually something malicious
611  }
612 
613  while (1)
614  {
615  // skip whitespace
616  while (*text && *text <= ' ')
617  {
618  text++;
619  }
620  if (!*text)
621  {
622  return; // all tokens parsed
623  }
624 
625  // skip // comments
626  if (text[0] == '/' && text[1] == '/')
627  {
628  // lets us put 'http://' in commandlines
629  if (text == text_in || (text > text_in && text[-1] != ':'))
630  {
631  return; // all tokens parsed
632  }
633  }
634 
635  // skip /* */ comments
636  if (text[0] == '/' && text[1] == '*')
637  {
638  while (*text && (text[0] != '*' || text[1] != '/'))
639  {
640  text++;
641  }
642  if (!*text)
643  {
644  return; // all tokens parsed
645  }
646  text += 2;
647  }
648  else
649  {
650  break; // we are ready to parse a token
651  }
652  }
653 
654  // handle quoted strings
655  // NOTE: this doesn't handle \" escaping
656  if (!ignoreQuotes && *text == '"')
657  {
658  cmd_argv[cmd_argc] = textOut;
659  cmd_argc++;
660  text++;
661  while (*text && *text != '"')
662  {
663  *textOut++ = *text++;
664  }
665  *textOut++ = 0;
666  if (!*text)
667  {
668  return; // all tokens parsed
669  }
670  text++;
671  continue;
672  }
673 
674  // regular token
675  cmd_argv[cmd_argc] = textOut;
676  cmd_argc++;
677 
678  // skip until whitespace, quote, or command
679  while (*text > ' ')
680  {
681  if (!ignoreQuotes && text[0] == '"')
682  {
683  break;
684  }
685 
686  if (text[0] == '/' && text[1] == '/')
687  {
688  // lets us put 'http://' in commandlines
689  if (text == text_in || (text > text_in && text[-1] != ':'))
690  {
691  break;
692  }
693  }
694 
695  // skip /* */ comments
696  if (text[0] == '/' && text[1] == '*')
697  {
698  break;
699  }
700 
701  *textOut++ = *text++;
702  }
703 
704  *textOut++ = 0;
705 
706  if (!*text)
707  {
708  return; // all tokens parsed
709  }
710  }
711 }

968 void Cmd_ExecuteString(const char *text)
969 {
970  cmd_function_t *cmd, **prev;
971 
972  // execute the command line
973  Cmd_TokenizeString(text);
974  if (!Cmd_Argc())
975  {
976  return; // no tokens
977  }
978 
979  // check registered command functions
980  for (prev = &cmd_functions ; *prev ; prev = &cmd->next)
981  {
982  cmd = *prev;
983  if (!Q_stricmp(cmd_argv[0], cmd->name))
984  {
985  // rearrange the links so that the command will be
986  // near the head of the list next time it is used
987  *prev = cmd->next;
988  cmd->next = cmd_functions;
989  cmd_functions = cmd;
990 
991  // perform the action
992  if (!cmd->function)
993  {
994  // let the cgame or game handle it
995  break;
996  }
997  else
998  {
