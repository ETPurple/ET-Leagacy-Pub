112  inUse = scs * sizeof(sndBuffer);

123  Com_Printf("Sound memory manager started\n");

146 static int ResampleSfx(sfx_t *sfx, int channels, int inrate, int inwidth, int samples, byte *data, qboolean compressed)
147 {
148  float stepscale = (float)inrate / dma.speed; // this is usually 0.5, 1, or 2
149  int outcount = (int)(samples / stepscale);
150  int srcsample;
151  int i, j;
152  int sample, samplefrac = 0, fracstep = (int)(stepscale * 256 * channels);
153  int part;
154  sndBuffer *chunk = sfx->soundData;
155 
156  for (i = 0 ; i < outcount ; i++)
157  {
158  srcsample = samplefrac >> 8;
159  samplefrac += fracstep;
160 
161  for (j = 0 ; j < channels ; j++)
162  {
163  if (inwidth == 2)
164  {
165  sample = (((short *)data)[srcsample + j]);
166  }
167  else
168  {
169  sample = (unsigned int)((unsigned char)(data[srcsample + j]) - 128) << 8;
170  }
171  part = (i * channels + j) & (SND_CHUNK_SIZE - 1);
172  if (part == 0)
173  {
174  sndBuffer *newchunk;
175  newchunk = SND_malloc();
176  if (chunk == NULL)
177  {
178  sfx->soundData = newchunk;
179  }
180  else
181  {
182  chunk->next = newchunk;
183  }
184  chunk = newchunk;
185  }
186 
187  chunk->sndChunk[part] = sample;
188  }
189  }
190 
191  return outcount;
192 }
