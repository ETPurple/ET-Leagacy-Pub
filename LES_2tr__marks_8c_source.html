60 static void R_ChopPolyBehindPlane(int numInPoints, vec3_t inPoints[MAX_VERTS_ON_POLY],
61  int *numOutPoints, vec3_t outPoints[MAX_VERTS_ON_POLY],
62  vec3_t normal, vec_t dist, vec_t epsilon)
63 {
64  float dists[MAX_VERTS_ON_POLY + 4];
65  int sides[MAX_VERTS_ON_POLY + 4];
66  int counts[3];
67  float dot;
68  int i, j;
69  float *p1, *p2, *clip;
70  float d;
71 
72  Com_Memset(dists, 0, MAX_VERTS_ON_POLY + 4);
73  Com_Memset(sides, 0, MAX_VERTS_ON_POLY + 4);
74 
75  // don't clip if it might overflow
76  if (numInPoints >= MAX_VERTS_ON_POLY - 2)
77  {
78  *numOutPoints = 0;
79  return;
80  }
81 
82  counts[0] = counts[1] = counts[2] = 0;
83 
84  // determine sides for each point
85  for (i = 0; i < numInPoints; i++)
86  {
87  dot = DotProduct(inPoints[i], normal);
88  dot -= dist;
89  dists[i] = dot;
90  if (dot > epsilon)
91  {
92  sides[i] = SIDE_FRONT;
93  }
94  else if (dot < -epsilon)
95  {
96  sides[i] = SIDE_BACK;
97  }
98  else
99  {
100  sides[i] = SIDE_ON;
101  }
102  counts[sides[i]]++;
103  }
104  sides[i] = sides[0];
105  dists[i] = dists[0];
106 
107  *numOutPoints = 0;
108 
109  if (!counts[0])
110  {
111  return;
112  }
113  if (!counts[1])
114  {
115  *numOutPoints = numInPoints;
116  Com_Memcpy(outPoints, inPoints, numInPoints * sizeof(vec3_t));
117  return;
118  }
119 
120  for (i = 0; i < numInPoints; i++)
121  {
122  p1 = inPoints[i];
123  clip = outPoints[*numOutPoints];
124 
125  if (sides[i] == SIDE_ON)
126  {
127  VectorCopy(p1, clip);
128  (*numOutPoints)++;
129  continue;
130  }
131 
132  if (sides[i] == SIDE_FRONT)
133  {
134  VectorCopy(p1, clip);
135  (*numOutPoints)++;
136  clip = outPoints[*numOutPoints];
137  }
138 
139  if (sides[i + 1] == SIDE_ON || sides[i + 1] == sides[i])
140  {
141  continue;
142  }
143 
144  // generate a split point
145  p2 = inPoints[(i + 1) % numInPoints];
146 
147  d = dists[i] - dists[i + 1];
148  if (d == 0.f)
149  {
150  dot = 0;
151  }
152  else
153  {
154  dot = dists[i] / d;
155  }
156 
157  // clip xyz
158 
159  for (j = 0; j < 3; j++)
160  {
161  clip[j] = p1[j] + dot * (p2[j] - p1[j]);
162  }
163 
164  (*numOutPoints)++;
165  }
166 }

178 void R_BoxSurfaces_r(mnode_t *node, vec3_t mins, vec3_t maxs, surfaceType_t **list, int listsize, int *listlength, vec3_t dir)
179 {
180  int s, c;
181  msurface_t *surf, **mark;
182 
183  // if this node hasn't been rendered recently, ignore it
184  if (node->visframe < tr.visCount - 2) // allow us to be a few frames behind
185  {
186  return;
187  }
188 
189  // do the tail recursion in a loop
190  while (node->contents == -1)
191  {
192  s = BoxOnPlaneSide(mins, maxs, node->plane);
193  if (s == 1)
194  {
195  node = node->children[0];
196  }
197  else if (s == 2)
198  {
199  node = node->children[1];
200  }
201  else
202  {
203  R_BoxSurfaces_r(node->children[0], mins, maxs, list, listsize, listlength, dir);
204  node = node->children[1];
205  }
206  }
207 
208  // don't mark alpha surfaces
209  if (node->contents & CONTENTS_TRANSLUCENT)
210  {
211  return;
212  }
213 
214  // add the individual surfaces
215  mark = node->firstmarksurface;
216  c = node->nummarksurfaces;
217  while (c--)
218  {
219  if (*listlength >= listsize)
220  {
221  break;
222  }
223 
224  surf = *mark;
225 
226  // check if the surface has NOIMPACT or NOMARKS set
227  if ((surf->shader->surfaceFlags & (SURF_NOIMPACT | SURF_NOMARKS))
228  || (surf->shader->contentFlags & CONTENTS_FOG))
229  {
230  surf->viewCount = tr.viewCount;
231  }
232  // extra check for surfaces to avoid list overflows
233  else if (*(surf->data) == SF_FACE)
234  {
235  if ((( srfSurfaceFace_t * ) surf->data)->plane.type != PLANE_NON_PLANAR)
236  {
237  // the face plane should go through the box
238  s = BoxOnPlaneSide(mins, maxs, &(( srfSurfaceFace_t * ) surf->data)->plane);
239  if (s == 1 || s == 2)
240  {
241  surf->viewCount = tr.viewCount;
242  }
243  else if (DotProduct((( srfSurfaceFace_t * ) surf->data)->plane.normal, dir) < -0.5f)
244  {
245  // don't add faces that make sharp angles with the projection direction
246  surf->viewCount = tr.viewCount;
247  }
248  }
249  }
250  else if (*( surfaceType_t * )(surf->data) != SF_GRID && *( surfaceType_t * )(surf->data) != SF_TRIANGLES)
251  {
252  surf->viewCount = tr.viewCount;
253  }
254  // check the viewCount because the surface may have
255  // already been added if it spans multiple leafs
256  if (surf->viewCount != tr.viewCount)
257  {
258  surf->viewCount = tr.viewCount;
259  list[*listlength] = (surfaceType_t *) surf->data;
260  (*listlength)++;
261  }
262  mark++;
263  }
264 }

451  {

453  // the bounding planes of the to be projected polygon.
454  // LOD is not taken into account, not such a big deal though.
455  //
456  // It's probably much nicer to chop the grid itself and deal
