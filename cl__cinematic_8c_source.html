314 cinHandle_t CIN_PlayCinematic(const char *name, int x, int y, int w, int h, int flags)
315 {
316  cinematic_t *cin;
317  cinHandle_t handle;
318  fileHandle_t file;
319  int size;
320  int i;
321  const char *fileExt;
322 
323  fileExt = COM_GetExtension(name);
324 
325  // See if already playing
326  for (i = 0, cin = cin_cinematics; i < MAX_CINEMATICS; i++, cin++)
327  {
328  if (!cin->playing)
329  {
330  continue;
331  }
332 
333  if (!Q_stricmp(cin->name, name))
334  {
335  if (cin->flags != flags)
336  {
337  continue;
338  }
339 
340  return i + 1;
341  }
342  }
343 
344  // Open the file
345  size = FS_FOpenFileRead(name, &file, qtrue);
346 
347  if (size <= 0)
348  {
349  Com_Printf("Warning: Cinematic '%s' file not found or can't be read\n", name);
350  return -1;
351  }
352 
353 
354  if (!file)
355  {
356  if (flags & CIN_system)
357  {
358  Com_Printf("Warning: Cinematic %s file not found\n", name);
359  }
360 
361  return -1;
362  }
363 
364  if (flags & CIN_system)
365  {
366  Com_Printf("Cinematic %s found\n", name);
367  }
368 
369  // Play the cinematic
370  cin = CIN_HandleForCinematic(&handle);
371 
372  // Fill it in
373  cin->playing = qtrue;
374  Q_strncpyz(cin->name, name, MAX_OSPATH);
375  cin->flags = flags;
376  cin->file = file;
377  cin->size = size;
378  cin->startTime = 0;
379  cin->frameWidth = 0;
380  cin->frameHeight = 0;
381  cin->frameCount = 0;
382  cin->frameBuffer[0] = NULL;
383  cin->frameBuffer[1] = NULL;
384  cin->rectangle.x = x;
385  cin->rectangle.y = y;
386  cin->rectangle.w = w;
387  cin->rectangle.h = h;
388 
389  for (i = 0; i < VIDEO_NUM_CODECS; i++)
390  {
391  if (!videoDecoders[i].fileExt[0])
392  {
393  continue;
394  }
395 
396  if (COM_CompareExtension(fileExt, videoDecoders[i].fileExt) ||
397  (videoDecoders[i].fileExt2[0] && COM_CompareExtension(fileExt, videoDecoders[i].fileExt2)))
398  {
399  if (!videoDecoders[i].Start)
400  {
401  Com_Error(ERR_FATAL, "Cinematic %s cannot be run, there is no start method defined", name);
402  }
403 
404  cin->videoType = i;
405 
406  if (!videoDecoders[i].Start(cin))
407  {
408  if (flags & CIN_system)
409  {
410  Com_Printf("Warning: Cinematic %s is not a valid %s file\n", name, videoDecoders[i].fileExt);
411  }
412 
413  goto video_playback_failed;
414  }
415  goto codec_found_valid;
416  }
417  }
418  Com_Printf("Could not find a codec for the video: %s\n", name);
419 video_playback_failed:
420  CIN_FreeCinematic(cin, qfalse);
421  return -1;
422 
423 codec_found_valid:
424 
425  if (flags & CIN_system)
426  {
427  Com_Printf("Playing cinematic %s\n", name);
428 
429  cls.state = CA_CINEMATIC;
430 
431  // Force console and GUI off
432  if (uivm)
433  {
434  VM_Call(uivm, UI_SET_ACTIVE_MENU, UIMENU_NONE);
435  }
436 
437  Con_Close();
438  }
439 
440  // Make sure sounds aren't playing
441  S_StopAllSounds();
442 
443  return handle;
444 }

482 e_status CIN_StopCinematic(cinHandle_t handle)
483 {
484  cinematic_t *cin;
485 
486  if (!CIN_HandleValid(handle))
487  {
488  return FMV_EOF;
489  }
490 
491  cin = CIN_GetCinematicByHandle(handle);
492 
493  if (cin->flags & CIN_system)
494  {
495  const char *s;
496 
497  cls.state = CA_DISCONNECTED;
498  // we can't just do a vstr nextmap, because
499  // if we are aborting the intro cinematic with

501  // the time it was referenced
502  s = Cvar_VariableString("nextmap");

505  Cbuf_ExecuteText(EXEC_APPEND, va("%s\n", s));
506  Cvar_Set("nextmap", "");
507  }
508  cls.cinematicHandle = -1;
509  }
510 
511  CIN_FreeCinematic(cin, qtrue);
512 
513  return FMV_EOF;
