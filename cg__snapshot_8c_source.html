524 void CG_ProcessSnapshots(void)
525 {
526  snapshot_t *snap;
527  int n;
528 
529  // see what the latest snapshot the client system has is
530  trap_GetCurrentSnapshotNumber(&n, &cg.latestSnapshotTime);
531  if (n != cg.latestSnapshotNum)
532  {
533  if (n < cg.latestSnapshotNum)
534  {
535  if (!cg.demoPlayback)
536  {
537  // this should never happen
538  CG_Error("CG_ProcessSnapshots: n < cg.latestSnapshotNum\n");
539  }
540  else
541  {
542  // These are here so we dont actually need to have
543  // the client call a timereset or something similar to the cgame
544  cg.snap = 0;
545  cg.nextSnap = 0;
546  cgs.processedSnapshotNum = -2;
547  cg.time = 0;
548  }
549  }
550  cg.latestSnapshotNum = n;
551  }
552 
553  // If we have yet to receive a snapshot, check for it.
554  // Once we have gotten the first snapshot, cg.snap will
555  // always have valid data for the rest of the game
556  while (!cg.snap)
557  {
558  snap = CG_ReadNextSnapshot();
559  if (!snap)
560  {
561  // we can't continue until we get a snapshot
562  return;
563  }
564 
565  // set our weapon selection to what
566  // the playerstate is currently using
567  if (!(snap->snapFlags & SNAPFLAG_NOT_ACTIVE))
568  {
569  CG_SetInitialSnapshot(snap);
570  }
571  }
572 
573  // loop until we either have a valid nextSnap with a serverTime
574  // greater than cg.time to interpolate towards, or we run
575  // out of available snapshots
576  do
577  {
578  // if we don't have a nextframe, try and read a new one in
579  if (!cg.nextSnap)
580  {
581  snap = CG_ReadNextSnapshot();
582 
583  // if we still don't have a nextframe, we will just have to
584  // extrapolate
585  if (!snap)
586  {
587  break;
588  }
589 
590  CG_SetNextSnap(snap);
591 
592  // if time went backwards, we have a level restart
593  if (cg.nextSnap->serverTime < cg.snap->serverTime)
594  {
595  if (!cg.demoPlayback)
596  {
597  CG_Error("CG_ProcessSnapshots: Server time went backwards\n");
598  }
599  }
600  }
601 
602  // if our time is < nextFrame's, we have a nice interpolating state
603  if (cg.time >= cg.snap->serverTime && cg.time < cg.nextSnap->serverTime)
604  {
605  break;
606  }
607 
608  // we have passed the transition from nextFrame to frame
609  CG_TransitionSnapshot();
610  }
611  while (1);
612 
613  // assert our valid conditions upon exiting
614  if (cg.snap == NULL)
615  {
616  if (!cg.demoPlayback)
617  {
618  CG_Error("CG_ProcessSnapshots: cg.snap == NULL\n");
619  }
620  return;
621  }
622  if (cg.time < cg.snap->serverTime)
623  {
