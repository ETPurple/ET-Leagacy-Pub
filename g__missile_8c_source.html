319 qboolean G_MissileImpact(gentity_t *ent, trace_t *trace, int impactDamage)
320 {
321  gentity_t *other = &g_entities[trace->entityNum];
322  vec3_t velocity;
323 
324  // handle func_explosives
325  if (other->classname && Q_stricmp(other->classname, "func_explosive") == 0)
326  {
327  // the damage is sufficient to "break" the ent (health == 0 is non-breakable)
328  if (other->health && impactDamage >= other->health)
329  {
330  // check for other->takedamage needs to be inside the health check since it is
331  // likely that, if successfully destroyed by the missile, in the next runmissile()
332  // update takedamage would be set to '0' and the func_explosive would not be
333  // removed yet, causing a bounce.
334  if (other->takedamage)
335  {
336  BG_EvaluateTrajectoryDelta(&ent->s.pos, level.time, velocity, qfalse, ent->s.effect2Time);
337  G_Damage(other, ent, &g_entities[ent->r.ownerNum], velocity, ent->s.origin, impactDamage, 0, ent->methodOfDeath);
338  }
339 
340  // its possible of the func_explosive not to die from this and it
341  // should reflect the missile or explode it not vanish into oblivion
342  if (other->health <= 0)
343  {
344  return qfalse;
345  }
346  }
347  }
348 
349  // check for bounce
350  if ((!other->takedamage || !ent->damage) && (ent->s.eFlags & (EF_BOUNCE | EF_BOUNCE_HALF)))
351  {
352  G_BounceMissile(ent, trace);
353 
354  G_AddEvent(ent, EV_GRENADE_BOUNCE, BG_FootstepForSurface(trace->surfaceFlags));
355 
356  return qfalse;
357  }
358 
359  // impact damage
360  if (other->takedamage || other->dmgparent)
361  {
362  if (ent->damage)
363  {
364  BG_EvaluateTrajectoryDelta(&ent->s.pos, level.time, velocity, qfalse, ent->s.effect2Time);
365  if (VectorLengthSquared(velocity) == 0.f)
366  {
367  velocity[2] = 1; // stepped on a grenade
368  }
369  G_Damage(other->dmgparent ? other->dmgparent : other, ent, &g_entities[ent->r.ownerNum], velocity, ent->s.origin, ent->damage, 0, ent->methodOfDeath);
370  }
371  else // if no damage value, then this is a splash damage grenade only
372  {
373  G_BounceMissile(ent, trace);
374  return qfalse;
375  }
376  }
377 
378  if (other->takedamage && other->client)
379  {
380  G_RealExplodedMissile(ent, trace->plane.normal, other);
381  }
382  else
383  {
384  // try projecting it in the direction it came from, for better decals
385  vec3_t dir;
386 
387  BG_EvaluateTrajectoryDelta(&ent->s.pos, level.time, dir, qfalse, ent->s.effect2Time);
388  BG_GetMarkDir(dir, trace->plane.normal, dir);
389 
390  G_RealExplodedMissile(ent, dir, NULL);
391  }
392 
393  return qtrue;
394 }

439 {

461

464  {

466  {

470

472  {

474  {

476  if (ent->r.currentOrigin[0] < level.mapcoordsMins[0] ||
477  ent->r.currentOrigin[1] > level.mapcoordsMins[1] ||
478  ent->r.currentOrigin[0] > level.mapcoordsMaxs[0] ||

490  }

494

496  groundFloor = BG_GetTracemapGroundFloor();
497 
498  // is ent under the ground limit, and ground valid

501  gentity_t *tent;
502 
503  tent = G_TempEntity(ent->r.currentOrigin, EV_MORTAR_MISS);
504  tent->s.clientNum = ent->r.ownerNum;
505  tent->r.svFlags |= SVF_BROADCAST;
506  tent->s.density = 0; // direct
507 
508  G_ExplodeMissile(ent);
509  return; // delete it and play explode sound
510  }
511

514  {
515  G_RunThink(ent);

518 
519  return; // keep flying
520  }
521

531  VectorCopy(angle, ent->r.currentAngles);

565  }
566  }

569  // ignoring interactions with the missile owner

571

578  }
579

581

583  {
584  qboolean exploded = qfalse;
585 
586  if (level.tracemapLoaded && ent->s.pos.trType == TR_GRAVITY && ent->r.contents != CONTENTS_CORPSE
587  && ((!tr.surfaceFlags && tr.startsolid) || (tr.surfaceFlags & SURF_SKY)))
588  {

591  trap_UnlinkEntity(ent);
592  G_RunThink(ent);
593  return; // keep flying
594  }
595 
596  if (tr.surfaceFlags & SURF_NOIMPACT) // never explode or bounce on sky
597  {
598  G_FreeEntity(ent);
599  return;
600  }
601 
602  if (ent->r.contents == CONTENTS_CORPSE)
603  {
604  if (ent->s.pos.trType != TR_STATIONARY)
605  {
606  exploded = G_MissileImpact(ent, &tr, GetWeaponFireTableData(ent->s.weapon)->impactDamage);

612  }
613

615  {
616  return; // exploded
617  }
618  }
619  else if (VectorLengthSquared(ent->s.pos.trDelta) != 0.f) // free fall/no intersection
620  {
621  ent->s.groundEntityNum = ENTITYNUM_NONE;
622  }
623 
624  // check think function after bouncing
625  G_RunThink(ent);
626 }
