75 int Com_AddToGrowList(growList_t *list, void *data)
76 {
77  void **old;
78 
79  if (list->currentElements != list->maxElements)
80  {
81  list->elements[list->currentElements] = data;
82  return list->currentElements++;
83  }
84 
85  // grow, reallocate and move
86  old = list->elements;
87 
88  if (list->maxElements < 0)
89  {
90  Ren_Fatal("Com_AddToGrowList: maxElements = %i", list->maxElements);
91  }
92

94  {
95  // initialize the list to hold 100 elements
96  Com_InitGrowList(list, 100);

99

101 
102  // whenever we are seeing this we might increase affected growlist
103  Com_DPrintf("Resizing growlist to %i maxElements\n", list->maxElements);
104 
105  list->elements = (void **)Com_Allocate(list->maxElements * sizeof(void *));
106 
107  if (!list->elements)
108  {
109  Ren_Drop("Growlist alloc failed");
110  }
111 
112  Com_Memcpy(list->elements, old, list->currentElements * sizeof(void *));
113 
114  Com_Dealloc(old);
115 
116  return Com_AddToGrowList(list, data);
