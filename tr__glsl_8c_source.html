227 programInfo_t *GLSL_ParseDefinition(char **text, const char *defname)
228 {
229  char *token;
230  programInfo_t *def;
231  void *valptr;
232 
233  token = COM_ParseExt(text, qtrue);
234  if (token[0] != '{')
235  {
236  Ren_Warning("WARNING: expecting '{', found '%s' instead in shader definition '%s'\n", token, defname);
237  return NULL;
238  }
239 
240  def = (programInfo_t *)Com_Allocate(sizeof(programInfo_t));
241  Com_Memset(def, 0, sizeof(programInfo_t));
242 
243  GLSL_CopyStringAlloc(&def->name, defname);
244  def->compiled = qfalse;
245 
246  while (1)
247  {
248  token = COM_ParseExt(text, qtrue);
249  if (!token[0])
250  {
251  Ren_Warning("WARNING: no concluding '}' in shader definition %s\n", defname);
252  goto parseerror;
253  }
254 
255  // end of shader definition
256  if (token[0] == '}')
257  {
258  break;
259  }
260  else if (!Q_stricmp(token, "filename"))
261  {
262  GLSL_CopyNextToken(text, &def->filename);
263  }
264  else if (!Q_stricmp(token, "fragfilename"))
265  {
266  GLSL_CopyNextToken(text, &def->fragFilename);
267  }
268  else if (!Q_stricmp(token, "macros"))
269  {
270  int macro;
271 
272  while ((token = COM_ParseExt(text, qfalse))[0])
273  {
274  macro = GLSL_GetMacroByName(token);
275  if (macro >= 0)
276  {
277  def->macros[def->numMacros] = macro;
278  def->numMacros++;
279  }
280  else
281  {
282  Ren_Warning("WARNING: Macro '%s' for shaderdef '%s' was not recognized\n", token, defname);
283  goto parseerror;
284  }
285  }
286  }
287  else if (!Q_stricmp(token, "extramacros"))
288  {
289  GLSL_CopyNextToken(text, &def->extraMacros);
290  }
291  else if (!Q_stricmp(token, "attribs"))
292  {
293  unsigned int attribs = 0;
294 
295  while ((token = COM_ParseExt(text, qfalse))[0])
296  {
297  attribs |= GLSL_GetAttribByName(token);
298  }
299  def->attributes = attribs;
300  }
301  else if (!Q_stricmp(token, "vertexLibraries"))
302  {
303  GLSL_CopyNextToken(text, &def->vertexLibraries);
304  }
305  else if (!Q_stricmp(token, "fragmentLibraries"))
306  {
307  GLSL_CopyNextToken(text, &def->fragmentLibraries);
308  }
309  else if (!Q_stricmp(token, "uniform"))
310  {
311  token = COM_ParseExt(text, qtrue);
312 
313  if (!Q_stricmp(token, "int"))
314  {
315  if (def->numUniformValues < MAX_UNIFORM_VALUES - 1)
316  {
317  def->uniformValues[def->numUniformValues].type.type = GLSL_INT;
318  GLSL_CopyNextToken(text, &def->uniformValues[def->numUniformValues].type.name);
319  token = COM_ParseExt(text, qtrue);
320  valptr = Com_Allocate(sizeof(int));
321  *((int *)valptr) = atoi(token);
322  def->uniformValues[def->numUniformValues].value = valptr;
323  //Ren_Print("%d\n",*((int*)valptr));
324  def->numUniformValues++;
325  }
326  else
327  {
328  Ren_Warning("GLSL_ParseDefinition: MAX_UNIFORM_VALUES reached.\n");
329  goto parseerror;
330  }
331  }
332  else if (!Q_stricmp(token, "bool"))
333  {
334  if (def->numUniformValues < MAX_UNIFORM_VALUES - 1)
335  {
336  def->uniformValues[def->numUniformValues].type.type = GLSL_BOOL;
337  GLSL_CopyNextToken(text, &def->uniformValues[def->numUniformValues].type.name);
338  token = COM_ParseExt(text, qtrue);
339  valptr = Com_Allocate(sizeof(qboolean));
340  *((qboolean *)valptr) = atoi(token);
341  def->uniformValues[def->numUniformValues].value = valptr;
342  //Ren_Print("%d\n",*((qboolean*)valptr));
343  def->numUniformValues++;
344  }
345  else
346  {
347  Ren_Warning("GLSL_ParseDefinition: MAX_UNIFORM_VALUES reached.\n");
348  goto parseerror;
349  }
350  }
351  else
352  {
353  // FIXME: implement other formats
354  Ren_Warning("GLSL_ParseDefinition: uniform format not implemented.\n");
355  goto parseerror;
356  }
357  }
358  }
359 
360  return def;
361 
362 parseerror:
363  Com_Dealloc(def);
364  return NULL;
365 }

409 programInfo_t *GLSL_FindShader(const char *name)
410 {
411  char strippedName[MAX_QPATH];
412  int hash;
413  char *shaderText;
414  programInfo_t *sh;
415 
416  if (name[0] == 0)
417  {
418  return DEFAULT_SHADER_DEF;
419  }
420 
421  COM_StripExtension(name, strippedName, sizeof(strippedName));
422  COM_FixPath(strippedName);
423 
424  hash = Q_GenerateHashValue(strippedName, FILE_HASH_SIZE, qfalse, qtrue);
425 
426  for (sh = hashTable[hash]; sh; sh = sh->next)
427  {
428  // index by name
429  // the original way was correct
430  if (!Q_stricmp(sh->name, strippedName))
431  {
432  // match found
433  return sh;
434  }
435  }
436 
437  shaderText = GLSL_FindDefinitionInText(strippedName);
438  if (!shaderText)
439  {
440  Ren_Print("Shader definition find failed: %s\n", name);
441  return DEFAULT_SHADER_DEF;
442  }
443 
444  sh = GLSL_ParseDefinition(&shaderText, strippedName);
445  if (!sh)
446  {
447  Ren_Print("Shader definition parsing failed: %s\n", name);
448  return DEFAULT_SHADER_DEF;
449  }
450  else
451  {
452  sh->next = hashTable[hash];
453  hashTable[hash] = sh;
454  return sh;
455  }

576 qboolean GLSL_LoadShaderBinary(programInfo_t *info, size_t programNum)
577 {
578  GLint success;
579  GLint fileLength;
580  void *binary;
581  byte *binaryptr;
582  GLShaderHeader_t shaderHeader;
583  shaderProgram_t *shaderProgram;
584  unsigned int i;
585 
586  // we need to recompile the shaders
587  if (r_recompileShaders->integer)
588  {
589  return qfalse;
590  }
591 
592  // don't even try if the necessary functions aren't available
593  if (!glConfig2.getProgramBinaryAvailable)
594  {
595  return qfalse;
596  }
597 
598  fileLength = ri.FS_ReadFile(va("glsl/%s/%s_%u.bin", info->name, info->name, ( unsigned int ) programNum), &binary);
599 
600  // file empty or not found
601  if (fileLength <= 0)
602  {
603  return qfalse;
604  }
605 
606  binaryptr = ( byte * )binary;
607 
608  // get the shader header from the file
609  Com_Memcpy(&shaderHeader, binaryptr, sizeof(shaderHeader));
610  binaryptr += sizeof(shaderHeader);
611 
612  // check if this shader binary is the correct format
613  if (shaderHeader.version != GL_SHADER_VERSION)
614  {
615  ri.FS_FreeFile(binary);
616  return qfalse;
617  }
618 
619  // make sure this shader uses the same number of macros
620  if (shaderHeader.numMacros != info->numMacros)
621  {
622  ri.FS_FreeFile(binary);
623  return qfalse;
624  }
625 
626  // make sure this shader uses the same macros
627  for (i = 0; i < shaderHeader.numMacros; i++)
628  {
629  if (info->macros[i] != shaderHeader.macros[i])
630  {
631  ri.FS_FreeFile(binary);
632  return qfalse;
633  }
634  }
635 
636  // make sure the checksums for the source code match
637  if (shaderHeader.checkSum != info->checkSum)
638  {
639  ri.FS_FreeFile(binary);
640  return qfalse;
641  }
642 
643  // load the shader
644  shaderProgram = &info->list->programs[programNum];
645  shaderProgram->program = glCreateProgram();
646  glProgramBinary(shaderProgram->program, shaderHeader.binaryFormat, binaryptr, shaderHeader.binaryLength);
647  glGetProgramiv(shaderProgram->program, GL_LINK_STATUS, &success);
648 
649  if (!success)
650  {
651  ri.FS_FreeFile(binary);
652  glDeleteProgram(shaderProgram->program);
653  return qfalse;
654  }
655 
656  ri.FS_FreeFile(binary);
657  return qtrue;
658 }
