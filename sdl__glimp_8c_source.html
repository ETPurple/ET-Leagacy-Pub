152 qboolean GLimp_GetModeInfo(int *width, int *height, float *windowAspect, int mode)
153 {
154  vidmode_t *vm;
155  float pixelAspect;
156 
157  if (mode < -1)
158  {
159  return qfalse;
160  }
161  if (mode >= s_numVidModes)
162  {
163  return qfalse;
164  }
165 
166  if (mode == -1)
167  {
168  *width = r_customwidth->integer;
169  *height = r_customheight->integer;
170  pixelAspect = r_customaspect->value;
171  }
172  else
173  {
174  vm = &glimp_vidModes[mode];
175 
176  *width = vm->width;
177  *height = vm->height;
178  pixelAspect = vm->pixelAspect;
179  }
180 
181  *windowAspect = (float)*width / (*height * pixelAspect);
182 
183  return qtrue;
184 }

272 static int GLimp_CompareModes(const void *a, const void *b)
273 {
274  const float ASPECT_EPSILON = 0.001f;
275  const SDL_Rect *modeA = (const SDL_Rect *)a;
276  const SDL_Rect *modeB = (const SDL_Rect *)b;
277  float aspectA = modeA->w / (float)modeA->h;
278  float aspectB = modeB->w / (float)modeB->h;
279  int areaA = modeA->w * modeA->h;
280  int areaB = modeB->w * modeB->h;
281  float aspectDiffA = Q_fabs(aspectA - displayAspect);
282  float aspectDiffB = Q_fabs(aspectB - displayAspect);
283  float aspectDiffsDiff = aspectDiffA - aspectDiffB;
284 
285  if (aspectDiffsDiff > ASPECT_EPSILON)
286  {
287  return 1;
288  }
289  else if (aspectDiffsDiff < -ASPECT_EPSILON)
290  {
291  return -1;
292  }
293  else
294  {
295  return areaA - areaB;
296  }
297 }

302 static void GLimp_DetectAvailableModes(void)
303 {
304  int i, j;
305  char buf[MAX_STRING_CHARS] = { 0 };
306  SDL_Rect modes[128];
307  int numModes = 0;
308  int display = 0;
309  SDL_DisplayMode windowMode;
310 
311  if (!main_window)
312  {
313  if (!SDL_GetNumVideoDisplays())
314  {
315  Com_Error(ERR_VID_FATAL, "There is no available display to open a game screen - %s", SDL_GetError());
316  }
317 
318  // Use the zero display index
319  display = 0;
320  }
321  else
322  {
323  // Detect the used display
324  display = SDL_GetWindowDisplayIndex(main_window);
325  }
326 
327  // was SDL_GetWindowDisplayMode
328  if (SDL_GetDesktopDisplayMode(display, &windowMode) < 0)
329  {
330  Com_Printf(S_COLOR_YELLOW "Couldn't get desktop display mode, no resolutions detected - %s\n", SDL_GetError());
331  return;
332  }
333 
334  for (i = 0; i < SDL_GetNumDisplayModes(display); i++)
335  {
336  SDL_DisplayMode mode;
337 
338  if (SDL_GetDisplayMode(display, i, &mode) < 0)
339  {
340  continue;
341  }
342 
343  if (!mode.w || !mode.h)
344  {
345  Com_Printf("Display supports any resolution\n");
346  return;
347  }
348 
349  if (windowMode.format != mode.format)
350  {
351  continue;
352  }
353 
354  // SDL can give the same resolution with different refresh rates.
355  // Only list resolution once.
356  for (j = 0; j < numModes; j++)
357  {
358  if (mode.w == modes[j].w && mode.h == modes[j].h)
359  {
360  break;
361  }
362  }
363 
364  if (j != numModes)
365  {
366  continue;
367  }
368 
369  modes[numModes].w = mode.w;
370  modes[numModes].h = mode.h;
371  numModes++;
372  }
373 
374  if (numModes > 1)
375  {
376  qsort(modes, numModes, sizeof(SDL_Rect), GLimp_CompareModes);
377  }
378 
379  for (i = 0; i < numModes; i++)
380  {
381  const char *newModeString = va("%ux%u ", modes[i].w, modes[i].h);
382 
383  if (strlen(newModeString) < (int)sizeof(buf) - strlen(buf))
384  {
385  Q_strcat(buf, sizeof(buf), newModeString);
386  }
387  else
388  {
389  Com_Printf(S_COLOR_YELLOW "Skipping mode %ux%u, buffer too small\n", modes[i].w, modes[i].h);
390  }
391  }
392 
393  if (*buf)
394  {
395  buf[strlen(buf) - 1] = 0;
396  Com_Printf("Available modes [%i]: '%s'\n", numModes, buf);
397  Cvar_Set("r_availableModes", buf);
398  }
399 }

410 static int GLimp_SetMode(glconfig_t *glConfig, int mode, qboolean fullscreen, qboolean noborder, windowContext_t *context)
411 {
412  int perChannelColorBits;
413  int colorBits, depthBits, stencilBits;
414  int samples;
415  int i = 0;
416  SDL_Surface *icon = NULL;
417  SDL_DisplayMode desktopMode;
418  int display = 0;
419  int x = SDL_WINDOWPOS_UNDEFINED, y = SDL_WINDOWPOS_UNDEFINED;
420 
421  Uint32 flags = SDL_WINDOW_OPENGL | SDL_WINDOW_INPUT_GRABBED;
422 
423  Com_Printf("Initializing OpenGL display\n");
424 
425  if (r_allowResize->integer && !fullscreen)
426  {
427  flags |= SDL_WINDOW_RESIZABLE;
428  }
429 
430  icon = SDL_CreateRGBSurfaceFrom(
431  (void *)CLIENT_WINDOW_ICON.pixel_data,
432  CLIENT_WINDOW_ICON.width,
433  CLIENT_WINDOW_ICON.height,
434  CLIENT_WINDOW_ICON.bytes_per_pixel * 8,
435  CLIENT_WINDOW_ICON.bytes_per_pixel * CLIENT_WINDOW_ICON.width,
436 #ifdef Q3_LITTLE_ENDIAN
437  0x000000FF, 0x0000FF00, 0x00FF0000, 0xFF000000
438 #else
439  0xFF000000, 0x00FF0000, 0x0000FF00, 0x000000FF
440 #endif
441  );
442 
443  // If a window exists, note its display index
444  if (main_window != NULL)
445  {
446  display = SDL_GetWindowDisplayIndex(main_window);
447  }
448 
449  if (SDL_GetDesktopDisplayMode(display, &desktopMode) == 0)
450  {
451  displayAspect = (double)desktopMode.w / (double)desktopMode.h;
452 
453  Com_Printf("Estimated display aspect: %.3f\n", displayAspect);
454  }
455  else
456  {
457  Com_Memset(&desktopMode, 0, sizeof(SDL_DisplayMode));
458 
459  Com_Printf("Cannot estimate display aspect, assuming 1.333\n");
460  }
461 
462  Com_Printf("...setting mode %d: ", mode);
463 
464  if (mode == -2)
465  {
466  // use desktop video resolution
467  if (desktopMode.h > 0)
468  {
469  glConfig->vidWidth = desktopMode.w;
470  glConfig->vidHeight = desktopMode.h;
471  }
472  else
473  {
474  glConfig->vidWidth = 640;
475  glConfig->vidHeight = 480;
476  Com_Printf("Cannot determine display resolution, assuming 640x480\n");
477  }
478 
479  glConfig->windowAspect = (float)glConfig->vidWidth / (float)glConfig->vidHeight;
480  }
481  else if (!GLimp_GetModeInfo(&glConfig->vidWidth, &glConfig->vidHeight, &glConfig->windowAspect, mode))
482  {
483  Com_Printf("invalid mode\n");
484  return RSERR_INVALID_MODE;
485  }
486  Com_Printf("%dx%d\n", glConfig->vidWidth, glConfig->vidHeight);
487 
488  // Center window
489  if (r_centerWindow->integer && !fullscreen)
490  {
491  x = (desktopMode.w / 2) - (glConfig->vidWidth / 2);
492  y = (desktopMode.h / 2) - (glConfig->vidHeight / 2);
493  }
494 
495  // Destroy existing state if it exists
496  if (SDL_glContext != NULL)
497  {
498  SDL_GL_DeleteContext(SDL_glContext);
499  SDL_glContext = NULL;
500  }
501 
502  if (main_window != NULL)
503  {
504  SDL_GetWindowPosition(main_window, &x, &y);
505  Com_Printf(S_COLOR_YELLOW "Existing window at %dx%d before being destroyed\n", x, y);
506  SDL_DestroyWindow(main_window);
507  main_window = NULL;
508  }
509 
510  if (fullscreen)
511  {
512  if (r_mode->integer == -2)
513  {
514  flags |= SDL_WINDOW_FULLSCREEN_DESKTOP;
515  }
516  else
517  {
518  flags |= SDL_WINDOW_FULLSCREEN;
519  }
520 
521  glConfig->isFullscreen = qtrue;
522  }
523  else
524  {
525  if (noborder)
526  {
527  flags |= SDL_WINDOW_BORDERLESS;
528  }
529 
530  glConfig->isFullscreen = qfalse;
531  }
532 
533  colorBits = r_colorbits->integer;
534  if ((!colorBits) || (colorBits >= 32))
535  {
536  colorBits = 24;
537  }
538 
539  if (r_depthbits->value == 0.f)
540  {
541  depthBits = 24;
542  }
543  else
544  {
545  depthBits = r_depthbits->integer;
546  }
547  stencilBits = r_stencilbits->integer;
548  samples = r_ext_multisample->integer;
549 
550  for (i = 0; i < 16; i++)
551  {
552  int testColorBits, testDepthBits, testStencilBits;
553 
554  // 0 - default
555  // 1 - minus colorBits
556  // 2 - minus depthBits
557  // 3 - minus stencil
558  if ((i % 4) == 0 && i)
559  {
560  // one pass, reduce
561  switch (i / 4)
562  {
563  case 2:
564  if (colorBits == 24)
565  {
566  colorBits = 16;
567  }
568  break;
569  case 1:
570  if (depthBits == 32)
571  {
572  depthBits = 24;
573  }
574  else if (depthBits == 24)
575  {
576  depthBits = 16;
577  }
578  else if (depthBits == 16)
579  {
580  depthBits = 8;
581  }
582  case 3: // fall through
583  if (stencilBits == 24)
584  {
585  stencilBits = 16;
586  }
587  else if (stencilBits == 16)
588  {
589  stencilBits = 8;
590  }
591  }
592  }
593 
594  testColorBits = colorBits;
595  testDepthBits = depthBits;
596  testStencilBits = stencilBits;
597 
598  if ((i % 4) == 3) // reduce colorbits
599  {
600  if (testColorBits == 24)
601  {
602  testColorBits = 16;
603  }
604  }
605 
606  if ((i % 4) == 2) // reduce depthbits
607  {
608  if (testDepthBits == 24)
609  {
610  testDepthBits = 16;
611  }
612  else if (testDepthBits == 16)
613  {
614  testDepthBits = 8;
615  }
616  }
617 
618  if ((i % 4) == 1) // reduce stencilbits
619  {
620  if (testStencilBits == 24)
621  {
622  testStencilBits = 16;
623  }
624  else if (testStencilBits == 16)
625  {
626  testStencilBits = 8;
627  }
628  else
629  {
630  testStencilBits = 0;
631  }
632  }
633 
634  if (testColorBits == 24)
635  {
636  perChannelColorBits = 8;
637  }
638  else
639  {
640  perChannelColorBits = 4;
641  }
642 
643 #ifdef __sgi // Fix for SGIs grabbing too many bits of color
644  if (perChannelColorBits == 4)
645  {
646  perChannelColorBits = 0; /* Use minimum size for 16-bit color */
647 
648  }

650  SDL_GL_SetAttribute(SDL_GL_ALPHA_SIZE, 1);
651 #endif
652 
653  SDL_GL_SetAttribute(SDL_GL_RED_SIZE, perChannelColorBits);
654  SDL_GL_SetAttribute(SDL_GL_GREEN_SIZE, perChannelColorBits);
655  SDL_GL_SetAttribute(SDL_GL_BLUE_SIZE, perChannelColorBits);
656  SDL_GL_SetAttribute(SDL_GL_DEPTH_SIZE, testDepthBits);
657  SDL_GL_SetAttribute(SDL_GL_STENCIL_SIZE, testStencilBits);
658 
659  SDL_GL_SetAttribute(SDL_GL_MULTISAMPLEBUFFERS, samples ? 1 : 0);
660  SDL_GL_SetAttribute(SDL_GL_MULTISAMPLESAMPLES, samples);
661 
662  // SDL2 uses opengl by default, if we want opengl es we need to set this attribute
663  //SDL_GL_SetAttribute(SDL_GL_CONTEXT_EGL, 1);
664 
665  SDL_GL_SetAttribute(SDL_GL_DOUBLEBUFFER, 1);
666 
667  // If not allowing software GL, demand accelerated
668  if (!r_allowSoftwareGL->integer)
669  {
670  SDL_GL_SetAttribute(SDL_GL_ACCELERATED_VISUAL, 1);
671  }
672 
673  main_window = SDL_CreateWindow(CLIENT_WINDOW_TITLE, x, y, glConfig->vidWidth, glConfig->vidHeight, flags | SDL_WINDOW_SHOWN);
674 
675  if (!main_window)
676  {
677  Com_Printf("SDL_CreateWindow failed: %s\n", SDL_GetError());
678  continue;
679  }
680 
681  if (fullscreen)
682  {
683  SDL_DisplayMode modefullScreen;
684 
685  switch (testColorBits)
686  {
687  case 16: modefullScreen.format = SDL_PIXELFORMAT_RGB565; break;
688  case 24: modefullScreen.format = SDL_PIXELFORMAT_RGB24; break;
689  default: Com_Printf("SDL_SetWindowDisplayMode failed: testColorBits is %d, can't fullscreen\n", testColorBits); continue;
690  }
691 
692  modefullScreen.w = glConfig->vidWidth;
693  modefullScreen.h = glConfig->vidHeight;
694  modefullScreen.refresh_rate = glConfig->displayFrequency = r_displayRefresh->integer;
695  modefullScreen.driverdata = NULL;
696 
697  if (SDL_SetWindowDisplayMode(main_window, &modefullScreen) < 0)
698  {
699  Com_Printf("SDL_SetWindowDisplayMode failed: %s\n", SDL_GetError());
700  continue;
701  }
702  }
703 
704  SDL_SetWindowIcon(main_window, icon);
705 
706  if (context && context->versionMajor > 0)
707  {
708  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, context->versionMajor);
709  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, context->versionMinor);
710 
711  switch (context->context)
712  {
713  case GL_CONTEXT_COMP:
714  SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_COMPATIBILITY);
715  break;
716  case GL_CONTEXT_CORE:
717  SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_CORE);
718  SDL_GL_SetAttribute(SDL_GL_CONTEXT_FLAGS, SDL_GL_CONTEXT_FORWARD_COMPATIBLE_FLAG);
719  break;
720  case GL_CONTEXT_EGL:

1020 void GLimp_SetGamma(unsigned char red[256], unsigned char green[256], unsigned char blue[256])
1021 {
1022  Uint16 table[3][256];
1023  int i, j;
1024 
1025  if (!cls.glconfig.deviceSupportsGamma || r_ignorehwgamma->integer > 0)
1026  {
1027  Com_Printf(S_COLOR_YELLOW "Device doesn't support gamma or r_ignorehwgamma is set.\n");
1028  return;
1029  }
1030 
1031  for (i = 0; i < 256; i++)
1032  {
1033  table[0][i] = ((( Uint16 ) red[i]) << 8) | red[i];
1034  table[1][i] = ((( Uint16 ) green[i]) << 8) | green[i];
1035  table[2][i] = ((( Uint16 ) blue[i]) << 8) | blue[i];
1036  }
1037 
1038 #ifdef _WIN32
1039 
1040  // Win2K and newer put this odd restriction on gamma ramps...
1041  {
1042  OSVERSIONINFO vinfo;
1043 
1044  vinfo.dwOSVersionInfoSize = sizeof(vinfo);
1045  GetVersionEx(&vinfo);
1046  if (vinfo.dwMajorVersion >= 5 && vinfo.dwPlatformId == VER_PLATFORM_WIN32_NT)
1047  {
1048  Com_DPrintf("performing gamma clamp.\n");
1049  for (j = 0 ; j < 3 ; j++)
1050  {
1051  for (i = 0 ; i < 128 ; i++)
1052  {
1053  if (table[j][i] > ((128 + i) << 8))
1054  {
1055  table[j][i] = (128 + i) << 8;
1056  }
1057  }
1058 
1059  if (table[j][127] > 254 << 8)
1060  {
1061  table[j][127] = 254 << 8;
1062  }
1063  }
1064  }
1065  }
1066 #endif
1067 
1068  // enforce constantly increasing
1069  for (j = 0; j < 3; j++)
1070  {
1071  for (i = 1; i < 256; i++)
1072  {
1073  if (table[j][i] < table[j][i - 1])
1074  {
1075  table[j][i] = table[j][i - 1];
1076  }
1077  }
1078  }
1079 
1080  SDL_SetWindowGammaRamp(main_window, table[0], table[1], table[2]);
1081 }
