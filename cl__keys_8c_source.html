256 void Field_VariableSizeDraw(field_t *edit, int x, int y, int width, int size, qboolean showCursor, qboolean noColorEscape)
257 {
258  int len = strlen(edit->buffer);
259  int drawLen = edit->widthInChars;
260  int prestep;
261  char str[MAX_STRING_CHARS];
262 
263  // guarantee that cursor will be visible
264  if (len <= drawLen)
265  {
266  prestep = 0;
267  }
268  else
269  {
270  if (edit->scroll + drawLen > len)
271  {
272  edit->scroll = len - drawLen;
273  if (edit->scroll < 0)
274  {
275  edit->scroll = 0;
276  }
277  }
278  prestep = edit->scroll;
279  }
280 
281  if (prestep + drawLen > len)
282  {
283  drawLen = len - prestep;
284  }
285 
286  // extract <drawLen> characters from the field at <prestep>
287  if (drawLen >= MAX_STRING_CHARS)
288  {
289  Com_Error(ERR_DROP, "drawLen >= MAX_STRING_CHARS");
290  return;
291  }
292 
293  Com_Memcpy(str, edit->buffer + prestep, drawLen);
294  str[drawLen] = 0;
295 
296  // draw it
297  if (size == SMALLCHAR_WIDTH)
298  {
299  SCR_DrawSmallString(x, y, str, colorWhite, qfalse, noColorEscape);
300  }
301  else
302  {
303  // draw big string with drop shadow
304  SCR_DrawBigString(x, y, str, colorWhite, noColorEscape);
305  }
306 
307  // draw the cursor
308  if (showCursor)
309  {
310  int cursorChar, i;
311 
312  if ((int)(cls.realtime >> 8) & 1)
313  {
314  return; // off blink
315  }
316 
317  if (key_overstrikeMode)
318  {
319  cursorChar = 11;
320  }
321  else
322  {
323  cursorChar = 10;
324  }
325 
326  if (!noColorEscape)
327  {
328  Q_CleanStr(str);
329  }
330  i = drawLen - strlen(str);
331 
332  if (size == SMALLCHAR_WIDTH)
333  {
334  SCR_DrawSmallChar(x + (edit->cursor - prestep - i) * size, y, cursorChar);
335  }
336  else
337  {
338  str[0] = cursorChar;
339  str[1] = 0;
340  SCR_DrawBigString(x + (edit->cursor - prestep - i) * size, y, str, colorWhite, qfalse);
341  }
342  }
343 }

410 void Field_KeyDownEvent(field_t *edit, int key)
411 {
412  int len;
413 
414  // shift-insert is paste
415  if (((key == K_INS) || (key == K_KP_INS)) && (keys[K_RSHIFT].down || keys[K_LSHIFT].down))
416  {
417  Field_Paste(edit);
418  return;
419  }
420 
421  key = tolower(key);
422  len = strlen(edit->buffer);
423 
424  switch (key)
425  {
426  case K_DEL:
427  case K_KP_DEL:
428  if (edit->cursor < len)
429  {
430  memmove(edit->buffer + edit->cursor,
431  edit->buffer + edit->cursor + 1, len - edit->cursor);
432  }
433  break;
434  case K_RIGHTARROW:
435  case K_KP_RIGHTARROW:
436  if (edit->cursor < len)
437  {
438  edit->cursor++;
439  }
440  break;
441  case K_LEFTARROW:
442  case K_KP_LEFTARROW:
443  if (edit->cursor > 0)
444  {
445  edit->cursor--;
446  }
447  break;
448  case K_HOME:
449  case K_KP_HOME:
450  edit->cursor = 0;
451  break;
452  case 'a':
453  if (keys[K_LCTRL].down || keys[K_RCTRL].down)
454  {
455  edit->cursor = 0;
456  }
457  break;
458  case K_END:
459  case K_KP_END:
460  edit->cursor = len;
461  break;
462  case 'e':
463  if (keys[K_LCTRL].down || keys[K_RCTRL].down)
464  {
465  edit->cursor = len;
466  }
467  break;
468  case K_INS:
469  case K_KP_INS:
470  key_overstrikeMode = !key_overstrikeMode;
471  break;
472  default:
473  break;
474  }
475 
476  // Change scroll if cursor is no longer visible
477  if (edit->cursor < edit->scroll)
478  {
479  edit->scroll = edit->cursor;
480  }
481  else if (edit->cursor >= edit->scroll + edit->widthInChars && edit->cursor <= len)
482  {
483  edit->scroll = edit->cursor - edit->widthInChars + 1;
484  }
485 }

492 void Field_CharEvent(field_t *edit, int ch)
493 {
494  int len, stringLen;
495  //int charWidth;
496  //char *value = NULL;
497 
498  //charWidth = Q_UTF8_WidthCP(ch);
499  //value = Q_UTF8_Encode(ch);
500 
501  if (ch == 'v' - 'a' + 1) // ctrl-v is paste
502  {
503  Field_Paste(edit);
504  return;
505  }
506 
507  if (ch == 'c' - 'a' + 1) // ctrl-c clears the field
508  {
509  Field_Clear(edit);
510  return;
511  }
512 
513  len = strlen(edit->buffer);
514  stringLen = Q_UTF8_Strlen(edit->buffer);
515 
516  if (ch == 'h' - 'a' + 1) // ctrl-h is backspace
517  {
518  if (edit->cursor > 0)
519  {
520  memmove(edit->buffer + edit->cursor - 1, edit->buffer + edit->cursor, len + 1 - edit->cursor);
521  edit->cursor--;
522  if (edit->cursor < edit->scroll)
523  {
524  edit->scroll--;
525  }
526  }
527  return;
528  }
529 
530  if (ch == 'a' - 'a' + 1) // ctrl-a is home
531  {
532  edit->cursor = 0;
533  edit->scroll = 0;
534  return;
535  }
536 
537  if (ch == 'e' - 'a' + 1) // ctrl-e is end
538  {
539  edit->cursor = len;
540  edit->scroll = edit->cursor - edit->widthInChars;
541  return;
542  }
543 
544  // ignore any other non printable chars
545  if (ch < 32)
546  {
547  return;
548  }
549 
550  // - 2 to leave room for the leading slash and trailing \0
551  if (len == MAX_EDIT_LINE - 2)
552  {
553  return; // all full
554  }
555 
556  Q_UTF8_Insert(edit->buffer, stringLen, edit->cursor, ch, key_overstrikeMode);
557  edit->cursor++;
558 
559 
560  if (edit->cursor >= edit->widthInChars)
561  {
562  edit->scroll++;
563  }
564 
565  if (edit->cursor == stringLen + 1)
566  {
567  edit->buffer[Q_UTF8_ByteOffset(edit->buffer, edit->cursor)] = 0;
568  }
569 }

581 void Console_Key(int key)
582 {
583  // ctrl-L clears screen
584  if (key == 'l' && (keys[K_LCTRL].down || keys[K_RCTRL].down))
585  {
586  Cbuf_AddText("clear\n");
587  return;
588  }
589 
590  // enter finishes the line
591  if (key == K_ENTER || key == K_KP_ENTER)
592  {
593  con.highlightOffset = 0;
594 
595 #if SLASH_COMMAND
596  // if not in the game explicitly prepend a slash if needed
597  if (cls.state != CA_ACTIVE && g_consoleField.buffer[0] != '\\'
598  && g_consoleField.buffer[0] != '/')
599  {
600  char temp[MAX_STRING_CHARS];
601 
602  Q_strncpyz(temp, g_consoleField.buffer, sizeof(temp));
603  Com_sprintf(g_consoleField.buffer, sizeof(g_consoleField.buffer), "\\%s", temp);
604  g_consoleField.cursor++;
605  }
606 #endif
607 
608  Com_Printf("]%s\n", g_consoleField.buffer);
609 
610 #if SLASH_COMMAND
611  // leading slash is an explicit command
612  if (g_consoleField.buffer[0] == '\\' || g_consoleField.buffer[0] == '/')
613  {
614  Cbuf_AddText(g_consoleField.buffer + 1); // valid command
615  Cbuf_AddText("\n");
616  }
617  else
618  {
619  // other text will be chat messages
620  if (!g_consoleField.buffer[0])
621  {
622  return; // empty lines just scroll the console without adding to history
623  }
624  else
625  {
626  Cbuf_AddText("cmd say ");
627  Cbuf_AddText(g_consoleField.buffer);
628  Cbuf_AddText("\n");
629  }
630  }
631 #else
632  Cbuf_AddText(g_consoleField.buffer); // valid command
633  Cbuf_AddText("\n");
634 
635  if (!g_consoleField.buffer[0])
636  {
637  return; // empty lines just scroll the console without adding to history
638  }
639 #endif
640 
641  // copy line to history buffer
642  historyEditLines[nextHistoryLine % COMMAND_HISTORY] = g_consoleField;
643  nextHistoryLine++;
644  historyLine = nextHistoryLine;
645 
646  Field_Clear(&g_consoleField);
647 
648  g_consoleField.widthInChars = g_console_field_width;
649 
650  if (cls.state == CA_DISCONNECTED)
651  {
652  SCR_UpdateScreen(); // force an update, because the command
653  } // may take some time
654  return;
655  }
656 
657  // command completion
658 
659  if (key == K_TAB)
660  {
661  Console_AutoComplete(&g_consoleField, &con.highlightOffset);
662  return;
663  }
664 
665  // clear autocompletion buffer on normal key input
666  if ((key >= K_SPACE && key <= K_BACKSPACE) || (key == K_LEFTARROW) || (key == K_RIGHTARROW) ||
667  (key >= K_KP_LEFTARROW && key <= K_KP_RIGHTARROW) ||
668  (key >= K_KP_SLASH && key <= K_KP_PLUS) || (key >= K_KP_STAR && key <= K_KP_EQUALS))
669  {
670  if (key == K_BACKSPACE && con.highlightOffset)
671  {
672  Console_RemoveHighlighted(&g_consoleField, &con.highlightOffset);
673  }
674  else
675  {
676  con.highlightOffset = 0;
677  }
678  }
679 
680  // command history (ctrl-p ctrl-n for unix style)
681 
682  // added some mousewheel functionality to the console
683  if ((key == K_MWHEELUP && (keys[K_RSHIFT].down || keys[K_LSHIFT].down)) || (key == K_UPARROW) || (key == K_KP_UPARROW) ||
684  ((tolower(key) == 'p') && (keys[K_LCTRL].down || keys[K_RCTRL].down)))
685  {
686  if (nextHistoryLine - historyLine < COMMAND_HISTORY
687  && historyLine > 0)
688  {
689  historyLine--;
690  }
691  g_consoleField = historyEditLines[historyLine % COMMAND_HISTORY];
692  con.highlightOffset = 0;
693  return;
694  }
695 
696  // added some mousewheel functionality to the console
697 
698  if ((key == K_MWHEELDOWN && (keys[K_RSHIFT].down || keys[K_LSHIFT].down)) || (key == K_DOWNARROW) || (key == K_KP_DOWNARROW) ||
699  ((tolower(key) == 'n') && (keys[K_LCTRL].down || keys[K_RCTRL].down)))
700  {
701  if (historyLine == nextHistoryLine)
702  {
703  return;
704  }
705  historyLine++;
706  g_consoleField = historyEditLines[historyLine % COMMAND_HISTORY];
707  con.highlightOffset = 0;
708  return;
709  }
710 
711  // console scrolling
712  if (key == K_PGUP || key == K_KP_PGUP)
713  {
714  Con_PageUp();
715  return;
716  }
717 
718  if (key == K_PGDN || key == K_KP_PGDN)
719  {
720  Con_PageDown();
721  return;
722  }
723 
724  if (key == K_MWHEELUP) // added some mousewheel functionality to the console
725  {
726  if (keys[K_LCTRL].down || keys[K_RCTRL].down) // hold <ctrl> to accelerate scrolling
727  {
728  Con_ScrollUp(con.visibleLines);
729  }
730  else
731  {
732  Con_ScrollUp(con.visibleLines / 4);
733  }
734  return;
735  }
736 
737  if (key == K_MWHEELDOWN) // added some mousewheel functionality to the console
738  {
739  if (keys[K_LCTRL].down || keys[K_RCTRL].down) // hold <ctrl> to accelerate scrolling
740  {
741  Con_ScrollDown(con.visibleLines);
742  }
743  else
744  {
745  Con_ScrollDown(con.visibleLines / 4);
746  }
747  return;
748  }
749 
750  // ctrl-home = top of console
751  if ((key == K_HOME || key == K_KP_HOME) && (keys[K_LCTRL].down || keys[K_RCTRL].down))
752  {
753  Con_ScrollTop();
754  return;
755  }
756 
757  // ctrl-end = bottom of console
758  if ((key == K_END || key == K_KP_END) && (keys[K_LCTRL].down || keys[K_RCTRL].down))
759  {
760  Con_ScrollBottom();
761  return;
762  }
763 
764  // pass to the normal editline routine
765  Field_KeyDownEvent(&g_consoleField, key);
766 }
