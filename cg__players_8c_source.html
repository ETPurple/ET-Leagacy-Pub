489 void CG_RunLerpFrame(centity_t *cent, clientInfo_t *ci, lerpFrame_t *lf, int newAnimation, float speedScale)
490 {
491  // debugging tool to get no animations
492  if (cg_animSpeed.integer == 0)
493  {
494  lf->oldFrame = lf->frame = 0;
495  lf->backlerp = 0;
496  return;
497  }
498 
499  // see if the animation sequence is switching
500  if (ci && (newAnimation != lf->animationNumber || !lf->animation))
501  {
502  CG_SetLerpFrameAnimation(cent, ci, lf, newAnimation);
503  }
504 
505  // if we have passed the current frame, move it to
506  // oldFrame and calculate a new frame
507  if (cg.time >= lf->frameTime)
508  {
509  animation_t *anim;
510  int f;
511 
512  lf->oldFrame = lf->frame;
513  lf->oldFrameTime = lf->frameTime;
514  lf->oldFrameModel = lf->frameModel;
515 
516  // get the next frame based on the animation
517  anim = lf->animation;
518  if (!anim || !anim->frameLerp)
519  {
520  CG_Printf("Warning: CG_RunLerpFrame w/o animation.\n");
521  return; // shouldn't happen
522  }
523  if (cg.time < lf->animationTime)
524  {
525  lf->frameTime = lf->animationTime; // initial lerp
526  }
527  else
528  {
529  lf->frameTime = lf->oldFrameTime + anim->frameLerp;
530  }
531  f = (lf->frameTime - lf->animationTime) / anim->frameLerp;
532  f *= speedScale; // adjust for haste, etc
533  if (f >= anim->numFrames)
534  {
535  f -= anim->numFrames;
536  if (anim->loopFrames)
537  {
538  f %= anim->loopFrames;
539  f += anim->numFrames - anim->loopFrames;
540  }
541  else
542  {
543  f = anim->numFrames - 1;
544  // the animation is stuck at the end, so it
545  // can immediately transition to another sequence
546  lf->frameTime = cg.time;
547  }
548  }
549  lf->frame = anim->firstFrame + f;
550  lf->frameModel = anim->mdxFile;
551 
552  if (cg.time > lf->frameTime)
553  {
554  lf->frameTime = cg.time;
555  if (cg_debugAnim.integer)
556  {
557  CG_Printf("Clamp lf->frameTime\n");
558  }
559  }
560  }
561 
562  if (lf->frameTime > cg.time + 200)
563  {
564  lf->frameTime = cg.time;
565  }
566 
567  if (lf->oldFrameTime > cg.time)
568  {
569  lf->oldFrameTime = cg.time;
570  }
571  // calculate current lerp value
572  if (lf->frameTime == lf->oldFrameTime)
573  {
574  lf->backlerp = 0;
575  }
576  else
577  {
578  lf->backlerp = 1.0f - (float)(cg.time - lf->oldFrameTime) / (lf->frameTime - lf->oldFrameTime);
579  }
580 }
