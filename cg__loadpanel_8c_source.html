511 float campaignNameTextScaleFactor(int len)
512 {
513  float scaleF = 1.f;
514 
515  //CG_Printf("CampaignNameText len: %i\n", len);
516 
517  if (len >= 27)
518  {
519  scaleF *= 0.8f;
520  return scaleF;
521  }
522  // in between scale is 1
523  else if (len <= 20 && len > 17)
524  {
525  scaleF *= 1.25f;
526  return scaleF;
527  }
528  else if (len <= 17 && len > 13)
529  {
530  scaleF *= 1.5f;
531  return scaleF;
532  }

535  scaleF *= 2;
536  return scaleF;

672 void CG_LoadPanel_DrawPin(const char *text, float px, float py, float sx, float sy, qhandle_t shader, float pinsize, float backheight)
673 {
674  static vec4_t colourFadedBlack = { 0.f, 0.f, 0.f, 0.4f };

677 
678  px += cgs.wideXoffset;

680  // Pin half width is 16
681  // Pin left margin is 4
682  // Pin right margin is 0
683  // Text margin is 4

686  // x - pinhwidth (16) - pin left margin (4) - w - text margin (4) => x - w - 24

689  }

692  // Width = pinhwidth (16) + pin right margin (0) + w + text margin (4) = 20 + w

694  DC->fillRect(px, py - (backheight / 2.f), 20 + w, backheight, colorBlack);
695  }
696

698

700  {
