186 void CG_LoadLocations(void)
187 {
188  fileHandle_t f; // handle of file on disk
189  int fLen; // length of the file
190  char fBuffer[MAX_BUFFER]; // buffer to read the file into
191  char message[128] = "\0"; // location description
192  char temp[128] = "\0"; // temporary buffer
193  int x = 0; // x-coord of the location
194  int y = 0; // y-coord of the location
195  int z = 0; // z-coord of the location
196  int p = 0; // current location in the file buffer
197  int t = 0; // current location in the temp buffer
198 
199  fLen = trap_FS_FOpenFile(va("maps/%s_loc_override.dat", cgs.rawmapname), &f, FS_READ);
200 
201  if (fLen <= 0)
202  {
203  // open the location .dat file that matches the map's name
204  fLen = trap_FS_FOpenFile(va("maps/%s_loc.dat", cgs.rawmapname), &f, FS_READ);
205 
206  if (fLen <= 0)
207  {
208  CG_Printf(S_COLOR_BLUE "LoadLocations: ^3Warning: ^9No location data found for map ^2%s^9.\n", cgs.rawmapname);
209  return;
210  }
211  }
212 
213  if (fLen >= MAX_BUFFER)
214  {
215  trap_FS_FCloseFile(f);
216  CG_Error("Location file is too big, make it smaller (max = %i bytes)\n", MAX_BUFFER);
217  }
218 
219  trap_FS_Read(&fBuffer, fLen, f); // read the file into the buffer
220  fBuffer[fLen] = '\0'; // make sure it's null-terminated
221  trap_FS_FCloseFile(f); // close the file, we're done with it
222 
223  CG_Printf(S_COLOR_BLUE "LoadLocations: ^9location data for map ^2%s ^9loaded\n", cgs.rawmapname);
224 
225  // start parsing!
226  while (p < fLen)
227  {
228  // check for the beginning of a comment
229  if (fBuffer[p++] == '/')
230  {
231  //check for single line comment
232  if (fBuffer[p] == '/')
233  {
234  while (p < fLen && (fBuffer[p] != '\n' && fBuffer[p] != '\r'))
235  {
236  p++;
237  }
238  }
239  // check for multiline comment
240  else if (fBuffer[p] == '*')
241  {
242  while (p < fLen && (fBuffer[p] != '*' && fBuffer[p + 1] != '/'))
243  {
244  p++;
245  }
246  }
247  }
248 
249  // parse the next line
250  while (p < fLen && (fBuffer[p] != '\n' && fBuffer[p] != '\r'))
251  {
252  // grab the x-coord
253  while (p < fLen && fBuffer[p] != ' ')
254  {
255  temp[t++] = fBuffer[p++];
256  }
257  temp[t] = '\0';
258  x = atoi(temp);
259  t = 0;
260  Com_Memset(&temp, 0, sizeof(temp));
261 
262  if (p > fLen)
263  {
264  break;
265  }
266 
267  p++;
268 
269  // grab the y-coord
270  while (p < fLen && fBuffer[p] != ' ')
271  {
272  temp[t++] = fBuffer[p++];
273  }
274  temp[t] = '\0';
275  y = atoi(temp);
276  t = 0;
277  Com_Memset(&temp, 0, sizeof(temp));
278 
279  if (p > fLen)
280  {
281  break;
282  }
283 
284  p++;
285 
286  // grab the z-coord
287  while (p < fLen && fBuffer[p] != ' ')
288  {
289  temp[t++] = fBuffer[p++];
290  }
291  temp[t] = '\0';
292  z = atoi(temp);
293  t = 0;
294 
295  Com_Memset(&temp, 0, sizeof(temp));
296  if (p > fLen)
297  {
298  break;
299  }
300 
301  p++;
302 
303  // grab the description
304  while (p < fLen && fBuffer[p] != '\n' && fBuffer[p] != '\r')
305  {
306  // ignore quotation marks
307  if (fBuffer[p] != '\"')
308  {
309  temp[t++] = fBuffer[p++];
310  }
311  else
312  {
313  p++;
314  }
315  }
316  temp[t] = '\0';
317  t = 0;
318 
319  // if @, then keep the previous location name, otherwise, update message
320  if (Q_stricmp(temp, "@"))
321  {
322  strcpy(message, temp);
323  }
324 
325  if (p > fLen)
326  {
327  break;
328  }
329 
330  if ((x != 0 || y != 0 || z != 0) && strlen(message) > 0)
331  {
332  location_t *loc = &cgs.location[cgs.numLocations];
333 
334  loc->index = cgs.numLocations;
335  strcpy(loc->message, message);
336  loc->origin[0] = x;
337  loc->origin[1] = y;
338  loc->origin[2] = z;
339  cgs.numLocations++;
340 
341  if (cgs.numLocations == MAX_C_LOCATIONS)
342  {
343  CG_Printf("^9Too many locations specifed.\n");
344  break;
345  }
346  }
347  }
348  }
349  // ok we are succesfull
350  CG_Printf("^2%i ^9locations loaded.\n", cgs.numLocations);
351  cgs.locationsLoaded = qtrue;
352 }
