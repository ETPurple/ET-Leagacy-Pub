225 void GL_State(unsigned long stateBits)
226 {
227  unsigned long diff = stateBits ^ glState.glStateBits;
228 
229  if (!diff)
230  {
231  return;
232  }
233 
234  // check depthFunc bits
235  if (diff & GLS_DEPTHFUNC_EQUAL)
236  {
237  if (stateBits & GLS_DEPTHFUNC_EQUAL)
238  {
239  qglDepthFunc(GL_EQUAL);
240  }
241  else
242  {
243  qglDepthFunc(GL_LEQUAL);
244  }
245  }
246 
247  // check blend bits
248  if (diff & (GLS_SRCBLEND_BITS | GLS_DSTBLEND_BITS))
249  {
250  GLenum srcFactor, dstFactor;
251 
252  if (stateBits & (GLS_SRCBLEND_BITS | GLS_DSTBLEND_BITS))
253  {
254  switch (stateBits & GLS_SRCBLEND_BITS)
255  {
256  case GLS_SRCBLEND_ZERO:
257  srcFactor = GL_ZERO;
258  break;
259  case GLS_SRCBLEND_ONE:
260  srcFactor = GL_ONE;
261  break;
262  case GLS_SRCBLEND_DST_COLOR:
263  srcFactor = GL_DST_COLOR;
264  break;
265  case GLS_SRCBLEND_ONE_MINUS_DST_COLOR:
266  srcFactor = GL_ONE_MINUS_DST_COLOR;
267  break;
268  case GLS_SRCBLEND_SRC_ALPHA:
269  srcFactor = GL_SRC_ALPHA;
270  break;
271  case GLS_SRCBLEND_ONE_MINUS_SRC_ALPHA:
272  srcFactor = GL_ONE_MINUS_SRC_ALPHA;
273  break;
274  case GLS_SRCBLEND_DST_ALPHA:

276  break;
277  case GLS_SRCBLEND_ONE_MINUS_DST_ALPHA:
278  srcFactor = GL_ONE_MINUS_DST_ALPHA;
279  break;
280  case GLS_SRCBLEND_ALPHA_SATURATE:
281  srcFactor = GL_SRC_ALPHA_SATURATE;
282  break;
283  default:
284  srcFactor = GL_ONE; // to get warning to shut up

286  }
287 
288  switch (stateBits & GLS_DSTBLEND_BITS)

291  dstFactor = GL_ZERO;
292  break;
293  case GLS_DSTBLEND_ONE:

297  dstFactor = GL_SRC_COLOR;
298  break;
299  case GLS_DSTBLEND_ONE_MINUS_SRC_COLOR:
300  dstFactor = GL_ONE_MINUS_SRC_COLOR;
301  break;
302  case GLS_DSTBLEND_SRC_ALPHA:
303  dstFactor = GL_SRC_ALPHA;
304  break;
305  case GLS_DSTBLEND_ONE_MINUS_SRC_ALPHA:
306  dstFactor = GL_ONE_MINUS_SRC_ALPHA;
307  break;
308  case GLS_DSTBLEND_DST_ALPHA:
309  dstFactor = GL_DST_ALPHA;
310  break;
311  case GLS_DSTBLEND_ONE_MINUS_DST_ALPHA:
312  dstFactor = GL_ONE_MINUS_DST_ALPHA;
313  break;
314  default:
315  dstFactor = GL_ONE; // to get warning to shut up
316  Ren_Drop("GL_State: invalid dst blend state bits\n");
317  }
318 
319  qglEnable(GL_BLEND);
320  qglBlendFunc(srcFactor, dstFactor);
321  }
322  else
323  {
324  qglDisable(GL_BLEND);
325  }
326  }
327 
328  // check depthmask
329  if (diff & GLS_DEPTHMASK_TRUE)
330  {
331  if (stateBits & GLS_DEPTHMASK_TRUE)
332  {
333  qglDepthMask(GL_TRUE);
334  }
335  else
336  {
337  qglDepthMask(GL_FALSE);
338  }
339  }
340 
341  // fill/line mode
342  if (diff & GLS_POLYMODE_LINE)
343  {
344  if (stateBits & GLS_POLYMODE_LINE)
345  {
346  qglPolygonMode(GL_FRONT_AND_BACK, GL_LINE);
347  }
348  else
349  {
350  qglPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
351  }
352  }
353 
354  // depthtest
355  if (diff & GLS_DEPTHTEST_DISABLE)
356  {
357  if (stateBits & GLS_DEPTHTEST_DISABLE)
358  {
359  qglDisable(GL_DEPTH_TEST);
360  }
361  else
362  {
363  qglEnable(GL_DEPTH_TEST);
364  }
365  }
366 
367  // alpha test
368  if (diff & GLS_ATEST_BITS)
369  {
370  switch (stateBits & GLS_ATEST_BITS)
371  {
372  case 0:
373  qglDisable(GL_ALPHA_TEST);
374  break;
375  case GLS_ATEST_GT_0:
376  qglEnable(GL_ALPHA_TEST);
377  qglAlphaFunc(GL_GREATER, 0.0f);
378  break;
379  case GLS_ATEST_LT_80:
380  qglEnable(GL_ALPHA_TEST);
381  qglAlphaFunc(GL_LESS, 0.5f);
382  break;
383  case GLS_ATEST_GE_80:
384  qglEnable(GL_ALPHA_TEST);
385  qglAlphaFunc(GL_GEQUAL, 0.5f);
386  break;
387  default:
388  etl_assert(0);
389  break;
390  }
391  }
392 
393  glState.glStateBits = stateBits;
394 }
