178 static void R_SetupEntityLightingGrid(trRefEntity_t *ent, vec3_t forcedOrigin)
179 {
180  vec3_t lightOrigin;
181  int pos[3];
182  int i, j;
183  bspGridPoint_t *gridPoint;
184  bspGridPoint_t *gridPoint2;
185  float frac[3];
186  int gridStep[3];
187  vec3_t direction;
188  float totalFactor;
189  float v;
190 
191  if (forcedOrigin)
192  {
193  VectorCopy(forcedOrigin, lightOrigin);
194  }
195  else
196  {
197  if (ent->e.renderfx & RF_LIGHTING_ORIGIN)
198  {
199  // seperate lightOrigins are needed so an object that is
200  // sinking into the ground can still be lit, and so
201  // multi-part models can be lit identically
202  VectorCopy(ent->e.lightingOrigin, lightOrigin);
203  }
204  else
205  {
206  VectorCopy(ent->e.origin, lightOrigin);
207  }
208  }
209 
210  VectorSubtract(lightOrigin, tr.world->lightGridOrigin, lightOrigin);
211  for (i = 0; i < 3; i++)
212  {
213  v = lightOrigin[i] * tr.world->lightGridInverseSize[i];
214  pos[i] = floor(v);
215  frac[i] = v - pos[i];
216  if (pos[i] < 0)
217  {
218  pos[i] = 0;
219  }
220  else if (pos[i] > tr.world->lightGridBounds[i] - 1)
221  {
222  pos[i] = tr.world->lightGridBounds[i] - 1;
223  }
224  }
225 
226  VectorClear(ent->ambientLight);
227  VectorClear(ent->directedLight);
228  VectorClear(direction);
229 
230  etl_assert(tr.world->lightGridData); // NULL with -nolight maps
231 
232  // trilerp the light value
233  gridStep[0] = 1; //sizeof(bspGridPoint_t);
234  gridStep[1] = tr.world->lightGridBounds[0]; // * sizeof(bspGridPoint_t);
235  gridStep[2] = tr.world->lightGridBounds[0] * tr.world->lightGridBounds[1]; // * sizeof(bspGridPoint_t);
236  gridPoint = tr.world->lightGridData + pos[0] * gridStep[0] + pos[1] * gridStep[1] + pos[2] * gridStep[2];
237 
238  totalFactor = 0;
239  for (i = 0; i < 8; i++)
240  {
241  float factor = 1.0;
242 
243  gridPoint2 = gridPoint;
244  for (j = 0; j < 3; j++)
245  {
246  if (i & (1 << j))
247  {
248  factor *= frac[j];
249  gridPoint2 += gridStep[j];
250  }
251  else
252  {
253  factor *= (1.0f - frac[j]);
254  }
255  }
256 
257  if (!(gridPoint2->ambientColor[0] + gridPoint2->ambientColor[1] + gridPoint2->ambientColor[2]))
258  {
259  continue; // ignore samples in walls
260  }
261 
262  totalFactor += factor;
263 
264  ent->ambientLight[0] += factor * gridPoint2->ambientColor[0];
265  ent->ambientLight[1] += factor * gridPoint2->ambientColor[1];
266  ent->ambientLight[2] += factor * gridPoint2->ambientColor[2];
267 
268  ent->directedLight[0] += factor * gridPoint2->directedColor[0];
269  ent->directedLight[1] += factor * gridPoint2->directedColor[1];
270  ent->directedLight[2] += factor * gridPoint2->directedColor[2];
271 
272  VectorMA(direction, factor, gridPoint2->direction, direction);
273  }
274 
275  if (totalFactor > 0 && totalFactor < 0.99f)
276  {
277  totalFactor = 1.0f / totalFactor;
278  VectorScale(ent->ambientLight, totalFactor, ent->ambientLight);
279  VectorScale(ent->directedLight, totalFactor, ent->directedLight);
280  }
281 
282  VectorNormalize2(direction, ent->lightDir);
283 
284  // cheats? check for single player?
285  if (tr.lightGridMulDirected != 0.f)
286  {
287  VectorScale(ent->directedLight, tr.lightGridMulDirected, ent->directedLight);
288  }
289  if (tr.lightGridMulAmbient != 0.f)
290  {
291  VectorScale(ent->ambientLight, tr.lightGridMulAmbient, ent->ambientLight);
292  }
293 }
