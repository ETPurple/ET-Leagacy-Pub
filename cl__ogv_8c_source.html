154 static qboolean OGV_LoadAudio(cinematic_t *cin)
155 {
156  qboolean anyDataTransferred = qtrue;
157  float **pcm;
158  int frames, frameNeeded;
159  int i, j;
160  short *ptr;
161  ogg_packet op;
162  vorbis_block vb;
163 
164  Com_Memset(&op, 0, sizeof(op));
165  Com_Memset(&vb, 0, sizeof(vb));
166  vorbis_block_init(&g_ogm->vd, &vb);
167 
168  while (anyDataTransferred && g_ogm->currentTime + MAX_AUDIO_PRELOAD > (int)(g_ogm->vd.granulepos * 1000 / g_ogm->vi.rate))
169  {
170  anyDataTransferred = qfalse;
171 
172  if ((frames = vorbis_synthesis_pcmout(&g_ogm->vd, &pcm)) > 0)
173  {
174  // vorbis -> raw
175  ptr = (short *)g_ogm->audioBuffer;
176 
177  frameNeeded = (SIZEOF_RAWBUFF) / (OGG_SAMPLEWIDTH * g_ogm->vi.channels);
178 
179  if (frames < frameNeeded)
180  {
181  frameNeeded = frames;
182  }
183 
184  for (i = 0; i < frameNeeded; i++)
185  {
186  for (j = 0; j < g_ogm->vi.channels; j++)
187  {
188  *(ptr++) = (short)((pcm[j][i] >= -1.0f && pcm[j][i] <= 1.0f) ? pcm[j][i] * 32767.f : 32767 * ((pcm[j][i] > 0.0f) - (pcm[j][i] < 0.0f)));
189  }
190  }
191 
192  // tell libvorbis how many samples we actually consumed (we ate them all!)
193  vorbis_synthesis_read(&g_ogm->vd, frameNeeded);
194 
195  if (!(cin->flags & CIN_silent))
196  {
197  S_RawSamples(0, frameNeeded, g_ogm->vi.rate, OGG_SAMPLEWIDTH, g_ogm->vi.channels, g_ogm->audioBuffer, 1.0f, 1.0f);
198  }
199 
200  anyDataTransferred = qtrue;
201  }
202 
203  if (!anyDataTransferred)
204  {
205  // op -> vorbis
206  if (ogg_stream_packetout(&g_ogm->os_audio, &op))
207  {
208  if (vorbis_synthesis(&vb, &op) == 0)
209  {
210  vorbis_synthesis_blockin(&g_ogm->vd, &vb);
211  }
212  anyDataTransferred = qtrue;
213  }
214  }
215  }
216 
217  vorbis_block_clear(&vb);
218 
219  return (qboolean)(g_ogm->currentTime + MIN_AUDIO_PRELOAD > (int)(g_ogm->vd.granulepos * 1000 / g_ogm->vi.rate));
220 }

279 static qboolean OGV_yuv_to_rgb24(yuv_buffer *yuv, theora_info *info, uint32_t *output)
280 {
281  int i, j;
282  int uv_ki = yuv->y_width / yuv->uv_width;
283  int uv_kj = yuv->y_height / yuv->uv_height;
284 
285  int y_offset = info->offset_x + yuv->y_stride * info->offset_y;
286  int uv_offset = info->offset_x / uv_ki + yuv->uv_stride * info->offset_y / uv_kj;
287  int y_p, uv_p;
288 
289  /*
290  if(!OGV_CheckFrame(yuv, info))
291  {
292  return qfalse;
293  }
294  */
295 
296  //FIXME: this is slow as fuck, might want to rewrite and use precalculated tables
297  for (j = 0; j < info->height; j++)
298  {
299  y_p = y_offset + j * yuv->y_stride;
300  uv_p = uv_offset + j / uv_kj * yuv->uv_stride;
301 
302  for (i = 0; i < info->width; i++)
303  {
304  //http://en.wikipedia.org/wiki/YUV
305  int y = yuv->y[y_p] - 16;
306  int u = yuv->u[uv_p] - 128;
307  int v = yuv->v[uv_p] - 128;
308  int r = Com_ByteClamp((y * 298 + 409 * v + 128) >> 8);
309  int g = Com_ByteClamp((y * 298 - 100 * v - 208 * u + 128) >> 8);
310  int b = Com_ByteClamp((y * 298 + 516 * u + 128) >> 8);
311 
312  *output = (uint32_t)LittleLong((r) | (g << 8) | (b << 16) | (255 << 24));
313  ++output;
314 
315  y_p++;
316  if (i % uv_ki == uv_ki - 1)
317  {
318  uv_p++;
319  }
320  }
321  }
322 
323  return qtrue;
324 }

342 {
343  int r = 0;

355 
356  if ((g_ogm->VFrameCount < th_frame && th_frame >= OGV_NextNeededVFrame(cin)) || !cin->frameBuffer[0])

361  }
362

368  }
369 
370  if (cin->frameBufferSize < g_ogm->th_info.width * g_ogm->th_info.height)
371  {
372 
373  cin->frameBufferSize = g_ogm->th_info.width * g_ogm->th_info.height;
374 
375  /* Free old output buffer */
376  if (cin->frameBuffer[0])
377  {
378  Com_Dealloc(cin->frameBuffer[0]);

398  {
399  r = -1;
400  }
401  }
402  }

412 static qboolean OGV_LoadFrame(cinematic_t *cin)
413 {
414  qboolean anyDataTransferred = qtrue;
415  qboolean needVOutputData = qtrue;
416  qboolean audioWantsMoreData = qfalse;
417  int status;
418 
419  while (anyDataTransferred && (needVOutputData || audioWantsMoreData))
420  {
421  anyDataTransferred = qfalse;
422  {
423  if (needVOutputData && (status = OGV_LoadVideoFrame(cin)))
424  {
425  needVOutputData = qfalse;
426  if (status > 0)
427  {
428  anyDataTransferred = qtrue;
429  }
430  else
431  {
432  // error (we don't need any videodata and we had no transferred)
433  anyDataTransferred = qfalse;
434  }
435  }
436 
437  if (needVOutputData || audioWantsMoreData)
438  {
439  // try to transfer Pages to the audio- and video-Stream
440  if (OGV_LoadPagesToStreams(cin))
441  {
442  // try to load a datablock from file
443  anyDataTransferred |= !OGV_LoadBlockToSync(cin);
444  }
445  else
446  {
447  // successful OGV_LoadPagesToStreams()
448  anyDataTransferred = qtrue;
449  }
450  }
451 
452  // load all Audio after loading new pages ...
453  if (g_ogm->VFrameCount > 1) // wait some videoframes (it's better to have some delay, than a lagy sound)
454  {
455  audioWantsMoreData = OGV_LoadAudio(cin);
456  }
457  }
458  }
459 
460  return (qboolean) !!anyDataTransferred;
461 }
