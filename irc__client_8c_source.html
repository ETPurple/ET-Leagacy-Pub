577 static qboolean IRC_Parser(char next)
578 {
579  qboolean has_msg = qfalse;
580 
581  switch (IRC_ParserState)
582  {
583 
584  /* Initial state; clear the message, then check input. ':'
585  * indicates there is a prefix, a digit indicates a numeric
586  * command, an upper-case letter indicates a string command.
587  * It's also possible we received an empty line - just skip
588  * it. Anything else is an error.
589  */
590  case IRC_PARSER_START:
591  IRC_ParserError = qfalse;
592  IRC_ParserInMessage = qfalse;
593  if (next == ':')
594  {
595  P_INIT_MESSAGE(PFX_NOS_START);
596  }
597  else if (next == '\r')
598  {
599  P_SET_STATE(LF);
600  }
601  else if (IS_DIGIT(next))
602  {
603  P_INIT_MESSAGE(NUM_COMMAND_2);
604  P_INIT_STRING(cmd_string);
605  }
606  else if (IS_UPPER(next))
607  {
608  P_INIT_MESSAGE(STR_COMMAND);
609  P_INIT_STRING(cmd_string);
610  }
611  else
612  {
613  P_ERROR(RECOVERY);
614  }
615  break;
616  /*
617  * Start of prefix; anything is accepted, except for '!', '@', ' '
618  * and control characters which all cause an error recovery.
619  */
620  case IRC_PARSER_PFX_NOS_START:
621  if (next == '!' || next == '@' || next == ' ' || IS_CNTRL(next))
622  {
623  P_AUTO_ERROR;
624  }
625  else
626  {
627  P_SET_STATE(PFX_NOS);
628  P_INIT_STRING(pfx_nickOrServer);
629  }
630  break;
631  /*
632  * Prefix, server or nick name. Control characters cause an error,
633  * ' ', '!' and '@' cause state changes.
634  */
635  case IRC_PARSER_PFX_NOS:
636  if (next == '!')
637  {
638  P_SET_STATE(PFX_USER_START);
639  }
640  else if (next == '@')
641  {
642  P_SET_STATE(PFX_HOST_START);
643  }
644  else if (next == ' ')
645  {
646  P_SET_STATE(COMMAND_START);
647  }
648  else if (IS_CNTRL(next))
649  {
650  P_AUTO_ERROR;
651  }
652  else
653  {
654  P_ADD_STRING(pfx_nickOrServer);
655  }
656  break;
657  /*
658  * Start of user name; anything goes, except for '!', '@', ' '
659  * and control characters which cause an error.
660  */
661  case IRC_PARSER_PFX_USER_START:
662  if (next == '!' || next == '@' || next == ' ' || IS_CNTRL(next))
663  {
664  P_AUTO_ERROR;
665  }
666  else
667  {
668  P_SET_STATE(PFX_USER);
669  P_INIT_STRING(pfx_user);
670  }
671  break;
672  /*
673  * User name; '@' will cause state changes, '!' , ' ' and
674  * control characters will cause errors.
675  */
676  case IRC_PARSER_PFX_USER:
677  if (next == '@')
678  {
679  P_SET_STATE(PFX_HOST_START);
680  }
681  else if (next == '!' || next == ' ' || IS_CNTRL(next))
682  {
683  P_AUTO_ERROR;
684  }
685  else
686  {
687  P_ADD_STRING(pfx_user);
688  }
689  break;
690  /*
691  * Start of host name; anything goes, except for '!', '@', ' '
692  * and control characters which cause an error.
693  */
694  case IRC_PARSER_PFX_HOST_START:
695  if (next == '!' || next == '@' || next == ' ' || IS_CNTRL(next))
696  {
697  P_AUTO_ERROR;
698  }
699  else
700  {
701  P_SET_STATE(PFX_HOST);
702  P_INIT_STRING(pfx_host);
703  }
704  break;
705  /*
706  * Host name; ' ' will cause state changes, '!' and control
707  * characters will cause errors.
708  */
709  case IRC_PARSER_PFX_HOST:
710  if (next == ' ')
711  {
712  P_SET_STATE(COMMAND_START);
713  }
714  else if (next == '!' || next == '@' || IS_CNTRL(next))
715  {
716  P_AUTO_ERROR;
717  }
718  else
719  {
720  P_ADD_STRING(pfx_host);
721  }
722  break;
723  /*
724  * Start of command, will accept start of numeric and string
725  * commands; anything else is an error.
726  */
727  case IRC_PARSER_COMMAND_START:
728  if (IS_DIGIT(next))
729  {
730  P_SET_STATE(NUM_COMMAND_2);
731  P_INIT_STRING(cmd_string);
732  }
733  else if (IS_UPPER(next))
734  {
735  P_SET_STATE(STR_COMMAND);
736  P_INIT_STRING(cmd_string);
737  }
738  else
739  {
740  P_AUTO_ERROR;
741  }
742  break;
743  /*
744  * String command. Uppercase letters will cause the parser
745  * to continue on string commands, ' ' indicates a parameter
746  * is expected, '\r' means we're done. Anything else is an
747  * error.
748  */
749  case IRC_PARSER_STR_COMMAND:
750  if (next == ' ')
751  {
752  P_SET_STATE(PARAM_START);
753  }
754  else if (next == '\r')
755  {
756  P_SET_STATE(LF);
757  }
758  else if (IS_UPPER(next))
759  {
760  P_ADD_STRING(cmd_string);
761  }
762  else
763  {
764  P_ERROR(RECOVERY);
765  }
766  break;
767  /*
768  * Second/third digit of numeric command; anything but a digit
769  * is an error.
770  */
771  case IRC_PARSER_NUM_COMMAND_2:
772  case IRC_PARSER_NUM_COMMAND_3:
773  if (IS_DIGIT(next))
774  {
775  IRC_ParserState++;
776  P_ADD_STRING(cmd_string);
777  }
778  else
779  {
780  P_AUTO_ERROR;
781  }
782  break;
783  /*
784  * End of numeric command, could be a ' ' or a '\r'.
785  */
786  case IRC_PARSER_NUM_COMMAND_4:
787  if (next == ' ')
788  {
789  P_SET_STATE(PARAM_START);
790  }
791  else if (next == '\r')
792  {
793  P_SET_STATE(LF);
794  }
795  else
796  {
797  P_ERROR(RECOVERY);
798  }
799  break;
800  /*
801  * Start of parameter. ':' means it's a trailing parameter,
802  * spaces and control characters shouldn't be here, and
803  * anything else is a "middle" parameter.
804  */
805  case IRC_PARSER_PARAM_START:
806  if (next == ':')
807  {
808  P_SET_STATE(TRAILING_PARAM);
809  P_NEXT_PARAM;
810  }
811  else if (next == '\r')
812  {
813  P_SET_STATE(LF);
814  }
815  else if (IS_CNTRL(next))
816  {
817  P_AUTO_ERROR;
818  }
819  else if (next != ' ')
820  {
821  if (next & 0x80)
822  {
823  next = '?';
824  }
825  P_SET_STATE(MID_PARAM);
826  P_START_PARAM;
827  }
828  break;
829  /*
830  * "Middle" parameter; ' ' means there's another parameter coming,
831  * '\r' means the end of the message, control characters are not
832  * accepted, anything else is part of the parameter.
833  */
834  case IRC_PARSER_MID_PARAM:
835  if (next == ' ')
836  {
837  P_SET_STATE(PARAM_START);
838  }
839  else if (next == '\r')
840  {
841  P_SET_STATE(LF);
842  }
843  else if (IS_CNTRL(next))
844  {
845  P_ERROR(RECOVERY);
846  }
847  else
848  {
849  if (next & 0x80)
850  {
851  next = '?';
852  }
853  P_ADD_PARAM;
854  }
855  break;
856  /*
857  * Trailing parameter; '\r' means the end of the command,
858  * and anything else is just added to the string.
859  */
860  case IRC_PARSER_TRAILING_PARAM:
861  if (next == '\r')
862  {
863  P_SET_STATE(LF);
864  }
865  else
866  {
867  if (next & 0x80)
868  {
869  next = '?';
870  }
871  P_ADD_PARAM;
872  }
873  break;
874  /*
875  * End of line, expect '\n'. If found, we may have a message
876  * to handle (unless there were errors). Anything else is an
877  * error.
878  */
879  case IRC_PARSER_LF:
880  if (next == '\n')
881  {
882  has_msg = IRC_ParserInMessage;
883  P_SET_STATE(START);
884  }
885  else
886  {
887  P_AUTO_ERROR;
888  }
889  break;
890  /*
891  * Error recovery: wait for an '\r'.
892  */
893  case IRC_PARSER_RECOVERY:
894  if (next == '\r')
895  {
896  P_SET_STATE(LF);
897  }
898  break;
899  }
900 
901  return has_msg && !IRC_ParserError;
902 }
