473 static void RB_TakeScreenshot(int x, int y, int width, int height, const char *fileName)
474 {
475  byte *allbuf, *buffer;
476  byte *srcptr, *destptr;
477  byte *endline, *endmem;
478  byte temp;
479  int linelen, padlen;
480  size_t offset = 18, memcount;
481 
482  allbuf = RB_ReadPixels(x, y, width, height, &offset, &padlen);
483  buffer = allbuf + offset - 18;
484 
485  Com_Memset(buffer, 0, 18);
486  buffer[2] = 2; // uncompressed type
487  buffer[12] = width & 255;
488  buffer[13] = width >> 8;
489  buffer[14] = height & 255;
490  buffer[15] = height >> 8;
491  buffer[16] = 24; // pixel size
492 
493  // swap rgb to bgr and remove padding from line endings
494  linelen = width * 3;
495 
496  srcptr = destptr = allbuf + offset;
497  endmem = srcptr + (linelen + padlen) * height;
498 
499  while (srcptr < endmem)
500  {
501  endline = srcptr + linelen;
502 
503  while (srcptr < endline)
504  {
505  temp = srcptr[0];
506  *destptr++ = srcptr[2];
507  *destptr++ = srcptr[1];
508  *destptr++ = temp;
509 
510  srcptr += 3;
511  }
512 
513  // Skip the pad
514  srcptr += padlen;
515  }
516 
517  memcount = linelen * height;
518 
519  // gamma correct
520  if (glConfig.deviceSupportsGamma)
521  {
522  R_GammaCorrect(allbuf + offset, memcount);
523  }
524 
525  ri.FS_WriteFile(fileName, buffer, memcount + 18);
526 
527  ri.Hunk_FreeTempMemory(allbuf);
528 }

823 const void *RB_TakeVideoFrameCmd(const void *data)
824 {
825  const videoFrameCommand_t *cmd = (const videoFrameCommand_t *)data;
826  GLint packAlign;
827  int lineLen, captureLineLen;
828  byte *pixels;
829  int i;
830  int outputSize;
831  int j;
832  int aviLineLen;
833 
834  // RB: it is possible to we still have a videoFrameCommand_t but we already stopped
835  // video recording
836  if (ri.CL_VideoRecording())
837  {
838  // take care of alignment issues for reading RGB images..
839  glGetIntegerv(GL_PACK_ALIGNMENT, &packAlign);
840 
841  lineLen = cmd->width * 3;
842  captureLineLen = PAD(lineLen, packAlign);
843 
844  pixels = (byte *)PADP(cmd->captureBuffer, packAlign);
845  glReadPixels(0, 0, cmd->width, cmd->height, GL_RGB, GL_UNSIGNED_BYTE, pixels);
846 
847  if (tr.overbrightBits > 0 && glConfig.deviceSupportsGamma)
848  {
849  // this also runs over the padding...
850  R_GammaCorrect(pixels, captureLineLen * cmd->height);
851  }
852 
853  if (cmd->motionJpeg)
854  {
855  // Drop alignment and line padding bytes
856  for (i = 0; i < cmd->height; ++i)
857  {
858  memmove(cmd->captureBuffer + i * lineLen, pixels + i * captureLineLen, lineLen);
859  }
860 
861  outputSize = RE_SaveJPGToBuffer(cmd->encodeBuffer, 3 * cmd->width * cmd->height, r_screenshotJpegQuality->integer, cmd->width, cmd->height, cmd->captureBuffer, 0);
862  ri.CL_WriteAVIVideoFrame(cmd->encodeBuffer, outputSize);
863  }
864  else
865  {
866  aviLineLen = PAD(lineLen, AVI_LINE_PADDING);
867 
868  for (i = 0; i < cmd->height; ++i)
869  {
870  for (j = 0; j < lineLen; j += 3)
871  {
872  cmd->encodeBuffer[i * aviLineLen + j + 0] = pixels[i * captureLineLen + j + 2];
873  cmd->encodeBuffer[i * aviLineLen + j + 1] = pixels[i * captureLineLen + j + 1];
874  cmd->encodeBuffer[i * aviLineLen + j + 2] = pixels[i * captureLineLen + j + 0];
875  }
876  while (j < aviLineLen)
877  {
878  cmd->encodeBuffer[i * aviLineLen + j++] = 0;
879  }
880  }
881 
882  ri.CL_WriteAVIVideoFrame(cmd->encodeBuffer, aviLineLen * cmd->height);
883  }
884  }
