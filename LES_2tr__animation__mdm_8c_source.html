97 static float RB_ProjectRadius(float r, vec3_t location)
98 {
99  float pr;
100  float dist;
101  float c;
102  vec3_t p;
103  float projected[4];
104 
105  c = DotProduct(backEnd.viewParms.orientation.axis[0], backEnd.viewParms.orientation.origin);
106  dist = DotProduct(backEnd.viewParms.orientation.axis[0], location) - c;
107 
108  if (dist <= 0)
109  {
110  return 0;
111  }
112 
113  p[0] = 0;
114  p[1] = Q_fabs(r);
115  p[2] = -dist;
116 
117  projected[0] = p[0] * backEnd.viewParms.projectionMatrix[0] +
118  p[1] * backEnd.viewParms.projectionMatrix[4] +
119  p[2] * backEnd.viewParms.projectionMatrix[8] +
120  backEnd.viewParms.projectionMatrix[12];
121 
122  projected[1] = p[0] * backEnd.viewParms.projectionMatrix[1] +
123  p[1] * backEnd.viewParms.projectionMatrix[5] +
124  p[2] * backEnd.viewParms.projectionMatrix[9] +
125  backEnd.viewParms.projectionMatrix[13];
126 
127  projected[2] = p[0] * backEnd.viewParms.projectionMatrix[2] +
128  p[1] * backEnd.viewParms.projectionMatrix[6] +
129  p[2] * backEnd.viewParms.projectionMatrix[10] +
130  backEnd.viewParms.projectionMatrix[14];
131 
132  projected[3] = p[0] * backEnd.viewParms.projectionMatrix[3] +
133  p[1] * backEnd.viewParms.projectionMatrix[7] +
134  p[2] * backEnd.viewParms.projectionMatrix[11] +
135  backEnd.viewParms.projectionMatrix[15];
136 
137 
138  pr = projected[1] / projected[3];
139 
140  if (pr > 1.0f)
141  {
142  pr = 1.0f;
143  }
144 
145  return pr;
146 }

153 static int R_CullModel(trRefEntity_t *ent)
154 {
155  vec3_t bounds[2];
156  mdxHeader_t *oldFrameHeader, *newFrameHeader;
157  mdxFrame_t *oldFrame, *newFrame;
158  int i;
159 
160  newFrameHeader = R_GetModelByHandle(ent->e.frameModel)->model.mdx;
161  oldFrameHeader = R_GetModelByHandle(ent->e.oldframeModel)->model.mdx;
162 
163  if (!newFrameHeader || !oldFrameHeader)
164  {
165  return CULL_OUT;
166  }
167 
168  // compute frame pointers
169  newFrame = (mdxFrame_t *) ((byte *) newFrameHeader + newFrameHeader->ofsFrames +
170  ent->e.frame * (int)(sizeof(mdxBoneFrameCompressed_t)) * newFrameHeader->numBones +
171  ent->e.frame * sizeof(mdxFrame_t));
172  oldFrame = (mdxFrame_t *) ((byte *) oldFrameHeader + oldFrameHeader->ofsFrames +
173  ent->e.oldframe * (int)(sizeof(mdxBoneFrameCompressed_t)) * oldFrameHeader->numBones +
174  ent->e.oldframe * sizeof(mdxFrame_t));
175 
176  // cull bounding sphere ONLY if this is not an upscaled entity
177  if (!ent->e.nonNormalizedAxes)
178  {
179  if (ent->e.frame == ent->e.oldframe && ent->e.frameModel == ent->e.oldframeModel)
180  {
181  switch (R_CullLocalPointAndRadius(newFrame->localOrigin, newFrame->radius))
182  {
183  case CULL_OUT:
184  tr.pc.c_sphere_cull_md3_out++;
185  return CULL_OUT;
186 
187  case CULL_IN:
188  tr.pc.c_sphere_cull_md3_in++;
189  return CULL_IN;
190 
191  case CULL_CLIP:
192  tr.pc.c_sphere_cull_md3_clip++;
193  break;
194  }
195  }
196  else
197  {
198  int sphereCull, sphereCullB;
199 
200  sphereCull = R_CullLocalPointAndRadius(newFrame->localOrigin, newFrame->radius);
201  if (newFrame == oldFrame)
202  {
203  sphereCullB = sphereCull;
204  }
205  else
206  {
207  sphereCullB = R_CullLocalPointAndRadius(oldFrame->localOrigin, oldFrame->radius);
208  }
209 
210  if (sphereCull == sphereCullB)
211  {
212  if (sphereCull == CULL_OUT)
213  {
214  tr.pc.c_sphere_cull_md3_out++;
215  return CULL_OUT;
216  }
217  else if (sphereCull == CULL_IN)
218  {
219  tr.pc.c_sphere_cull_md3_in++;
220  return CULL_IN;
221  }
222  else
223  {
224  tr.pc.c_sphere_cull_md3_clip++;
225  }
226  }
227  }
228  }
229 
230  // calculate a bounding box in the current coordinate system
231  for (i = 0 ; i < 3 ; i++)
232  {
233  bounds[0][i] = oldFrame->bounds[0][i] < newFrame->bounds[0][i] ? oldFrame->bounds[0][i] : newFrame->bounds[0][i];
234  bounds[1][i] = oldFrame->bounds[1][i] > newFrame->bounds[1][i] ? oldFrame->bounds[1][i] : newFrame->bounds[1][i];
235  }
236 
237  switch (R_CullLocalBox(bounds))
238  {
239  case CULL_IN:
240  tr.pc.c_box_cull_md3_in++;
241  return CULL_IN;
242  case CULL_CLIP:
243  tr.pc.c_box_cull_md3_clip++;
244  return CULL_CLIP;
245  case CULL_OUT:
246  default:
247  tr.pc.c_box_cull_md3_out++;
248  return CULL_OUT;
249  }
250 }
