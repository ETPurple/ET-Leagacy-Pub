75 unsigned long GeoIP_addr_to_num(const char *addr)
76 {
77  char tok[4];
78  int i, octet, j = 0, k = 0;
79  unsigned long ipnum = 0;
80  char c = 0;
81 
82  for (i = 0; i < 4; i++)
83  {
84  for (;; )
85  {
86  c = addr[k++];
87  if (c == '.' || c == ':' || c == '\0')
88  {
89  tok[j] = '\0';
90  octet = atoi(tok);
91  if (octet > 255)
92  {
93  return 0;
94  }
95  ipnum += (octet << ((3 - i) * 8));
96  j = 0;
97  break;
98  }
99  else if (c >= '0' && c <= '9')
100  {
101  if (j > 2)
102  {
103  return 0;
104  }
105  tok[j++] = c;
106  }
107  else
108  {
109  return 0;
110  }
111  }
112  if (c == '\0' && i < 3)
113  {
114  return 0;
115  }
116  }
117  return ipnum;
118 }

126 unsigned int GeoIP_seek_record(GeoIP *gi, unsigned long ipnum)
127 {
128  int depth;
129  unsigned int x = 0;
130  unsigned int step = 0;
131  const unsigned char *buf = NULL;
132 
133  for (depth = 31; depth >= 0; depth--)
134  {
135  step = 6 * x;
136 
137  if (step + 6 >= gi->memsize)
138  {
139  G_Printf("GeoIP: Error Traversing Database for ipnum = %lu - Perhaps database is corrupt?\n", ipnum);
140  return 255;
141  }
142 
143  buf = gi->cache + step;
144 
145  if (ipnum & (1 << depth))
146  {
147  x = (buf[3] << 0) + (buf[4] << 8) + (buf[5] << 16);
148  }
149  else
150  {
151  x = (buf[0] << 0) + (buf[1] << 8) + (buf[2] << 16);
152  }
153 
154  if (x >= 16776960)
155  {
156  return x - 16776960;
157  }
158  }
159 
160  G_Printf("GeoIP: Error Traversing Database for ipnum = %lu - Perhaps database is corrupt?\n", ipnum);
161  return 255;
