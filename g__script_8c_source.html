303 
304  if (len > 0)

307  }
308  }
309

311  {

317  Q_strcat(filename, sizeof(filename), "_lms");
318  }
319

321  len = trap_FS_FOpenFile(filename, &f, FS_READ);
322  }
323

344 void G_Script_ScriptParse(gentity_t *ent)
345 {
346  char *pScript;
347  char *token;
348  qboolean wantName;
349  qboolean wantScript;
350  qboolean inScript;
351  int eventNum;
352  g_script_event_t events[G_MAX_SCRIPT_STACK_ITEMS];
353  unsigned int numEventItems;
354  g_script_event_t *curEvent;
355  char params[MAX_INFO_STRING]; // was MAX_QPATH some of our multiplayer script commands have longer parameters
356  g_script_stack_action_t *action;
357  int i;
358  int bracketLevel;
359  unsigned int len;
360  qboolean buildScript;
361 
362  if (!ent->scriptName)
363  {
364  return;
365  }
366  if (!level.scriptEntity)
367  {
368  return;
369  }
370 
371  buildScript = (qboolean)(trap_Cvar_VariableIntegerValue("com_buildScript"));
372 
373  pScript = level.scriptEntity;
374  wantName = qtrue;
375  wantScript = qfalse;
376  inScript = qfalse;
377  COM_BeginParseSession("G_Script_ScriptParse");
378  bracketLevel = 0;
379  numEventItems = 0;
380 
381  Com_Memset(events, 0, sizeof(events));
382 
383  while (1)
384  {
385  token = COM_Parse(&pScript);
386 
387  if (!token[0])
388  {
389  if (!wantName)
390  {
391  G_Error("G_Script_ScriptParse(), Error (line %d): '}' expected, end of script found.\n", COM_GetCurrentParseLine());
392  }
393  break;
394  }
395 
396  // end of script
397  if (token[0] == '}')
398  {
399  if (inScript)
400  {
401  break;
402  }
403  if (wantName)
404  {
405  G_Error("G_Script_ScriptParse(), Error (line %d): '}' found, but not expected.\n", COM_GetCurrentParseLine());
406  }
407  wantName = qtrue;
408  }
409  else if (token[0] == '{')
410  {
411  if (wantName)
412  {
413  G_Error("G_Script_ScriptParse(), Error (line %d): '{' found, NAME expected.\n", COM_GetCurrentParseLine());
414  }
415  wantScript = qfalse;
416  }
417  else if (wantName)
418  {
419  if (!Q_stricmp(token, "entity"))
420  {
421  // this is an entity, so go back to look for a name
422  continue;
423  }
424  if (!Q_stricmp(ent->scriptName, token))
425  {
426  inScript = qtrue;
427  numEventItems = 0;
428  }
429  wantName = qfalse;
430  wantScript = qtrue;
431  }
432  else if (inScript)
433  {
434  Q_strlwr(token);
435  eventNum = G_Script_EventForString(token);
436  if (eventNum < 0)
437  {
438  G_Error("G_Script_ScriptParse(), Error (line %d): unknown event: %s.\n", COM_GetCurrentParseLine(), token);
439  }
440 
441  if (numEventItems >= G_MAX_SCRIPT_STACK_ITEMS)
442  {
443  G_Error("G_Script_ScriptParse(), Error (line %d): G_MAX_SCRIPT_STACK_ITEMS reached (%d)\n", COM_GetCurrentParseLine(), G_MAX_SCRIPT_STACK_ITEMS);
444  }
445 
446  curEvent = &events[numEventItems];
447  curEvent->eventNum = eventNum;
448  Com_Memset(params, 0, sizeof(params));
449 
450  // parse any event params before the start of this event's actions
451  while ((token = COM_Parse(&pScript)) != NULL && (token[0] != '{'))
452  {
453  if (!token[0])
454  {
455  G_Error("G_Script_ScriptParse(), Error (line %d): '}' expected, end of script found.\n", COM_GetCurrentParseLine());
456  }
457 
458  if (strlen(params)) // add a space between each param
459  {
460  Q_strcat(params, sizeof(params), " ");
461  }
462  Q_strcat(params, sizeof(params), token);
463  }
464 
465  len = strlen(params);
466 
467  if (len) // copy the params into the event
468  {
469  curEvent->params = G_Alloc(len + 1);
470  Q_strncpyz(curEvent->params, params, len + 1);
471  }
472 
473  // parse the actions for this event
474  while ((token = COM_Parse(&pScript)) != NULL && (token[0] != '}'))
475  {
476  if (!token[0])
477  {
478  G_Error("G_Script_ScriptParse(), Error (line %d): '}' expected, end of script found.\n", COM_GetCurrentParseLine());
479  }
480 
481  action = G_Script_ActionForString(token);
482  if (!action)
483  {
484  G_Error("G_Script_ScriptParse(), Error (line %d): unknown action: %s.\n", COM_GetCurrentParseLine(), token);
485  }
486 
487  curEvent->stack.items[curEvent->stack.numItems].action = action;
488 
489  Com_Memset(params, 0, sizeof(params));
490 
491  // Parse for {}'s if this is a set command
492  if (action->hash == SET_HASH || action->hash == DELETE_HASH || action->hash == CREATE_HASH)
493  {
494  token = COM_Parse(&pScript);
495  if (token[0] != '{')
496  {
497  COM_ParseError("'{' expected, found: %s.\n", token);
498  }
499 
500  while ((token = COM_Parse(&pScript)) && (token[0] != '}'))
501  {
502  if (strlen(params)) // add a space between each param
503  {
504  Q_strcat(params, sizeof(params), " ");
505  }
506 
507  if (strrchr(token, ' ')) // need to wrap this param in quotes since it has more than one word
508  {
509  Q_strcat(params, sizeof(params), "\"");
510  }
511 
512  Q_strcat(params, sizeof(params), token);
513 
514  if (strrchr(token, ' ')) // need to wrap this param in quotes since it has mor
515  {
516  Q_strcat(params, sizeof(params), "\"");
517  }
518  }
519  }
520  else
521  {
522  token = COM_ParseExt(&pScript, qfalse);
523  for (i = 0; token[0]; i++)
524  {
525  if (strlen(params)) // add a space between each param
526  {
527  Q_strcat(params, sizeof(params), " ");
528  }
529 
530  if (i == 0)
531  {
532  // Special case: playsound's need to be cached on startup to prevent in-game pauses
533  if (action->hash == PLAYSOUND_HASH)
534  {
535  G_SoundIndex(token);
536  }
537  else if (action->hash == CHANGEMODEL_HASH)
538  {
539  G_ModelIndex(token);
540  }
541  else if (buildScript && (action->hash == MU_START_HASH || action->hash == MU_PLAY_HASH || action->hash == MU_QUEUE_HASH))
542  {
543  if (strlen(token)) // we know there's a [0], but don't know if it's '0'
544  {
545  trap_SendServerCommand(-1, va("addToBuild %s\n", token));
546  }
547  }
548  }
549 
550  if (i == 0 || i == 1)
551  {
552  if (action->hash == REMAPSHADER_HASH)
553  {
554  G_ShaderIndex(token);
555  }
556  }
557 
558  if (strrchr(token, ' ')) // need to wrap this param in quotes since it has more than one word
559  {
560  Q_strcat(params, sizeof(params), "\"");
561  }
562 
563  Q_strcat(params, sizeof(params), token);
564 
565  if (strrchr(token, ' ')) // need to wrap this param in quotes since it has more than one word
566  {
567  Q_strcat(params, sizeof(params), "\"");
568  }
569 
570  token = COM_ParseExt(&pScript, qfalse);
571  }
572  }
573 
574  if (strlen(params)) // copy the params into the event
575  {
576  curEvent->stack.items[curEvent->stack.numItems].params = G_Alloc(strlen(params) + 1);
577  Q_strncpyz(curEvent->stack.items[curEvent->stack.numItems].params, params, strlen(params) + 1);
578  }
579 
580  curEvent->stack.numItems++;
581 
582  if (curEvent->stack.numItems >= G_MAX_SCRIPT_STACK_ITEMS)
583  {
584  G_Error("G_Script_ScriptParse(): script exceeded G_MAX_SCRIPT_STACK_ITEMS (%d), line %d\n", G_MAX_SCRIPT_STACK_ITEMS, COM_GetCurrentParseLine());
585  }
586  }
587 
588  numEventItems++;
589  }
590  else // skip this character completely
591  {
592  if (wantScript)
593  {
594  // note: There are maps available which don't start anymore
595  // Fix the mapscripts!
596  G_Error("G_Script_ScriptParse(), Error (line %d): '{' expected, but found '%s'.\n", COM_GetCurrentParseLine(), token);
597  }
598 
599  while ((token = COM_Parse(&pScript)) != NULL)
600  {
601  if (!token[0])
602  {
603  G_Error("G_Script_ScriptParse(), Error (line %d): '}' expected, end of script found.\n", COM_GetCurrentParseLine());
604  }
605  else if (token[0] == '{')
606  {
607  bracketLevel++;
608  }
609  else if (token[0] == '}')
610  {
611  if (!--bracketLevel)
612  {
613  break;
614  }
615  }
616  }
617  }

795 qboolean G_Script_ScriptRun(gentity_t *ent)
796 {
797  g_script_stack_t *stack;
798  int oldScriptId;
799 
800  if (!ent->scriptEvents)
801  {
802  ent->scriptStatus.scriptEventIndex = -1;
803  return qtrue;
804  }
805 
806  // if we are still doing a gotomarker, process the movement
807  if (ent->scriptStatus.scriptFlags & SCFL_GOING_TO_MARKER)
808  {
809  G_ScriptAction_GotoMarker(ent, NULL);
810  }
811 
812  // if we are animating, do the animation
813  if (ent->scriptStatus.scriptFlags & SCFL_ANIMATING)
814  {
815  G_ScriptAction_PlayAnim(ent, ent->scriptStatus.animatingParams);
816  }
817 
818  if (ent->scriptStatus.scriptEventIndex < 0)
819  {
820  return qtrue;
821  }
822 
823  stack = &ent->scriptEvents[ent->scriptStatus.scriptEventIndex].stack;
824 
825  if (!stack->numItems)
826  {
827  ent->scriptStatus.scriptEventIndex = -1;
828  return qtrue;
829  }
830 
831  // show debugging info
832  if (g_scriptDebug.integer && ent->scriptStatus.scriptStackChangeTime == level.time &&
833  (!g_scriptDebugTarget.string[0] || G_MatchString(g_scriptDebugTarget.string, ent->scriptName, qfalse)))
834  {
835  if (ent->scriptStatus.scriptStackHead < stack->numItems)
836  {
837  G_Printf("^7%i : (^5%s^7) ^9GScript Action: ^d%s %s\n", level.time, ent->scriptName, stack->items[ent->scriptStatus.scriptStackHead].action->actionString, (stack->items[ent->scriptStatus.scriptStackHead].params ? stack->items[ent->scriptStatus.scriptStackHead].params : ""));
838  }
839  }
840 
841  while (ent->scriptStatus.scriptStackHead < stack->numItems)
842  {
843  oldScriptId = ent->scriptStatus.scriptId;
844  if (!stack->items[ent->scriptStatus.scriptStackHead].action->actionFunc(ent, stack->items[ent->scriptStatus.scriptStackHead].params))
845  {
846  ent->scriptStatus.scriptFlags &= ~SCFL_FIRST_CALL;
847  return qfalse;
848  }
849  // if the scriptId changed, a new event was triggered in our scripts which did not finish
850  if (oldScriptId != ent->scriptStatus.scriptId)
851  {
852  return qfalse;
853  }
854  // move to the next action in the script
855  ent->scriptStatus.scriptStackHead++;
856  // record the time that this new item became active
857  ent->scriptStatus.scriptStackChangeTime = level.time;
858  //
859  ent->scriptStatus.scriptFlags |= SCFL_FIRST_CALL;
860  // show debugging info
861  if (g_scriptDebug.integer &&
862  (!g_scriptDebugTarget.string[0] || G_MatchString(g_scriptDebugTarget.string, ent->scriptName, qfalse)))
863  {
864  if (ent->scriptStatus.scriptStackHead < stack->numItems)
865  {
866  G_Printf("^7%i : (^5%s^7) ^9GScript Action: ^d%s %s\n", level.time, ent->scriptName, stack->items[ent->scriptStatus.scriptStackHead].action->actionString, (stack->items[ent->scriptStatus.scriptStackHead].params ? stack->items[ent->scriptStatus.scriptStackHead].params : ""));
867  }
868  }
869  }
870 
871  ent->scriptStatus.scriptEventIndex = -1;
872 
873  return qtrue;
874 }
