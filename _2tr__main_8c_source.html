439 void R_CalcTBN2(vec3_t tangent, vec3_t binormal, vec3_t normal,
440  const vec3_t v1, const vec3_t v2, const vec3_t v3, const vec2_t t1, const vec2_t t2, const vec2_t t3)
441 {

443  vec3_t v3v1;
444  float c2c1_T;
445  float c2c1_B;
446  float c3c1_T;
447  float c3c1_B;
448  float denominator;
449  float scale1, scale2;
450  vec3_t T, B, N, C;
451 
452  // Calculate the tangent basis for each vertex of the triangle
453  // UPDATE: In the 3rd edition of the accompanying article, the for-loop located here has
454  // been removed as it was redundant (the entire TBN matrix was calculated three times
455  // instead of just one).
456  //
457  // Please note, that this function relies on the fact that the input geometry are triangles
458  // and the tangent basis for each vertex thus is identical!
459  //
460 
461  // Calculate the vectors from the current vertex to the two other vertices in the triangle
462  VectorSubtract(v2, v1, v2v1);
463  VectorSubtract(v3, v1, v3v1);
464 
465  // The equation presented in the article states that:
466  // c2c1_T = V2.texcoord.x â€“ V1.texcoord.x
467  // c2c1_B = V2.texcoord.y â€“ V1.texcoord.y
468  // c3c1_T = V3.texcoord.x â€“ V1.texcoord.x
469  // c3c1_B = V3.texcoord.y â€“ V1.texcoord.y
470 
471  // Calculate c2c1_T and c2c1_B
472  c2c1_T = t2[0] - t1[0];
473  c2c1_B = t2[1] - t2[1];
474 
475  // Calculate c3c1_T and c3c1_B
476  c3c1_T = t3[0] - t1[0];
477  c3c1_B = t3[1] - t1[1];
478 
479  denominator = c2c1_T * c3c1_B - c3c1_T * c2c1_B;
480  //if(ROUNDOFF(fDenominator) == 0.0f)
481  if (denominator == 0.0f)
482  {
483  // We won't risk a divide by zero, so set the tangent matrix to the identity matrix
484  VectorSet(tangent, 1, 0, 0);
485  VectorSet(binormal, 0, 1, 0);
486  VectorSet(normal, 0, 0, 1);
487  }
488  else
489  {
490  // Calculate the reciprocal value once and for all (to achieve speed)
491  scale1 = 1.0f / denominator;
492 
493  // T and B are calculated just as the equation in the article states
494  VectorSet(T, (c3c1_B * v2v1[0] - c2c1_B * v3v1[0]) * scale1,
495  (c3c1_B * v2v1[1] - c2c1_B * v3v1[1]) * scale1,
496  (c3c1_B * v2v1[2] - c2c1_B * v3v1[2]) * scale1);
497 
498  VectorSet(B, (-c3c1_T * v2v1[0] + c2c1_T * v3v1[0]) * scale1,
499  (-c3c1_T * v2v1[1] + c2c1_T * v3v1[1]) * scale1,
500  (-c3c1_T * v2v1[2] + c2c1_T * v3v1[2]) * scale1);
501 
502  // The normal N is calculated as the cross product between T and B
503  CrossProduct(T, B, N);
504 
505 #if 0
506  VectorCopy(T, tangent);
507  VectorCopy(B, binormal);
508  VectorCopy(N, normal);
509 #else
510  // Calculate the reciprocal value once and for all (to achieve speed)
511  scale2 = 1.0f / ((T[0] * B[1] * N[2] - T[2] * B[1] * N[0]) +
512  (B[0] * N[1] * T[2] - B[2] * N[1] * T[0]) +
513  (N[0] * T[1] * B[2] - N[2] * T[1] * B[0]));
514 
515  // Calculate the inverse if the TBN matrix using the formula described in the article.
516  // We store the basis vectors directly in the provided TBN matrix: pvTBNMatrix
517  CrossProduct(B, N, C); tangent[0] = C[0] * scale2;
518  CrossProduct(N, T, C); tangent[1] = -C[0] * scale2;
519  CrossProduct(T, B, C); tangent[2] = C[0] * scale2;
520  VectorNormalize(tangent);
521 
522  CrossProduct(B, N, C); binormal[0] = -C[1] * scale2;
523  CrossProduct(N, T, C); binormal[1] = C[1] * scale2;
524  CrossProduct(T, B, C); binormal[2] = -C[1] * scale2;
525  VectorNormalize(binormal);
526

528  CrossProduct(N, T, C); normal[1] = -C[2] * scale2;
529  CrossProduct(T, B, C); normal[2] = C[2] * scale2;

531 #endif
532  }
