324 static qboolean Sys_StringToSockaddr(const char *s, struct sockaddr *sadr, int sadr_len, sa_family_t family)
325 {
326  struct addrinfo hints; // provides hints about the type of socket the caller supports
327  struct addrinfo *res = NULL; // contains response information about the host
328 #ifdef FEATURE_IPV6
329  struct addrinfo *search = NULL;
330  struct addrinfo *hintsp;
331  int retval;
332 
333  Com_Memset(sadr, '\0', sizeof(*sadr));
334  Com_Memset(&hints, '\0', sizeof(hints));
335 
336  hintsp = &hints;
337  hintsp->ai_family = family;
338  hintsp->ai_socktype = SOCK_DGRAM;
339 
340  retval = getaddrinfo(s, NULL, hintsp, &res);
341 
342  if (!retval)
343  {
344  if (family == AF_UNSPEC)
345  {
346  // Decide here and now which protocol family to use
347  if (net_enabled->integer & NET_PRIOV6)
348  {
349  if (net_enabled->integer & NET_ENABLEV6)
350  {
351  search = SearchAddrInfo(res, AF_INET6);
352  }
353 
354  if (!search && (net_enabled->integer & NET_ENABLEV4))
355  {
356  search = SearchAddrInfo(res, AF_INET);
357  }
358  }
359  else
360  {
361  if (net_enabled->integer & NET_ENABLEV4)
362  {
363  search = SearchAddrInfo(res, AF_INET);
364  }
365 
366  if (!search && (net_enabled->integer & NET_ENABLEV6))
367  {
368  search = SearchAddrInfo(res, AF_INET6);
369  }
370  }
371  }
372  else
373  {
374  search = SearchAddrInfo(res, family);
375  }
376 
377  if (search)
378  {
379  if (res->ai_addrlen > sadr_len)
380  {
381  res->ai_addrlen = sadr_len;
382  }
383 
384  Com_Memcpy(sadr, res->ai_addr, res->ai_addrlen);
385  freeaddrinfo(res);
386 
387  return qtrue;
388  }
389  else
390  {
391  Com_Printf("Sys_StringToSockaddr: Error resolving %s: No address of required type found\n", s);
392  }
393  }
394  else
395  {
396  Com_Printf("Sys_StringToSockaddr: Error resolving %s: %s\n", s, gai_strerror(retval));
397  }
398 
399  if (res)
400  {
401  freeaddrinfo(res);
402  }
403 
404  return qfalse;
405 #else // IPV4
406 
407  Com_Memset(sadr, 0, sizeof(*sadr));
408  ((struct sockaddr_in *)sadr)->sin_family = AF_INET;
409  ((struct sockaddr_in *)sadr)->sin_port = 0;
410 
411  Com_Memset(&hints, '\0', sizeof(hints));
412  hints.ai_family = family;
413  hints.ai_socktype = SOCK_DGRAM;
414 
415  if (s[0] >= '0' && s[0] <= '9')
416  {
417  *(int *)&((struct sockaddr_in *)sadr)->sin_addr = inet_addr(s);
418  }
419  else
420  {
421  if (getaddrinfo(s, NULL, &hints, &res))
422  {
423  return qfalse;
424  }
425  Com_Memcpy(sadr, res->ai_addr, res->ai_addrlen);
426  freeaddrinfo(res);
427  }
428 
429  return qtrue;
430 #endif
431 }

504 qboolean NET_CompareBaseAdrMask(netadr_t a, netadr_t b, int netmask)
505 {
506  qboolean differed;
507  byte cmpmask, *addra, *addrb;
508  int curbyte;
509 
510  if (a.type != b.type)
511  {
512  return qfalse;
513  }
514 
515  switch (a.type)
516  {
517  case NA_LOOPBACK:
518  return qtrue;
519  case NA_IP:
520  {
521  addra = (byte *) &a.ip;
522  addrb = (byte *) &b.ip;
523 
524  if (netmask < 0 || netmask > 32)
525  {
526  netmask = 32;
527  }
528  }
529  break;
530 #ifdef FEATURE_IPV6
531  case NA_IP6:
532  {
533  addra = (byte *) &a.ip6;
534  addrb = (byte *) &b.ip6;
535 
536  if (netmask < 0 || netmask > 128)
537  {
538  netmask = 128;
539  }
540  }
541  break;
542 #endif
543  default:
544  Com_Printf("NET_CompareBaseAdrMask: bad address type a: %i (b: %i, netmask: %i)\n", a.type, b.type, netmask);
545  return qfalse;
546  }
547 
548  differed = qfalse;
549  curbyte = 0;
550 
551  while (netmask > 7)
552  {
553  if (addra[curbyte] != addrb[curbyte])
554  {
555  differed = qtrue;
556  break;
557  }
558 
559  curbyte++;
560  netmask -= 8;
561  }
562 
563  if (differed)
564  {
565  return qfalse;
566  }
567 
568  if (netmask)
569  {
570  cmpmask = (1 << netmask) - 1;
571  cmpmask <<= 8 - netmask;
572 
573  if ((addra[curbyte] & cmpmask) == (addrb[curbyte] & cmpmask))
574  {
575  return qtrue;
576  }
577  }
578  else
579  {
580  return qtrue;
581  }
582 
583  return qfalse;
584 }
