81 static void CG_FreeTrailJunc(trailJunc_t *junc)
82 {
83  // kill any juncs after us, so they aren't left hanging
84  if (junc->nextJunc)
85  {
86  CG_KillTrail(junc);
87  }
88 
89  // make it non-active
90  junc->inuse = qfalse;
91  junc->freed = qtrue;
92  if (junc->nextGlobal)
93  {
94  junc->nextGlobal->prevGlobal = junc->prevGlobal;
95  }
96  if (junc->prevGlobal)
97  {
98  junc->prevGlobal->nextGlobal = junc->nextGlobal;
99  }
100  if (junc == activeTrails)
101  {
102  activeTrails = junc->nextGlobal;
103  }
104 
105  // if it's a head, remove it
106  if (junc == headTrails)
107  {
108  headTrails = junc->nextHead;
109  }
110  if (junc->nextHead)
111  {
112  junc->nextHead->prevHead = junc->prevHead;
113  }
114  if (junc->prevHead)
115  {
116  junc->prevHead->nextHead = junc->nextHead;
117  }
118  junc->nextHead = NULL;
119  junc->prevHead = NULL;
120 
121  // stick it in the free list
122  junc->prevGlobal = NULL;
123  junc->nextGlobal = freeTrails;
124  if (freeTrails)
125  {
126  freeTrails->prevGlobal = junc;
127  }
128  freeTrails = junc;
129 
130  numTrailsInuse--;
131 }

166 static trailJunc_t *CG_SpawnTrailJunc(trailJunc_t *headJunc)
167 {
168  trailJunc_t *j;
169 
170  if (!freeTrails)
171  {
172  return NULL;
173  }
174 
175  if (cg_paused.integer)
176  {
177  return NULL;
178  }
179 
180  // select the first free trail, and remove it from the list
181  j = freeTrails;
182  freeTrails = j->nextGlobal;
183  if (freeTrails)
184  {
185  freeTrails->prevGlobal = NULL;
186  }
187 
188  j->nextGlobal = activeTrails;
189  if (activeTrails)
190  {
191  activeTrails->prevGlobal = j;
192  }
193  activeTrails = j;
194  j->prevGlobal = NULL;
195  j->inuse = qtrue;
196  j->freed = qfalse;
197 
198  // if this owner has a headJunc, add us to the start
199  if (headJunc)
200  {
201  // remove the headJunc from the list of heads
202  if (headJunc == headTrails)
203  {
204  headTrails = headJunc->nextHead;
205  if (headTrails)
206  {
207  headTrails->prevHead = NULL;
208  }
209  }
210  else
211  {
212  if (headJunc->nextHead)
213  {
214  headJunc->nextHead->prevHead = headJunc->prevHead;
215  }
216  if (headJunc->prevHead)
217  {
218  headJunc->prevHead->nextHead = headJunc->nextHead;
219  }
220  }
221  headJunc->prevHead = NULL;
222  headJunc->nextHead = NULL;
223  }
224  // make us the headTrail
225  if (headTrails)
226  {
227  headTrails->prevHead = j;
228  }
229  j->nextHead = headTrails;
230  j->prevHead = NULL;
231  headTrails = j;
232 
233  j->nextJunc = headJunc; // if headJunc is NULL, then we'll just be the end of the list
234 
235  numTrailsInuse++;
236 
237  // debugging
238  //CG_Printf( "NumTrails: %i\n", numTrailsInuse );
239 
240  return j;
241 }

527 static void CG_AddTrailToScene(trailJunc_t *trail, int iteration, int numJuncs)
528 {
529  int k, i, n, l, numOutVerts;
530  polyVert_t mid;
531  float mod[4];
532  float sInc, s;
533  trailJunc_t *j, *jNext;

535  // clipping vars
536  vec3_t viewProj;
537  float viewDist, fadeAlpha;
538 
539  // add spark shader at head position
540  if (trail->flags & TJFL_SPARKHEADFLARE)
541  {
542  polyBuffer_t *pPolyBuffer = CG_PB_FindFreePolyBuffer(cgs.media.sparkFlareShader, 4, 6);
543 
544  if (pPolyBuffer)
545  {
546  int pos = pPolyBuffer->numVerts;
547 
548  j = trail;
549 
550  VectorCopy(j->pos, pPolyBuffer->xyz[pos]);
551  VectorMA(pPolyBuffer->xyz[pos], -j->width * 2, vup, pPolyBuffer->xyz[pos]);
552  VectorMA(pPolyBuffer->xyz[pos], -j->width * 2, vright, pPolyBuffer->xyz[pos]);
553  pPolyBuffer->st[pos][0] = 0;
554  pPolyBuffer->st[pos][1] = 0;
555  pos++;
556 
557  VectorCopy(j->pos, pPolyBuffer->xyz[pos]);
558  VectorMA(pPolyBuffer->xyz[pos], -j->width * 2, vup, pPolyBuffer->xyz[pos]);
559  VectorMA(pPolyBuffer->xyz[pos], j->width * 2, vright, pPolyBuffer->xyz[pos]);
560  pPolyBuffer->st[pos][0] = 0;
561  pPolyBuffer->st[pos][1] = 1;
562  pos++;
563 
564  VectorCopy(j->pos, pPolyBuffer->xyz[pos]);
565  VectorMA(pPolyBuffer->xyz[pos], j->width * 2, vup, pPolyBuffer->xyz[pos]);
566  VectorMA(pPolyBuffer->xyz[pos], j->width * 2, vright, pPolyBuffer->xyz[pos]);
567  pPolyBuffer->st[pos][0] = 1;
568  pPolyBuffer->st[pos][1] = 1;
569  pos++;
570 
571  VectorCopy(j->pos, pPolyBuffer->xyz[pos]);
572  VectorMA(pPolyBuffer->xyz[pos], j->width * 2, vup, pPolyBuffer->xyz[pos]);
573  VectorMA(pPolyBuffer->xyz[pos], -j->width * 2, vright, pPolyBuffer->xyz[pos]);
574  pPolyBuffer->st[pos][0] = 1;
575  pPolyBuffer->st[pos][1] = 0;
576  pos++;
577 
578  for (i = 0; i < 4; i++)
579  {
580  pPolyBuffer->color[pPolyBuffer->numVerts + i][0] = 255;
581  pPolyBuffer->color[pPolyBuffer->numVerts + i][1] = 255;
582  pPolyBuffer->color[pPolyBuffer->numVerts + i][2] = 255;
583  pPolyBuffer->color[pPolyBuffer->numVerts + i][3] = ( unsigned char )(j->alpha * 255.0f);
584  }
585 
586  pPolyBuffer->indicies[pPolyBuffer->numIndicies + 0] = (unsigned)pPolyBuffer->numVerts + 0;
587  pPolyBuffer->indicies[pPolyBuffer->numIndicies + 1] = (unsigned)pPolyBuffer->numVerts + 1;
588  pPolyBuffer->indicies[pPolyBuffer->numIndicies + 2] = (unsigned)pPolyBuffer->numVerts + 2;
589 
590  pPolyBuffer->indicies[pPolyBuffer->numIndicies + 3] = (unsigned)pPolyBuffer->numVerts + 2;
591  pPolyBuffer->indicies[pPolyBuffer->numIndicies + 4] = (unsigned)pPolyBuffer->numVerts + 3;
592  pPolyBuffer->indicies[pPolyBuffer->numIndicies + 5] = (unsigned)pPolyBuffer->numVerts + 0;
593 
594  pPolyBuffer->numVerts += 4;
595  pPolyBuffer->numIndicies += 6;
596  }
597  }
598 
599  //if (trail->flags & TJFL_CROSSOVER && iteration < 1) {
600  // iteration = 1;
601  //}
602 
603  sInc = 0;
604 
605  if (!numJuncs)
606  {
607  // first count the number of juncs in the trail
608  j = trail;
609  numJuncs = 0;
610  sInc = 0;
611  while (j)
612  {
613  numJuncs++;
614 
615  // check for a dead next junc
616  if (!j->inuse && j->nextJunc && !j->nextJunc->inuse)
617  {
618  CG_KillTrail(j);
619  }
620  else if (j->nextJunc && j->nextJunc->freed)
621  {
622  // not sure how this can happen, but it does, and causes infinite loops
623  j->nextJunc = NULL;
624  }
625 
626  if (j->nextJunc)
627  {
628  sInc += VectorDistance(j->nextJunc->pos, j->pos);
629  }
630 
631  j = j->nextJunc;
632  }
633  }

636  {
637  return;
638  }
639

641  if (trail->sType == STYPE_STRETCH)
642  {
643  //sInc = ((1.0 - 0.1) / (float)(numJuncs)); // hack, the end of funnel shows a bit of the start (looping)
644  s = 0.05f;
645  }
646  else if (trail->sType == STYPE_REPEAT)
647  {
648  s = trail->sTex;
649  }
650 
651  // now traverse the list
652  j = trail;
653  jNext = j->nextJunc;
654  i = 0;
655  while (jNext)
656  {
657  // first get the directional vectors to the next junc
658  GetPerpendicularViewVector(cg.refdef_current->vieworg, j->pos, jNext->pos, up);
659

661  if (j->flags & TJFL_CROSSOVER)
662  {
663  if (iteration > 0)
664  {

666  VectorSubtract(cg.refdef_current->vieworg, viewProj, v);
667  VectorNormalize(v);
668 
669  if (iteration == 1)
670  {
671  VectorMA(up, 0.3f, v, up);
672  }
673  else
674  {
675  VectorMA(up, -0.3f, v, up);
676  }
677  VectorNormalize(up);
678  }
679  }
680  // do fading when moving towards the projection point onto the trail segment vector
681  else if (!(j->flags & TJFL_NOCULL) && (j->widthEnd > 4 || jNext->widthEnd > 4))
682  {
683  ProjectPointOntoVector(cg.refdef_current->vieworg, j->pos, jNext->pos, viewProj);
684  viewDist = Distance(viewProj, cg.refdef_current->vieworg);
685  if (viewDist < (TRAIL_FADE_CLOSE_DIST * TRAIL_FADE_FAR_SCALE))
686  {
687  if (viewDist < TRAIL_FADE_CLOSE_DIST)
688  {
689  fadeAlpha = 0.0;
690  }
691  else
692  {
693  fadeAlpha = (viewDist - TRAIL_FADE_CLOSE_DIST) / (TRAIL_FADE_CLOSE_DIST * TRAIL_FADE_FAR_SCALE);
694  }
695  if (fadeAlpha < j->alpha)
696  {
697  j->alpha = fadeAlpha;
698  }
699  if (fadeAlpha < jNext->alpha)
700  {
701  jNext->alpha = fadeAlpha;
702  }
703  }
704  }
705 
706  // now output the QUAD for this segment
707 
708  // 1 ----
709  VectorMA(j->pos, 0.5f * j->width, up, p);
710  VectorCopy(p, verts[i].xyz);
711  verts[i].st[0] = s;
712  verts[i].st[1] = 1.0;
713  for (k = 0; k < 3; k++)
714  {
715  verts[i].modulate[k] = ( unsigned char )(j->color[k] * 255.0f);
716  }
717  verts[i].modulate[3] = ( unsigned char )(j->alpha * 255.0f);
718 
719  // blend this with the previous junc
720  if (j != trail)
721  {
722  VectorAdd(verts[i].xyz, verts[i - 1].xyz, verts[i].xyz);
723  VectorScale(verts[i].xyz, 0.5f, verts[i].xyz);
724  VectorCopy(verts[i].xyz, verts[i - 1].xyz);
725  }
726  else if (j->flags & TJFL_FADEIN)
727  {
728  verts[i].modulate[3] = 0; // fade in
729  }
730 
731  i++;
732 
733  // 2 ----
734  VectorMA(p, -1 * j->width, up, p);
735  VectorCopy(p, verts[i].xyz);
736  verts[i].st[0] = s;
737  verts[i].st[1] = 0;
738  for (k = 0; k < 3; k++)
739  {
740  verts[i].modulate[k] = ( unsigned char )(j->color[k] * 255.0f);
741  }
742  verts[i].modulate[3] = ( unsigned char )(j->alpha * 255.0f);
743 
744  // blend this with the previous junc
745  if (j != trail)
746  {
747  VectorAdd(verts[i].xyz, verts[i - 3].xyz, verts[i].xyz);
748  VectorScale(verts[i].xyz, 0.5f, verts[i].xyz);
749  VectorCopy(verts[i].xyz, verts[i - 3].xyz);
750  }
751  else if (j->flags & TJFL_FADEIN)
752  {
753  verts[i].modulate[3] = 0; // fade in
754  }
755 
756  i++;
757 
758  if (trail->sType == STYPE_REPEAT)
759  {
760  s = jNext->sTex;
761  }
762  else
763  {
764  //s += sInc;
765  s += VectorDistance(j->pos, jNext->pos) / sInc; // FIXME: div/0!?
766  if (s > 1.0f)
767  {
768  s = 1.0f;
769  }
770  }
771 
772  // 3 ----
773  VectorMA(jNext->pos, -0.5f * jNext->width, up, p);
774  VectorCopy(p, verts[i].xyz);

832  {

844  }

863  }
864  }
865  }
