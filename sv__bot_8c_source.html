53 int SV_BotAllocateClient(int clientNum)
54 {
55  int i;
56  client_t *cl;
57 
58  // added possibility to request a clientnum
59  if (clientNum > 0 && clientNum >= sv_privateClients->integer)
60  {
61  if (clientNum >= sv_maxclients->integer)
62  {
63  return -1;
64  }
65 
66  cl = &svs.clients[clientNum];
67  if (cl->state != CS_FREE)
68  {
69  return -1;
70  }
71  else
72  {
73  i = clientNum;
74  }
75  }
76  else
77  {
78  // start searching above private slots for a free client slot
79  for (i = sv_privateClients->integer, cl = svs.clients + sv_privateClients->integer; i < sv_maxclients->integer; i++, cl++)
80  {
81  // Wolfenstein, never use the first slot, otherwise if a bot connects before the first client on a listen server, game won't start
82  if (i < 1)
83  {
84  continue;
85  }
86 
87  if (cl->state == CS_FREE)
88  {
89  break; // done.
90  }
91  }
92  }
93 
94  if (i == sv_maxclients->integer)
95  {
96  Com_DPrintf("SV_BotAllocateClient: can't allocate a bot client.\n");
97  return -1;
98  }
99 
100  cl->gentity = SV_GentityNum(i);
101  cl->gentity->s.number = i;
102  cl->state = CS_ACTIVE;
103  cl->lastPacketTime = svs.time;
104  cl->netchan.remoteAddress.type = NA_BOT;
105  cl->rate = 16384;
106 
107  return i;
108 }
