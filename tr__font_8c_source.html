113 void R_GetGlyphInfo(FT_GlyphSlot glyph, int *left, int *right, int *width, int *top, int *bottom, int *height, int *pitch)
114 {
115  *left = _FLOOR(glyph->metrics.horiBearingX - 1);
116  *right = _CEIL(glyph->metrics.horiBearingX + glyph->metrics.width + 1);
117  *width = _TRUNC(*right - *left);

180 static glyphInfo_t *RE_ConstructGlyphInfo(int imageSize, unsigned char *imageOut, int *xOut, int *yOut, int *maxHeight, FT_Face face, unsigned long codepoint, qboolean calcHeight)
181 {

183  unsigned char *src, *dst;
184  FT_Bitmap *bitmap = NULL;

186

188  // make sure everything is here
189  if (face != NULL)
190  {
191  FT_UInt index = FT_Get_Char_Index(face, codepoint);
192  float scaled_width, scaled_height;
193  int i;
194 
195  if (index == 0)
196  {
197  return &glyph; // nothing to render
198  }
199 
200  flags |= FT_LOAD_FORCE_AUTOHINT;
201 
202  // Test filtering
203  FT_Library_SetLcdFilter(ftLibrary, FT_LCD_FILTER_LIGHT);
204  flags |= FT_LOAD_TARGET_LCD;
205

207  bitmap = R_RenderGlyph(face->glyph, &glyph);
208  if (!bitmap)
209  {
210  return &glyph;
211  }
212 
213  if (glyph.height > *maxHeight)
214  {
215  *maxHeight = glyph.height;
216  }
217 
218  if (calcHeight)
219  {
220  ri.Free(bitmap->buffer);
221  ri.Free(bitmap);
222  return &glyph;
223  }
224 
225  scaled_width = glyph.pitch;
226  scaled_height = glyph.height;
227 
228  // we need to make sure we fit
229  if (*xOut + scaled_width + 1 >= imageSize - 1)
230  {
231  *xOut = 0;
232  *yOut += *maxHeight + 1;
233  }
234 
235  if (*yOut + *maxHeight + 1 >= imageSize - 1)
236  {
237  *yOut = -1;
238  *xOut = -1;
239  ri.Free(bitmap->buffer);
240  ri.Free(bitmap);
241  return &glyph;
242  }
243 
244  src = bitmap->buffer;
245  dst = imageOut + (*yOut * imageSize) + *xOut;
246 
247  if (bitmap->pixel_mode == FT_PIXEL_MODE_MONO)
248  {
249  for (i = 0; i < glyph.height; i++)
250  {
251  int j;
252  unsigned char *_src = src;
253  unsigned char *_dst = dst;
254  unsigned char mask = 0x80;
255  unsigned char val = *_src;
256 
257  for (j = 0; j < glyph.pitch; j++)
258  {
259  if (mask == 0x80)
260  {
261  val = *_src++;
262  }
263  if (val & mask)
264  {
265  *_dst = 0xff;
266  }
267  mask >>= 1;
268 
269  if (mask == 0)
270  {
271  mask = 0x80;
272  }
273  _dst++;
274  }
275 
276  src += glyph.pitch;
277  dst += imageSize;
278  }
279  }
280  else
281  {
282  for (i = 0; i < glyph.height; i++)
283  {
284  Com_Memcpy(dst, src, glyph.pitch);
285  src += glyph.pitch;
286  dst += imageSize;
287  }
288  }
289 
290  // we now have an 8 bit per pixel grey scale bitmap
291  // that is width wide and pf->ftSize->metrics.y_ppem tall
292 
293  glyph.imageHeight = scaled_height;
294  glyph.imageWidth = scaled_width;
295  glyph.s = (float)*xOut / imageSize;
296  glyph.t = (float)*yOut / imageSize;
297  glyph.s2 = glyph.s + (float)scaled_width / imageSize;
298  glyph.t2 = glyph.t + (float)scaled_height / imageSize;
299 
300  // ET uses pitch as a horizontal BearingX so we need to change this at this point for the font to be usable in game
301  // Super stupid btw
302  glyph.pitch = _TRUNC(face->glyph->metrics.horiBearingX);
303 
304  *xOut += scaled_width + 1;
305  }
306 
307  if (bitmap && bitmap->buffer)
308  {
309  ri.Free(bitmap->buffer);
310  }
311 
312  if (bitmap)
313  {
314  ri.Free(bitmap);
315  }
316 
317  return &glyph;
318 }

477  }
478

482  return qfalse;
483  }
484 
485  // Check if this font isn't loaded already
486  for (i = 0; i < registeredFontCount; i++)
487  {
488  if (Q_stricmp(va("fonts/%s_%i.dat", fontName, pointSize), registeredFont[i].datName) == 0)
489  {

491  {
492  Com_Memcpy(font, &registeredFont[i], sizeof(fontInfo_extra_t));

500  }

520
