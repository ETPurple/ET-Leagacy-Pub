50 qboolean GL_CheckForExtension(const char *ext)
51 {
52 #ifdef FEATURE_RENDERER2
53  int i = 0, exts = 0;
54 
55  glGetIntegerv(GL_NUM_EXTENSIONS, &exts);
56  for (i = 0; i < exts; i++)
57  {
58  if (!Q_stricmp(ext, (char *)glGetStringi(GL_EXTENSIONS, i)))
59  {
60  return qtrue;
61  }
62  }
63  return qfalse;
64 #else
65  const char *ptr = Q_stristr(glConfig.extensions_string, ext);
66 
67  if (ptr == NULL)
68  {
69  return qfalse;
70  }
71  ptr += strlen(ext);
72  return ((*ptr == ' ') || (*ptr == '\0')); // verify it's complete string.
73 #endif

180 static qboolean GLimp_CheckForVersionExtension(const char *ext, int coresince, qboolean required, cvar_t *var)
181 {
182  qboolean result = qfalse;
183 
184  if ((coresince >= 0 && coresince <= glConfig2.contextCombined) || GL_CheckForExtension(ext))
185  {
186  if (var && var->integer)
187  {
188  result = qtrue;
189  }
190  else if (!var)
191  {
192  result = qtrue;
193  }
194  }
195 
196  if (required && !result)
197  {
198  Ren_Fatal(MSG_ERR_OLD_VIDEO_DRIVER "\nYour GL driver is missing support for: %s\n", ext);
199  }
200 
201  if (result)
202  {
203  Com_Printf("...found OpenGL extension - %s\n", ext);
204  }
205  else
206  {
207  if (var)
208  {
209  Com_Printf("...ignoring %s\n", ext);
210  }
211  else
212  {
213  Com_Printf("...%s not found\n", ext);
214  }
215  }
216 
217  return result;
218 }
