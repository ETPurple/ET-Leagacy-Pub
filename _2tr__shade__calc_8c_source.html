245 {
246 #if 1
247  int i;
248  expOperation_t op;
249  expOperation_t ops[MAX_EXPRESSION_OPS];
250  int numOps = 0;
251  float value = 0;
252  float value1 = 0;
253  float value2 = 0;
254  extern const opstring_t opStrings[];
255 
256  if (!exp || !exp->active)
257  {
258  return defaultValue;
259  }
260 
261  // http://www.qiksearch.com/articles/cs/postfix-evaluation/
262  // http://www.kyz.uklinux.net/evaluate/
263 
264  for (i = 0; i < exp->numOps; i++)
265  {
266  op = exp->ops[i];

268  switch (op.type)

281  numOps--;
282 
283  value = -value1;
284 
285  // push result
286  op.type = OP_NUM;
287  op.value = value;
288  ops[numOps++] = op;
289  break;
290  }
291  case OP_NUM:
292  case OP_TIME:
293  case OP_PARM0:
294  case OP_PARM1:
295  case OP_PARM2:
296  case OP_PARM3:
297  case OP_PARM4:
298  case OP_PARM5:
299  case OP_PARM6:
300  case OP_PARM7:
301  case OP_PARM8:
302  case OP_PARM9:
303  case OP_PARM10:
304  case OP_PARM11:
305  case OP_GLOBAL0:
306  case OP_GLOBAL1:
307  case OP_GLOBAL2:
308  case OP_GLOBAL3:
309  case OP_GLOBAL4:
310  case OP_GLOBAL5:
311  case OP_GLOBAL6:
312  case OP_GLOBAL7:

314  case OP_FRAMEBUFFEROBJECTS:
315  case OP_SOUND:
316  case OP_DISTANCE:
317  ops[numOps++] = op;
318  break;
319  case OP_TABLE:
320  {
321  shaderTable_t *table;
322  int numValues;
323  float index;
324  float lerp;
325  int oldIndex;
326  int newIndex;
327 
328  if (numOps < 1)
329  {
330  Ren_Print("WARNING: shader %s has numOps < 1 for table operator\n", tess.surfaceShader->name);
331  return defaultValue;

333 
334  value1 = GetOpValue(&ops[numOps - 1]);
335  numOps--;
336 
337  table = tr.shaderTables[(int)op.value];
338 
339  numValues = table->numValues;
340 
341  index = value1 * numValues; // float index into the table?s elements
342  lerp = index - floor(index); // being inbetween two elements of the table
343 
344  oldIndex = (int)index;
345  newIndex = (int)index + 1;
346 
347  if (table->clamp)
348  {
349  // clamp indices to table-range
350  Q_clamp(oldIndex, 0, numValues - 1);
351  Q_clamp(newIndex, 0, numValues - 1);
352  }
353  else
354  {
355  // wrap around indices
356  oldIndex %= numValues;
357  newIndex %= numValues;
358  }
359 
360  if (table->snap)
361  {
362  // use fixed value
363  value = table->values[oldIndex];
364  }
365  else
366  {
367  // lerp value
368  value = table->values[oldIndex] + ((table->values[newIndex] - table->values[oldIndex]) * lerp);
369  }
370 
371  //Ren_Print("%s: %i %i %f\n", table->name, oldIndex, newIndex, value);
372 
373  // push result
374  op.type = OP_NUM;
375  op.value = value;
376  ops[numOps++] = op;
377  break;
378  }
379  default:
380  {
381  if (numOps < 2)
382  {
383  Ren_Print("WARNING: shader %s has numOps < 2 for binary operator %s\n", tess.surfaceShader->name,
384  opStrings[op.type].s);
385  return defaultValue;
386  }
387 
388  value2 = GetOpValue(&ops[numOps - 1]);
389  numOps--;
390 
391  value1 = GetOpValue(&ops[numOps - 1]);
392  numOps--;
393 
394  switch (op.type)
395  {
396  case OP_LAND:
397  value = value1 && value2;
398  break;
399  case OP_LOR:
400  value = value1 || value2;
401  break;
402  case OP_GE:
403  value = value1 >= value2;
404  break;
405  case OP_LE:
406  value = value1 <= value2;
407  break;
408  case OP_LEQ:
409  value = value1 == value2;
410  break;
411  case OP_LNE:
412  value = value1 != value2;
413  break;
414  case OP_ADD:
415  value = value1 + value2;
416  break;
417  case OP_SUB:
418  value = value1 - value2;
419  break;
420  case OP_DIV:
421  if (value2 == 0)
422  {
423  // don't divide by zero
424  value = value1;
425  }
426  else
427  {
428  value = value1 / value2;
429  }
430  break;
431  case OP_MOD:
432  value = (float)((int)value1 % (int)value2);
433  break;
434  case OP_MUL:
435  value = value1 * value2;
436  break;
437  case OP_LT:
438  value = value1 < value2;
439  break;
440  case OP_GT:
441  value = value1 > value2;
442  break;
443  default:
444  value = value1 = value2 = 0;
445  break;
446  }
447 
448  //Ren_Print("%s: %f %f %f\n", opStrings[op.type].s, value, value1, value2);
449 
450  // push result
451  op.type = OP_NUM;
452  op.value = value;
453  ops[numOps++] = op;
454  break;
455  }
456  }
457  }
458 
459  return GetOpValue(&ops[0]);
460 #else
461  return defaultValue;
462 #endif
463 }

483

487

489

491  {
492  off = (xyz[0] + xyz[1] + xyz[2]) * ds->deformationSpread;
493

495  ds->deformationWave.amplitude, ds->deformationWave.phase + off, ds->deformationWave.frequency);
496 
497  VectorScale(normal, scale, offset);
498 
499  xyz[0] += offset[0];

501  xyz[2] += offset[2];
502  }
503  }
504  else if (ds->deformationWave.frequency == 0.f)
505  {
506  scale = RB_EvalWaveForm(&ds->deformationWave);
507 
508  for (i = 0; i < tess.numVertexes; i++, xyz += 4, normal += 4)
509  {
510  VectorScale(normal, scale, offset);
511 
512  xyz[0] += offset[0];
513  xyz[1] += offset[1];
514  xyz[2] += offset[2];
515  }
516  }
517  else //if (ds->deformationWave.frequency < 0)
518  {
519  float off;

523 
524  if (VectorCompare(backEnd.currentEntity->e.fireRiseDir, vec3_origin))

546  inverse = qtrue;
547  ds->deformationWave.frequency -= 999;
548  }
549 
550  table = TableForFunc(ds->deformationWave.func);
551 
552  for (i = 0; i < tess.numVertexes; i++, xyz += 4, normal += 4)

555

557  ds->deformationWave.amplitude, ds->deformationWave.phase + off, ds->deformationWave.frequency);
558

560 
561  if (dot * scale > 0)
562  {
563  if (inverse)
564  {
565  scale *= -1;
566  }
567  VectorMA(xyz, dot * scale, worldUp, xyz);
568  }
569  }
570 
571  if (inverse)
572  {
573  ds->deformationWave.frequency += 999;
574  }
575  ds->deformationWave.frequency *= -1;
576  }
577 }

752 static void AutospriteDeform(void)

769  }

772  tess.numVertexes = 0;

775

779  GlobalVectorToLocal(backEnd.viewParms.orientation.axis[2], upDir);
780  }
781  else

784  VectorCopy(backEnd.viewParms.orientation.axis[2], upDir);
785  }

789  // find the midpoint

798

805  }

808  if (backEnd.currentEntity->e.nonNormalizedAxes)
809  {
810  float axisLength;

842 static void Autosprite2Deform(void)
843 {
844  int i, j, k;
845  int indexes;
846  float *xyz;
847  vec3_t forward;
848 
849  if (tess.numVertexes & 3)
850  {
851  Ren_Warning("Autosprite2 shader %s had odd vertex count\n", tess.surfaceShader->name);
852  }
853  if (tess.numIndexes != (tess.numVertexes >> 2) * 6)
854  {
855  Ren_Warning("Autosprite2 shader %s had odd index count\n", tess.surfaceShader->name);
856  }
857 
858  if (backEnd.currentEntity != &tr.worldEntity)
859  {
860  GlobalVectorToLocal(backEnd.viewParms.orientation.axis[0], forward);
861  }
862  else
863  {
864  VectorCopy(backEnd.viewParms.orientation.axis[0], forward);
865  }
866 
867  // this is a lot of work for two triangles...
868  // we could precalculate a lot of it is an issue, but it would mess up
869  // the shader abstraction
870  for (i = 0, indexes = 0; i < tess.numVertexes; i += 4, indexes += 6)
871  {
872  float lengths[2];
873  int nums[2];
874  vec3_t mid[2];
875  vec3_t major, minor;
876  float *v1, *v2;
877 
878  // find the midpoint
879  xyz = tess.xyz[i];
880 
881  // identify the two shortest edges
882  nums[0] = nums[1] = 0;
883  lengths[0] = lengths[1] = 999999;
884 
885  for (j = 0; j < 6; j++)
886  {
887  float l;
888  vec3_t temp;
889 
890  v1 = xyz + 4 * edgeVerts[j][0];
891  v2 = xyz + 4 * edgeVerts[j][1];
892 
893  VectorSubtract(v1, v2, temp);
894 
895  l = DotProduct(temp, temp);
896  if (l < lengths[0])
897  {
898  nums[1] = nums[0];
899  lengths[1] = lengths[0];
900  nums[0] = j;
901  lengths[0] = l;
902  }
903  else if (l < lengths[1])
904  {
905  nums[1] = j;
906  lengths[1] = l;
907  }
908  }
909 
910  for (j = 0; j < 2; j++)
911  {
912  v1 = xyz + 4 * edgeVerts[nums[j]][0];
913  v2 = xyz + 4 * edgeVerts[nums[j]][1];
914 
915  mid[j][0] = 0.5f * (v1[0] + v2[0]);
916  mid[j][1] = 0.5f * (v1[1] + v2[1]);
917  mid[j][2] = 0.5f * (v1[2] + v2[2]);
918  }
919 
920  // find the vector of the major axis
921  VectorSubtract(mid[1], mid[0], major);
922 
923  // cross this with the view direction to get minor axis
924  CrossProduct(major, forward, minor);
925  VectorNormalize(minor);
926 
927  // re-project the points
928  for (j = 0; j < 2; j++)
929  {
930  float l;
931 
932  v1 = xyz + 4 * edgeVerts[nums[j]][0];
933  v2 = xyz + 4 * edgeVerts[nums[j]][1];
934 
935  l = 0.5 * sqrt(lengths[j]);
936 
937  // we need to see which direction this edge
938  // is used to determine direction of projection
939  for (k = 0; k < 5; k++)
940  {
941  if (tess.indexes[indexes + k] == i + edgeVerts[nums[j]][0]
942  && tess.indexes[indexes + k + 1] == i + edgeVerts[nums[j]][1])
943  {
944  break;
945  }
946  }
947 
948  if (k == 5)
949  {
950  VectorMA(mid[j], l, minor, v1);
951  VectorMA(mid[j], -l, minor, v2);
952  }
953  else
954  {
955  VectorMA(mid[j], -l, minor, v1);
956  VectorMA(mid[j], l, minor, v2);
957  }
958  }
959  }
960 }
