446 qboolean CG_ParseSpawnVars(void)
447 {
448  char keyname[MAX_TOKEN_CHARS];
449  char com_token[MAX_TOKEN_CHARS];
450 
451  cg.numSpawnVars = 0;
452  cg.numSpawnVarChars = 0;
453 
454  // parse the opening brace
455  if (!trap_GetEntityToken(com_token, sizeof(com_token)))
456  {
457  // end of spawn string
458  return qfalse;
459  }
460  if (com_token[0] != '{')
461  {
462  CG_Error("CG_ParseSpawnVars: found %s when expecting {\n", com_token);
463  }
464 
465  // go through all the key / value pairs
466  while (1)
467  {
468  // parse key
469  if (!trap_GetEntityToken(keyname, sizeof(keyname)))
470  {
471  CG_Error("CG_ParseSpawnVars: EOF without closing brace\n");
472  }
473 
474  if (keyname[0] == '}')
475  {
476  break;
477  }
478 
479  // parse value
480  if (!trap_GetEntityToken(com_token, sizeof(com_token)))
481  {
482  CG_Error("CG_ParseSpawnVars: EOF without closing brace\n");
483  }
484 
485  if (com_token[0] == '}')
486  {
487  CG_Error("CG_ParseSpawnVars: closing brace without data\n");
488  }
489  if (cg.numSpawnVars == MAX_SPAWN_VARS)
490  {
491  CG_Error("CG_ParseSpawnVars: MAX_SPAWN_VARS\n");
492  }
493  cg.spawnVars[cg.numSpawnVars][0] = CG_AddSpawnVarToken(keyname);
494  cg.spawnVars[cg.numSpawnVars][1] = CG_AddSpawnVarToken(com_token);
495  cg.numSpawnVars++;
496  }
497 
498  return qtrue;
499 }
