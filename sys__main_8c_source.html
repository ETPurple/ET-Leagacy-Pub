267 void Sys_AnsiColorPrint(const char *msg)
268 {
269  static char buffer[MAXPRINTMSG];
270  int i, j, _found, length = 0;
271 
272  // colors hash from http://wolfwiki.anime.net/index.php/Color_Codes
273  static int etAnsiHash[][6] =
274  {
275  // here should be black, but it's invisible in terminal
276  // so you see dark grey here. like in 9 color
277  { 1, 30, 0, '0', 'p', 'P' },
278  { 1, 31, 0, '1', 'q', 'Q' },
279  { 1, 32, 0, '2', 'r', 'R' },
280  { 1, 33, 0, '3', 's', 'S' },
281  { 1, 34, 0, '4', 't', 'T' },
282  { 1, 36, 0, '5', 'u', 'U' },
283  { 1, 35, 0, '6', 'v', 'V' },
284  { 1, 37, 0, '7', 'w', 'W' },
285 
286  { 38, 5, 208, '8', 'x', 'X' },
287  { 1, 30, 0, '9', 'y', 'Y' },
288  { 0, 37, 0, 'z', 'Z', ':' }, // the same
289  { 0, 37, 0, '[', '{', ';' }, // --------
290  { 0, 32, 0, '<', '\\', '|' },
291  { 0, 33, 0, '=', ']', '}' },
292  { 0, 34, 0, '>', '~', '^' },
293  { 0, 31, 0, '?', '_', 0 },
294  { 38, 5, 94, '@', '`', 0 },
295 
296  { 38, 5, 214, 'a', 'A', '!' },
297  { 38, 5, 30, 'b', 'B', '"' },
298  { 38, 5, 90, 'c', 'C', '#' },
299  { 38, 5, 33, 'd', 'D', '$' },
300  { 38, 5, 93, 'e', 'E', '%' },
301  { 38, 5, 38, 'f', 'F', '&' },
302  { 38, 5, 194, 'g', 'G', '\'' },
303  { 38, 5, 29, 'h', 'H', '(' },
304 
305  { 38, 5, 197, 'i', 'I', ')' },
306  { 38, 5, 124, 'j', 'J', '*' },
307  { 38, 5, 130, 'k', 'K', '+' },
308  { 38, 5, 179, 'l', 'L', ',' },
309  { 38, 5, 143, 'm', 'M', '-' },
310  { 38, 5, 229, 'n', 'N', '.' },
311  { 38, 5, 228, 'o', 'O', '/' }
312  };
313 
314  while (*msg)
315  {
316  if (Q_IsColorString(msg) || *msg == '\n' || *msg == '\\')
317  {
318  // First empty the buffer
319  if (length > 0)
320  {
321  buffer[length] = '\0';
322  fputs(buffer, stderr);
323  length = 0;
324  }
325 
326  if (*msg == '\n')
327  {
328  // Issue a reset and then the newline
329  fputs("\033[0m\n", stderr);
330  msg++;
331  }
332  else if (*msg == '\\')
333  {
334  fputs("\033[0m\\", stderr);
335  msg++;
336  }
337  else if (*(msg + 1) == '7' && (*(msg + 2) == '"' || *(msg + 2) == '\'' || *(msg + 2) == '\0'))
338  {
339  fputs("\033[0m", stderr);
340  msg += 2;
341  }
342  else
343  {
344  // Print the color code
345  _found = 0;
346  for (i = 0; i < 32; i++)
347  {
348  if (_found)
349  {
350  break;
351  }
352 
353  for (j = 3; j < 6; j++)
354  {
355  if (etAnsiHash[i][j] == 0)
356  {
357  break;
358  }
359 
360  if (*(msg + 1) == etAnsiHash[i][j])
361  {
362  if (etAnsiHash[i][2] == 0)
363  {
364  Com_sprintf(buffer, sizeof(buffer), "\033[%d;%dm",
365  etAnsiHash[i][0], etAnsiHash[i][1]);
366  _found = 1;
367  break;
368  }
369  else
370  {
371  Com_sprintf(buffer, sizeof(buffer), "\033[%d;%d;%dm",
372  etAnsiHash[i][0], etAnsiHash[i][1], etAnsiHash[i][2]);
373  _found = 1;
374  break;
375  }
376  }
377  }
378  }
379 
380  fputs(buffer, stderr);
381  msg += 2;
382  }
383  }
384  else
385  {
386  if (length >= MAXPRINTMSG - 1)
387  {
388  break;
389  }
390 
391  buffer[length] = *msg;
392  length++;
393  msg++;
394  }
395  }
396 
397  // Empty anything still left in the buffer
398  if (length > 0)
399  {
400  buffer[length] = '\0';
401  fputs(buffer, stderr);
402  }
403 }

575 static void *Sys_TryLibraryLoad(const char *base, const char *gamedir, const char *fname)
576 {
577  void *libHandle;
578  char *fn;
579 
580 #ifdef __APPLE__
581  Com_Printf("Sys_LoadDll -> Sys_TryLibraryLoad(%s, %s, %s)... \n", base, gamedir, fname);
582  libHandle = NULL;
583 
584  // Incoming is (for example) "cgame_mac"
585  // What we may actually have is:
586  // 1) A zipped .bundle package
587  // 2) A .dylib without an extension
588  // 3) A .dylib with an extension we need to append.
589  //
590  // In older versions of OS X (pre 10.5), dylibs could not be unloaded allowing the host process to reclaim that address space.
591  // This is why .bundles were used instead. When W:ET originally shipped, it used used .bundle packages for the VM libraries,
592  // but to make them single files, it zipped that .bundle package and had no extension at all (ie just "cgame_mac"). But now
593  // that dylibs can be unloaded, there's no practical difference between the two (for W:ET's purposes), so using a single file
594  // dylib is simpler. That's why we now support a dylib with some backward compatibility to allow .bundles.
595 
596  // 1: The zipped .bundle package
597  {
598  Com_Printf("-- Trying zipped .bundle... ");
599  fn = FS_BuildOSPath(base, gamedir, fname);
600  if (FS_Unzip(fn, qtrue))
601  {
602  char buffer[MAX_OSPATH];
603  Com_sprintf(buffer, sizeof(buffer), "%s.bundle/Contents/MacOS/%s", fname, fname);
604  fn = FS_BuildOSPath(Cvar_VariableString("fs_homepath"), gamedir, buffer);
605 
606  libHandle = Sys_LoadLibrary(fn);
607  if (!libHandle)
608  {
609  Com_Printf("failed: %s", Sys_LibraryError());
610  }
611  else
612  {
613  Com_Printf("succeeded\n");
614  }
615  }
616  else
617  {
618  Com_Printf("failed (not a valid zip)\n");
619  }
620  }
621 
622  // 2: The dylib without an extension
623  if (!libHandle)
624  {
625  fn = FS_BuildOSPath(base, gamedir, fname);
626 
627  Com_Printf("-- Trying extension-less dylib... ");
628  libHandle = Sys_LoadLibrary(fn);
629  if (!libHandle)
630  {
631  Com_Printf("failed: %s", Sys_LibraryError());
632  }
633  else
634  {
635  Com_Printf("succeeded\n");
636  }
637  }
638 
639  // 3: The dylib with an extension
640  if (!libHandle)
641  {
642  char buffer[MAX_OSPATH];
643  Com_sprintf(buffer, sizeof(buffer), "%s.dylib", fname);
644  fn = FS_BuildOSPath(Cvar_VariableString("fs_homepath"), gamedir, buffer);
645 
646  Com_Printf("-- Trying dylib with extension... ");
647  libHandle = Sys_LoadLibrary(fn);
648  if (!libHandle)
649  {
650  Com_Printf("failed: %s", Sys_LibraryError());
651  }
652  else
653  {
654  Com_Printf("succeeded\n");
655  }
656  }
657 
658 #else // __APPLE__
659 
660  fn = FS_BuildOSPath(base, gamedir, fname);
661  Com_Printf("Sys_LoadDll(%s)... ", fn);
662 
663  libHandle = Sys_LoadLibrary(fn);
664 
665  if (!libHandle)
666  {
667  Com_Printf("failed: %s", Sys_LibraryError());
668  return NULL;
669  }
670 
671  Com_Printf("succeeded\n");
672 
673 #endif // __APPLE__
674 
675  return libHandle;
676 }

691 void *Sys_LoadGameDll(const char *name, qboolean extract,
692  intptr_t(**entryPoint) (int, ...),
693  intptr_t (*systemcalls)(intptr_t, ...))
694 {
695  void *libHandle;
696  void (*dllEntry)(intptr_t (*syscallptr)(intptr_t, ...));
697  char fname[MAX_OSPATH];
698  char *basepath;
699  char *homepath;
700  char *gamedir;
701 
702  etl_assert(name);
703 
704  Com_sprintf(fname, sizeof(fname), Sys_GetDLLName("%s"), name);
705 
706  // TODO: use fs_searchpaths from files.c
707  basepath = Cvar_VariableString("fs_basepath");
708  homepath = Cvar_VariableString("fs_homepath");
709  gamedir = Cvar_VariableString("fs_game");
710 
711  // STORY TIME
712  //
713  //When doing an debug build just load the mod lib from the basepath as that will have the debug pointers
714  //
715  // Now the code always just unpacks new libraries from the packs on release builds.
716  // So the libraryfiles in the homepath are always refreshed with latest from the packs.
717  // This fixes many issues with clients loading old libraries from the mod paths.
718  //
719  // The way it used to work is described under (or in debug mode)..
720  //
721  // if the server is pure, extract the dlls from the mod_bin.pk3 so
722  // that they can be referenced
723  //
724  // If the server is not pure, then the
725  // lib must either already be in the homepath, or in the basepath,
726  // without being in a pak. For a pure server, it always grabs the lib
727  // from within a pak. This means there must be two copies of the lib!
728  //
729  // So now, if connecting to an impure server, and the lib was not
730  // loaded from homepath or the basepath, let's pull it out of the pak.
731  // This means we only *need* the copy that's in the pak, and will use
732  // it if another copy isn't found first.
733 #ifdef LEGACY_DEBUG
734 #define SEARCHPATH1 basepath
735 #define SEARCHPATH2 homepath
736 #define LIB_DO_UNPACK Cvar_VariableIntegerValue("sv_pure")
737 #else
738 #define LIB_DO_UNPACK qtrue
739 #define SEARCHPATH1 homepath
740 #define SEARCHPATH2 basepath
741 #endif
742 
743 #ifndef DEDICATED
744  if (LIB_DO_UNPACK && extract)
745  {
746  if (!FS_CL_ExtractFromPakFile(homepath, gamedir, fname))
747  {
748  // no drama, we still check SEARCHPATH2
749  Com_Printf("Sys_LoadDll(%s/%s) failed to extract library from fs_homepath\n", gamedir, name);
750  }
751  else
752  {
753  Com_Printf("Sys_LoadGameDll -> FS_CL_ExtractFromPakFile(%s, %s, %s)\n", homepath, gamedir, fname);
754  }
755  }
756 #endif
757 
758  libHandle = Sys_TryLibraryLoad(SEARCHPATH1, gamedir, fname);
759 
760  if (!libHandle && SEARCHPATH2)
761  {
762  libHandle = Sys_TryLibraryLoad(SEARCHPATH2, gamedir, fname);
763  }
764 
765 #ifndef DEDICATED
766  if (extract)
767  {
768  if (!libHandle && !LIB_DO_UNPACK)
769  {
770  if (!FS_CL_ExtractFromPakFile(homepath, gamedir, fname))
771  {
772  Com_Printf("Sys_LoadDll(%s/%s) failed to extract library\n", gamedir, name);
773  return NULL;
774  }
775 
776  Com_Printf("Sys_LoadGameDll -> FS_CL_ExtractFromPakFile(%s, %s, %s)\n", homepath, gamedir, fname);
777  libHandle = Sys_TryLibraryLoad(homepath, gamedir, fname);
778  }
779 
780  // use legacy ui for download process (mod binary pk3 isn't extracted)
781  if (!strcmp(name, "ui") && !libHandle && strcmp(gamedir, DEFAULT_MODGAME))
782  {
783  Com_Printf("Sys_LoadDll: mod initialisation - ui fallback\n");
784 
785  libHandle = Sys_TryLibraryLoad(homepath, DEFAULT_MODGAME, fname);
786 
787  if (!libHandle)
788  {
789  libHandle = Sys_TryLibraryLoad(basepath, DEFAULT_MODGAME, fname);
790  }
791  }
792  }
793 #endif
794 
795  if (!libHandle)
796  {
797  Com_Printf("Sys_LoadDll(%s/%s) failed to load library\n", gamedir, name);
798  return NULL;
799  }
800 
801  dllEntry = (void(QDECL *)(intptr_t(QDECL *)(intptr_t, ...)))Sys_LoadFunction(libHandle, "dllEntry");
802  *entryPoint = (intptr_t(QDECL *)(int, ...))Sys_LoadFunction(libHandle, "vmMain");
803 
804  if (!*entryPoint || !dllEntry)
