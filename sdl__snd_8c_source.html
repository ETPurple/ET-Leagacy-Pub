64 static void SNDDMA_AudioCallback(void *userdata, Uint8 *stream, int len)
65 {
66  int pos = (dmapos * (dma.samplebits / 8));
67 
68  if (pos >= dmasize)
69  {
70  dmapos = pos = 0;
71  }
72 
73  if (!snd_inited) /* shouldn't happen, but just in case... */
74  {
75  Com_Memset(stream, '\0', len);
76  return;
77  }
78  else
79  {
80  int tobufend = dmasize - pos; /* bytes to buffer's end. */
81  int len1 = len;
82  int len2 = 0;
83 
84  if (len1 > tobufend)
85  {
86  len1 = tobufend;
87  len2 = len - len1;
88  }
89  Com_Memcpy(stream, dma.buffer + pos, len1);
90  if (len2 <= 0)
91  {
92  dmapos += (len1 / (dma.samplebits / 8));
93  }
94  else /* wraparound? */
95  {
96  Com_Memcpy(stream + len1, dma.buffer, len2);
97  dmapos = (len2 / (dma.samplebits / 8));
98  }
99  }
100 
101  if (dmapos >= dmasize)
102  {
103  dmapos = 0;
104  }
105 }
