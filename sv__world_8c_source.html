119 worldSector_t *SV_CreateworldSector(int depth, vec3_t mins, vec3_t maxs)
120 {
121  worldSector_t *anode = &sv_worldSectors[sv_numworldSectors];
122  vec3_t size;
123  vec3_t mins1, maxs1, mins2, maxs2;
124 
125  sv_numworldSectors++;
126 
127  if (depth == AREA_DEPTH)
128  {
129  anode->axis = -1;
130  anode->children[0] = anode->children[1] = NULL;
131  return anode;
132  }
133 
134  VectorSubtract(maxs, mins, size);
135  if (size[0] > size[1])
136  {
137  anode->axis = 0;
138  }
139  else
140  {
141  anode->axis = 1;
142  }
143 
144  anode->dist = 0.5f * (maxs[anode->axis] + mins[anode->axis]);
145  VectorCopy(mins, mins1);
146  VectorCopy(mins, mins2);
147  VectorCopy(maxs, maxs1);
148  VectorCopy(maxs, maxs2);
149 
150  maxs1[anode->axis] = mins2[anode->axis] = anode->dist;
151 
152  anode->children[0] = SV_CreateworldSector(depth + 1, mins2, maxs2);
153  anode->children[1] = SV_CreateworldSector(depth + 1, mins1, maxs1);
154

220 void SV_LinkEntity(sharedEntity_t *gEnt)
221 {
222  worldSector_t *node;
223  int leafs[MAX_TOTAL_ENT_LEAFS];
224  int cluster;
225  int num_leafs;
226  int i, j, k;
227  int area;
228  int lastLeaf;
229  float *origin, *angles;
230  svEntity_t *ent;
231 
232  ent = SV_SvEntityForGentity(gEnt);
233 
234  // sanity check for possible currentOrigin being reset bug
235  if (!gEnt->r.bmodel && vec3_compare(gEnt->r.currentOrigin, vec3_origin))
236  {
237  // I've seen this warning a lot - let map makers know which entity is affected.
238  // FIXME: - Clarify if this warning is false positive for some ents
239  // - print gEnt->s.eType as string
240  Com_DPrintf("WARNING: BBOX entity %i (type: %i) is being linked at world origin, this is probably a bug - see /entitylist cmd\n", gEnt->s.number, gEnt->s.eType);
241  }
242 
243  if (ent->worldSector)
244  {
245  SV_UnlinkEntity(gEnt); // unlink from old position
246  }
247 
248  // encode the size into the entityState_t for client prediction
249  if (gEnt->r.bmodel)
250  {
251  gEnt->s.solid = SOLID_BMODEL; // a solid_box will never create this value
252 
253  // for the origin only bmodel checks
254  ent->originCluster = CM_LeafCluster(CM_PointLeafnum(gEnt->r.currentOrigin));
255  }
256  else if (gEnt->r.contents & (CONTENTS_SOLID | CONTENTS_BODY))
257  {
258  // assume that x/y are equal and symetric
259  i = gEnt->r.maxs[0];
260  if (i < 1)
261  {
262  i = 1;
263  }
264  if (i > 255)
265  {
266  i = 255;
267  }
268 
269  // z is not symetric
270  j = (-gEnt->r.mins[2]);
271  if (j < 1)
272  {
273  j = 1;
274  }
275  if (j > 255)
276  {
277  j = 255;
278  }
279 
280  // and z maxs can be negative...
281  k = (gEnt->r.maxs[2] + 32);
282  if (k < 1)
283  {
284  k = 1;
285  }
286  if (k > 255)
287  {
288  k = 255;
289  }
290 
291  gEnt->s.solid = (k << 16) | (j << 8) | i;
292  }
293  else
294  {
295  gEnt->s.solid = 0;
296  }
297 
298  // get the position
299  origin = gEnt->r.currentOrigin;
300  angles = gEnt->r.currentAngles;
301 
302  // set the abs box
303  if (gEnt->r.bmodel && (angles[0] != 0.f || angles[1] != 0.f || angles[2] != 0.f))
304  {
305  // expand for rotation
306  float max;
307 
308  max = RadiusFromBounds(gEnt->r.mins, gEnt->r.maxs);
309  for (i = 0 ; i < 3 ; i++)
310  {
311  gEnt->r.absmin[i] = origin[i] - max;
312  gEnt->r.absmax[i] = origin[i] + max;
313  }
314  }
315  else
316  {
317  // normal
318  VectorAdd(origin, gEnt->r.mins, gEnt->r.absmin);
319  VectorAdd(origin, gEnt->r.maxs, gEnt->r.absmax);
320  }
321 
322  // because movement is clipped an epsilon away from an actual edge,
323  // we must fully check even when bounding boxes don't quite touch
324  gEnt->r.absmin[0] -= 1;
325  gEnt->r.absmin[1] -= 1;
326  gEnt->r.absmin[2] -= 1;
327  gEnt->r.absmax[0] += 1;
328  gEnt->r.absmax[1] += 1;
329  gEnt->r.absmax[2] += 1;
330 
331  // link to PVS leafs
332  ent->numClusters = 0;
333  ent->lastCluster = 0;
334  ent->areanum = -1;
335  ent->areanum2 = -1;
336 
337  //get all leafs, including solids
338  num_leafs = CM_BoxLeafnums(gEnt->r.absmin, gEnt->r.absmax,
339  leafs, MAX_TOTAL_ENT_LEAFS, &lastLeaf);
340 
341  // if none of the leafs were inside the map, the
342  // entity is outside the world and can be considered unlinked
343  if (!num_leafs)
344  {
345  return;
346  }
347 
348  // set areas, even from clusters that don't fit in the entity array
349  for (i = 0 ; i < num_leafs ; i++)
350  {
351  area = CM_LeafArea(leafs[i]);
352  if (area != -1)
353  {
354  // doors may legally straggle two areas,
355  // but nothing should evern need more than that
356  if (ent->areanum != -1 && ent->areanum != area)
357  {
358  if (ent->areanum2 != -1 && ent->areanum2 != area && sv.state == SS_LOADING)
359  {
360  Com_DPrintf("Entity %i touching 3 areas at %f %f %f\n",
361  gEnt->s.number,
362  (double)gEnt->r.absmin[0], (double)gEnt->r.absmin[1], (double)gEnt->r.absmin[2]);
363  }
364 
365  ent->areanum2 = area;
366  }
367  else
368  {
369  ent->areanum = area;
370  }
371  }
372  }
373 
374  // store as many explicit clusters as we can
375  ent->numClusters = 0;
376  for (i = 0 ; i < num_leafs ; i++)
377  {
378  cluster = CM_LeafCluster(leafs[i]);
379  if (cluster != -1)
380  {
381  ent->clusternums[ent->numClusters++] = cluster;
382  if (ent->numClusters == MAX_ENT_CLUSTERS)
383  {
384  break;
385  }
386  }
387  }
388 
389  // store off a last cluster if we need to
390  if (i != num_leafs)
391  {
392  ent->lastCluster = CM_LeafCluster(lastLeaf);
393  }
394 
395  gEnt->r.linkcount++;
396 
397  // find the first world sector node that the ent's box crosses
398  node = sv_worldSectors;
399  while (1)
400  {
401  if (node->axis == -1)
402  {
403  break;
404  }
405  if (gEnt->r.absmin[node->axis] > node->dist)
406  {
407  node = node->children[0];
408  }
409  else if (gEnt->r.absmax[node->axis] < node->dist)
410  {
411  node = node->children[1];
412  }
413  else
414  {
415  break; // crosses the node
416  }
417  }
418 
419  // link it in
420  ent->worldSector = node;
421  ent->nextEntityInWorldSector = node->entities;
422  node->entities = ent;
423 
424  gEnt->r.linked = qtrue;
425 }

449 void SV_AreaEntities_r(worldSector_t *node, areaParms_t *ap)
450 {
451  svEntity_t *check, *next;
452  sharedEntity_t *gcheck;
453 
454  for (check = node->entities ; check ; check = next)
455  {
456  next = check->nextEntityInWorldSector;
457 
458  gcheck = SV_GEntityForSvEntity(check);
459 
460  if (!gcheck->r.linked)
461  {
462  continue;
463  }
464 
465  if (gcheck->r.absmin[0] > ap->maxs[0]
466  || gcheck->r.absmin[1] > ap->maxs[1]
467  || gcheck->r.absmin[2] > ap->maxs[2]
468  || gcheck->r.absmax[0] < ap->mins[0]
469  || gcheck->r.absmax[1] < ap->mins[1]
470  || gcheck->r.absmax[2] < ap->mins[2])
471  {
472  continue;
473  }
474 
475  if (ap->count == ap->maxcount)
476  {
477  Com_Printf("SV_AreaEntities: MAXCOUNT\n");
478  return;
479  }
480 
481  ap->list[ap->count] = check - sv.svEntities;
482  ap->count++;
483  }
484 
485  if (node->axis == -1)
486  {
487  return; // terminal node
488  }
489 
490  // recurse down both sides
491  if (ap->maxs[node->axis] > node->dist)
492  {
493  SV_AreaEntities_r(node->children[0], ap);
494  }
495  if (ap->mins[node->axis] < node->dist)
496  {
497  SV_AreaEntities_r(node->children[1], ap);
498  }
499 }

609  {
610  passOwnerNum = (SV_GentityNum(clip->passEntityNum))->r.ownerNum;
611  if (passOwnerNum == ENTITYNUM_NONE)
612  {
613  passOwnerNum = -1;
614  }
615  }
616  else
617  {
618  passOwnerNum = -1;
619  }
620 
621  for (i = 0 ; i < num ; i++)
622  {
623  if (clip->trace.allsolid)
624  {
625  return;
626  }
627  touch = SV_GentityNum(touchlist[i]);
628 
629  // see if we should ignore this entity
630  if (clip->passEntityNum != ENTITYNUM_NONE)
631  {
632  if (touchlist[i] == clip->passEntityNum)
633  {
634  continue; // don't clip against the pass entity
635  }
636  if (touch->r.ownerNum == clip->passEntityNum)
637  {
638  continue; // don't clip against own missiles
639  }
640  if (touch->r.ownerNum == passOwnerNum)
641  {
642  continue; // don't clip against other missiles from our owner
643  }
644  }
645 
646  // if it doesn't have any brushes of a type we
647  // are looking for, ignore it
648  if (!(clip->contentmask & touch->r.contents))
649  {
650  continue;
651  }
652 
653  // might intersect, so do an exact clip
654  clipHandle = SV_ClipHandleForEntity(touch);
655 
656  // non-worldspawn entities must not use world as clip model!
657  if (clipHandle == 0)
658  {
659  continue;
660  }
661 
662  // If clipping against BBOX, set to correct contents
663  if (clipHandle == BOX_MODEL_HANDLE)
664  {
665  CM_SetTempBoxModelContents(touch->r.contents);
666  }
667 
668  origin = touch->r.currentOrigin;
669  angles = touch->r.currentAngles;
670 
671 
672  if (!touch->r.bmodel)
673  {
674  angles = vec3_origin; // boxes don't rotate
675  }
676 
677  CM_TransformedBoxTrace(&trace, clip->start, clip->end,
678  clip->mins, clip->maxs, clipHandle, clip->contentmask,
679  origin, angles, clip->capsule);
680 
681  if (trace.allsolid)
682  {
683  clip->trace.allsolid = qtrue;
684  trace.entityNum = touch->s.number;
685 
686  // FIXME: entity damage, see CG_ClipMoveToEntities
687  //clip->trace.entityNum = touch->s.number;
688  }
689  else if (trace.startsolid)

693 
694  // FIXME: entity damage, see CG_ClipMoveToEntities

696  }

702 
703  trace.entityNum = touch->s.number;

729 void SV_Trace(trace_t *results, const vec3_t start, const vec3_t mins, const vec3_t maxs, const vec3_t end, int passEntityNum, int contentmask, qboolean capsule)
730 {
731  moveclip_t clip;
732  int i;
733 
734  if (!mins)
735  {
736  mins = vec3_origin;
737  }
738  if (!maxs)
739  {
740  maxs = vec3_origin;
741  }
742 
743  Com_Memset(&clip, 0, sizeof(moveclip_t));
744 
745  // clip to world
746  CM_BoxTrace(&clip.trace, start, end, mins, maxs, 0, contentmask, capsule);
747  clip.trace.entityNum = clip.trace.fraction != 1.0f ? ENTITYNUM_WORLD : ENTITYNUM_NONE;
748  if (clip.trace.fraction == 0.f || passEntityNum == -2)
749  {
750  *results = clip.trace;
751  return; // blocked immediately by the world
752  }
753 
754  clip.contentmask = contentmask;
755  clip.start = start;
756  //VectorCopy(clip.trace.endpos, clip.end);
757  VectorCopy(end, clip.end);
758  clip.mins = mins;
759  clip.maxs = maxs;
760  clip.passEntityNum = passEntityNum;
761  clip.capsule = capsule;
762 
763  // create the bounding box of the entire move
764  // we can limit it to the part of the move not
765  // already clipped off by the world, which can be
766  // a significant savings for line of sight and shot traces
767  for (i = 0 ; i < 3 ; i++)
768  {
769  if (end[i] > start[i])
770  {
771  clip.boxmins[i] = clip.start[i] + clip.mins[i] - 1;
772  clip.boxmaxs[i] = clip.end[i] + clip.maxs[i] + 1;
773  }
774  else
775  {
776  clip.boxmins[i] = clip.end[i] + clip.mins[i] - 1;
777  clip.boxmaxs[i] = clip.start[i] + clip.maxs[i] + 1;
778  }
779  }
780 
781  // clip to other solid entities
782  SV_ClipMoveToEntities(&clip);
783 
784  *results = clip.trace;
785 }
