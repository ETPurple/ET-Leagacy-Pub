163 static void R_AddPolysToScene(qhandle_t hShader, int numVerts, const polyVert_t *verts, int numPolys)
164 {
165  srfPoly_t *poly;
166  int i, j;
167  int fogIndex;
168  fog_t *fog;
169  vec3_t bounds[2];
170 
171  if (!tr.registered)
172  {
173  return;
174  }
175 
176  if (!r_drawpolies->integer)
177  {
178  return;
179  }
180 
181  if (!hShader)
182  {
183  Ren_Developer("WARNING: RE_AddPolyToScene: NULL poly shader\n");
184  return;
185  }
186 
187  for (j = 0; j < numPolys; j++)
188  {
189  if (r_numPolyVerts + numVerts >= r_maxPolyVerts->integer || r_numPolys >= r_maxPolys->integer)
190  {
191  /*
192  NOTE TTimo this was initially a PRINT_WARNING
193  but it happens a lot with high fighting scenes and particles
194  since we don't plan on changing the const and making for room for those effects
195  simply cut this message to developer only
196  */
197  Ren_Developer("WARNING: RE_AddPolyToScene: r_maxPolyVerts or r_maxPolys reached\n");
198  return;
199  }
200 
201  poly = &backEndData->polys[r_numPolys];
202  poly->surfaceType = SF_POLY;
203  poly->hShader = hShader;
204  poly->numVerts = numVerts;
205  poly->verts = &backEndData->polyVerts[r_numPolyVerts];
206 
207  Com_Memcpy(poly->verts, &verts[numVerts * j], numVerts * sizeof(*verts));
208 
209  // done.
210  r_numPolys++;
211  r_numPolyVerts += numVerts;
212 
213  // if no world is loaded
214  if (tr.world == NULL)
215  {
216  fogIndex = 0;
217  }
218  // see if it is in a fog volume
219  else if (tr.world->numFogs == 1)
220  {
221  fogIndex = 0;
222  }
223  else
224  {
225  // find which fog volume the poly is in
226  VectorCopy(poly->verts[0].xyz, bounds[0]);
227  VectorCopy(poly->verts[0].xyz, bounds[1]);
228 
229  for (i = 1; i < poly->numVerts; i++)
230  {
231  AddPointToBounds(poly->verts[i].xyz, bounds[0], bounds[1]);
232  }
233 
234  for (fogIndex = 1; fogIndex < tr.world->numFogs; fogIndex++)
235  {
236  fog = &tr.world->fogs[fogIndex];
237 
238 
239  if (BoundsIntersect(bounds[0], bounds[1], fog->bounds[0], fog->bounds[1]))
240  {
241  break;
242  }
243  }
244 
245  if (fogIndex == tr.world->numFogs)
246  {
247  fogIndex = 0;
248  }
249  }
250  poly->fogIndex = fogIndex;
251  }
252 }

602 void RE_RenderScene(const refdef_t *fd)
603 {
604  viewParms_t parms;
605  int startTime;
606 
607  if (!tr.registered)
608  {
609  return;
610  }
611  Ren_LogComment("====== RE_RenderScene =====\n");
612 
613  if (r_noreFresh->integer)
614  {
615  return;
616  }
617 
618  startTime = ri.Milliseconds();
619 
620  if (!tr.world && !(fd->rdflags & RDF_NOWORLDMODEL))
621  {
622  Ren_Drop("R_RenderScene: NULL worldmodel");
623  }
624 
625  Com_Memcpy(tr.refdef.text, fd->text, sizeof(tr.refdef.text));
626 
627  tr.refdef.x = fd->x;
628  tr.refdef.y = fd->y;
629  tr.refdef.width = fd->width;
630  tr.refdef.height = fd->height;
631  tr.refdef.fov_x = fd->fov_x;
632  tr.refdef.fov_y = fd->fov_y;
633 
634  VectorCopy(fd->vieworg, tr.refdef.vieworg);
635  VectorCopy(fd->viewaxis[0], tr.refdef.viewaxis[0]);
636  VectorCopy(fd->viewaxis[1], tr.refdef.viewaxis[1]);
637  VectorCopy(fd->viewaxis[2], tr.refdef.viewaxis[2]);
638 
639  tr.refdef.time = fd->time;
640  tr.refdef.rdflags = fd->rdflags;
641 
642  if(fd->rdflags & RDF_SKYBOXPORTAL)
643  {
644  tr.world->hasSkyboxPortal = qtrue; // see int skyboxportal var tr_scene.c/r1
645  }
646 
647  // copy the areamask data over and note if it has changed, which
648  // will force a reset of the visible leafs even if the view hasn't moved
649  tr.refdef.areamaskModified = qfalse;
650  if (!(tr.refdef.rdflags & RDF_NOWORLDMODEL) && !((tr.refdef.rdflags & RDF_SKYBOXPORTAL) && tr.world && tr.world->numSkyNodes > 0))
651  {
652  int areaDiff;
653  int i;
654 
655  // compare the area bits
656  areaDiff = 0;
657  for (i = 0; i < MAX_MAP_AREA_BYTES / 4; i++)
658  {
659  areaDiff |= ((int *)tr.refdef.areamask)[i] ^ ((int *)fd->areamask)[i];
660  ((int *)tr.refdef.areamask)[i] = ((int *)fd->areamask)[i];
661  }
662 
663  if (areaDiff)
664  {
665  // a door just opened or something
666  tr.refdef.areamaskModified = qtrue;
667  }
668  }
669 
670  R_AddWorldLightsToScene();
671 
672  // derived info
673  tr.refdef.floatTime = tr.refdef.time * 0.001;
674 
675  tr.refdef.numDrawSurfs = r_firstSceneDrawSurf;
676  tr.refdef.drawSurfs = backEndData->drawSurfs;
677 
678  tr.refdef.numInteractions = r_firstSceneInteraction;
679  tr.refdef.interactions = backEndData->interactions;
680 
681  tr.refdef.numEntities = r_numEntities - r_firstSceneEntity;
682  tr.refdef.entities = &backEndData->entities[r_firstSceneEntity];
683 
684  tr.refdef.numLights = r_numLights - r_firstSceneLight;
685  tr.refdef.lights = &backEndData->lights[r_firstSceneLight];
686 
687  tr.refdef.num_coronas = r_numcoronas - r_firstSceneCorona;
688  tr.refdef.coronas = &backEndData->coronas[r_firstSceneCorona];
689 
690  tr.refdef.numPolys = r_numPolys - r_firstScenePoly;
691  tr.refdef.polys = &backEndData->polys[r_firstScenePoly];
692 
693  tr.refdef.numPolybuffers = r_numPolybuffers - r_firstScenePolybuffer;
694  tr.refdef.polybuffers = &backEndData->polybuffers[r_firstScenePolybuffer];
695 
696  tr.refdef.numDecalProjectors = r_numDecalProjectors - r_firstSceneDecalProjector;
697  tr.refdef.decalProjectors = &backEndData->decalProjectors[r_firstSceneDecalProjector];
698 
699  tr.refdef.numDecals = 0;
700  tr.refdef.decals = &backEndData->decals[r_firstSceneDecal];
701 
702 
703  // a single frame may have multiple scenes draw inside it --
704  // a 3D game view, 3D status bar renderings, 3D menus, etc.
705  // They need to be distinguished by the light flare code, because
706  // the visibility state for a given surface may be different in
707  // each scene / view.
708  tr.frameSceneNum++;
709  tr.sceneCount++;
710 
711  // a scene can have multiple views caused by mirrors or portals
712  // the number of views is restricted so we can use hardware occlusion queries
713  // and put them into the BSP nodes for each view
714  tr.viewCount = -1;
715 
716  // setup view parms for the initial view
717  //
718  // set up viewport
719  // The refdef takes 0-at-the-top y coordinates, so
720  // convert to GL's 0-at-the-bottom space
721  //
722  Com_Memset(&parms, 0, sizeof(parms));
723 
724 #if 1
725  if (tr.refdef.pixelTarget == NULL)
726  {
727  parms.viewportX = tr.refdef.x;
728  parms.viewportY = glConfig.vidHeight - (tr.refdef.y + tr.refdef.height);
729  }
730  else
731  {
732  // Driver bug, if we try and do pixel target work along the top edge of a window
733  // we can end up capturing part of the status bar. (see screenshot corruption..)
734  // Soooo.. use the middle.
735  parms.viewportX = glConfig.vidWidth / 2;
736  parms.viewportY = glConfig.vidHeight / 2;
737  }
738 #else
739  parms.viewportX = tr.refdef.x;
740  parms.viewportY = glConfig.vidHeight - (tr.refdef.y + tr.refdef.height);
741 #endif
742 
743  parms.viewportWidth = tr.refdef.width;
744  parms.viewportHeight = tr.refdef.height;
745 
746  Vector4Set(parms.viewportVerts[0], parms.viewportX, parms.viewportY, 0, 1);
747  Vector4Set(parms.viewportVerts[1], parms.viewportX + parms.viewportWidth, parms.viewportY, 0, 1);
748  Vector4Set(parms.viewportVerts[2], parms.viewportX + parms.viewportWidth, parms.viewportY + parms.viewportHeight, 0, 1);
749  Vector4Set(parms.viewportVerts[3], parms.viewportX, parms.viewportY + parms.viewportHeight, 0, 1);
750 
751  parms.isPortal = qfalse;
752 
753  parms.fovX = tr.refdef.fov_x;
754  parms.fovY = tr.refdef.fov_y;
755 
756  VectorCopy(fd->vieworg, parms.orientation.origin);
757  VectorCopy(fd->viewaxis[0], parms.orientation.axis[0]);
758  VectorCopy(fd->viewaxis[1], parms.orientation.axis[1]);
759  VectorCopy(fd->viewaxis[2], parms.orientation.axis[2]);
760 
761  VectorCopy(fd->vieworg, parms.pvsOrigin);
762 
763  R_RenderView(&parms);
764 
765  // the next scene rendered in this frame will tack on after this one
766  r_firstSceneDrawSurf = tr.refdef.numDrawSurfs;
767  r_firstSceneInteraction = tr.refdef.numInteractions;
768  r_firstSceneDecal += tr.refdef.numDecals;
769  r_firstSceneEntity = r_numEntities;
770  r_firstSceneLight = r_numLights;
771  r_firstScenePoly = r_numPolys;
772  r_firstScenePolybuffer = r_numPolybuffers;
773
