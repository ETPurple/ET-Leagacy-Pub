526 void Use_Shooter(gentity_t *ent, gentity_t *other, gentity_t *activator)
527 {
528  vec3_t dir;
529  float deg;
530  vec3_t up, right;
531 
532  // see if we have a target
533  if (ent->enemy)
534  {
535  VectorSubtract(ent->enemy->r.currentOrigin, ent->s.origin, dir);
536  VectorNormalize(dir);
537  }
538  else
539  {
540  VectorCopy(ent->movedir, dir);
541  }
542 
543  if (ent->s.weapon == WP_MAPMORTAR)
544  {
545  AimAtTarget(ent); // store in ent->s.origin2 the direction/force needed to pass through the target
546  VectorCopy(ent->s.origin2, dir);
547  }
548 
549  // randomize a bit
550  PerpendicularVector(up, dir);
551  CrossProduct(up, dir, right);
552 
553  deg = crandom() * ent->random;
554  VectorMA(dir, deg, up, dir);
555 
556  deg = crandom() * ent->random;
557  VectorMA(dir, deg, right, dir);
558 
559  VectorNormalize(dir);
560 
561  if (GetWeaponTableData(ent->s.weapon)->type & WEAPON_TYPE_GRENADE)
562  {
563  VectorScale(dir, 700, dir); // had to add this as fire_grenade now expects a non-normalized direction vector
564  // FIXME: why we do normalize the vector before this switch? See comment of fire_grenade
565  fire_missile(ent, ent->s.origin, dir, ent->s.weapon);
566  }
567  else if (GetWeaponTableData(ent->s.weapon)->type & WEAPON_TYPE_PANZER)
568  {
569  VectorNormalize(dir);
570  VectorScale(dir, 5000, dir);
571  fire_missile(ent, ent->s.origin, dir, ent->s.weapon);
572  }
573  else if (ent->s.weapon == WP_MAPMORTAR)
574  {
575  VectorScale(dir, VectorLength(ent->s.origin2), dir);
576  fire_missile(ent, ent->s.origin, dir, ent->s.weapon);
577 
578  if (ent->spawnflags)
579  {
580  gentity_t *tent;
581 
582  tent = G_TempEntity(ent->s.pos.trBase, EV_MORTAREFX);
583  tent->s.density = ent->spawnflags; // send smoke and muzzle flash flags
584  VectorCopy(ent->s.pos.trBase, tent->s.origin);
585  VectorCopy(ent->s.apos.trBase, tent->s.angles);
586  }
587  }
588 
589  G_AddEvent(ent, EV_FIRE_WEAPON, 0);
590 }
