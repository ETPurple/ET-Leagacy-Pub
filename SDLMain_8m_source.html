168  * Catch document open requests...this lets us notice files when the app
169  * was launched by double-clicking a document, or when a document was
170  * dragged/dropped on the app's icon. You need to have a
171  * CFBundleDocumentTypes section in your Info.plist to get this message,
172  * apparently.
173  *
174  * Files are added to gArgv, so to the app, they'll look like command line
175  * arguments. Previously, apps launched from the finder had nothing but
176  * an argv[0].

189  }
190 
191  [self addArgc:filename];
192 
193  return _gArgc != gArgc ? TRUE : FALSE;
194 }
