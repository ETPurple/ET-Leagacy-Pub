169 static qboolean SV_CheckTransitionGameState(gamestate_t new_gs, gamestate_t old_gs)
170 {
171  if (old_gs == new_gs && new_gs != GS_PLAYING)
172  {
173  return qfalse;
174  }
175 
176  if (old_gs == GS_WAITING_FOR_PLAYERS && new_gs != GS_WARMUP)
177  {
178  return qfalse;
179  }
180 
181  if (old_gs == GS_INTERMISSION && new_gs != GS_WARMUP)
182  {
183  return qfalse;
184  }
185 
186  if (old_gs == GS_RESET && (new_gs != GS_WAITING_FOR_PLAYERS && new_gs != GS_WARMUP))
187  {
188  return qfalse;
189  }
190 
191  return qtrue;
192 }

201 static qboolean SV_TransitionGameState(gamestate_t new_gs, gamestate_t old_gs, int delay)
202 {
203  // we always do a warmup before starting match
204  if (old_gs == GS_INTERMISSION && new_gs == GS_PLAYING)
205  {
206  new_gs = GS_WARMUP;
207  }
208 
209  // check if its a valid state transition
210  if (!SV_CheckTransitionGameState(new_gs, old_gs))
211  {
212  return qfalse;
213  }
214 
215  if (new_gs == GS_RESET)
216  {
217  new_gs = GS_WARMUP;
218  }
219 
220  Cvar_Set("gamestate", va("%i", new_gs));
221 
222  return qtrue;
223 }

241 static void SV_MapRestart_f(void)
242 {
243  int i;
244  client_t *client;
245  char *denied;
246  qboolean isBot;
247  int delay = 0;
248  gamestate_t new_gs, old_gs;
249 
250  // make sure we aren't restarting twice in the same frame
251  if (com_frameTime == sv.serverId)
252  {
253  return;
254  }
255 
256  // make sure server is running
257  if (!com_sv_running->integer)
258  {
259  Com_Printf("Server is not running.\n");
260  return;
261  }
262 
263  if (Cmd_Argc() > 1)
264  {
265  delay = atoi(Cmd_Argv(1));
266  }
267 
268  if (delay)
269  {
270  sv.restartTime = svs.time + delay * 1000;
271  SV_SetConfigstring(CS_WARMUP, va("%i", sv.restartTime));
272  return;
273  }
274 
275  // read in gamestate or just default to GS_PLAYING
276  old_gs = atoi(Cvar_VariableString("gamestate"));
277 
278  if (Cmd_Argc() > 2)
279  {
280  new_gs = atoi(Cmd_Argv(2));
281  }
282  else
283  {
284  new_gs = GS_PLAYING;
285  }
286 
287  if (!SV_TransitionGameState(new_gs, old_gs, delay))
288  {
289  return;
290  }
291 
292  // check for changes in variables that can't just be restarted
293  // check for maxclients change
294  if (sv_maxclients->modified)
295  {
296  char mapname[MAX_QPATH];
297 
298  Com_Printf("sv_maxclients variable change -- restarting.\n");
299  // restart the map the slow way
300  Q_strncpyz(mapname, Cvar_VariableString("mapname"), sizeof(mapname));
301 
302  SV_SpawnServer(mapname);
303  return;
304  }
305 
306  // toggle the server bit so clients can detect that a
307  // map_restart has happened
308  svs.snapFlagServerBit ^= SNAPFLAG_SERVERCOUNT;
309 
310  // generate a new serverid
311  // don't update restartedserverId there, otherwise we won't deal correctly with multiple map_restart
312  sv.serverId = com_frameTime;
313  Cvar_Set("sv_serverid", va("%i", sv.serverId));
314 
315  // reset all the vm data in place without changing memory allocation
316  // note that we do NOT set sv.state = SS_LOADING, so configstrings that
317  // had been changed from their default values will generate broadcast updates
318  sv.state = SS_LOADING;
319  sv.restarting = qtrue;
320 
321  SV_RestartGameProgs();
322 
323  // run a few frames to allow everything to settle
324  for (i = 0; i < GAME_INIT_FRAMES; i++)
325  {
326  VM_Call(gvm, GAME_RUN_FRAME, svs.time);
327  svs.time += FRAMETIME;
328  }
329 
330  sv.state = SS_GAME;
331  sv.restarting = qfalse;
332 
333  // connect and begin all the clients
334  for (i = 0 ; i < sv_maxclients->integer ; i++)
335  {
336  client = &svs.clients[i];
337 
338  // send the new gamestate to all connected clients
339  if (client->state < CS_CONNECTED)
340  {
341  continue;
342  }
343 
344  if (client->netchan.remoteAddress.type == NA_BOT)
345  {
346  isBot = qtrue;
347  }
348  else
349  {
350  isBot = qfalse;
351  }
352 
353  // add the map_restart command
354  SV_AddServerCommand(client, "map_restart\n");
355 
356  // connect the client again, without the firstTime flag
357  denied = VM_ExplicitArgPtr(gvm, VM_Call(gvm, GAME_CLIENT_CONNECT, i, qfalse, isBot));
358  if (denied)
359  {
360  // this generally shouldn't happen, because the client
361  // was connected before the level change
362  SV_DropClient(client, denied);
363  if (!isBot)
364  {
365  Com_Printf("SV_MapRestart_f(%d): dropped client %i - denied!\n", delay, i); // bk010125
366  }
367 
368  continue;
369  }
370 
371  // Player won't enter the world until the download is done
372  if (client->download == 0 && client->bWWWing == qfalse)
373  {
374  if (client->state == CS_ACTIVE)
375  {
376  SV_ClientEnterWorld(client, &client->lastUsercmd);
377  }
378  else
379  {
380  // If we don't reset client->lastUsercmd and are restarting during map load,
381  // the client will hang because we'll use the last Usercmd from the previous map,
382  // which is wrong obviously.
383  SV_ClientEnterWorld(client, NULL);
384  }
385  }
386  }
387 
388  // run another frame to allow things to look at all the players
389  VM_Call(gvm, GAME_RUN_FRAME, svs.time);
390  svs.time += FRAMETIME;
391 }

400 void SV_TempBanNetAddress(netadr_t address, int length)
401 {
402  int i;
403  int oldesttime = 0;
404  int oldest = -1;
405 
406  for (i = 0; i < MAX_TEMPBAN_ADDRESSES; i++)
407  {
408  if (!svs.tempBanAddresses[i].endtime || svs.tempBanAddresses[i].endtime < svs.time)
409  {
410  // found a free slot
411  svs.tempBanAddresses[i].adr = address;
412  svs.tempBanAddresses[i].endtime = svs.time + (length * 1000);
413 
414  return;
415  }
416  else
417  {
418  if (oldest == -1 || oldesttime > svs.tempBanAddresses[i].endtime)
419  {
420  oldesttime = svs.tempBanAddresses[i].endtime;
421  oldest = i;
422  }
423  }
424  }
425 
426  svs.tempBanAddresses[oldest].adr = address;
427  svs.tempBanAddresses[oldest].endtime = svs.time + length;
428 }
