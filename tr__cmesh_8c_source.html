46 static float ProjectRadius(float r, vec3_t location)
47 {
48  float pr;
49  float dist;
50  float c;
51  vec3_t p;
52  float projected[4];
53 
54  c = DotProduct(tr.viewParms.orientation.axis[0], tr.viewParms.orientation.origin);
55  dist = DotProduct(tr.viewParms.orientation.axis[0], location) - c;
56 
57  if (dist <= 0)
58  {
59  return 0;
60  }
61 
62  p[0] = 0;
63  p[1] = Q_fabs(r);
64  p[2] = -dist;
65 
66  projected[0] = p[0] * tr.viewParms.projectionMatrix[0] +
67  p[1] * tr.viewParms.projectionMatrix[4] +
68  p[2] * tr.viewParms.projectionMatrix[8] +
69  tr.viewParms.projectionMatrix[12];
70 
71  projected[1] = p[0] * tr.viewParms.projectionMatrix[1] +
72  p[1] * tr.viewParms.projectionMatrix[5] +
73  p[2] * tr.viewParms.projectionMatrix[9] +
74  tr.viewParms.projectionMatrix[13];
75 
76  projected[2] = p[0] * tr.viewParms.projectionMatrix[2] +
77  p[1] * tr.viewParms.projectionMatrix[6] +
78  p[2] * tr.viewParms.projectionMatrix[10] +
79  tr.viewParms.projectionMatrix[14];
80 
81  projected[3] = p[0] * tr.viewParms.projectionMatrix[3] +
82  p[1] * tr.viewParms.projectionMatrix[7] +
83  p[2] * tr.viewParms.projectionMatrix[11] +
84  tr.viewParms.projectionMatrix[15];
85 
86 
87  pr = projected[1] / projected[3];
88 
89  if (pr > 1.0f)
90  {
91  pr = 1.0f;
92  }
93 
94  return pr;
95 }

103 static int R_CullModel(mdcHeader_t *header, trRefEntity_t *ent)
104 {
105  vec3_t bounds[2];
106  int i;
107  // compute frame pointers
108  md3Frame_t *newFrame = ( md3Frame_t * )(( byte * ) header + header->ofsFrames) + ent->e.frame;
109  md3Frame_t *oldFrame = ( md3Frame_t * )(( byte * ) header + header->ofsFrames) + ent->e.oldframe;
110 
111  // cull bounding sphere ONLY if this is not an upscaled entity
112  if (!ent->e.nonNormalizedAxes)
113  {
114  if (ent->e.frame == ent->e.oldframe)
115  {
116  switch (R_CullLocalPointAndRadius(newFrame->localOrigin, newFrame->radius))
117  {
118  case CULL_OUT:
119  tr.pc.c_sphere_cull_md3_out++;
120  return CULL_OUT;
121 
122  case CULL_IN:
123  tr.pc.c_sphere_cull_md3_in++;
124  return CULL_IN;
125 
126  case CULL_CLIP:
127  tr.pc.c_sphere_cull_md3_clip++;
128  break;
129  }
130  }
131  else
132  {
133  int sphereCull, sphereCullB;
134 
135  sphereCull = R_CullLocalPointAndRadius(newFrame->localOrigin, newFrame->radius);
136  if (newFrame == oldFrame)
137  {
138  sphereCullB = sphereCull;
139  }
140  else
141  {
142  sphereCullB = R_CullLocalPointAndRadius(oldFrame->localOrigin, oldFrame->radius);
143  }
144 
145  if (sphereCull == sphereCullB)
146  {
147  if (sphereCull == CULL_OUT)
148  {
149  tr.pc.c_sphere_cull_md3_out++;
150  return CULL_OUT;
151  }
152  else if (sphereCull == CULL_IN)
153  {
154  tr.pc.c_sphere_cull_md3_in++;
155  return CULL_IN;
156  }
157  else
158  {
159  tr.pc.c_sphere_cull_md3_clip++;
160  }
161  }
162  }
163  }
164 
165  // calculate a bounding box in the current coordinate system
166  for (i = 0 ; i < 3 ; i++)
167  {
168  bounds[0][i] = oldFrame->bounds[0][i] < newFrame->bounds[0][i] ? oldFrame->bounds[0][i] : newFrame->bounds[0][i];
169  bounds[1][i] = oldFrame->bounds[1][i] > newFrame->bounds[1][i] ? oldFrame->bounds[1][i] : newFrame->bounds[1][i];
170  }
171 
172  switch (R_CullLocalBox(bounds))
173  {
174  case CULL_IN:
175  tr.pc.c_box_cull_md3_in++;
176  return CULL_IN;
177  case CULL_CLIP:
178  tr.pc.c_box_cull_md3_clip++;
179  return CULL_CLIP;
180  case CULL_OUT:
181  default:
182  tr.pc.c_box_cull_md3_out++;
183  return CULL_OUT;
184  }
185 }

192 static int R_ComputeLOD(trRefEntity_t *ent)
193 {
194  int lod;
195 
196  if (tr.currentModel->numLods < 2)
197  {
198  // model has only 1 LOD level, skip computations and bias
199  lod = 0;
200  }
201  else
202  {
203  float projectedRadius;
204  float radius;
205  float flod;
206  md3Frame_t *frame;
207 
208  // multiple LODs exist, so compute projected bounding sphere
209  // and use that as a criteria for selecting LOD
210 
211  // checked for a forced lowest LOD
212  if (ent->e.reFlags & REFLAG_FORCE_LOD)
213  {
214  return (tr.currentModel->numLods - 1);
215  }
216 
217  frame = ( md3Frame_t * )((( unsigned char * ) tr.currentModel->model.mdc[0]) + tr.currentModel->model.mdc[0]->ofsFrames);
218 
219  frame += ent->e.frame;
220 
221  radius = RadiusFromBounds(frame->bounds[0], frame->bounds[1]);
222 
223  // testing
224  //if (ent->e.reFlags & REFLAG_ORIENT_LOD)
225  //{
226  // right now this is for trees, and pushes the lod distance way in.
227  // this is not the intended purpose, but is helpful for the new
228  // terrain level that has loads of trees
229  // radius = radius/2.0f;
230  //}
231 
232  if ((projectedRadius = ProjectRadius(radius, ent->e.origin)) != 0.f)
233  {
234  float lodscale = r_lodScale->value;
235 
236  if (lodscale > 20)
237  {
238  lodscale = 20;
239  }
240  flod = 1.0f - projectedRadius * lodscale;
241  }
242  else
243  {
244  // object intersects near view plane, e.g. view weapon
245  flod = 0;
246  }
247 
248  flod *= tr.currentModel->numLods;
249  lod = round(flod);
250 
251  if (lod < 0)
252  {
253  lod = 0;
254  }
255  else if (lod >= tr.currentModel->numLods)
256  {
257  lod = tr.currentModel->numLods - 1;
258  }
259  }
260 
261  lod += r_lodBias->integer;
262 
263  if (lod >= tr.currentModel->numLods)
264  {
265  lod = tr.currentModel->numLods - 1;
266  }
267  if (lod < 0)
268  {
269  lod = 0;
270  }
271 
272  return lod;
273 }

282 {

295  VectorAdd(ent->e.origin, md3Frame->localOrigin, localOrigin);

297  {

309  }

312  return i;
313  }
314  }
315 
316  return 0;

328  md3Shader_t *md3Shader = 0;

334

336  {
337  ent->e.frame %= tr.currentModel->model.mdc[0]->numFrames;
338  ent->e.oldframe %= tr.currentModel->model.mdc[0]->numFrames;
339  }
340 
341  // Validate the frames so there is no chance of a crash.
342  // This will write directly into the entity structure, so
343  // when the surfaces are rendered, they don't need to be
344  // range checked again.
345  if ((ent->e.frame >= tr.currentModel->model.mdc[0]->numFrames)
346  || (ent->e.frame < 0)
347  || (ent->e.oldframe >= tr.currentModel->model.mdc[0]->numFrames)
348  || (ent->e.oldframe < 0))
349  {
350  Ren_Developer("R_AddMDCSurfaces: no such frame %d to %d for '%s'\n",
351  ent->e.oldframe, ent->e.frame,
352  tr.currentModel->name);
353  ent->e.frame = 0;
354  ent->e.oldframe = 0;
355  }
356 
357  // compute LOD
358  if (ent->e.renderfx & RF_FORCENOLOD)
359  {
360  lod = 0;
361  }
362  else
363  {
364  lod = R_ComputeLOD(ent);
365  }
366 
367  header = tr.currentModel->model.mdc[lod];
368 
369  // cull the entire model if merged bounding box of both frames
370  // is outside the view frustum.
371  cull = R_CullModel(header, ent);
372  if (cull == CULL_OUT)
373  {
374  return;
375  }
376 
377  // set up lighting now that we know we aren't culled
378  if (!personalModel || r_shadows->integer > 1)
379  {
380  R_SetupEntityLighting(&tr.refdef, ent);
381  }
382 
383  // see if we are in a fog volume
384  fogNum = R_ComputeFogNum(header, ent);
385 
386  // draw all surfaces
387  surface = ( mdcSurface_t * )((byte *)header + header->ofsSurfaces);
388  for (i = 0 ; i < header->numSurfaces ; i++)
389  {
390  if (ent->e.customShader)
391  {
392  shader = R_GetShaderByHandle(ent->e.customShader);
393  }
394  else if (ent->e.customSkin > 0 && ent->e.customSkin < tr.numSkins)
395  {
396  skin_t *skin;
397  int hash;
398  int j;
399 
400  skin = R_GetSkinByHandle(ent->e.customSkin);
401 
402  // match the surface name to something in the skin file
403  shader = tr.defaultShader;
404  // added blink
405  if (ent->e.renderfx & RF_BLINK)
406  {
407  char *s = va("%s_b", surface->name); // append '_b' for 'blink'
408 
409  hash = Com_HashKey(s, strlen(s));
410  for (j = 0 ; j < skin->numSurfaces ; j++)
411  {
412  if (hash != skin->surfaces[j].hash)
413  {
414  continue;
415  }
416  if (!strcmp(skin->surfaces[j].name, s))
417  {
418  shader = skin->surfaces[j].shader;
419  break;
420  }
421  }
422  }
423 
424  if (shader == tr.defaultShader) // blink reference in skin was not found
425  {
426  hash = Com_HashKey(surface->name, sizeof(surface->name));
427  for (j = 0 ; j < skin->numSurfaces ; j++)
428  {
429  // the names have both been lowercased
430  if (hash != skin->surfaces[j].hash)
431  {
432  continue;
433  }
434  if (!strcmp(skin->surfaces[j].name, surface->name))
435  {
436  shader = skin->surfaces[j].shader;
437  break;
438  }
439  }
440  }
441  }
442  else if (surface->numShaders <= 0)
443  {
444  shader = tr.defaultShader;
445  }
446  else
447  {
448  md3Shader = ( md3Shader_t * )((byte *)surface + surface->ofsShaders);
449  md3Shader += ent->e.skinNum % surface->numShaders;
450  shader = tr.shaders[md3Shader->shaderIndex];
451  }
452 
453  // we will add shadows even if the main object isn't visible in the view
454 
455  // stencil shadows can't do personal models unless I polyhedron clip
456  if (!personalModel
457  && r_shadows->integer == 2
458  && fogNum == 0
459  && !(ent->e.renderfx & (RF_NOSHADOW | RF_DEPTHHACK))
460  && shader->sort == SS_OPAQUE)
461  {
462  R_AddDrawSurf((void *)surface, tr.shadowShader, 0, 0, 0);
463  }
464 
465  // projection shadows work fine with personal models
466  if (r_shadows->integer == 3
467  && fogNum == 0
468  && (ent->e.renderfx & RF_SHADOW_PLANE)
469  && shader->sort == SS_OPAQUE)
470  {
471  R_AddDrawSurf((void *)surface, tr.projectionShadowShader, 0, 0, 0);
472  }
473 
474  // for testing polygon shadows (on /all/ models)
475  if (r_shadows->integer == 4)
476  {
477  R_AddDrawSurf((void *)surface, tr.projectionShadowShader, 0, 0, 0);
478  }
479 
480  // don't add third_person objects if not viewing through a portal
481  if (!personalModel)
482  {
483  R_AddDrawSurf((void *)surface, shader, fogNum, 0, 0);
484  }
485 
486  surface = ( mdcSurface_t * )((byte *)surface + surface->ofsEnd);
487  }
488 }
