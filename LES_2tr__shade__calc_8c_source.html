506 static void Autosprite2Deform(void)
507 {
508  int i, j, k;
509  int indexes;
510  int nums[2];
511  float *xyz;
512  float lengths[2];
513  float l;
514  vec3_t forward;
515  vec3_t temp;
516  vec3_t mid[2];
517  vec3_t major, minor;
518  float *v1, *v2;
519 
520  if (tess.numVertexes & 3)
521  {
522  Ren_Developer("Autosprite2 shader %s had odd vertex count\n", tess.shader->name);
523  }
524  if (tess.numIndexes != (tess.numVertexes >> 2) * 6)
525  {
526  Ren_Developer("Autosprite2 shader %s had odd index count\n", tess.shader->name);
527  }
528 
529  if (backEnd.currentEntity != &tr.worldEntity)
530  {
531  GlobalVectorToLocal(backEnd.viewParms.orientation.axis[0], forward);
532  }
533  else
534  {
535  VectorCopy(backEnd.viewParms.orientation.axis[0], forward);
536  }
537 
538  // this is a lot of work for two triangles...
539  // we could precalculate a lot of it is an issue, but it would mess up
540  // the shader abstraction
541  for (i = 0, indexes = 0 ; i < tess.numVertexes ; i += 4, indexes += 6)
542  {
543  // find the midpoint
544  xyz = tess.xyz[i];
545 
546  // identify the two shortest edges
547  nums[0] = nums[1] = 0;
548  lengths[0] = lengths[1] = 999999;
549 
550  for (j = 0 ; j < 6 ; j++)
551  {
552  v1 = xyz + 4 * edgeVerts[j][0];
553  v2 = xyz + 4 * edgeVerts[j][1];
554 
555  VectorSubtract(v1, v2, temp);
556 
557  l = DotProduct(temp, temp);
558  if (l < lengths[0])
559  {
560  nums[1] = nums[0];
561  lengths[1] = lengths[0];
562  nums[0] = j;
563  lengths[0] = l;
564  }
565  else if (l < lengths[1])
566  {
567  nums[1] = j;
568  lengths[1] = l;
569  }
570  }
571 
572  for (j = 0 ; j < 2 ; j++)
573  {
574  v1 = xyz + 4 * edgeVerts[nums[j]][0];
575  v2 = xyz + 4 * edgeVerts[nums[j]][1];
576 
577  mid[j][0] = 0.5f * (v1[0] + v2[0]);
578  mid[j][1] = 0.5f * (v1[1] + v2[1]);
579  mid[j][2] = 0.5f * (v1[2] + v2[2]);
580  }
581 
582  // find the vector of the major axis
583  VectorSubtract(mid[1], mid[0], major);
584 
585  // cross this with the view direction to get minor axis
586  CrossProduct(major, forward, minor);
587  VectorNormalize(minor);
588 
589  // re-project the points
590  for (j = 0 ; j < 2 ; j++)
591  {
592  v1 = xyz + 4 * edgeVerts[nums[j]][0];
593  v2 = xyz + 4 * edgeVerts[nums[j]][1];
594 
595  l = 0.5 * sqrt(lengths[j]);
596 
597  // we need to see which direction this edge
598  // is used to determine direction of projection
599  for (k = 0 ; k < 5 ; k++)
600  {
601  if (tess.indexes[indexes + k] == i + edgeVerts[nums[j]][0]
602  && tess.indexes[indexes + k + 1] == i + edgeVerts[nums[j]][1])
603  {
604  break;
605  }
606  }
607 
608  if (k == 5)
609  {
610  VectorMA(mid[j], l, minor, v1);
611  VectorMA(mid[j], -l, minor, v2);
612  }
613  else
614  {
615  VectorMA(mid[j], -l, minor, v1);
616  VectorMA(mid[j], l, minor, v2);
617  }
618  }
619  }
