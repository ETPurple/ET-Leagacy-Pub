132 void CG_ParseFireteams()
133 {
134  int i, j;
135  char *s;
136  const char *p;
137  int clnts[2];
138 
139  for (i = 0; i < MAX_CLIENTS; i++)
140  {
141  cgs.clientinfo[i].fireteamData = NULL;
142  }
143 
144  for (i = 0; i < MAX_FIRETEAMS; i++)
145  {
146  char hexbuffer[11] = "0x00000000";
147 
148  p = CG_ConfigString(CS_FIRETEAMS + i);
149 
150  j = atoi(Info_ValueForKey(p, "id"));
151  if (j == -1)
152  {
153  cg.fireTeams[i].inuse = qfalse;
154  continue;
155  }
156  else
157  {
158  cg.fireTeams[i].inuse = qtrue;
159  cg.fireTeams[i].ident = j;
160  }
161 
162  s = Info_ValueForKey(p, "l");
163  cg.fireTeams[i].leader = atoi(s);
164 
165  s = Info_ValueForKey(p, "p");
166  cg.fireTeams[i].priv = (qboolean)atoi(s);
167 
168  s = Info_ValueForKey(p, "c");
169  Q_strncpyz(hexbuffer + 2, s, 9);
170  sscanf(hexbuffer, "%x", &clnts[1]);
171  Q_strncpyz(hexbuffer + 2, s + 8, 9);
172  sscanf(hexbuffer, "%x", &clnts[0]);
173 
174  for (j = 0; j < cgs.maxclients; j++)
175  {
176  if (COM_BitCheck(clnts, j))
177  {
178  cg.fireTeams[i].joinOrder[j] = qtrue;
179  cgs.clientinfo[j].fireteamData = &cg.fireTeams[i];
180  //CG_Printf("%s\n", cgs.clientinfo[j].name);
181  }
182  else
183  {
184  cg.fireTeams[i].joinOrder[j] = qfalse;
185  }
186  }
187  }
188 
189  CG_SortClientFireteam();
190 }
