50 static char *CommaParse(char **data_p)
51 {
52  int c = 0, len = 0;
53  char *data = *data_p;
54  static char com_token[MAX_TOKEN_CHARS];
55 
56  com_token[0] = 0;
57 
58  // make sure incoming data is valid
59  if (!data)
60  {
61  *data_p = NULL;
62  return com_token;
63  }
64 
65  while (1)
66  {
67  // skip whitespace
68  while ((c = *data) <= ' ')
69  {
70  if (!c)
71  {
72  break;
73  }
74  data++;
75  }
76 
77  c = *data;
78 
79  // skip double slash comments
80  if (c == '/' && data[1] == '/')
81  {
82  while (*data && *data != '\n')
83  data++;
84  }
85  // skip /* */ comments
86  else if (c == '/' && data[1] == '*')
87  {
88  while (*data && (*data != '*' || data[1] != '/'))
89  {
90  data++;
91  }
92  if (*data)
93  {
94  data += 2;
95  }
96  }
97  else
98  {
99  break;
100  }
101  }
102 
103  if (c == 0)
104  {
105  return "";
106  }
107 
108  // handle quoted strings
109  if (c == '\"')
110  {
111  data++;
112  while (1)
113  {
114  c = *data++;
115  if (c == '\"' || !c)
116  {
117  com_token[len] = 0;
118  *data_p = (char *)data;
119  return com_token;
120  }
121  if (len < MAX_TOKEN_CHARS - 1)
122  {
123  com_token[len] = c;
124  len++;
125  }
126  }
127  }
128 
129  // parse a regular word
130  do
131  {
132  if (len < MAX_TOKEN_CHARS - 1)
133  {
134  com_token[len] = c;
135  len++;
136  }
137  data++;
138  c = *data;
139  }
140  while (c > 32 && c != ',');
141 
142  if (len == MAX_TOKEN_CHARS)
143  {
144  //Ren_Print ("Token exceeded %i chars, discarded.\n", MAX_TOKEN_CHARS);
145  len = 0;
146  }
147  com_token[len] = 0;
148 
149  *data_p = (char *)data;
150  return com_token;
151 }

343  return 0;
344  }
345 
346  tr.numSkins++;
347  skin = (skin_t *)ri.Hunk_Alloc(sizeof(skin_t), h_low);
348  tr.skins[hSkin] = skin;

352

354  while (text_p && *text_p)

365  Q_strlwr(surfName);
366

370  }
371

373  {
374  continue;
375  }
376 
377  if (!Q_stricmpn(token, "md3_", 4))
378  {
379  if (skin->numModels >= MAX_PART_MODELS)
380  {
381  Ren_Warning("WARNING: Ignoring models in '%s', the max is %d!\n", name, MAX_PART_MODELS);
382  break;
383  }
384 
385  // this is specifying a model
386  model = skin->models[skin->numModels] = ri.Hunk_Alloc(sizeof(skinModel_t), h_low);
387  Q_strncpyz(model->type, token, sizeof(model->type));
388  model->hash = Com_HashKey(model->type, sizeof(model->type));
389 
390  // get the model name
391  token = CommaParse(&text_p);
392 
393  Q_strncpyz(model->model, token, sizeof(model->model));
394 
395  skin->numModels++;
396  continue;
397  }
398 
399  // parse the shader name
400  token = CommaParse(&text_p);
401 
402  if (skin->numSurfaces < MAX_SKIN_SURFACES)
403  {
404  surf = &parseSurfaces[skin->numSurfaces];
405  Q_strncpyz(surf->name, surfName, sizeof(surf->name));
406  // FIXME: bspSurface not not have ::hash yet
407  //surf->hash = Com_HashKey(surf->name, sizeof(surf->name));
408  surf->shader = R_FindShader(token, SHADER_3D_DYNAMIC, qtrue);
409  skin->numSurfaces++;
410  }
411  totalSurfaces++;
412  }
413 
414  ri.FS_FreeFile(text);
415 
416  if (totalSurfaces > MAX_SKIN_SURFACES)
417  {
418  ri.Printf(PRINT_WARNING, "WARNING: Ignoring excess surfaces (found %d, max is %d) in skin '%s'!\n",
419  totalSurfaces, MAX_SKIN_SURFACES, name);
420  }
421 
422  // never let a skin have 0 shaders
423  if (skin->numSurfaces == 0)
424  {
425  return 0; // use default skin
426  }
427 
428  // copy surfaces to skin
429  skin->surfaces = ri.Hunk_Alloc(skin->numSurfaces * sizeof(skinSurface_t), h_low);
430  Com_Memcpy(skin->surfaces, parseSurfaces, skin->numSurfaces * sizeof(skinSurface_t));
431 
432  return hSkin;
433 }
