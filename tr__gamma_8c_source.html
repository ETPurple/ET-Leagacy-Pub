67 static void R_BuildGammaProgram(void)
68 {
69  GLint compiled;
70 
71  gammaProgram.vertexShader = glCreateShaderObjectARB(GL_VERTEX_SHADER_ARB);
72  gammaProgram.fragmentShader = glCreateShaderObjectARB(GL_FRAGMENT_SHADER_ARB);
73 
74  glShaderSourceARB(gammaProgram.vertexShader, 1, (const GLcharARB **)&simpleGammaVert, NULL);
75  glShaderSourceARB(gammaProgram.fragmentShader, 1, (const GLcharARB **)&simpleGammaFrag, NULL);
76 
77  glCompileShaderARB(gammaProgram.vertexShader);
78  glCompileShaderARB(gammaProgram.fragmentShader);
79 
80  glGetObjectParameterivARB(gammaProgram.vertexShader, GL_COMPILE_STATUS, &compiled);
81  if (!compiled)
82  {
83  GLint blen = 0;
84  GLsizei slen = 0;
85 
86  glGetShaderiv(gammaProgram.vertexShader, GL_INFO_LOG_LENGTH, &blen);
87  if (blen > 1)
88  {
89  GLchar *compiler_log;
90 
91  compiler_log = (GLchar *) Com_Allocate(blen);
92 
93  glGetInfoLogARB(gammaProgram.vertexShader, blen, &slen, compiler_log);
94  Ren_Fatal("Failed to compile the gamma vertex shader reason: %s\n", compiler_log);
95  }
96  else
97  {
98  Ren_Fatal("Failed to compile the gamma vertex shader\n");
99  }
100  }
101 
102  glGetObjectParameterivARB(gammaProgram.fragmentShader, GL_COMPILE_STATUS, &compiled);
103  if (!compiled)
104  {
105  Ren_Fatal("Failed to compile the gamma fragment shader\n");
106  }
107 
108  gammaProgram.program = glCreateProgramObjectARB();
109  if (!gammaProgram.program)
110  {
111  Ren_Fatal("Failed to create program\n");
112  }
113 
114  glAttachObjectARB(gammaProgram.program, gammaProgram.vertexShader);
115  glAttachObjectARB(gammaProgram.program, gammaProgram.fragmentShader);
116 
117  glLinkProgramARB(gammaProgram.program);
118 
119  glGetObjectParameterivARB(gammaProgram.program, GL_OBJECT_LINK_STATUS_ARB, &compiled);
120 
121  if (!compiled)
122  {
123  Ren_Fatal("Failed to link gamma shaders\n");
124  }
125 
126  glUseProgramObjectARB(gammaProgram.program);
127 
128  gammaProgram.currentMapUniform = glGetUniformLocation(gammaProgram.program, "u_CurrentMap");
129  gammaProgram.gammaUniform = glGetUniformLocation(gammaProgram.program, "u_gamma");
130 
131  glUseProgramObjectARB(0);
132 }

137 void R_ScreenGamma(void)

142 
143  qglActiveTextureARB(GL_TEXTURE0_ARB);

145 
146  GL_Bind(screenImage);
147  // We will copy the current buffer into the screenImage
148  qglCopyTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA8, 0, 0, glConfig.vidWidth, glConfig.vidHeight, 0);
149 
150  qglTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
151  qglTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
152 
153  qglTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
154  qglTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
155 
156  glUniform1f(gammaProgram.gammaUniform, r_gamma->value);
157 
158  // Draw a simple quad, We could have done this in the GLSL code directly but that is version 130 upwards
159  // and we want to be sure that R1 runs even with a toaster.
160  glBegin(GL_QUADS);
161  {
162  glTexCoord2f(0.0, 0.0);
163  glVertex3f(-1.0f, -1.0f, 0.0f);
164  glTexCoord2f(1.0, 0.0);
165  glVertex3f(1.0f, -1.0f, 0.0f);
166  glTexCoord2f(1.0, 1.0);
167  glVertex3f(1.0f, 1.0f, 0.0f);
168  glTexCoord2f(0.0, 1.0);

172 
173  glUseProgramObjectARB(0);
