359  {

367

384  }

398  {

400  if (w->effects & WFX_SCROLLUP)
401  {

403  {

419  else

421  w->state = WSTATE_COMPLETE;
422  }
423  }
424  }
425  else if (w->state == WSTATE_SHUTDOWN)
426  {
427  tmp = w->targetTime - t_offset;
428  if (w->effects & WFX_SCROLLUP)
429  {
430  if (tmp > 0)
431  {
432  y = (int)(w->curY + (SCREEN_HEIGHT - w->y) * t_offset / w->targetTime); //(100 * t_offset / w->targetTime) / 100;
433  }
434  if (tmp < 0 || y >= SCREEN_HEIGHT)
435  {

444  {
445  textColor[3] -= ((float)t_offset / (float)w->targetTime);

452  }
453  }

460 
461  x += 5;

476  }
477  }
478  }

480 #ifdef FEATURE_MULTIVIEW
481  // Wedge in MV info overlay
482  if (cg.mvTotalClients > 0 && fAllowMV)
483  {
484  CG_mvOverlayDisplay();

486 #endif
487

491
