549 void GL_State(uint32_t stateBits)
550 {
551  uint32_t diff = stateBits ^ glState.glStateBits;
552 
553  if (!diff)
554  {
555  return;
556  }
557 
558  // check depthFunc bits
559  if (diff & GLS_DEPTHFUNC_BITS)
560  {
561  switch (stateBits & GLS_DEPTHFUNC_BITS)
562  {
563  default:
564  GL_DepthFunc(GL_LEQUAL);
565  break;
566  case GLS_DEPTHFUNC_LESS:
567  GL_DepthFunc(GL_LESS);
568  break;
569  case GLS_DEPTHFUNC_EQUAL:
570  GL_DepthFunc(GL_EQUAL);
571  break;
572  }
573  }
574 
575  // check blend bits
576  if (diff & (GLS_SRCBLEND_BITS | GLS_DSTBLEND_BITS))
577  {
578  GLenum srcFactor, dstFactor;
579 
580  if (stateBits & (GLS_SRCBLEND_BITS | GLS_DSTBLEND_BITS))
581  {
582  switch (stateBits & GLS_SRCBLEND_BITS)
583  {
584  case GLS_SRCBLEND_ZERO:
585  srcFactor = GL_ZERO;
586  break;
587  case GLS_SRCBLEND_ONE:
588  srcFactor = GL_ONE;
589  break;
590  case GLS_SRCBLEND_DST_COLOR:
591  srcFactor = GL_DST_COLOR;
592  break;
593  case GLS_SRCBLEND_ONE_MINUS_DST_COLOR:
594  srcFactor = GL_ONE_MINUS_DST_COLOR;
595  break;
596  case GLS_SRCBLEND_SRC_ALPHA:
597  srcFactor = GL_SRC_ALPHA;
598  break;
599  case GLS_SRCBLEND_ONE_MINUS_SRC_ALPHA:
600  srcFactor = GL_ONE_MINUS_SRC_ALPHA;
601  break;
602  case GLS_SRCBLEND_DST_ALPHA:
603  srcFactor = GL_DST_ALPHA;
604  break;
605  case GLS_SRCBLEND_ONE_MINUS_DST_ALPHA:
606  srcFactor = GL_ONE_MINUS_DST_ALPHA;
607  break;
608  case GLS_SRCBLEND_ALPHA_SATURATE:
609  srcFactor = GL_SRC_ALPHA_SATURATE;
610  break;
611  default:
612  srcFactor = GL_ONE; // to get warning to shut up
613  Ren_Drop("GL_State: invalid src blend state bits\n");
614  }

620  break;
621  case GLS_DSTBLEND_ONE:

626  break;

628  dstFactor = GL_ONE_MINUS_SRC_COLOR;
629  break;
630  case GLS_DSTBLEND_SRC_ALPHA:
631  dstFactor = GL_SRC_ALPHA;
632  break;
633  case GLS_DSTBLEND_ONE_MINUS_SRC_ALPHA:
634  dstFactor = GL_ONE_MINUS_SRC_ALPHA;
635  break;
636  case GLS_DSTBLEND_DST_ALPHA:
637  dstFactor = GL_DST_ALPHA;
638  break;
639  case GLS_DSTBLEND_ONE_MINUS_DST_ALPHA:
640  dstFactor = GL_ONE_MINUS_DST_ALPHA;
641  break;
642  default:
643  dstFactor = GL_ONE; // to get warning to shut up
644  Ren_Drop("GL_State: invalid dst blend state bits\n");
645  }
646 
647  glEnable(GL_BLEND);
648  GL_BlendFunc(srcFactor, dstFactor);
649  }
650  else
651  {
652  glDisable(GL_BLEND);
653  }
654  }
