675  {
676  break;
677  }
678  }

1221 static qboolean SV_WriteDownloadToClient(client_t *cl, msg_t *msg)
1222 {
1223  int curindex;
1224 
1225  if (!*cl->downloadName)
1226  {
1227  return qfalse; // Nothing being downloaded
1228  }
1229  if (cl->bWWWing)
1230  {
1231  return qfalse; // The client acked and is downloading with ftp/http
1232  }
1233 
1234  // CVE-2006-2082: validate the download against the list of pak files
1235  if (!FS_VerifyPak(cl->downloadName))
1236  {
1237  // will drop the client and leave it hanging on the other side. good for him
1238  SV_DropClient(cl, "illegal download request");
1239  return qfalse;
1240  }
1241 
1242  // set up the file to be downloaded
1243  if (!cl->download && SV_SetupDownloadFile(cl, msg))
1244  {
1245  return qtrue;
1246  }
1247 
1248  // Perform any reads that we need to
1249  while (cl->downloadCurrentBlock - cl->downloadClientBlock < MAX_DOWNLOAD_WINDOW && cl->downloadSize != cl->downloadCount)
1250  {
1251  curindex = (cl->downloadCurrentBlock % MAX_DOWNLOAD_WINDOW);
1252 
1253  if (!cl->downloadBlocks[curindex])
1254  {
1255  cl->downloadBlocks[curindex] = Z_Malloc(MAX_DOWNLOAD_BLKSIZE);
1256  }
1257 
1258  cl->downloadBlockSize[curindex] = FS_Read(cl->downloadBlocks[curindex], MAX_DOWNLOAD_BLKSIZE, cl->download);
1259 
1260  if (cl->downloadBlockSize[curindex] < 0)
1261  {
1262  // EOF right now
1263  cl->downloadCount = cl->downloadSize;
1264  break;
1265  }
1266 
1267  cl->downloadCount += cl->downloadBlockSize[curindex];
1268 
1269  // Load in next block
1270  cl->downloadCurrentBlock++;
1271  }
1272 
1273  // Check to see if we have eof condition and add the EOF block
1274  if (cl->downloadCount == cl->downloadSize && !cl->downloadEOF &&
1275  cl->downloadCurrentBlock - cl->downloadClientBlock < MAX_DOWNLOAD_WINDOW)
1276  {
1277 
1278  cl->downloadBlockSize[cl->downloadCurrentBlock % MAX_DOWNLOAD_WINDOW] = 0;
1279  cl->downloadCurrentBlock++;
1280 
1281  cl->downloadEOF = qtrue; // We have added the EOF block
1282  }
1283 
1284  if (cl->downloadClientBlock == cl->downloadCurrentBlock)
1285  {
1286  return qfalse; // Nothing to transmit
1287  }
1288 
1289  // Write out the next section of the file, if we have already reached our window,
1290  // automatically start retransmitting
1291  if (cl->downloadXmitBlock == cl->downloadCurrentBlock)
1292  {
1293  // We have transmitted the complete window, should we start resending?
1294  if (svs.time - cl->downloadSendTime > 1000)
1295  {
1296  cl->downloadXmitBlock = cl->downloadClientBlock;
1297  }
1298  else
1299  {
1300  return qfalse;
1301  }
1302  }
1303 
1304  // Send current block
1305  curindex = (cl->downloadXmitBlock % MAX_DOWNLOAD_WINDOW);
1306 
1307  MSG_WriteByte(msg, svc_download);
1308  MSG_WriteShort(msg, cl->downloadXmitBlock);
1309 
1310  // block zero is special, contains file size
1311  if (cl->downloadXmitBlock == 0)
1312  {
1313  MSG_WriteLong(msg, cl->downloadSize);
1314  }
1315 
1316  MSG_WriteShort(msg, cl->downloadBlockSize[curindex]);
1317 
1318  // Write the block
1319  if (cl->downloadBlockSize[curindex])
1320  {
1321  MSG_WriteData(msg, cl->downloadBlocks[curindex], cl->downloadBlockSize[curindex]);
1322  }
1323 
1324  Com_DPrintf("clientDownload: %d : writing block %d\n", (int)(cl - svs.clients), cl->downloadXmitBlock);
1325 
1326  // Move on to the next block
1327  // It will get sent with next snap shot. The rate will keep us in line.
1328  cl->downloadXmitBlock++;
1329  cl->downloadSendTime = svs.time;
1330 
1331  return qtrue;
1332 }
