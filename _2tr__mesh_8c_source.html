44 static void R_CullMDV(mdvModel_t *model, trRefEntity_t *ent)
45 {
46  int i;
47  // compute frame pointers
48  mdvFrame_t *newFrame = model->frames + ent->e.frame;
49  mdvFrame_t *oldFrame = model->frames + ent->e.oldframe;
50 
51  // calculate a bounding box in the current coordinate system
52  for (i = 0; i < 3; i++)
53  {
54  ent->localBounds[0][i] =
55  oldFrame->bounds[0][i] < newFrame->bounds[0][i] ? oldFrame->bounds[0][i] : newFrame->bounds[0][i];
56  ent->localBounds[1][i] =
57  oldFrame->bounds[1][i] > newFrame->bounds[1][i] ? oldFrame->bounds[1][i] : newFrame->bounds[1][i];
58  }
59 
60  // setup world bounds for intersection tests
61  R_SetupEntityWorldBounds(ent);
62 
63  // cull bounding sphere ONLY if this is not an upscaled entity
64  if (!ent->e.nonNormalizedAxes)
65  {
66  if (ent->e.frame == ent->e.oldframe)
67  {
68  switch (R_CullLocalPointAndRadius(newFrame->localOrigin, newFrame->radius))
69  {
70  case CULL_OUT:
71  tr.pc.c_sphere_cull_mdx_out++;
72  ent->cull = CULL_OUT;
73  return;
74  case CULL_IN:
75  tr.pc.c_sphere_cull_mdx_in++;
76  ent->cull = CULL_IN;
77  return;
78  case CULL_CLIP:
79  tr.pc.c_sphere_cull_mdx_clip++;
80  break;
81  }
82  }
83  else
84  {
85  int sphereCull, sphereCullB;
86 
87  sphereCull = R_CullLocalPointAndRadius(newFrame->localOrigin, newFrame->radius);
88  if (newFrame == oldFrame)
89  {
90  sphereCullB = sphereCull;
91  }
92  else
93  {
94  sphereCullB = R_CullLocalPointAndRadius(oldFrame->localOrigin, oldFrame->radius);
95  }
96 
97  if (sphereCull == sphereCullB)
98  {
99  if (sphereCull == CULL_OUT)
100  {
101  tr.pc.c_sphere_cull_mdx_out++;
102  ent->cull = CULL_OUT;
103  return;
104  }
105  else if (sphereCull == CULL_IN)
106  {
107  tr.pc.c_sphere_cull_mdx_in++;
108  ent->cull = CULL_IN;
109  return;
110  }
111  else
112  {
113  tr.pc.c_sphere_cull_mdx_clip++;
114  }
115  }
116  }
117  }
118 
119  switch (R_CullBox(ent->worldBounds))
120  {
121  case CULL_IN:
122  tr.pc.c_box_cull_mdx_in++;
123  ent->cull = CULL_IN;
124  return;
125  case CULL_CLIP:
126  tr.pc.c_box_cull_mdx_clip++;
127  ent->cull = CULL_CLIP;
128  return;
129  case CULL_OUT:
130  default:
131  tr.pc.c_box_cull_mdx_out++;
132  ent->cull = CULL_OUT;
133  return;
134  }
135 }

142 int R_ComputeLOD(trRefEntity_t *ent)
143 {
144  float radius;
145  float flod, lodscale;
146  float projectedRadius;
147  mdvFrame_t *frame;
148  int lod;
149 
150  if (tr.currentModel->numLods < 2)
151  {
152  // model has only 1 LOD level, skip computations and bias
153  lod = 0;
154  }
155  else
156  {
157  // multiple LODs exist, so compute projected bounding sphere
158  // and use that as a criteria for selecting LOD
159 
160  frame = tr.currentModel->mdv[0]->frames;
161  frame += ent->e.frame;
162 
163  radius = RadiusFromBounds(frame->bounds[0], frame->bounds[1]);
164 
165  if ((projectedRadius = R_ProjectRadius(radius, ent->e.origin)) != 0.f)
166  {
167  lodscale = r_lodScale->value;
168  if (lodscale > 20)
169  {
170  lodscale = 20;
171  }
172  flod = 1.0f - projectedRadius * lodscale;
173  }
174  else
175  {
176  // object intersects near view plane, e.g. view weapon
177  flod = 0;
178  }
179 
180  flod *= tr.currentModel->numLods;
181  lod = Q_ftol(flod);
182 
183  if (lod < 0)
184  {
185  lod = 0;
186  }
187  else if (lod >= tr.currentModel->numLods)
188  {
189  lod = tr.currentModel->numLods - 1;
190  }
191  }
192 
193  lod += r_lodBias->integer;
194 
195  if (lod >= tr.currentModel->numLods)
196  {
197  lod = tr.currentModel->numLods - 1;
198  }
199  if (lod < 0)
200  {
201  lod = 0;
202  }
203 
204  return lod;
205 }

213 static shader_t *GetMDVSurfaceShader(const trRefEntity_t *ent, mdvSurface_t *mdvSurface)
214 {
215  shader_t *shader = 0;
216 
217  if (ent->e.customShader)
218  {
219  shader = R_GetShaderByHandle(ent->e.customShader);
220  }
221  else if (ent->e.customSkin > 0 && ent->e.customSkin < tr.numSkins)
222  {
223  skin_t *skin;
224  int j;
225 
226  skin = R_GetSkinByHandle(ent->e.customSkin);
227 
228  // match the surface name to something in the skin file
229  shader = tr.defaultShader;
230  for (j = 0; j < skin->numSurfaces; j++)
231  {
232  // the names have both been lowercased
233  if (!strcmp(skin->surfaces[j].name, mdvSurface->name))
234  {
235  shader = skin->surfaces[j].shader;
236  break;
237  }
238  }
239  if (shader == tr.defaultShader)
240  {
241  Ren_Developer("WARNING: no shader for surface %s in skin %s\n", mdvSurface->name, skin->name);
242  }
243  else if (shader->defaultShader)
244  {
245  Ren_Developer("WARNING: shader %s in skin %s not found\n", shader->name, skin->name);
246  }
247  }
248  else
249  {
250  shader = mdvSurface->shader;
251  }
252 
253  return shader;
254 }

260 void R_AddMDVSurfaces(trRefEntity_t *ent)
261 {
262  int i;
263  mdvModel_t *model = 0;
264  mdvSurface_t *mdvSurface = 0;
265  shader_t *shader = 0;
266  int lod;
267  qboolean personalModel;
268  int fogNum;
269 
270  // don't add third_person objects if not in a portal
271  personalModel = (ent->e.renderfx & RF_THIRD_PERSON) && !tr.viewParms.isPortal;
272 
273  if (ent->e.renderfx & RF_WRAP_FRAMES)
274  {
275  ent->e.frame %= tr.currentModel->mdv[0]->numFrames;
276  ent->e.oldframe %= tr.currentModel->mdv[0]->numFrames;
277  }
278 
279  // compute LOD
280  if (ent->e.renderfx & RF_FORCENOLOD)
281  {
282  lod = 0;
283  }
284  else
285  {
286  lod = R_ComputeLOD(ent);
287  }
288 
289  // Validate the frames so there is no chance of a crash.
290  // This will write directly into the entity structure, so
291  // when the surfaces are rendered, they don't need to be
292  // range checked again.
293  if ((ent->e.frame >= tr.currentModel->mdv[lod]->numFrames)
294  || (ent->e.frame < 0) || (ent->e.oldframe >= tr.currentModel->mdv[lod]->numFrames) || (ent->e.oldframe < 0))
295  {
296  //Only spam if the lod level is 0 (lods usually don't have animation frames as they are only seen from a far)
297  if (lod == 0 || tr.currentModel->mdv[lod]->numFrames != 1)
298  {
299  Ren_Developer("R_AddMDVSurfaces: no such frame %d to %d for '%s' (%d)\n",
300  ent->e.oldframe, ent->e.frame, tr.currentModel->name, tr.currentModel->mdv[lod]->numFrames);
301  }
302  ent->e.frame = 0;
303  ent->e.oldframe = 0;
304  }
305 
306  model = tr.currentModel->mdv[lod];
307 
308  // cull the entire model if merged bounding box of both frames
309  // is outside the view frustum.
310  R_CullMDV(model, ent);
311  if (ent->cull == CULL_OUT)
312  {
313  return;
314  }
315 
316  // set up lighting now that we know we aren't culled
317  if (!personalModel || r_shadows->integer > SHADOWING_BLOB)
318  {
319  R_SetupEntityLighting(&tr.refdef, ent, NULL);
320  }
321 
322  // see if we are in a fog volume
323  fogNum = R_FogWorldBox(ent->worldBounds);
324 
325  // draw all surfaces
326  if (r_vboModels->integer && model->numVBOSurfaces)
327  {
328  int i;
329  srfVBOMDVMesh_t *vboSurface;
330  shader_t *shader;
331 
332  for (i = 0; i < model->numVBOSurfaces; i++)
333  {
334  vboSurface = model->vboSurfaces[i];
335  mdvSurface = vboSurface->mdvSurface;
336 
337  shader = GetMDVSurfaceShader(ent, mdvSurface);
338 
339  // don't add third_person objects if not viewing through a portal
340  if (!personalModel)
341  {
342  R_AddDrawSurf((surfaceType_t *)vboSurface, shader, LIGHTMAP_NONE, fogNum);
343  }
344  }
345  }
346  else
347  {
348  for (i = 0, mdvSurface = model->surfaces; i < model->numSurfaces; i++, mdvSurface++)
349  {
350  shader = GetMDVSurfaceShader(ent, mdvSurface);
351 
352  // we will add shadows even if the main object isn't visible in the view
353 
354  // don't add third_person objects if not viewing through a portal
355  if (!personalModel)
356  {
357  R_AddDrawSurf((surfaceType_t *)mdvSurface, shader, LIGHTMAP_NONE, fogNum);
358  }
359  }
360  }
361 }

368 void R_AddMDVInteractions(trRefEntity_t *ent, trRefLight_t *light)
369 {
370  int i;
371  mdvModel_t *model = 0;
372  mdvSurface_t *mdvSurface = 0;
373  shader_t *shader = 0;
374  int lod;
375  qboolean personalModel;
376  byte cubeSideBits;
377  interactionType_t iaType = IA_DEFAULT;
378 
379  // cull the entire model if merged bounding box of both frames
380  // is outside the view frustum and we don't care about proper shadowing
381  if (ent->cull == CULL_OUT)
382  {
383  if (r_shadows->integer <= SHADOWING_BLOB || light->l.noShadows)
384  {
385  return;
386  }
387  else
388  {
389  iaType = IA_SHADOWONLY;
390  }
391  }
392 
393  // avoid drawing of certain objects
394 #if defined(USE_REFENTITY_NOSHADOWID)
395  if (light->l.inverseShadows)
396  {
397  if (iaType != IA_LIGHTONLY && (light->l.noShadowID && (light->l.noShadowID != ent->e.noShadowID)))
398  {
399  return;
400  }
401  }
402  else
403  {
404  if (iaType != IA_LIGHTONLY && (light->l.noShadowID && (light->l.noShadowID == ent->e.noShadowID)))
405  {
406  return;
407  }
408  }
409 #endif
410 
411  // don't add third_person objects if not in a portal
412  personalModel = (ent->e.renderfx & RF_THIRD_PERSON) && !tr.viewParms.isPortal;
413 
414  // compute LOD
415  lod = R_ComputeLOD(ent);
416 
417  model = tr.currentModel->mdv[lod];
418 
419  // do a quick AABB cull
420  if (!BoundsIntersect(light->worldBounds[0], light->worldBounds[1], ent->worldBounds[0], ent->worldBounds[1]))
421  {
422  tr.pc.c_dlightSurfacesCulled += model->numSurfaces;
423  return;
424  }
425 
426  // do a more expensive and precise light frustum cull
427  if (!r_noLightFrustums->integer)
428  {
429  if (R_CullLightWorldBounds(light, ent->worldBounds) == CULL_OUT)
430  {
431  tr.pc.c_dlightSurfacesCulled += model->numSurfaces;
432  return;
433  }
434  }
435 
436  cubeSideBits = R_CalcLightCubeSideBits(light, ent->worldBounds);
437 
438  // generate interactions with all surfaces
439  if (r_vboModels->integer && model->numVBOSurfaces)
440  {
441  // new brute force method: just render everthing with static VBOs
442  int i;
443  srfVBOMDVMesh_t *vboSurface;
444  shader_t *shader;
445 
446 
447  // static VBOs are fine for lighting and shadow mapping
448  for (i = 0; i < model->numVBOSurfaces; i++)
449  {
450  vboSurface = model->vboSurfaces[i];
451  mdvSurface = vboSurface->mdvSurface;
452 
453  shader = GetMDVSurfaceShader(ent, mdvSurface);
454 
455  // skip all surfaces that don't matter for lighting only pass
456  if (shader->isSky || (!shader->interactLight && shader->noShadows))
457  {
458  continue;
459  }
460 
461  // we will add shadows even if the main object isn't visible in the view
462 
463  // don't add third_person objects if not viewing through a portal
464  if (!personalModel)
465  {
466  R_AddLightInteraction(light, (surfaceType_t *)vboSurface, shader, cubeSideBits, iaType);
467  tr.pc.c_dlightSurfaces++;
468  }
469  }
470  }
471  else
472  {
473  for (i = 0, mdvSurface = model->surfaces; i < model->numSurfaces; i++, mdvSurface++)
474  {
475  shader = GetMDVSurfaceShader(ent, mdvSurface);
476 
477  // skip all surfaces that don't matter for lighting only pass
478  if (shader->isSky || (!shader->interactLight && shader->noShadows))
479  {
480  continue;
481  }
482 
483  // we will add shadows even if the main object isn't visible in the view
484 
485  // don't add third_person objects if not viewing through a portal
486  if (!personalModel)
487  {
488  R_AddLightInteraction(light, (surfaceType_t *)mdvSurface, shader, cubeSideBits, iaType);
489  tr.pc.c_dlightSurfaces++;
490  }
491  }
492  }
